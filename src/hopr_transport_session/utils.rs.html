48/// Convenience function to copy data in both directions between a [`Session`](crate::HoprSession) and arbitrary
49/// async IO stream.
50/// This function is only available with Tokio and will panic with other runtimes.
51///
52/// The `abort_stream` will terminate the transfer from the `stream` side, i.e.:
53/// 1. Initiates graceful shutdown of `stream`
54/// 2. Once done, initiates a graceful shutdown of `session`
55/// 3. The function terminates, returning the number of bytes transferred in both directions.
56#[cfg(feature = "runtime-tokio")]
57pub async fn transfer_session<S>(
58    session: &mut crate::HoprSession,
59    stream: &mut S,
60    max_buffer: usize,
61    abort_stream: Option<futures::future::AbortRegistration>,
62) -> std::io::Result<(usize, usize)>
63where
64    S: tokio::io::AsyncRead + tokio::io::AsyncWrite + Unpin,
65{
66    // We can use equally sized buffer for both directions
67    tracing::debug!(
68        session_id = ?session.id(),
69        egress_buffer = max_buffer,
70        ingress_buffer = max_buffer,
71        "session buffers"
72    );
73
74    if let Some(abort_stream) = abort_stream {
75        // We only allow aborting from the "stream" side, not from the "session side"
76        // This is useful for UDP-like streams on the "stream" side, which cannot be terminated
77        // by a signal from outside (e.g.: for TCP sockets such signal is socket closure).
78        let (_, dummy) = futures::future::AbortHandle::new_pair();
79        hopr_network_types::utils::copy_duplex_abortable(
80            session,
81            stream,
82            (max_buffer, max_buffer),
83            (dummy, abort_stream),
84        )
85        .await
86        .map(|(a, b)| (a as usize, b as usize))
87    } else {
88        hopr_network_types::utils::copy_duplex(session, stream, (max_buffer, max_buffer))
89            .await
90            .map(|(a, b)| (a as usize, b as usize))
91    }
92}

94/// This function will use the given generator to generate an initial seeding key.
95/// It will check whether the given cache already contains a value for that key, and if not,
96/// calls the generator (with the previous value) to generate a new seeding key and retry.
97/// The function either finds a suitable free slot, inserting the `value` and returns the found key,
98/// or terminates with `None` when `gen` returns the initial seed again.
99pub(crate) async fn insert_into_next_slot<K, V, F>(
100    cache: &moka::future::Cache<K, V>,
101    generator: F,
102    value: V,
103) -> Option<K>
104where
105    K: Copy + std::hash::Hash + Eq + Send + Sync + 'static,
106    V: Clone + Send + Sync + 'static,
107    F: Fn(Option<K>) -> K,
108{
109    cache.run_pending_tasks().await;
110
111    let initial = generator(None);
112    let mut next = initial;
113    loop {
114        let insertion_result = cache
115            .entry(next)
116            .and_try_compute_with(|e| {
117                if e.is_none() {
118                    futures::future::ok::<_, ()>(moka::ops::compute::Op::Put(value.clone()))
119                } else {
120                    futures::future::ok::<_, ()>(moka::ops::compute::Op::Nop)
121                }
122            })
123            .await;
124
125        // If we inserted successfully, break the loop and return the insertion key
126        if let Ok(moka::ops::compute::CompResult::Inserted(_)) = insertion_result {
127            return Some(next);
128        }
129
130        // Otherwise, generate the next key
131        next = generator(Some(next));
132
133        // If generated keys made it to full loop, return failure
134        if next == initial {
135            return None;
136        }
137    }
138}

140pub(crate) fn spawn_keep_alive_stream<S>(
141    session_id: SessionId,
142    sender: S,
143    routing: DestinationRouting,
144) -> (SurbControllerWithCorrection, AbortHandle)
145where
146    S: futures::Sink<(DestinationRouting, ApplicationDataOut)> + Clone + Send + Sync + Unpin + 'static,
147    S::Error: std::error::Error + Send + Sync + 'static,
148{
149    let elem = HoprStartProtocol::KeepAlive(session_id.into());
150
151    // The stream is suspended until the caller sets a rate via the Controller
152    let controller = RateController::new(0, Duration::from_secs(1));
153
154    let (ka_stream, abort_handle) =
155        futures::stream::abortable(futures::stream::repeat(elem).rate_limit_with_controller(&controller));
156
157    let sender_clone = sender.clone();
158    let fwd_routing_clone = routing.clone();
159
160    // This task will automatically terminate once the returned abort handle is used.
161    debug!(%session_id, "spawning keep-alive stream");
162    hopr_async_runtime::prelude::spawn(
163        ka_stream
164            .map(move |msg| {
165                ApplicationData::try_from(msg)
166                    .map(|data| (fwd_routing_clone.clone(), ApplicationDataOut::with_no_packet_info(data)))
167            })
168            .map_err(TransportSessionError::from)
169            .try_for_each_concurrent(None, move |msg| {
170                let mut sender_clone = sender_clone.clone();
171                async move {
172                    sender_clone
173                        .send(msg)
174                        .await
175                        .map_err(|e| TransportSessionError::PacketSendingError(e.to_string()))
176                }
177            })
178            .then(move |res| {
179                match res {
180                    Ok(_) => tracing::trace!(
181                        component = "session",
182                        %session_id,
183                        task = "session keepalive",
184                        "background task finished"
185                    ),
186                    Err(error) => error!(%session_id, %error, "keep-alive stream failed"),
187                }
188                futures::future::ready(())
189            }),
190    );
191
192    // Currently, a keep-alive message can bear `HoprPacket::MAX_SURBS_IN_PACKET` SURBs,
193    // so the correction by this factor is applied.
194    (
195        SurbControllerWithCorrection(controller, HoprPacket::MAX_SURBS_IN_PACKET as u32),
196        abort_handle,
197    )
198}
