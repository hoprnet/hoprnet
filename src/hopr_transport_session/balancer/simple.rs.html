1use hopr_primitive_types::{prelude::SMA, sma::NoSumSMA};
2
3use crate::balancer::{BalancerControllerBounds, SurbBalancerController};

5/// Controller that uses the simple linear formula `limit * min(current / setpoint, 1.0)` to
6/// compute the control output.
7///
8/// The controller also allows optionally increasing the setpoint if its simple moving average
9/// over a given number of samples is above a certain threshold.
10/// See [`SimpleBalancerController::with_increasing_setpoint`] for details.
11///
12/// If the controller is constructed via the `Default` constructor, no setpoint increase takes place.
13#[derive(Clone, Debug, Default)]
14pub struct SimpleBalancerController {
15    bounds: BalancerControllerBounds,
16    increasing: Option<(f64, NoSumSMA<f64>)>,
17}
18
19impl SimpleBalancerController {
20    /// Constructs the controller with increasing setpoint.
21    ///
22    /// If the simple moving average of the `current / setpoint` ratio over `window_size` of samples
23    /// is greater than the `ratio_threshold + 1`, the setpoint adjusted by multiplying
24    /// by the value of the moving average.
25    ///
26    /// The given `ratio_threshold` must be between 0 and 1, otherwise it is clamped to this range.
27    /// The `window_size` must be greater or equal to 1, otherwise it is set to 1.
28    pub fn with_increasing_setpoint(ratio_threshold: f64, window_size: usize) -> Self {
29        Self {
30            bounds: BalancerControllerBounds::default(),
31            increasing: Some((ratio_threshold.clamp(0.0, 1.0), NoSumSMA::new(window_size.max(1)))),
32        }
33    }
34}
35
36impl SurbBalancerController for SimpleBalancerController {
37    fn bounds(&self) -> BalancerControllerBounds {
38        self.bounds
39    }
40
41    fn set_target_and_limit(&mut self, bounds: BalancerControllerBounds) {
42        self.bounds = bounds;
43    }
44
45    fn next_control_output(&mut self, current_buffer_level: u64) -> u64 {
46        let ratio = current_buffer_level as f64 / self.bounds.target() as f64;
47
48        if let Some((threshold, sma)) = self.increasing.as_mut() {
49            sma.push(ratio);
50            if let Some(avg) = sma.average().filter(|avg| *avg >= *threshold + 1.0) {
51                let new_setpoint = (avg * self.bounds.target() as f64).round() as u64;
52                let new_limit = (avg * self.bounds.output_limit() as f64).floor() as u64;
53                tracing::debug!(
54                    old_setpoint = self.bounds.target(),
55                    new_setpoint,
56                    old_limit = self.bounds.output_limit(),
57                    new_limit,
58                    avg,
59                    "setpoint increased"
60                );
61                self.bounds = BalancerControllerBounds::new(new_setpoint, new_limit);
62            }
63        }
64
65        (self.bounds.output_limit() as f64 * ratio.clamp(0.0, 1.0)).floor() as u64
66    }
67}

69#[cfg(test)]
70mod tests {
71    use super::*;

73    #[test]
74    fn test_simple_balancer() {
75        let mut controller = SimpleBalancerController::default();
76        controller.set_target_and_limit(BalancerControllerBounds::new(100, 100));
77        assert_eq!(100, controller.bounds.target());
78        assert_eq!(controller.next_control_output(10), 10);
79        assert_eq!(controller.next_control_output(100), 100);
80        assert_eq!(controller.next_control_output(101), 100);
81        assert_eq!(100, controller.bounds.target());
82    }
83
84    #[test_log::test]
85    fn test_simple_balance_with_increasing_setpoint() {
86        let mut controller = SimpleBalancerController::with_increasing_setpoint(0.2, 3);
87        controller.set_target_and_limit(BalancerControllerBounds::new(100, 100));
88        assert_eq!(100, controller.bounds.target());
89
90        assert_eq!(100, controller.next_control_output(101));
91        assert_eq!(100, controller.bounds.target());
92        assert_eq!(100, controller.next_control_output(120));
93        assert_eq!(100, controller.bounds.target());
94        assert_eq!(140, controller.next_control_output(200));
95        assert_eq!(140, controller.bounds.target());
96    }
