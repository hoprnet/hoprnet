5/// Controller that uses the simple linear formula `limit * min(current / setpoint, 1.0)` to
6/// compute the control output.
7///
8/// The controller also allows optionally increasing the setpoint if its simple moving average
9/// over a given number of samples is above a certain threshold.
10/// See [`SimpleBalancerController::with_increasing_setpoint`] for details.
11///
12/// If the controller is constructed via the `Default` constructor, no setpoint increase takes place.
13#[derive(Clone, Debug, Default)]
14pub struct SimpleBalancerController {
15    bounds: BalancerControllerBounds,
16    increasing: Option<(f64, NoSumSMA<f64>)>,
17}
18
19impl SimpleBalancerController {
20    /// Constructs the controller with increasing setpoint.
21    ///
22    /// If the simple moving average of the `current / setpoint` ratio over `window_size` of samples
23    /// is greater than the `ratio_threshold + 1`, the setpoint adjusted by multiplying
24    /// by the value of the moving average.
25    ///
26    /// The given `ratio_threshold` must be between 0 and 1, otherwise it is clamped to this range.
27    /// The `window_size` must be greater or equal to 1, otherwise it is set to 1.
28    pub fn with_increasing_setpoint(ratio_threshold: f64, window_size: usize) -> Self {
29        let window_size = window_size.max(1);
30        Self {
31            bounds: BalancerControllerBounds::default(),
32            increasing: Some((
33                ratio_threshold.clamp(0.0, 1.0),
34                NoSumSMA::new_with_samples(window_size, std::iter::repeat_n(1.0, window_size)),
35            )),
36        }
37    }
38}
39
40impl SurbBalancerController for SimpleBalancerController {
41    fn bounds(&self) -> BalancerControllerBounds {
42        self.bounds
43    }
44
45    fn set_target_and_limit(&mut self, bounds: BalancerControllerBounds) {
46        self.bounds = bounds;
47    }
48
49    fn next_control_output(&mut self, current_buffer_level: u64) -> u64 {
50        let ratio = current_buffer_level as f64 / self.bounds.target() as f64;
51
52        if let Some((threshold, sma)) = self.increasing.as_mut() {
53            sma.push(ratio);
54            if let Some(avg) = sma.average().filter(|avg| *avg >= *threshold + 1.0) {
55                let new_setpoint = (avg * self.bounds.target() as f64).round() as u64;
56                let new_limit = (avg * self.bounds.output_limit() as f64).floor() as u64;
57                tracing::debug!(
58                    old_setpoint = self.bounds.target(),
59                    new_setpoint,
60                    old_limit = self.bounds.output_limit(),
61                    new_limit,
62                    avg,
63                    "setpoint increased"
64                );
65                self.bounds = BalancerControllerBounds::new(new_setpoint, new_limit);
66            }
67        }
68
69        (self.bounds.output_limit() as f64 * ratio.clamp(0.0, 1.0)).floor() as u64
70    }
71}

74mod tests {

77    #[test]
78    fn test_simple_balancer() {
79        let mut controller = SimpleBalancerController::default();
80        controller.set_target_and_limit(BalancerControllerBounds::new(100, 100));
81        assert_eq!(100, controller.bounds.target());
82        assert_eq!(controller.next_control_output(10), 10);
83        assert_eq!(controller.next_control_output(100), 100);
84        assert_eq!(controller.next_control_output(101), 100);
85        assert_eq!(100, controller.bounds.target());
86    }
87
88    #[test_log::test]
89    fn test_simple_balance_with_increasing_setpoint() {
90        let mut controller = SimpleBalancerController::with_increasing_setpoint(0.2, 3);
91        controller.set_target_and_limit(BalancerControllerBounds::new(100, 100));
92        assert_eq!(100, controller.bounds.target());
93
94        assert_eq!(100, controller.next_control_output(101));
95        assert_eq!(100, controller.bounds.target());
96        assert_eq!(100, controller.next_control_output(120));
97        assert_eq!(100, controller.bounds.target());
98        assert_eq!(140, controller.next_control_output(200));
99        assert_eq!(140, controller.bounds.target());
100    }
