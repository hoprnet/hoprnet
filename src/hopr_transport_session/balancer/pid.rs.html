11/// Carries finite Proportional, Integral and Derivative controller gains for a PID controller.
12#[derive(Clone, Copy, Debug, PartialEq)]
13pub struct PidControllerGains(f64, f64, f64);
14
15impl PidControllerGains {
16    /// Creates PID controller gains, returns an error if the gains are not finite.
17    pub fn new(p: f64, i: f64, d: f64) -> errors::Result<Self> {
18        if p.is_finite() && i.is_finite() && d.is_finite() {
19            Ok(Self(p, i, d))
20        } else {
21            Err(SessionManagerError::Other("gains must be finite".into()).into())
22        }
23    }
24
25    /// Uses PID controller gains from the env variables or uses the defaults if not set.
26    pub fn from_env_or_default() -> Self {
27        let default = Self::default();
28        Self(
29            std::env::var("HOPR_BALANCER_PID_P_GAIN")
30                .ok()
31                .and_then(|v| f64::from_str(&v).ok())
32                .unwrap_or(default.0),
33            std::env::var("HOPR_BALANCER_PID_I_GAIN")
34                .ok()
35                .and_then(|v| f64::from_str(&v).ok())
36                .unwrap_or(default.1),
37            std::env::var("HOPR_BALANCER_PID_D_GAIN")
38                .ok()
39                .and_then(|v| f64::from_str(&v).ok())
40                .unwrap_or(default.2),
41        )
42    }
43
44    /// P gain.
45    #[inline]
46    pub fn p(&self) -> f64 {
47        self.0
48    }
49
50    /// I gain.
51    #[inline]
52    pub fn i(&self) -> f64 {
53        self.1
54    }
55
56    /// D gain.
57    #[inline]
58    pub fn d(&self) -> f64 {
59        self.2
60    }
61}
62
63// Safe to implement Eq, because the floats are finite
64impl Eq for PidControllerGains {}
65
66// Default coefficients for the PID controller
67// This might be tweaked in the future.
68const DEFAULT_P_GAIN: f64 = 0.6;
69const DEFAULT_I_GAIN: f64 = 0.7;
70const DEFAULT_D_GAIN: f64 = 0.2;
71
72impl Default for PidControllerGains {
73    fn default() -> Self {
74        Self(DEFAULT_P_GAIN, DEFAULT_I_GAIN, DEFAULT_D_GAIN)
75    }
76}

78impl TryFrom<(f64, f64, f64)> for PidControllerGains {
79    type Error = errors::TransportSessionError;
80
81    fn try_from(value: (f64, f64, f64)) -> Result<Self, Self::Error> {
82        Self::new(value.0, value.1, value.2)
83    }
84}

86/// Implementation of [`SurbBalancerController`] using a PID controller.
87#[derive(Clone, Copy, Debug)]
88pub struct PidBalancerController(Pid<f64>);
89
90impl PidBalancerController {
91    /// Creates new instance given the `setpoint`, `output_limit` and PID gains (P,I and D).
92    pub fn new(setpoint: u64, output_limit: u64, gains: PidControllerGains) -> Self {
93        let mut pid = Pid::new(setpoint as f64, output_limit as f64);
94        pid.p(gains.p(), output_limit as f64);
95        pid.i(gains.i(), output_limit as f64);
96        pid.d(gains.d(), output_limit as f64);
97        Self(pid)
98    }
99
100    /// Creates new instance with setpoint and output limit set to 0.
101    ///
102    /// Needs to be [reconfigured](SurbBalancerController::set_target_and_limit) in order to function
103    /// correctly.
104    pub fn from_gains(gains: PidControllerGains) -> Self {
105        Self::new(0, 0, gains)
106    }
107}
108
109impl Default for PidBalancerController {
110    /// The default instance does nothing unless [reconfigured](SurbBalancerController::set_target_and_limit).
111    fn default() -> Self {
112        Self::new(0, 0, PidControllerGains::default())
113    }
114}
115
116impl SurbBalancerController for PidBalancerController {
117    fn bounds(&self) -> BalancerControllerBounds {
118        BalancerControllerBounds::new(self.0.setpoint as u64, self.0.output_limit as u64)
119    }
120
121    fn set_target_and_limit(&mut self, bounds: BalancerControllerBounds) {
122        let mut pid = Pid::new(bounds.target() as f64, bounds.output_limit() as f64);
123        pid.p(self.0.kp, bounds.output_limit() as f64);
124        pid.i(self.0.ki, bounds.output_limit() as f64);
125        pid.d(self.0.kd, bounds.output_limit() as f64);
126        self.0 = pid;
127    }
128
129    fn next_control_output(&mut self, current_buffer_level: u64) -> u64 {
130        self.0.next_control_output(current_buffer_level as f64).output.max(0.0) as u64
131    }
