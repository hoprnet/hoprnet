1//! [`HoprSession`] object providing the session functionality over the HOPR transport
2//!
3//! The session proxies the user interactions with the transport to hide the
4//! advanced interactions and functionality.
5//!
6//! The [`SessionManager`] allows for automatic management of sessions via the Start protocol.

27/// Number of bytes that can be sent in a single Session protocol payload.
28///
29/// In other words, this is the effective payload capacity of a single Session segment.
30pub const SESSION_MTU: usize =
31    hopr_protocol_session::session_socket_mtu::<{ hopr_protocol_app::v1::ApplicationData::PAYLOAD_SIZE }>();

33/// Size of the HOPR SURB in bytes.

35/// This is the re-export of [`hopr_crypto_packet::HoprSurb::SIZE`].

38flagset::flags! {
39    /// Individual capabilities of a Session.
40    #[repr(u8)]
41    #[derive(PartialOrd, Ord, strum::EnumString, strum::Display, serde_repr::Serialize_repr, serde_repr::Deserialize_repr)]
42    pub enum Capability : u8 {
43        /// Frame segmentation.
44        Segmentation = 0b0000_1000,
45        /// Frame retransmission (ACK-based)
46        ///
47        /// Implies [`Segmentation`].
48        RetransmissionAck = 0b0000_1100,
49        /// Frame retransmission (NACK-based)
50        ///
51        /// Implies [`Segmentation`].
52        RetransmissionNack = 0b000_1010,
53        /// Disable packet buffering.
54        ///
55        /// Implies [`Segmentation`].
56        NoDelay = 0b0000_1001,
57        /// Disable SURB-based egress rate control.
58        ///
59        /// This applies only to the recipient of the Session (Exit).
60        NoRateControl = 0b0001_0000,
61    }
62}
63
64/// Set of Session [capabilities](Capability).
65pub type Capabilities = flagset::FlagSet<Capability>;

67/// Configuration for the session.
68///
69/// Relevant primarily for the client, since the server is only
70/// a reactive component in regard to the session concept.
71#[derive(Debug, PartialEq, Clone, smart_default::SmartDefault)]
72pub struct SessionClientConfig {
73    /// The forward path options for the session.
74    #[default(RoutingOptions::Hops(hopr_primitive_types::bounded::BoundedSize::MIN))]
75    pub forward_path_options: RoutingOptions,
76    /// The return path options for the session.
77    #[default(RoutingOptions::Hops(hopr_primitive_types::bounded::BoundedSize::MIN))]
78    pub return_path_options: RoutingOptions,
79    /// Capabilities offered by the session.
80    #[default(_code = "Capability::Segmentation.into()")]
81    pub capabilities: Capabilities,
82    /// Optional pseudonym used for the session. Mostly useful for testing only.
83    #[default(None)]
84    pub pseudonym: Option<hopr_internal_types::protocol::HoprPseudonym>,
85    /// Enable automatic SURB management for the Session.
86    #[default(Some(SurbBalancerConfig::default()))]
87    pub surb_management: Option<SurbBalancerConfig>,
88    /// If set, the maximum number of possible SURBs will always be sent with Session data packets (if they fit).
89    ///
90    /// This does not affect `KeepAlive` messages used with SURB balancing, as they will always
91    /// carry the maximum number of SURBs possible. Setting this to `true` will put additional CPU
92    /// pressure on the local node as it will generate the maximum number of SURBs for each data packet.
93    ///
94    /// Set this to `true` only when the underlying traffic is highly asymmetric.
95    ///
96    /// Default is `false`.
97    #[default(false)]
98    pub always_max_out_surbs: bool,
99}

121    #[test]
122    fn hopr_start_protocol_messages_must_fit_within_hopr_packet() -> anyhow::Result<()> {
123        let msg = HoprStartProtocol::StartSession(StartInitiation {
124            challenge: StartChallenge::MAX,
125            target: SessionTarget::TcpStream(SealedHost::Plain(
126                "example-of-a-very-very-long-second-level-name.on-a-very-very-long-domain-name.info:65530".parse()?,
127            )),
128            capabilities: Capabilities::full().into(),
129            additional_data: 0xffffffff,
130        });
131
132        assert!(
133            msg.encode()?.1.len() <= HoprPacket::PAYLOAD_SIZE,
134            "StartSession must fit within {}",
135            HoprPacket::PAYLOAD_SIZE
136        );
137
138        let msg = HoprStartProtocol::SessionEstablished(StartEstablished {
139            orig_challenge: StartChallenge::MAX,
140            session_id: SessionId::new(u64::MAX, HoprPseudonym::random()),
141        });
142
143        assert!(
144            msg.encode()?.1.len() <= HoprPacket::PAYLOAD_SIZE,
145            "SessionEstablished must fit within {}",
146            HoprPacket::PAYLOAD_SIZE
147        );
148
149        let msg = HoprStartProtocol::SessionError(StartErrorType {
150            challenge: StartChallenge::MAX,
151            reason: StartErrorReason::NoSlotsAvailable,
152        });
153
154        assert!(
155            msg.encode()?.1.len() <= HoprPacket::PAYLOAD_SIZE,
156            "SessionError must fit within {}",
157            HoprPacket::PAYLOAD_SIZE
158        );
159
160        let msg = HoprStartProtocol::KeepAlive(KeepAliveMessage {
161            session_id: SessionId::new(u64::MAX, HoprPseudonym::random()),
162            flags: 0xff,
163            additional_data: 0xffffffff,
164        });
165        assert!(
166            msg.encode()?.1.len() <= HoprPacket::PAYLOAD_SIZE,
167            "KeepAlive must fit within {}",
168            HoprPacket::PAYLOAD_SIZE
169        );
170
171        Ok(())
172    }

195    #[test]
196    fn hopr_start_protocol_message_keep_alive_message_should_allow_for_maximum_surbs() -> anyhow::Result<()> {
197        let msg = HoprStartProtocol::KeepAlive(KeepAliveMessage {
198            session_id: SessionId::new(u64::MAX, HoprPseudonym::random()),
199            flags: 0xff,
200            additional_data: 0xffffffff,
201        });
202        let len = msg.encode()?.1.len();
203        assert_eq!(
204            KeepAliveMessage::<SessionId>::MIN_SURBS_PER_MESSAGE,
205            HoprPacket::MAX_SURBS_IN_PACKET
206        );
207        assert!(
208            HoprPacket::max_surbs_with_message(len) >= HoprPacket::MAX_SURBS_IN_PACKET,
209            "Hopr KeepAlive message size ({}) must allow for at least {} SURBs in packet",
210            len,
211            HoprPacket::MAX_SURBS_IN_PACKET
212        );
213
214        Ok(())
215    }
