1//! [`Session`] object providing the session functionality over the HOPR transport
2//!
3//! The session proxies the user interactions with the transport to hide the
4//! advanced interactions and functionality.
5//!
6//! The [`SessionManager`] allows for automatic management of sessions via the Start protocol.

33flagset::flags! {
34    /// Individual capabilities of a Session.
35    #[repr(u8)]
36    #[derive(strum::EnumString, strum::Display)]
37    #[cfg_attr(feature = "serde", derive(serde_repr::Serialize_repr, serde_repr::Deserialize_repr))]
38    pub enum Capability : u8 {
39        /// Frame segmentation.
40        Segmentation = 0b1000,
41        /// Frame retransmission (ACK-based)
42        ///
43        /// Implies [`Segmentation`].
44        RetransmissionAck = 0b1100,
45        /// Frame retransmission (NACK-based)
46        ///
47        /// Implies [`Segmentation`].
48        RetransmissionNack = 0b1010,
49        /// Disable packet buffering
50        NoDelay = 0b0001,
51    }
52}
53
54/// Set of Session [capabilities](Capability).
55pub type Capabilities = flagset::FlagSet<Capability>;
56
57/// Configuration for the session.
58///
59/// Relevant primarily for the client, since the server is only
60/// a reactive component in regard to the session concept.
61#[derive(Debug, PartialEq, Clone, smart_default::SmartDefault)]
62pub struct SessionClientConfig {
63    /// The forward path options for the session.
64    #[default(RoutingOptions::Hops(hopr_primitive_types::bounded::BoundedSize::MIN))]
65    pub forward_path_options: RoutingOptions,
66    /// The return path options for the session.
67    #[default(RoutingOptions::Hops(hopr_primitive_types::bounded::BoundedSize::MIN))]
68    pub return_path_options: RoutingOptions,
69    /// Capabilities offered by the session.
70    #[default(_code = "Capability::Segmentation.into()")]
71    pub capabilities: Capabilities,
72    /// Optional pseudonym used for the session. Mostly useful for testing only.
73    #[default(None)]
74    pub pseudonym: Option<HoprPseudonym>,
75    /// Enable automatic SURB management for the Session.
76    #[default(Some(SurbBalancerConfig::default()))]
77    pub surb_management: Option<SurbBalancerConfig>,
78}
