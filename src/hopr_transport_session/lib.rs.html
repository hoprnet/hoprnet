1//! [`HoprSession`] object providing the session functionality over the HOPR transport
2//!
3//! The session proxies the user interactions with the transport to hide the
4//! advanced interactions and functionality.
5//!
6//! The [`SessionManager`] allows for automatic management of sessions via the Start protocol.

33/// Number of bytes that can be sent in a single Session protocol payload.
34///
35/// In other words, this is the effective payload capacity of a single Session segment.
36pub const SESSION_MTU: usize =
37    hopr_protocol_session::session_socket_mtu::<{ hopr_protocol_app::v1::ApplicationData::PAYLOAD_SIZE }>();
38
39/// Size of the HOPR SURB in bytes.
40///
41/// This is the re-export of [`hopr_crypto_packet::HoprSurb::SIZE`].
42pub const SURB_SIZE: usize = hopr_crypto_packet::HoprSurb::SIZE;

44flagset::flags! {
45    /// Individual capabilities of a Session.
46    #[repr(u8)]
47    #[derive(PartialOrd, Ord, strum::EnumString, strum::Display, serde_repr::Serialize_repr, serde_repr::Deserialize_repr)]
48    pub enum Capability : u8 {
49        /// Frame segmentation.
50        Segmentation = 0b0000_1000,
51        /// Frame retransmission (ACK-based)
52        ///
53        /// Implies [`Segmentation`].
54        RetransmissionAck = 0b0000_1100,
55        /// Frame retransmission (NACK-based)
56        ///
57        /// Implies [`Segmentation`].
58        RetransmissionNack = 0b000_1010,
59        /// Disable packet buffering.
60        ///
61        /// Implies [`Segmentation`].
62        NoDelay = 0b0000_1001,
63        /// Disable SURB-based egress rate control.
64        ///
65        /// This applies only to the recipient of the Session (Exit).
66        NoRateControl = 0b0001_0000,
67    }
68}
69
70/// Set of Session [capabilities](Capability).
71pub type Capabilities = flagset::FlagSet<Capability>;

73/// Configuration for the session.
74///
75/// Relevant primarily for the client, since the server is only
76/// a reactive component in regard to the session concept.
77#[derive(Debug, PartialEq, Clone, smart_default::SmartDefault)]
78pub struct SessionClientConfig {
79    /// The forward path options for the session.
80    #[default(RoutingOptions::Hops(hopr_primitive_types::bounded::BoundedSize::MIN))]
81    pub forward_path_options: RoutingOptions,
82    /// The return path options for the session.
83    #[default(RoutingOptions::Hops(hopr_primitive_types::bounded::BoundedSize::MIN))]
84    pub return_path_options: RoutingOptions,
85    /// Capabilities offered by the session.
86    #[default(_code = "Capability::Segmentation.into()")]
87    pub capabilities: Capabilities,
88    /// Optional pseudonym used for the session. Mostly useful for testing only.
89    #[default(None)]
90    pub pseudonym: Option<hopr_internal_types::protocol::HoprPseudonym>,
91    /// Enable automatic SURB management for the Session.
92    #[default(Some(SurbBalancerConfig::default()))]
93    pub surb_management: Option<SurbBalancerConfig>,
94    /// If set, the maximum number of possible SURBs will always be sent with Session data packets (if they fit).
95    ///
96    /// This does not affect `KeepAlive` messages used with SURB balancing, as they will always
97    /// carry the maximum number of SURBs possible. Setting this to `true` will put additional CPU
98    /// pressure on the local node as it will generate the maximum number of SURBs for each data packet.
99    ///
100    /// Set this to `true` only when the underlying traffic is highly asymmetric.
101    ///
102    /// Default is `false`.
103    #[default(false)]
104    pub always_max_out_surbs: bool,
105}

127    #[test]
128    fn hopr_start_protocol_messages_must_fit_within_hopr_packet() -> anyhow::Result<()> {
129        let msg = HoprStartProtocol::StartSession(StartInitiation {
130            challenge: StartChallenge::MAX,
131            target: SessionTarget::TcpStream(SealedHost::Plain(
132                "example-of-a-very-very-long-second-level-name.on-a-very-very-long-domain-name.info:65530".parse()?,
133            )),
134            capabilities: Capabilities::full().into(),
135            additional_data: 0xffffffff,
136        });
137
138        assert!(
139            msg.encode()?.1.len() <= HoprPacket::PAYLOAD_SIZE,
140            "StartSession must fit within {}",
141            HoprPacket::PAYLOAD_SIZE
142        );
143
144        let msg = HoprStartProtocol::SessionEstablished(StartEstablished {
145            orig_challenge: StartChallenge::MAX,
146            session_id: SessionId::new(u64::MAX, HoprPseudonym::random()),
147        });
148
149        assert!(
150            msg.encode()?.1.len() <= HoprPacket::PAYLOAD_SIZE,
151            "SessionEstablished must fit within {}",
152            HoprPacket::PAYLOAD_SIZE
153        );
154
155        let msg = HoprStartProtocol::SessionError(StartErrorType {
156            challenge: StartChallenge::MAX,
157            reason: StartErrorReason::NoSlotsAvailable,
158        });
159
160        assert!(
161            msg.encode()?.1.len() <= HoprPacket::PAYLOAD_SIZE,
162            "SessionError must fit within {}",
163            HoprPacket::PAYLOAD_SIZE
164        );
165
166        let msg = HoprStartProtocol::KeepAlive(KeepAliveMessage {
167            session_id: SessionId::new(u64::MAX, HoprPseudonym::random()),
168            flags: 0xff,
169            additional_data: 0xffffffff,
170        });
171        assert!(
172            msg.encode()?.1.len() <= HoprPacket::PAYLOAD_SIZE,
173            "KeepAlive must fit within {}",
174            HoprPacket::PAYLOAD_SIZE
175        );
176
177        Ok(())
178    }

180    #[test]
181    fn hopr_start_protocol_message_session_initiation_message_should_allow_for_at_least_one_surb() -> anyhow::Result<()>
182    {
183        let msg = HoprStartProtocol::StartSession(StartInitiation {
184            challenge: StartChallenge::MAX,
185            target: SessionTarget::TcpStream(SealedHost::Plain(
186                "example-of-a-very-very-long-second-level-name.on-a-very-very-long-domain-name.info:65530".parse()?,

189            additional_data: 0xffffffff,

195            len

198        Ok(())

201    #[test]
202    fn hopr_start_protocol_message_keep_alive_message_should_allow_for_maximum_surbs() -> anyhow::Result<()> {
203        let msg = HoprStartProtocol::KeepAlive(KeepAliveMessage {
204            session_id: SessionId::new(u64::MAX, HoprPseudonym::random()),
205            flags: 0xff,
206            additional_data: 0xffffffff,
207        });
208        let len = msg.encode()?.1.len();
209        assert_eq!(
210            KeepAliveMessage::<SessionId>::MIN_SURBS_PER_MESSAGE,
211            HoprPacket::MAX_SURBS_IN_PACKET
212        );
213        assert!(
214            HoprPacket::max_surbs_with_message(len) >= HoprPacket::MAX_SURBS_IN_PACKET,
215            "Hopr KeepAlive message size ({}) must allow for at least {} SURBs in packet",
216            len,
217            HoprPacket::MAX_SURBS_IN_PACKET
218        );
219
220        Ok(())
221    }
