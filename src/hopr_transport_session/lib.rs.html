30/// Capabilities of a session.
31#[repr(u8)]
32#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, strum::EnumIter, strum::Display, strum::EnumString)]
33#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
34pub enum Capability {
35    /// Frame segmentation
36    Segmentation,
37    /// Frame retransmission (ACK and NACK-based)
38    Retransmission,
39    /// Frame retransmission (only ACK-based)
40    RetransmissionAckOnly,
41    /// Disable packet buffering
42    NoDelay,
43}
44
45impl IntoIterator for Capability {
46    type Item = SessionFeature;
47    type IntoIter = std::vec::IntoIter<Self::Item>;
48
49    fn into_iter(self) -> Self::IntoIter {
50        match self {
51            Capability::Segmentation => vec![],
52            Capability::Retransmission => vec![
53                SessionFeature::AcknowledgeFrames,
54                SessionFeature::RequestIncompleteFrames,
55                SessionFeature::RetransmitFrames,
56            ],
57            Capability::RetransmissionAckOnly => {
58                vec![SessionFeature::AcknowledgeFrames, SessionFeature::RetransmitFrames]
59            }
60            Capability::NoDelay => vec![SessionFeature::NoDelay],
61        }
62        .into_iter()
63    }
64}

66/// Configuration for the session.
67///
68/// Relevant primarily for the client, since the server is only
69/// a reactive component in regard to the session concept.
70#[derive(Debug, PartialEq, Clone, smart_default::SmartDefault)]
71pub struct SessionClientConfig {
72    /// The forward path options for the session.
73    #[default(RoutingOptions::Hops(hopr_primitive_types::bounded::BoundedSize::MIN))]
74    pub forward_path_options: RoutingOptions,
75    /// The return path options for the session.
76    #[default(RoutingOptions::Hops(hopr_primitive_types::bounded::BoundedSize::MIN))]
77    pub return_path_options: RoutingOptions,
78    /// Capabilities offered by the session.
79    #[default(_code = "vec![Capability::Segmentation]")]
80    pub capabilities: Vec<Capability>,
81    /// Optional pseudonym used for the session. Mostly useful for testing only.
82    #[default(None)]
83    pub pseudonym: Option<HoprPseudonym>,
84    /// Enable automatic SURB management for the Session.
85    #[default(Some(SurbBalancerConfig::default()))]
86    pub surb_management: Option<SurbBalancerConfig>,
87}
