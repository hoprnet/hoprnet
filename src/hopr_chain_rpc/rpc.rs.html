38/// Configuration of the RPC related parameters.
39#[derive(Clone, Debug, Copy, PartialEq, Eq, smart_default::SmartDefault, Serialize, Deserialize, Validate)]
40pub struct RpcOperationsConfig {
41    /// Blockchain id
42    ///
43    /// Default is 100.
44    #[default = 100]
45    pub chain_id: u64,
46    /// Addresses of all deployed contracts
47    ///
48    /// Default contains empty (null) addresses.
49    pub contract_addrs: ContractAddresses,
50    /// Address of the node's module.
51    ///
52    /// Defaults to null address.
53    pub module_address: Address,
54    /// Address of the node's safe contract.
55    ///
56    /// Defaults to null address.
57    pub safe_address: Address,
58    /// Expected block time of the blockchain
59    ///
60    /// Defaults to 5 seconds
61    #[default(Duration::from_secs(5))]
62    pub expected_block_time: Duration,
63    /// The largest amount of blocks to fetch at once when fetching a range of blocks.
64    ///
65    /// If the requested block range size is N, then the client will always fetch `min(N, max_block_range_fetch_size)`
66    ///
67    /// Defaults to 2000 blocks
68    #[validate(range(min = 1))]
69    #[default = 2000]
70    pub max_block_range_fetch_size: u64,
71    /// Interval for polling on TX submission
72    ///
73    /// Defaults to 7 seconds.
74    #[default(Duration::from_secs(7))]
75    pub tx_polling_interval: Duration,
76    /// Finalization chain length
77    ///
78    /// The number of blocks including and decreasing from the chain HEAD
79    /// that the logs will be buffered for before being considered
80    /// successfully joined to the chain.
81    ///
82    /// Defaults to 8
83    #[validate(range(min = 1, max = 100))]
84    #[default = 8]
85    pub finality: u32,
86}

90/// Implementation of `HoprRpcOperations` and `HoprIndexerRpcOperations` trait via `ethers`
91#[derive(Debug)]
92pub struct RpcOperations<P: JsonRpcClient + 'static> {
93    pub(crate) provider: Arc<HoprMiddleware<P>>,
94    pub(crate) cfg: RpcOperationsConfig,
95    contract_instances: Arc<ContractInstances<HoprMiddleware<P>>>,
96    node_module: HoprNodeManagementModule<HoprMiddleware<P>>,
97    node_safe: SafeSingleton<HoprMiddleware<P>>,
98}

103        Self {

113impl<P: JsonRpcClient + 'static> RpcOperations<P> {
114    pub fn new(json_rpc: P, chain_key: &ChainKeypair, cfg: RpcOperationsConfig) -> Result<Self> {
115        let wallet = LocalWallet::from_bytes(chain_key.secret().as_ref())?.with_chain_id(cfg.chain_id);
116
117        let provider = Arc::new(
118            Provider::new(json_rpc)
119                .interval(cfg.tx_polling_interval)
120                .with_signer(wallet)
121                .nonce_manager(chain_key.public().to_address().into()),
122        );
123
124        debug!("{:?}", cfg.contract_addrs);
125
126        Ok(Self {
127            contract_instances: Arc::new(ContractInstances::new(
128                &cfg.contract_addrs,
129                provider.clone(),
130                cfg!(test),
131            )),
132            cfg,
133            node_module: HoprNodeManagementModule::new(cfg.module_address, provider.clone()),
134            node_safe: SafeSingleton::new(cfg.safe_address, provider.clone()),
135            provider,
136        })
137    }

139    pub(crate) async fn get_block_number(&self) -> Result<u64> {
140        Ok(self
141            .provider
142            .get_block_number()
143            .await?
144            .as_u64()
145            .saturating_sub(self.cfg.finality as u64))
146    }
147
148    pub(crate) async fn get_block(
149        &self,
150        block_number: u64,
151    ) -> Result<Option<ethers::types::Block<ethers::types::H256>>> {
152        let sanitized_block_number = block_number.saturating_sub(self.cfg.finality as u64);
153        Ok(self
154            .provider
155            .get_block(BlockId::Number(sanitized_block_number.into()))
156            .await?)
157    }
158}

166    async fn get_balance(&self, address: Address, balance_type: BalanceType) -> Result<Balance> {
167        match balance_type {
168            BalanceType::Native => {
169                let native = self
170                    .provider
171                    .get_balance(
172                        NameOrAddress::Address(address.into()),
173                        Some(BlockId::Number(self.get_block_number().await?.into())),
174                    )
175                    .await?;
176
177                Ok(Balance::new(native, BalanceType::Native))
178            }
179            BalanceType::HOPR => match self.contract_instances.token.balance_of(address.into()).call().await {
180                Ok(token_balance) => Ok(Balance::new(token_balance, BalanceType::HOPR)),
181                Err(e) => Err(ContractError(
182                    "HoprToken".to_string(),
183                    "balance_of".to_string(),
184                    e.to_string(),
185                )),
186            },
187        }
188    }

216    async fn get_eligibility_status(&self, address: Address) -> Result<bool> {
217        let mut multicall = Multicall::new(self.provider.clone(), Some(MULTICALL_ADDRESS))
218            .await
219            .map_err(|e| RpcError::MulticallError(e.to_string()))?;
220
221        multicall
222            // 1) check if the staking threshold set in the implementation is above zero
223            .add_call(
224                match &self.contract_instances.network_registry_proxy {
225                    NetworkRegistryProxy::Dummy(c) => c.method::<H160, U256>("maxAllowedRegistrations", address.into())
226                    .map_err(|e| {
227                        RpcError::MulticallError(format!(
228                            "Error in checking maxAllowedRegistrations function of HoprDummyProxyForNetworkRegistry contract in get_eligibility_status, due to {e}"
229                        ))
230                    })?,
231                    NetworkRegistryProxy::Safe(c) => c.method::<_, U256>("stakeThreshold", ())
232                    .map_err(|e| {
233                        RpcError::MulticallError(format!(
234                            "Error in checking stakeThreshold function of HoprSafeProxyForNetworkRegistry contract in get_eligibility_status, due to {e}"
235                        ))
236                    })?,
237                },
238                false,
239            )
240            // 2) check if the node is registered to an account. In case the selfRegister is disabled (i.e. when staking threshold is zero), this value is used to check if the node is eligible
241            .add_call(
242                self
243                    .contract_instances
244                    .network_registry
245                    .method::<H160, H160>("nodeRegisterdToAccount", address.into())
246                    .map_err(|e| {
247                        RpcError::MulticallError(format!(
248                            "Error in checking nodeRegisterdToAccount function of NetworkRegistry contract in get_eligibility_status, due to {e}"
249                        ))
250                    })?,
251                false,
252            )
253            // 3) check if the node is registered and eligible
254            .add_call(
255                self
256                    .contract_instances
257                    .network_registry
258                    .method::<H160, bool>("isNodeRegisteredAndEligible", address.into())
259                    .map_err(|e| {
260                        RpcError::MulticallError(format!(
261                            "Error in checking isNodeRegisteredAndEligible function of NetworkRegistry contract in get_eligibility_status, due to {e}"
262                        ))
263                    })?,
264                true,
265            );
266
267        match multicall.call_raw().await {
268            Ok(result_token_vec) => match &result_token_vec[2] {
269                // Ok(result_token_vec) => match Token::from_token(result_token_vec[2].map_err(Into::into)?) {
270                Ok(explicit_result) => Ok(explicit_result.clone().into_bool().unwrap_or(false)),
271                Err(e) => {
272                    // The "division by zero" error is caused by the self-registration,
273                    // which is forbidden to the public and thus returns false
274                    // therefore the eligibility check should be ignored
275                    // In EVM it returns `Panic(0x12)` error
276                    // https://docs.soliditylang.org/en/v0.8.12/control-structures.html#panic-via-assert-and-error-via-require
277                    if e.to_string().contains(DIV_BY_ZERO) {
278                        // let stake_threshold = result_token_vec[0].unwrap_or(Token::Uint(U256::zero()));
279                        let stake_threshold = result_token_vec[0]
280                            .clone()
281                            .map_err(|e| {
282                                RpcError::MulticallError(format!(
283                                    "Error in getting stake_threshold from get_eligibility_status, due to {e}"
284                                ))
285                            })?
286                            .into_uint()
287                            .unwrap_or(U256::zero());
288                        let registered_account = result_token_vec[1]
289                            .clone()
290                            .map_err(|e| {
291                                RpcError::MulticallError(format!(
292                                    "Error in getting registered_account from get_eligibility_status, due to {e}"
293                                ))
294                            })?
295                            .into_address()
296                            .unwrap_or(H160::zero());
297
298                        Ok(stake_threshold.is_zero() && !registered_account.is_zero())
299                    } else {
300                        // return a definite error
301                        Err(RpcError::MulticallError(format!(
302                            "Error in getting result from multicall in get_eligibility_status, due to {e}"
303                        )))
304                    }
305                }
306            },
307            Err(e) => {
308                // return a definite error
309                Err(RpcError::MulticallError(format!(
310                    "Error in getting result from get_eligibility_status, due to {e}"
311                )))
312            }
313        }
314    }

373    // Check on-chain status of, node, safe, and module
374    async fn check_node_safe_module_status(&self, node_address: Address) -> Result<NodeSafeModuleStatus> {
375        let mut multicall = Multicall::new(self.provider.clone(), Some(MULTICALL_ADDRESS))
376            .await
377            .map_err(|e| RpcError::MulticallError(e.to_string()))?;
378
379        multicall
380            // 1) check if the node is already included into the module
381            .add_call(
382                self.node_module
383                    .method::<H160, bool>("isNode", node_address.into())
384                    .map_err(|e| {
385                        RpcError::MulticallError(format!(
386                            "Error in checking isNode function in check_node_safe_module_status, due to {e}"
387                        ))
388                    })?,
389                false,
390            )
391            // 2) if the module is enabled in the safe
392            .add_call(
393                self.node_safe
394                    .method::<H160, bool>("isModuleEnabled", self.cfg.module_address.into())
395                    .map_err(|e| {
396                        RpcError::MulticallError(format!(
397                            "Error in checking isModuleEnabled function in check_node_safe_module_status, due to {e}"
398                        ))
399                    })?,
400                false,
401            )
402            // 3) if the safe is the owner of the module
403            .add_call(
404                self.node_module.method::<_, H160>("owner", ()).map_err(|e| {
405                    RpcError::MulticallError(format!(
406                        "Error in checking owner function in check_node_safe_module_status, due to {e}"
407                    ))
408                })?,
409                false,
410            );
411
412        let results: (bool, bool, H160) = multicall.call().await.map_err(|e| {
413            RpcError::MulticallError(format!(
414                "Error in getting result from check_node_safe_module_status, due to {e}"
415            ))
416        })?;
417        Ok(NodeSafeModuleStatus {
418            is_node_included_in_module: results.0,
419            is_module_enabled_in_safe: results.1,
420            is_safe_owner_of_module: results.2 == self.cfg.safe_address.into(),
421        })
422    }

424    async fn send_transaction(&self, tx: TypedTransaction) -> Result<PendingTransaction> {
425        // This only sets the nonce on the first TX, otherwise it is a no-op
426        let _ = self.provider.initialize_nonce(None).await;
427        debug!("send outgoing tx: {:?}", tx);
428
429        // Also fills the transaction including the EIP1559 fee estimates from the provider
430        let sent_tx = self
431            .provider
432            .send_transaction(tx, None)
433            .await?
434            .confirmations(self.cfg.finality as usize)
435            .interval(self.cfg.tx_polling_interval);
436
437        Ok(sent_tx.into())
438    }

470    pub const ETH_VALUE_FOR_MULTICALL3_DEPLOYER: u128 = 100_000_000_000_000_000; // 0.1 (anvil) ETH

472    pub async fn wait_until_tx(pending: PendingTransaction<'_>, timeout: Duration) {
473        let tx_hash = pending.tx_hash();
474        sleep(timeout).await;
475        pending
476            .await
477            .unwrap_or_else(|_| panic!("timeout awaiting tx hash {tx_hash} after {} seconds", timeout.as_secs()));
478    }

480    /// Deploy a MULTICALL contract into Anvil local chain for testing
481    pub async fn deploy_multicall3_to_anvil<M: Middleware>(provider: Arc<M>) -> Result<(), ContractError<M>> {
482        // Fund Multicall3 deployer and deploy ERC1820Registry
483        let mut tx = Eip1559TransactionRequest::new();
484        tx = tx.to(H160::from_str(MULTICALL3_DEPLOYER).unwrap());
485        tx = tx.value(ETH_VALUE_FOR_MULTICALL3_DEPLOYER);
486
487        provider
488            .send_transaction(tx, None)
489            .await
490            .map_err(|e| ContractError::MiddlewareError { e })?
491            .await?;
492
493        provider
494            .send_raw_transaction(Bytes::from_static(&*MULTICALL3_DEPLOY_CODE))
495            .await
496            .map_err(|e| ContractError::MiddlewareError { e })?
497            .await?;
498        Ok(())
499    }
