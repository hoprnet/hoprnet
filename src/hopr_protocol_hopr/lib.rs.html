54    pub fn create_blokli_client() -> anyhow::Result<BlokliTestClient<StaticState>> {
55        Ok(BlokliTestStateBuilder::default()
56            .with_accounts(PEERS.iter().enumerate().map(|(i, (chain_key, offchain_key))| {
57                (
58                    AccountEntry {
59                        public_key: *offchain_key.public(),
60                        chain_addr: chain_key.public().to_address(),
61                        entry_type: AccountType::NotAnnounced,
62                        safe_address: Some([1u8; 20].into()),
63                        key_id: ((i + 1) as u32).into(),
64                    },
65                    HoprBalance::new_base(100),
66                    XDaiBalance::new_base(1),
67                )
68            }))
69            .with_channels(PEERS.iter().enumerate().map(|(i, (chain_key, _))| {
70                ChannelEntry::new(
71                    chain_key.public().to_address(),
72                    PEERS[(i + 1) % PEERS.len()].0.public().to_address(),
73                    HoprBalance::new_base(100),
74                    0,
75                    ChannelStatus::Open,
76                    1,
77                )
78            }))
79            .build_static_client())
80    }

82    pub async fn create_node(index: usize, blokli_client: &BlokliTestClient<StaticState>) -> anyhow::Result<Node> {
83        let mut chain_api = create_trustful_hopr_blokli_connector(
84            &PEERS[index].0,
85            Default::default(),
86            blokli_client.clone(),
87            [10u8; 20].into(),
88        )
89        .await?;
90        chain_api.connect().await?;
91
92        Ok(Node {
93            chain_key: PEERS[index].0.clone(),
94            offchain_key: PEERS[index].1.clone(),
95            node_db: HoprNodeDb::new_in_memory().await?,
96            chain_api: Arc::new(chain_api),
97        })
98    }
