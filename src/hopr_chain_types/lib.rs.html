24/// Holds addresses of all smart contracts.
25#[serde_with::serde_as]
26#[derive(Clone, Copy, Debug, PartialEq, Eq, Serialize, Deserialize, Default)]
27pub struct ContractAddresses {
28    /// Token contract
29    #[serde_as(as = "serde_with::DisplayFromStr")]
30    pub token: Address,
31    /// Channels contract
32    #[serde_as(as = "serde_with::DisplayFromStr")]
33    pub channels: Address,
34    /// Announcement contract
35    #[serde_as(as = "serde_with::DisplayFromStr")]
36    pub announcements: Address,
37    /// Safe registry contract
38    #[serde_as(as = "serde_with::DisplayFromStr")]
39    pub node_safe_registry: Address,
40    /// Price oracle contract
41    #[serde_as(as = "serde_with::DisplayFromStr")]
42    pub ticket_price_oracle: Address,
43    /// Minimum ticket winning probability contract
44    #[serde_as(as = "serde_with::DisplayFromStr")]
45    pub winning_probability_oracle: Address,
46    /// Migration helper for node safes and modules
47    #[serde_as(as = "serde_with::DisplayFromStr")]
48    pub node_safe_migration: Address,
49    /// Stake factory contract
50    #[serde_as(as = "serde_with::DisplayFromStr")]
51    pub node_stake_factory: Address,
52    /// Node management module contract (can be zero if safe is not used)
53    #[serde_as(as = "serde_with::DisplayFromStr")]
54    pub module_implementation: Address,
55}
56
57impl IntoIterator for &ContractAddresses {
58    type IntoIter = std::vec::IntoIter<Address>;
59    type Item = Address;
60
61    fn into_iter(self) -> Self::IntoIter {
62        vec![
63            self.token,
64            self.channels,
65            self.announcements,
66            self.node_safe_registry,
67            self.ticket_price_oracle,
68            self.winning_probability_oracle,
69            self.node_stake_factory,
70            self.module_implementation,
71        ]
72        .into_iter()
73    }
74}

76#[cfg(feature = "use-bindings")]
77impl From<hopr_bindings::ContractAddresses> for ContractAddresses {
78    fn from(value: hopr_bindings::ContractAddresses) -> Self {
79        Self {
80            token: Address::new(&value.token.0.0),
81            channels: Address::new(&value.channels.0.0),
82            announcements: Address::new(&value.announcements.0.0),
83            node_safe_registry: Address::new(&value.node_safe_registry.0.0),
84            ticket_price_oracle: Address::new(&value.ticket_price_oracle.0.0),
85            winning_probability_oracle: Address::new(&value.winning_probability_oracle.0.0),
86            node_safe_migration: Address::new(&value.node_safe_migration.0.0),
87            node_stake_factory: Address::new(&value.node_stake_factory.0.0),
88            module_implementation: Address::new(&value.module_implementation.0.0),
89        }
90    }
91}

93#[cfg(feature = "use-bindings")]
94impl ContractAddresses {
95    /// Returns contract addresses for the given HOPR network `name`
96    /// or `None` if the network does not exist.
97    pub fn for_network(name: &str) -> Option<Self> {
98        hopr_bindings::config::NetworksWithContractAddresses::default()
99            .networks
100            .get(name)
101            .cloned()
102            .map(|n| n.addresses.into())
103    }
