51/// Used for testing. When block time is given, new blocks are mined periodically.

53///
54/// Uses a fixed mnemonic to make generated accounts deterministic.

56    let mut anvil = ethers::utils::Anvil::new()
57        .mnemonic("gentle wisdom move brush express similar canal dune emotion series because parrot");

59        anvil = anvil.block_time(bt.as_secs());
60    }
61
62    anvil.spawn()

188/// Prepare a safe transaction
189pub async fn get_safe_tx<M: Middleware>(
190    safe_contract: SafeContract<M>,
191    target: Address,
192    inner_tx_data: Bytes,
193    wallet: Wallet<SigningKey>,
194) -> Result<TypedTransaction, ContractError<M>> {
195    let nonce = safe_contract.nonce().call().await.unwrap();
196
197    let data_hash = safe_contract
198        .get_transaction_hash(
199            target.into(),
200            U256::zero(),
201            inner_tx_data.clone(),
202            0,
203            U256::zero(),
204            U256::zero(),
205            U256::zero(),
206            Address::default().into(),
207            wallet.address(),
208            nonce,
209        )
210        .call()
211        .await
212        .unwrap();
213    // Outer tx payload: execute as safe tx
214    let mut safe_tx = create_eip1559_transaction();
215
216    let signed_data_hash = wallet.sign_hash(data_hash.into()).unwrap();
217
218    safe_tx.set_data(
219        ExecTransactionCall {
220            to: target.into(),
221            value: U256::zero(),
222            data: inner_tx_data,
223            operation: 0,
224            safe_tx_gas: U256::zero(),
225            base_gas: U256::zero(),
226            gas_price: U256::zero(),
227            gas_token: Address::default().into(),
228            refund_receiver: wallet.address(),
229            signatures: encode_packed(&[Token::Bytes(signed_data_hash.to_vec())])
230                .unwrap()
231                .into(),
232        }
233        .encode()
234        .into(),
235    );
236    safe_tx.set_to(NameOrAddress::Address(safe_contract.address()));
237
238    Ok(safe_tx)
239}

241/// Send a Safe transaction to the module to include node to the module
242pub async fn include_node_to_module_by_safe<M: Middleware>(
243    provider: Arc<M>,
244    safe_address: Address,
245    module_address: Address,
246    node_address: Address,
247    deployer: &ChainKeypair, // also node address
248) -> Result<(), ContractError<M>> {
249    // prepare default permission for node.
250    // - Clearance: Function
251    // - TargetType: SEND
252    // - TargetPermission: allow all
253    // - NodeDefaultPermission: None
254    let node_target_permission = format!("{:?}010203000000000000000000", node_address);
255
256    // Inner tx payload: include node to the module
257    let inner_tx_data: ethers::types::Bytes = IncludeNodeCall {
258        node_default_target: U256::from_str(&node_target_permission).unwrap(),
259    }
260    .encode()
261    .into();
262
263    let safe_contract = SafeContract::new(safe_address, provider.clone());
264    let wallet =
265        ethers::signers::LocalWallet::from_bytes(deployer.secret().as_ref()).expect("failed to construct wallet");
266    let safe_tx = get_safe_tx(safe_contract, module_address, inner_tx_data, wallet)
267        .await
268        .unwrap();
269
270    provider
271        .send_transaction(safe_tx, None)
272        .await
273        .map_err(|e| ContractError::MiddlewareError { e })?
274        .await?;
275
276    Ok(())
277}

468    let typed_tx = instances
469        .stake_factory
470        .clone(
471            instances.module_implementation.address(),
472            vec![self_address],
473            nonce.into(),
474            U256::from_str(&default_target).unwrap().into(),
475        )
476        .tx;
477    debug!("typed_tx {:?}", typed_tx);
478
479    // deploy one safe and one module
480    let instance_deployment_tx_logs = instances
481        .stake_factory
482        .clone(
483            instances.module_implementation.address(),
484            vec![self_address],
485            nonce.into(),
486            U256::from_str(&default_target).unwrap().into(),
487        )
488        .send()
489        .await?
490        .await?
491        .unwrap()
492        .logs;
493
494    // decode the log.
495    let module_log = instance_deployment_tx_logs
496        .iter()
497        .find(|log| {
498            log.topics[0].eq(&"0x813d391dc490d6c1dae7d3fdd555f337533d1da2c908c6efd36d4cf557a63206"
499                .parse::<H256>()
500                .unwrap())
501        })
502        .ok_or(ContractError::ContractNotDeployed)?; // "NewHoprNodeStakeModule(address,address)"
503
504    let safe_log = instance_deployment_tx_logs
505        .iter()
506        .find(|log| {
507            log.topics[0].eq(&"0x8231d169f416b666ae7fa43faa24a18899738075a53f32c97617d173b189e386"
508                .parse::<H256>()
509                .unwrap())
510        })
511        .ok_or(ContractError::ContractNotDeployed)?; // "NewHoprNodeStakeSafe(address)"
512
513    // get address of deployed instances
514    let deployed_module_address: Address = instances
515        .stake_factory
516        .decode_event::<NewHoprNodeStakeModuleFilter>(
517            "NewHoprNodeStakeModule",
518            module_log.topics.clone(),
519            module_log.data.clone(),
520        )
521        .unwrap()
522        .instance
523        .into();
524
525    // note that using the following snippet would cause error
526    // "DetokenizationError(InvalidOutputType("Expected Tuple, got Address(0x8819c5bab7d63c61d72f65b19b05a6772f55391b)"))"
527    // ```
528    // let decoded_safe_log: NewHoprNodeStakeSafeFilter = instances
529    // .stake_factory
530    // .decode_event("NewHoprNodeStakeSafe", safe_log.topics.clone(), safe_log.data.clone())
531    // .unwrap();
532    // ```
533    let deployed_safe_address: Address = instances
534        .stake_factory
535        .abi()
536        .event("NewHoprNodeStakeSafe")
537        .unwrap()
538        .parse_log(RawLog {
539            topics: safe_log.topics.clone(),
540            data: safe_log.data.clone().to_vec(),
541        })?
542        .params
543        .into_iter()
544        .map(|param| param.value)
545        .collect::<Vec<_>>()
546        .pop()
547        .unwrap()
548        .into_address()
549        .unwrap()
550        .into();
551
552    debug!("instance_deployment_tx module instance {:?}", deployed_module_address);
553    debug!("instance_deployment_tx safe instance {:?}", deployed_safe_address);
554
555    Ok((deployed_module_address, deployed_safe_address))
