18pub(crate) struct RetryToken {
19    pub num_retry: usize,
20    pub started_at: Instant,
21    backoff_base: f64,
22    created_at: Instant,
23}
24
25#[derive(Debug, Clone, Copy, PartialEq)]
26pub(crate) enum RetryResult {
27    Wait(Duration),
28    RetryNow(RetryToken),
29    Expired,
30}
31
32impl RetryToken {
33    pub fn new(now: Instant, backoff_base: f64) -> Self {
34        Self {
35            num_retry: 0,
36            started_at: now,
37            created_at: Instant::now(),
38            backoff_base,
39        }
40    }
41
42    pub fn replenish(self, now: Instant, backoff_base: f64) -> Self {
43        Self {
44            num_retry: 0,
45            started_at: now,
46            created_at: self.created_at,
47            backoff_base,
48        }
49    }
50
51    fn retry_in(&self, base: Duration, max_duration: Duration, jitter_dev: f64) -> Option<Duration> {
52        let jitter_coeff = if jitter_dev > 0.0 {
53            // Should not use jitter with sigma > 0.25
54            rand_distr::Normal::new(1.0, jitter_dev.min(0.25))
55                .unwrap()
56                .sample(&mut thread_rng())
57                .abs()
58        } else {
59            1.0
60        };
61
62        // jitter * base * backoff_base ^ num_retry
63        let duration = base.mul_f64(jitter_coeff * self.backoff_base.powi(self.num_retry as i32));
64        (duration < max_duration).then_some(duration)
65    }
66
67    pub fn check(&self, now: Instant, base: Duration, max: Duration, jitter_dev: f64) -> RetryResult {
68        match self.retry_in(base, max, jitter_dev) {
69            None => RetryResult::Expired,
70            Some(retry_in) if self.started_at + retry_in >= now => RetryResult::Wait(self.started_at + retry_in - now),
71            _ => RetryResult::RetryNow(Self {
72                num_retry: self.num_retry + 1,
73                started_at: self.started_at,
74                backoff_base: self.backoff_base,
75                created_at: self.created_at,
76            }),
77        }
78    }
79
80    pub fn time_since_creation(&self) -> Duration {
81        self.created_at.elapsed()
82    }
83}

113/// Network simulator used for testing.
114pub struct FaultyNetwork<'a, const C: usize> {
115    ingress: UnboundedSender<Box<[u8]>>,
116    egress: BoxStream<'a, Box<[u8]>>,
117    stats: Option<NetworkStats>,
118}
119
120impl<const C: usize> AsyncWrite for FaultyNetwork<'_, C> {
121    fn poll_write(self: Pin<&mut Self>, _cx: &mut Context<'_>, buf: &[u8]) -> Poll<std::io::Result<usize>> {
122        if buf.len() > C {
123            return Poll::Ready(Err(std::io::Error::new(
124                std::io::ErrorKind::InvalidInput,
125                format!("data length passed to downstream must be less or equal to {C}"),
126            )));
127        }
128
129        if let Err(e) = self.ingress.unbounded_send(buf.into()) {
130            return Poll::Ready(Err(std::io::Error::new(
131                std::io::ErrorKind::BrokenPipe,
132                format!("failed to send data: {e}"),
133            )));
134        }
135
136        if let Some(stats) = &self.stats {
137            stats.bytes_sent.fetch_add(buf.len(), Ordering::Relaxed);
138            stats.packets_sent.fetch_add(1, Ordering::Relaxed);
139        }
140
141        tracing::trace!("FaultyNetwork::poll_write {} bytes", buf.len());
142        Poll::Ready(Ok(buf.len()))
143    }
144
145    fn poll_flush(self: Pin<&mut Self>, _cx: &mut Context<'_>) -> Poll<std::io::Result<()>> {
146        Poll::Ready(Ok(()))
147    }
148
149    fn poll_close(self: Pin<&mut Self>, _cx: &mut Context<'_>) -> Poll<std::io::Result<()>> {
150        self.ingress.close_channel();
151        Poll::Ready(Ok(()))
152    }
153}

155impl<const C: usize> AsyncRead for FaultyNetwork<'_, C> {
156    fn poll_read(mut self: Pin<&mut Self>, cx: &mut Context<'_>, buf: &mut [u8]) -> Poll<std::io::Result<usize>> {
157        match self.egress.poll_next_unpin(cx) {
158            Poll::Ready(Some(item)) => {
159                let len = buf.len().min(item.len());
160                buf[..len].copy_from_slice(&item.as_ref()[..len]);
161
162                if let Some(stats) = &self.stats {
163                    stats.bytes_received.fetch_add(len, Ordering::Relaxed);
164                    stats.packets_received.fetch_add(1, Ordering::Relaxed);
165                }
166
167                tracing::trace!("FaultyNetwork::poll_read: {len} bytes ready");
168                Poll::Ready(Ok(len))
169            }
170            Poll::Ready(None) => Poll::Ready(Ok(0)),
171            Poll::Pending => Poll::Pending,
172        }
173    }
174}

176impl<const C: usize> FaultyNetwork<'_, C> {
177    #[allow(dead_code)]
178    pub fn new(cfg: FaultyNetworkConfig, stats: Option<NetworkStats>) -> Self {
179        let (ingress, egress) = futures::channel::mpsc::unbounded::<Box<[u8]>>();
180
181        let mut rng = StdRng::from_seed(cfg.rng_seed);
182        let bernoulli = Bernoulli::new(1.0 - cfg.fault_prob).unwrap();
183        let egress = egress.filter(move |_| futures::future::ready(bernoulli.sample(&mut rng)));
184
185        let egress = if cfg.mixing_factor > 0 {
186            let mut rng = StdRng::from_seed(cfg.rng_seed);
187            egress
188                .map(move |e| {
189                    let wait = rng.gen_range(0..20);
190                    async move {
191                        hopr_async_runtime::prelude::sleep(Duration::from_micros(wait)).await;
192                        e
193                    }
194                })
195                .buffer_unordered(cfg.mixing_factor)
196                .boxed()
197        } else {
198            egress.boxed()
199        };
200
201        Self { ingress, egress, stats }
202    }
203}

214    fn spawn_single_byte_read_write<C>(
215        channel: C,
216        data: Vec<u8>,
217    ) -> (
218        impl Future<Output = std::result::Result<Vec<u8>, JoinError>>,
219        impl Future<Output = std::result::Result<Vec<u8>, JoinError>>,
220    )
221    where
222        C: AsyncRead + AsyncWrite + Unpin + Send + 'static,
223    {
224        let (mut recv, mut send) = channel.split();
225
226        let len = data.len();
227        let read = tokio::task::spawn(async move {
228            let mut out = Vec::with_capacity(len);
229            for _ in 0..len {
230                let mut bytes = [0u8; 1];
231                if recv.read(&mut bytes).await.unwrap() > 0 {
232                    out.push(bytes[0]);
233                } else {
234                    break;
235                }
236            }
237            out
238        });
239
240        let written = tokio::task::spawn(async move {
241            let mut out = Vec::with_capacity(len);
242            for byte in data {
243                let _ = send.write(&[byte]).await.unwrap();
244                out.push(byte);
245            }
246            send.close().await.unwrap();
247            out
248        });
249
250        (read, written)
251    }
