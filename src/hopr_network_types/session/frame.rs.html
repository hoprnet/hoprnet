1//! This module implements segmentation of [frames][Frame] into [segments][Segment] and
2//! their [reassembly](FrameReassembler) back into [`Frames`](Frame) and their sequencing.
3//!
4//! ## Frames
5//! Contain data of arbitrary length up to 65536 bytes, differently sized frames are supported.
6//! Each frame carries a [`frame_id`](FrameId) which
7//! should be unique within some higher level session. Frame ID ranges from 1 to 2^32-1.
8//! Frame ID of 0 is not allowed, and its segments cannot be pushed to the reassembler.
9//!
10//! ## Segmentation
11//! A [frame](Frame) can be [segmented](Frame::segment) into equally sized [`Segments`](Segment),
12//! each of them carrying its [sequence number](SeqNum).
13//! This operation runs in linear time with respect to the size of the frame.
14//! There can be up to 256 segments per frame.
15//! Frame segments are uniquely identified via [`SegmentId`].
16//!
17//! ## Reassembly
18//! This is an inverse operation to segmentation. Reassembly is performed by a [`FrameReassembler`]
19//! and is implemented lock-free. The reassembler acts as a [`Sink`] for [`Segments`](Segment) and
20//! is always paired with a [`Stream`] that outputs the reassembled [`Frames`](Frame).
21//!
22//! ### Ordering
23//! The reassembled frames will always have the segments in correct order, and complete frames emitted
24//! from the reassembler will also be ordered correctly according to their frame IDs.
25//! If the next frame in sequence cannot be completed within the `max_age` period given
26//! upon [construction](FrameReassembler::new) of the reassembler, [`NetworkTypeError::FrameDiscarded`]
27//! error will be emitted by the reassembler (see the next section).
28//!
29//! ### Expiration
30//! The reassembler also implements segment expiration. Upon [construction](FrameReassembler::new), the maximum
31//! incomplete frame age can be specified. If a frame is not completed in the reassembler within
32//! this period, it can be [evicted](FrameReassembler::evict) from the reassembler, so that it will be lost
33//! forever.
34//! The eviction operation is supposed to be run periodically, so that the space could be freed up in the
35//! reassembler, and the reassembler does not wait indefinitely for the next frame in sequence.
36//!
37//! Beware that once eviction is performed and an incomplete frame with ID `n` is destroyed;
38//! the caller should make sure that frames with ID <= `n` will not arrive into the reassembler,
39//! otherwise the [NetworkTypeError::OldSegment] error will be thrown.

88/// Helper function to segment `data` into segments of a given ` max_segment_size ` length.
89/// All segments are tagged with the same `frame_id`.
90pub fn segment(data: &[u8], max_segment_size: usize, frame_id: u32) -> crate::session::errors::Result<Vec<Segment>> {
91    if frame_id == 0 {
92        return Err(SessionError::InvalidFrameId);
93    }
94
95    if max_segment_size == 0 {
96        return Err(SessionError::InvalidSegmentSize);
97    }
98
99    let num_chunks = data.len().div_ceil(max_segment_size);
100    if num_chunks > SeqNum::MAX as usize {
101        return Err(SessionError::DataTooLong);
102    }
103
104    let chunks = data.chunks(max_segment_size);
105
106    let seq_len = chunks.len() as SeqNum;
107    Ok(chunks
108        .enumerate()
109        .map(|(idx, data)| Segment {
110            frame_id,
111            seq_len,
112            seq_idx: idx as u8,
113            data: data.into(),
114        })
115        .collect())
116}

143/// Represents a frame segment.
144/// Besides the data, a segment carries information about the total number of
145/// segments in the original frame, its index within the frame and
146/// ID of that frame.
147#[derive(Clone, Eq, PartialEq)]
148#[cfg_attr(feature = "serde", derive(serde::Serialize), derive(serde::Deserialize))]
149pub struct Segment {
150    /// ID of the [Frame] this segment belongs to.
151    pub frame_id: FrameId,
152    /// Index of this segment within the segment sequence.
153    pub seq_idx: SeqNum,
154    /// Total number of segments within this segment sequence.
155    pub seq_len: SeqNum,
156    /// Data in this segment.
157    #[cfg_attr(feature = "serde", serde(with = "serde_bytes"))]
158    pub data: Box<[u8]>,
159}
160
161impl Segment {
162    /// Size of the segment header.
163    pub const HEADER_SIZE: usize = mem::size_of::<FrameId>() + 2 * mem::size_of::<SeqNum>();
164    /// The minimum size of a segment: [`Segment::HEADER_SIZE`] + 1 byte of data.
165    pub const MINIMUM_SIZE: usize = Self::HEADER_SIZE + 1;
166
167    /// Returns the [SegmentId] for this segment.
168    pub fn id(&self) -> SegmentId {
169        SegmentId(self.frame_id, self.seq_idx)
170    }
171}

178            .field("seq_len", &self.seq_len)

181    }

199impl From<Segment> for Vec<u8> {
200    fn from(value: Segment) -> Self {
201        let mut ret = Vec::with_capacity(Segment::HEADER_SIZE + value.data.len());
202        ret.extend_from_slice(value.frame_id.to_be_bytes().as_ref());
203        ret.extend_from_slice(value.seq_idx.to_be_bytes().as_ref());
204        ret.extend_from_slice(value.seq_len.to_be_bytes().as_ref());
205        ret.extend_from_slice(value.data.as_ref());
206        ret
207    }
208}

210impl TryFrom<&[u8]> for Segment {
211    type Error = SessionError;
212
213    fn try_from(value: &[u8]) -> Result<Self, Self::Error> {
214        let (header, data) = value.split_at(Self::HEADER_SIZE);
215        let segment = Segment {
216            frame_id: FrameId::from_be_bytes(header[0..4].try_into().map_err(|_| SessionError::InvalidSegment)?),
217            seq_idx: SeqNum::from_be_bytes(header[4..5].try_into().map_err(|_| SessionError::InvalidSegment)?),
218            seq_len: SeqNum::from_be_bytes(header[5..6].try_into().map_err(|_| SessionError::InvalidSegment)?),
219            data: data.into(),
220        };
221        (segment.frame_id > 0 && segment.seq_idx < segment.seq_len)
222            .then_some(segment)
223            .ok_or(SessionError::InvalidSegment)
224    }
225}

227/// Rebuilds the [Frame] from [Segments](Segment).
228#[derive(Debug)]
229struct FrameBuilder {
230    frame_id: FrameId,
231    _initiated: std::time::Instant,
232    segments: Vec<OnceLock<Box<[u8]>>>,
233    remaining: AtomicU8,
234    last_ts: AtomicU64,
235}
236
237impl FrameBuilder {
238    /// Creates a new builder with the given `initial` [Segment] and its timestamp `ts`.
239    fn new(initial: Segment, ts: SystemTime) -> Self {
240        let ret = Self::empty(initial.frame_id, initial.seq_len);
241        ret.put(initial, ts).unwrap();
242        ret
243    }
244
245    /// Creates a new empty builder for the given frame.
246    fn empty(frame_id: FrameId, seq_len: SeqNum) -> Self {
247        Self {
248            frame_id,
249            _initiated: std::time::Instant::now(),
250            segments: vec![OnceLock::new(); seq_len as usize],
251            remaining: AtomicU8::new(seq_len),
252            last_ts: AtomicU64::new(0),
253        }
254    }
255
256    /// Adds a new [`segment`](Segment) to the builder with a timestamp `ts`.
257    /// Returns the number of segments remaining in this builder.
258    fn put(&self, segment: Segment, ts: SystemTime) -> crate::session::errors::Result<SeqNum> {
259        if self.frame_id == segment.frame_id {
260            if !self.is_complete() {
261                if self.segments[segment.seq_idx as usize].set(segment.data).is_ok() {
262                    // A new segment has been added, decrease the remaining number and update timestamp
263                    self.remaining.fetch_sub(1, Ordering::Relaxed);
264                    self.last_ts
265                        .fetch_max(ts.as_unix_timestamp().as_millis() as u64, Ordering::Relaxed);
266                }
267                Ok(self.remaining.load(Ordering::SeqCst))
268            } else {
269                // Silently throw away segments of a frame that is already complete
270                Ok(0)
271            }
272        } else {
273            Err(SessionError::InvalidFrameId)
274        }
275    }
276
277    /// Checks if the builder contains all segments of the frame.
278    fn is_complete(&self) -> bool {
279        self.remaining.load(Ordering::SeqCst) == 0
280    }
281
282    /// Checks if the last added segment to this frame happened before `cutoff`.
283    /// In other words, the frame under construction is considered expired if the last
284    /// segment was added before `cutoff`.
285    fn is_expired(&self, cutoff: u64) -> bool {
286        self.last_ts.load(Ordering::SeqCst) < cutoff
287    }
288
289    /// Returns information about the frame that is being built by this builder.
290    pub fn info(&self) -> FrameInfo {
291        let mut missing_segments = bitarr![0; 256];
292        self.segments
293            .iter()
294            .enumerate()
295            .filter_map(|(i, s)| s.get().is_none().then_some(i))
296            .for_each(|i| missing_segments.set(i, true));
297
298        FrameInfo {
299            frame_id: self.frame_id,
300            missing_segments,
301            total_segments: self.segments.len() as SeqNum,
302            last_update: SystemTime::UNIX_EPOCH.add(Duration::from_millis(self.last_ts.load(Ordering::SeqCst))),
303        }
304    }
305
306    /// Reassembles the [Frame]. Returns [`NetworkTypeError::IncompleteFrame`] if not
307    /// [complete](FrameBuilder::is_complete).
308    fn reassemble(self) -> crate::session::errors::Result<Frame> {
309        if self.is_complete() {
310            Ok(Frame {
311                frame_id: self.frame_id,
312                data: self
313                    .segments
314                    .into_iter()
315                    .map(|lock| lock.into_inner().unwrap())
316                    .collect::<Vec<Box<[u8]>>>()
317                    .concat()
318                    .into_boxed_slice(),
319            })
320        } else {
321            Err(SessionError::IncompleteFrame(self.frame_id))
322        }
323    }
324}

326/// Contains information about a frame that being built.
327/// The instances are totally ordered as most recently used first.
328#[derive(Debug, Clone, PartialEq, Eq)]
329pub struct FrameInfo {
330    /// ID of the frame.
331    pub frame_id: FrameId,
332    /// Indices of segments that are missing. Empty if the frame is complete.
333    pub missing_segments: BitArr!(for 256),
334    /// The total number of segments in this frame.
335    pub total_segments: SeqNum,
336    /// Time of the last received segment in this frame.
337    pub last_update: SystemTime,
338}
339
340impl FrameInfo {
341    /// Transform self into iterator of missing segment numbers.
342    pub fn iter_missing_sequence_indices(&self) -> impl Iterator<Item = SeqNum> + '_ {
343        self.missing_segments
344            .iter()
345            .by_vals()
346            .enumerate()
347            .filter(|(i, s)| *s && *i <= SeqNum::MAX as usize)
348            .map(|(s, _)| s as SeqNum)
349    }
350
351    pub fn into_missing_segments(self) -> impl Iterator<Item = SegmentId> {
352        self.missing_segments
353            .into_iter()
354            .enumerate()
355            .filter(|(i, s)| *s && *i <= SeqNum::MAX as usize)
356            .map(move |(i, _)| SegmentId(self.frame_id, i as SeqNum))
357    }
358}
359
360impl PartialOrd for FrameInfo {
361    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
362        Some(self.cmp(other))
363    }
364}
365
366impl Ord for FrameInfo {
367    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
368        match self.last_update.cmp(&other.last_update) {
369            std::cmp::Ordering::Equal => self.frame_id.cmp(&self.frame_id),
370            cmp => cmp,
371        }
372        .reverse()
373    }

376/// Represents a frame reassembler.
377///
378/// The [`FrameReassembler`] behaves as a [`Sink`] for [`Segment`].
379/// Upon creation, also [`Stream`] for reassembled [Frames](Frame) is created.
380/// The corresponding stream is closed either when the reassembler is dropped or
381/// [`futures::SinkExt::close`] is called.
382///
383/// As new segments are [pushed](FrameReassembler::push_segment) into the reassembler,
384/// the frames get reassembled, and once they are completed, they are automatically pushed out into
385/// the outgoing frame stream.
386///
387/// The reassembler can also have a `max_age` of frames that are under construction.
388/// The [`evict`](FrameReassembler::evict) method then can be called to remove
389/// the incomplete frames over `max_age`. The timestamps are measured with millisecond precision.
390///
391/// Note that the reassembler is also evicted when dropped.
392///
393/// ````rust
394/// # use std::time::Duration;
395/// futures::executor::block_on(async {
396/// use hopr_network_types::session::{Frame, FrameReassembler};
397/// use futures::{pin_mut, StreamExt, TryStreamExt};
398///
399/// let bytes = b"deadbeefcafe00112233";
400///
401/// // Build Frame and segment it
402/// let frame = Frame { frame_id: 1, data: bytes.as_ref().into() };
403/// let segments = frame.segment(2).unwrap();
404/// assert_eq!(bytes.len() / 2, segments.len());
405///
406/// // Create FrameReassembler and feed the segments to it
407/// let (fragmented, reassembled) = FrameReassembler::new(Duration::from_secs(10));
408///
409/// for segment in segments {
410///     fragmented.push_segment(segment).unwrap();
411/// }
412///
413/// drop(fragmented);
414/// pin_mut!(reassembled);
415///
416/// assert!(matches!(reassembled.try_next().await, Ok(Some(frame))));
417/// # });
418/// ````
419#[derive(Debug)]
420pub struct FrameReassembler {
421    sequences: DashMap<FrameId, FrameBuilder>,
422    highest_buffered_frame: AtomicU32,
423    next_emitted_frame: AtomicU32,
424    last_emission: AtomicU64,
425    reassembled: futures::channel::mpsc::UnboundedSender<crate::session::errors::Result<Frame>>,
426    max_age: Duration,
427}
428
429impl FrameReassembler {
430    /// Creates a new frame reassembler and a corresponding stream
431    /// for reassembled [Frames](Frame).
432    /// An optional `max_age` of segments can be specified,
433    /// which allows the [`evict`](FrameReassembler::evict) method to remove stale incomplete segments.
434    pub fn new(max_age: Duration) -> (Self, impl Stream<Item = crate::session::errors::Result<Frame>>) {
435        let (reassembled, reassembled_recv) = futures::channel::mpsc::unbounded();
436        (
437            Self {
438                sequences: DashMap::new(),
439                highest_buffered_frame: AtomicU32::new(0),
440                next_emitted_frame: AtomicU32::new(1),
441                last_emission: AtomicU64::new(u64::MAX),
442                reassembled,
443                max_age,
444            },
445            reassembled_recv,
446        )
447    }
448
449    /// Emits the frame if it is the next in sequence and complete.
450    /// If it is not next in the sequence or incomplete, it is discarded forever.
451    fn emit_if_complete_discard_otherwise(&self, builder: FrameBuilder) -> crate::session::errors::Result<()> {
452        let time_spent = builder._initiated.elapsed();
453        let frame_id = builder.frame_id;
454
455        if self.next_emitted_frame.fetch_add(1, Ordering::SeqCst) == builder.frame_id && builder.is_complete() {
456            self.reassembled
457                .unbounded_send(builder.reassemble())
458                .map_err(|_| SessionError::ReassemblerClosed)?;
459        } else {
460            self.reassembled
461                .unbounded_send(Err(SessionError::FrameDiscarded(builder.frame_id)))
462                .map_err(|_| SessionError::ReassemblerClosed)?;
463        }
464        self.last_emission
465            .store(current_time().as_unix_timestamp().as_millis() as u64, Ordering::Relaxed);
466
467        tracing::trace!(frame_id, ?time_spent, "frame finished");
468
469        Ok(())
470    }
471
472    /// Pushes a new [Segment] for reassembly.
473    /// This function also pushes out the reassembled frame if this segment completed it.
474    /// Pushing a segment belonging to a frame ID that has been already
475    /// previously completed or [evicted](FrameReassembler::evict) will fail.
476    pub fn push_segment(&self, segment: Segment) -> crate::session::errors::Result<()> {
477        if self.reassembled.is_closed() {
478            return Err(SessionError::ReassemblerClosed);
479        }
480
481        let start = std::time::Instant::now();
482
483        // Check if this frame has not been emitted yet.
484        let frame_id = segment.frame_id;
485        if frame_id < self.next_emitted_frame.load(Ordering::SeqCst) {
486            tracing::trace!("trying to push segment of a frame that has been emitted");
487            return Err(SessionError::OldSegment(frame_id));
488        }
489
490        let ts = current_time();
491        let mut cascade = false;
492
493        match self.sequences.entry(frame_id) {
494            Entry::Occupied(e) => {
495                // No more segments missing in this frame, check if it is the next on to emit
496                if e.get().put(segment, ts)? == 0
497                    && self
498                        .next_emitted_frame
499                        .compare_exchange(frame_id, frame_id + 1, Ordering::SeqCst, Ordering::Relaxed)
500                        .is_ok()
501                {
502                    let builder = e.remove();
503                    let time_spent = builder._initiated.elapsed();
504
505                    // Emit this complete frame
506                    self.reassembled
507                        .unbounded_send(builder.reassemble())
508                        .map_err(|_| SessionError::ReassemblerClosed)?;
509                    self.last_emission
510                        .store(current_time().as_unix_timestamp().as_millis() as u64, Ordering::Relaxed);
511
512                    tracing::trace!(frame_id, ?time_spent, "frame finished");
513
514                    cascade = true; // Try to emit next frames in sequence
515                }
516            }
517            Entry::Vacant(v) => {
518                let builder = FrameBuilder::new(segment, ts);
519                // If this frame is already complete, check if it is the next one to emit
520                if builder.is_complete()
521                    && self
522                        .next_emitted_frame
523                        .compare_exchange(frame_id, frame_id + 1, Ordering::SeqCst, Ordering::Relaxed)
524                        .is_ok()
525                {
526                    let time_spent = builder._initiated.elapsed();
527
528                    // Emit this frame if already complete
529                    self.reassembled
530                        .unbounded_send(builder.reassemble())
531                        .map_err(|_| SessionError::ReassemblerClosed)?;
532                    self.last_emission
533                        .store(current_time().as_unix_timestamp().as_millis() as u64, Ordering::Relaxed);
534
535                    tracing::trace!(frame_id, ?time_spent, "frame finished");
536
537                    cascade = true; // Try to emit the next frames in sequence
538                } else {
539                    // If not complete nor the next one to be emitted, just start building it
540                    v.insert(builder);
541                    self.highest_buffered_frame.fetch_max(frame_id, Ordering::Relaxed);
542                }
543            }
544        }
545
546        // As long as there are more in-sequence frames completed, emit them
547        if cascade {
548            while let Some((_, builder)) = self
549                .sequences
550                .remove_if(&self.next_emitted_frame.load(Ordering::SeqCst), |_, b| b.is_complete())
551            {
552                // If the frame is complete, push it out as reassembled
553                self.emit_if_complete_discard_otherwise(builder)?;
554            }
555        }
556
557        let push_time = start.elapsed();
558        if push_time > Duration::from_millis(PUSH_TIME_THRESHOLD_MS) {
559            tracing::trace!(?push_time, "segment push done");
560        }
561
562        Ok(())
563    }

565    /// Returns [information](FrameInfo) about the incomplete frames.
566    /// The returned collection is ordered by frame IDs.
567    pub fn incomplete_frames(&self) -> BinaryHeap<FrameInfo> {
568        (self.next_emitted_frame.load(Ordering::SeqCst)..=self.highest_buffered_frame.load(Ordering::SeqCst))
569            .filter_map(|frame_id| match self.sequences.get(&frame_id) {
570                Some(e) => (!e.is_complete()).then(|| e.info()),
571                None => Some({
572                    let mut missing_segments = BitArray::ZERO;
573                    missing_segments.set(0, true);
574                    FrameInfo {
575                        frame_id,
576                        missing_segments,
577                        total_segments: 1,
578                        last_update: SystemTime::UNIX_EPOCH,
579                    }
580                }),
581            })
582            .collect()
583    }

585    /// According to the [max_age](FrameReassembler::new) set during construction, evicts
586    /// leading incomplete frames that are expired at the time this method was called.
587    /// Returns that total number of frames that were evicted.
588    pub fn evict(&self) -> crate::session::errors::Result<usize> {
589        if self.reassembled.is_closed() {
590            return Err(SessionError::ReassemblerClosed);
591        }
592
593        if self.sequences.is_empty() {
594            return Ok(0);
595        }
596
597        let start = std::time::Instant::now();
598
599        let cutoff = current_time().sub(self.max_age).as_unix_timestamp().as_millis() as u64;
600        let mut count = 0;
601        loop {
602            let next = self.next_emitted_frame.load(Ordering::SeqCst);
603            if let Some((_, builder)) = self
604                .sequences
605                .remove_if(&next, |_, b| b.is_complete() || b.is_expired(cutoff))
606            {
607                // If the frame is complete, push it out as reassembled or discard it as expired
608                self.emit_if_complete_discard_otherwise(builder)?;
609                count += 1;
610            } else if !self.sequences.contains_key(&next) && self.last_emission.load(Ordering::SeqCst) < cutoff {
611                // Do not stall the sequencer too long if we haven't seen this frame at all
612                self.next_emitted_frame.fetch_add(1, Ordering::Relaxed);
613                self.last_emission
614                    .store(current_time().as_unix_timestamp().as_millis() as u64, Ordering::Relaxed);
615                count += 1;
616            } else {
617                // Break on the first incomplete and non-expired frame
618                tracing::trace!(incomplete = self.sequences.len(), "incomplete frames in reassembler");
619                break;
620            }
621        }
622
623        let eviction_time = start.elapsed();
624        if eviction_time > Duration::from_millis(EVICTION_TIME_THRESHOLD_MS) {
625            tracing::trace!(?eviction_time, count, "eviction done");
626        }
627
628        Ok(count)
629    }

723    /// Sample an index between `0` and `len - 1` using the given distribution and RNG.
724    pub fn sample_index<T: Distribution<f64>, R: Rng>(dist: &mut T, rng: &mut R, len: usize) -> usize {
725        let f: f64 = dist.sample(rng);
726        (f.max(0.0).round() as usize).min(len - 1)
727    }

729    /// Shuffles the given `vec` by taking a next element with index `|N(0,factor^2)`|, where
730    /// `N` denotes normal distribution.
731    /// When used on frame segments vector, it will shuffle the segments in a controlled manner;
732    /// such that an entire frame can unlikely swap position with another, if `factor` ~ frame length.
733    fn linear_half_normal_shuffle<T, R: Rng>(rng: &mut R, mut vec: VecDeque<T>, factor: f64) -> Vec<T> {
734        if factor == 0.0 || vec.is_empty() {
735            return vec.into(); // no mixing
736        }
737
738        let mut dist = Normal::new(0.0, factor).unwrap();
739        let mut ret = Vec::new();
740        while !vec.is_empty() {
741            ret.push(vec.remove(sample_index(&mut dist, rng, vec.len())).unwrap());
742        }
743        ret
744    }

1103    /// Creates `num_frames` out of which `num_corrupted` will have missing segments.
1104    fn corrupt_frames(
1105        num_frames: u32,
1106        corrupted_ratio: f32,
1107    ) -> (Vec<Segment>, Vec<&'static Frame>, HashSet<SegmentId>) {
1108        assert!((0.0..=1.0).contains(&corrupted_ratio));
1109
1110        let mut rng = rand::rngs::StdRng::from_seed(*RAND_SEED);
1111
1112        let (excluded_frame_ids, excluded_segments): (HashSet<FrameId>, HashSet<SegmentId>) = (1..num_frames + 1)
1113            .choose_multiple(&mut rng, ((num_frames as f32) * corrupted_ratio) as usize)
1114            .into_iter() // Must be sequentially generated due RNG determinism
1115            .map(|frame_id| {
1116                (
1117                    frame_id,
1118                    SegmentId(
1119                        frame_id,
1120                        rng.gen_range(0..SEGMENTS.iter().find(|s| s.frame_id == frame_id).unwrap().seq_len),
1121                    ),
1122                )
1123            })
1124            .unzip();
1125
1126        let segments = SEGMENTS
1127            .par_iter()
1128            .filter(|s| s.frame_id < num_frames && !excluded_segments.contains(&SegmentId(s.frame_id, s.seq_idx)))
1129            .cloned()
1130            .collect::<Vec<_>>();
1131
1132        let expected_frames = FRAMES
1133            .par_iter()
1134            .filter(|f| f.frame_id < num_frames && !excluded_frame_ids.contains(&f.frame_id))
1135            .collect::<Vec<_>>();
1136
1137        (segments, expected_frames, excluded_segments)
1138    }

1253                .par_iter()
