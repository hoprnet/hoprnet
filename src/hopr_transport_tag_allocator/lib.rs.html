14/// Identifies which component a partition belongs to.
15#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
16pub enum Usage {
17    /// Long-lived session tags (e.g. hundreds).
18    Session,
19    /// Session terminal telemetry tags (e.g. hundreds to low thousands).
20    SessionTerminalTelemetry,
21    /// Probing telemetry tags — high volume, short-lived (e.g. configurable via env, default ~1000).
22    ProvingTelemetry,
23}
24
25/// Lower bound of the application tag range (exclusive end of [`ReservedTag`] range).
26pub const TAG_RANGE_START: u64 = ReservedTag::UPPER_BOUND;
27/// Upper bound of the application tag range (exclusive).
28pub const TAG_RANGE_END: u64 = u16::MAX as u64 + 1;
29/// Total number of available application tags.
30pub const TAG_RANGE_SIZE: u64 = TAG_RANGE_END - TAG_RANGE_START;
31
32// Compile-time assert: the tag range must fit within u16. A bitmap for a
33// larger range would require a different allocation strategy.
34const _: () = assert!(
35    TAG_RANGE_END <= u16::MAX as u64 + 1,
36    "tag range exceeds u16 — a different allocation strategy is needed"
37);

39/// Allocates unique tags from a fixed partition of the tag range.
40pub trait TagAllocator {
41    /// Obtain the next available tag, or `None` if the partition is exhausted.
42    fn allocate(&self) -> Option<AllocatedTag>;
43
44    /// The total number of tags managed by this allocator.
45    fn capacity(&self) -> u64;
46
47    /// The tag value range `[base, base + capacity)` managed by this allocator.
48    fn tag_range(&self) -> Range<u64>;
49}
50
51/// Result type returned by [`create_allocators`].
52pub type CreateAllocatorsResult = Result<Vec<(Usage, Arc<dyn TagAllocator + Send + Sync>)>, TagAllocatorError>;
53
54/// Create allocators from a [`TagAllocatorConfig`].
55///
56/// Uses [`TagAllocatorConfig::tag_range`] as the available range and
57/// partitions it according to the configured capacities.
58///
59/// # Errors
60///
61/// Returns [`TagAllocatorError`] if any partition has zero capacity or the
62/// total requested capacity exceeds the range.
63pub fn create_allocators_from_config(cfg: &TagAllocatorConfig) -> CreateAllocatorsResult {
64    create_allocators(
65        TAG_RANGE_START..TAG_RANGE_END,
66        [
67            (Usage::Session, cfg.session),
68            (Usage::SessionTerminalTelemetry, cfg.session_probing),
69            (Usage::ProvingTelemetry, cfg.probing_telemetry),
70        ],
71    )
72}
73
74/// Create one [`TagAllocator`] per partition from a contiguous tag range.
75///
76/// The `range` is divided into non-overlapping sub-ranges according to the
77/// capacities carried by each [`Usage`] variant.  Each returned allocator
78/// yields tags exclusively from its own sub-range.
79///
80/// # Errors
81///
82/// Returns [`TagAllocatorError`] if the range is empty, any partition has
83/// zero capacity, or the total requested capacity exceeds the range.
84pub fn create_allocators(range: Range<u64>, partitions: [(Usage, u64); 3]) -> CreateAllocatorsResult {
85    let range_size = range.end.saturating_sub(range.start);
86    if range_size == 0 {
87        return Err(TagAllocatorError::EmptyRange);
88    }
89
90    for (usage, capacity) in &partitions {
91        if *capacity == 0 {
92            return Err(TagAllocatorError::ZeroCapacity(*usage));
93        }
94    }
95
96    let total_requested: u64 = partitions.iter().map(|(_, cap)| cap).sum();
97    if total_requested > range_size {
98        return Err(TagAllocatorError::CapacityExceedsRange {
99            total_requested,
100            range_size,
101        });
102    }
103
104    let mut base = range.start;
105    Ok(partitions
106        .iter()
107        .map(|(usage, capacity)| {
108            let alloc = Arc::new(allocator::PartitionAllocator::new(base, *capacity));
109            base += capacity;
110            (*usage, alloc as Arc<dyn TagAllocator + Send + Sync>)
111        })
112        .collect())
113}

121    #[test]
122    fn partitions_are_non_overlapping() {
123        let allocators = create_allocators(
124            ReservedTag::range().end..u16::MAX as u64 + 1,
125            [
126                (Usage::Session, 100),
127                (Usage::SessionTerminalTelemetry, 200),
128                (Usage::ProvingTelemetry, 300),
129            ],
130        )
131        .unwrap();
132
133        let mut all_tags = Vec::new();
134        let counts = [100, 200, 300];
135        for (i, (_, alloc)) in allocators.iter().enumerate() {
136            for _ in 0..counts[i] {
137                all_tags.push(alloc.allocate().unwrap());
138            }
139        }
140
141        // Check uniqueness while tags are still alive (not returned to pool).
142        let all_values: HashSet<u64> = all_tags.iter().map(|t| t.value()).collect();
143        assert_eq!(all_values.len(), 600);
144
145        // Verify ranges
146        let base = ReservedTag::range().end;
147        let session_tags: Vec<u64> = (base..base + 100).collect();
148        let telemetry_tags: Vec<u64> = (base + 100..base + 300).collect();
149        let probing_tags: Vec<u64> = (base + 300..base + 600).collect();
150
151        for t in &session_tags {
152            assert!(all_values.contains(t));
153        }
154        for t in &telemetry_tags {
155            assert!(all_values.contains(t));
156        }
157        for t in &probing_tags {
158            assert!(all_values.contains(t));
159        }
160    }
