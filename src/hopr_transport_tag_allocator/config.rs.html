8/// Default number of tags reserved for sessions.
9pub const DEFAULT_SESSION_CAPACITY: u64 = 2048;
10/// Default number of tags reserved for session terminal telemetry.
11pub const DEFAULT_SESSION_PROBING_CAPACITY: u64 = 4000;
12/// Default number of tags reserved for probing telemetry (remainder of range).
13pub const DEFAULT_PROBING_TELEMETRY_CAPACITY: u64 =
14    TAG_RANGE_SIZE - DEFAULT_SESSION_CAPACITY - DEFAULT_SESSION_PROBING_CAPACITY;
15
16/// Configuration for the tag allocator partitions.
17///
18/// Each field specifies the number of tags reserved for that usage category.
19#[derive(Debug, Clone, Copy, PartialEq, Eq, smart_default::SmartDefault)]
20#[cfg_attr(
21    feature = "serde",
22    derive(serde::Serialize, serde::Deserialize),
23    serde(deny_unknown_fields)
24)]
25pub struct TagAllocatorConfig {
26    /// Number of tags reserved for long-lived sessions.
27    ///
28    /// This also determines the maximum number of concurrent sessions.
29    #[default(DEFAULT_SESSION_CAPACITY)]
30    pub session: u64,
31
32    /// Number of tags reserved for session terminal telemetry.
33    #[default(DEFAULT_SESSION_PROBING_CAPACITY)]
34    pub session_probing: u64,
35
36    /// Number of tags reserved for probing telemetry.
37    ///
38    /// Defaults to the remainder of the available tag range when using the
39    /// default values for `session` and `session_probing`. When overriding
40    /// those fields, ensure the sum of all three capacities does not exceed
41    /// [`TAG_RANGE_SIZE`]; this is validated at allocator creation time.
42    #[default(DEFAULT_PROBING_TELEMETRY_CAPACITY)]
43    pub probing_telemetry: u64,
44}
45
46impl TagAllocatorConfig {
47    /// Returns the tag range for the given [`Usage`] partition.
48    ///
49    /// Partitions are laid out contiguously starting at
50    /// [`ReservedTag::UPPER_BOUND`] in the order: Session,
51    /// SessionTerminalTelemetry, ProvingTelemetry.
52    pub fn range_for(&self, usage: Usage) -> Range<u64> {
53        let base = ReservedTag::UPPER_BOUND;
54        match usage {
55            Usage::Session => base..base + self.session,
56            Usage::SessionTerminalTelemetry => {
57                let start = base + self.session;
58                start..start + self.session_probing
59            }
60            Usage::ProvingTelemetry => {
61                let start = base + self.session + self.session_probing;
62                start..start + self.probing_telemetry
63            }
64        }
65    }
66
67    /// The full tag range covered by this configuration.
68    ///
69    /// Starts at [`ReservedTag::UPPER_BOUND`] and spans the sum of all
70    /// configured partition capacities.
71    pub fn tag_range(&self) -> Range<u64> {
72        let start = ReservedTag::UPPER_BOUND;
73        start..start + self.session + self.session_probing + self.probing_telemetry
74    }
75}

77impl Validate for TagAllocatorConfig {
78    fn validate(&self) -> Result<(), ValidationErrors> {
79        let mut errors = ValidationErrors::new();
80
81        if self.session == 0 {
82            errors.add("session", ValidationError::new("capacity must be greater than zero"));
83        }
84        if self.session_probing == 0 {
85            errors.add(
86                "session_probing",
87                ValidationError::new("capacity must be greater than zero"),
88            );
89        }
90        if self.probing_telemetry == 0 {
91            errors.add(
92                "probing_telemetry",
93                ValidationError::new("capacity must be greater than zero"),
94            );
95        }
96
97        let total = self.session + self.session_probing + self.probing_telemetry;
98        if total > TAG_RANGE_SIZE {
99            let mut err = ValidationError::new("total capacity exceeds available tag range");
100            err.add_param(std::borrow::Cow::Borrowed("total"), &total);
101            err.add_param(std::borrow::Cow::Borrowed("available"), &TAG_RANGE_SIZE);
102            errors.add("probing_telemetry", err);
103        }
104
105        if errors.is_empty() { Ok(()) } else { Err(errors) }
106    }
107}
