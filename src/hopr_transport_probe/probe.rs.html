26#[inline(always)]
27fn to_nonce(message: &Message) -> String {
28    match message {
29        Message::Probe(NeighborProbe::Ping(ping)) => hex::encode(ping),
30        Message::Probe(NeighborProbe::Pong(ping)) => hex::encode(ping),
31        _ => "<telemetry>".to_string(),
32    }
33}

35#[inline(always)]
36fn to_pseudonym(path: &ResolvedTransportRouting) -> HoprPseudonym {
37    match path {
38        ResolvedTransportRouting::Forward { pseudonym, .. } => *pseudonym,
39        ResolvedTransportRouting::Return(pseudonym, _) => pseudonym.pseudonym(),
40    }

43#[derive(Clone, Debug)]
44struct Sender<T> {
45    downstream: T,
46}
47
48impl<T> Sender<T>
49where
50    T: futures::Sink<(ApplicationDataOut, ResolvedTransportRouting, PacketSendFinalizer)>
51        + Clone
52        + Send
53        + Sync
54        + 'static,
55{
56    #[tracing::instrument(level = "debug", skip(self, path, message), fields(message=%message, nonce=%to_nonce(&message), pseudonym=%to_pseudonym(&path)), ret(level = tracing::Level::TRACE), err(Display))]
57    async fn send_message(self, path: ResolvedTransportRouting, message: Message) -> crate::errors::Result<()> {
58        let (packet_sent_tx, packet_sent_rx) = channel::<std::result::Result<(), PacketError>>();
59
60        let push_to_network = self.downstream;
61        pin_mut!(push_to_network);
62        if push_to_network
63            .as_mut()
64            .send((
65                ApplicationDataOut::with_no_packet_info(message.try_into()?),
66                path,
67                packet_sent_tx.into(),
68            ))
69            .await
70            .is_ok()
71        {
72            packet_sent_rx
73                .await
74                .map_err(|error| ProbeError::SendError(error.to_string()))?
75                .map_err(|error| ProbeError::SendError(error.to_string()))
76        } else {
77            Err(ProbeError::SendError("transport error".to_string()))
78        }
79    }
80}

98    /// The main function that assembles and starts the probing process.
99    pub async fn continuously_scan<T, U, V, W, C, Up>(
100        self,
101        api: (T, U),      // lower (tx, rx) channels for sending and receiving messages
102        manual_events: V, // explicit requests from the API
103        store: W,         // peer store
104        db: C,            // database for SURB & peer resolution
105        move_up: Up,      // forward up non-probing messages from the network
106    ) -> HashMap<HoprProbeProcess, hopr_async_runtime::AbortHandle>
107    where
108        T: futures::Sink<(ApplicationDataOut, ResolvedTransportRouting, PacketSendFinalizer)>
109            + Clone
110            + Send
111            + Sync
112            + 'static,
113        T::Error: Send,
114        U: futures::Stream<Item = (HoprPseudonym, ApplicationDataIn)> + Send + Sync + 'static,
115        W: PeerDiscoveryFetch + ProbeStatusUpdate + Clone + Send + Sync + 'static,
116        C: DbOperations + std::fmt::Debug + Clone + Send + Sync + 'static,
117        V: futures::Stream<Item = (PeerId, PingQueryReplier)> + Send + Sync + 'static,
118        Up: futures::Sink<(HoprPseudonym, ApplicationDataIn)> + Clone + Send + Sync + 'static,
119    {
120        let max_parallel_probes = self.cfg.max_parallel_probes;
121        let interval_between_rounds = self.cfg.interval;
122
123        // For each probe target a cached version of transport routing is stored
124        let cache_peer_routing: moka::future::Cache<PeerId, ResolvedTransportRouting> = moka::future::Cache::builder()
125            .time_to_live(std::time::Duration::from_secs(600))
126            .max_capacity(100_000)
127            .build();
128
129        // Currently active probes
130        let store_eviction = store.clone();
131        let timeout = self.cfg.timeout;
132        let active_probes: moka::future::Cache<
133            (HoprPseudonym, NeighborProbe),
134            (PeerId, std::time::Duration, Option<PingQueryReplier>),
135        > = moka::future::Cache::builder()
136            .time_to_live(timeout)
137            .max_capacity(100_000)
138            .async_eviction_listener(
139                move |k: Arc<(HoprPseudonym, NeighborProbe)>,
140                      v: (PeerId, std::time::Duration, Option<PingQueryReplier>),
141                      cause|
142                      -> moka::notification::ListenerFuture {
143                    if matches!(cause, moka::notification::RemovalCause::Expired) {
144                        // If the eviction cause is expiration => record as a failed probe
145                        let store = store_eviction.clone();
146                        let (peer, _start, notifier) = v;
147
148                        tracing::debug!(%peer, pseudonym = %k.0, probe = %k.1, reason = "timeout", "probe failed");
149                        if let Some(replier) = notifier {
150                            replier.notify(Err(ProbeError::Timeout(timeout.as_secs())));
151                        };
152                        async move {
153                            store
154                                .on_finished(&peer, &Err(ProbeError::Timeout(timeout.as_secs())))
155                                .await;
156                        }
157                        .boxed()
158                    } else {
159                        // If the eviction cause is not expiration, nothing needs to be done
160                        futures::future::ready(()).boxed()
161                    }
162                },
163            )
164            .build();
165
166        let active_probes_rx = active_probes.clone();
167        let db_rx = db.clone();
168        let push_to_network = api.0.clone();
169
170        let mut processes = HashMap::<HoprProbeProcess, hopr_async_runtime::AbortHandle>::new();
171
172        // -- Emit probes --
173        let direct_neighbors = neighbors_to_probe(store.clone(), self.cfg)
174            .map(|peer| (peer, None))
175            .merge(manual_events.map(|(peer, notifier)| (peer, Some(notifier))));
176
177        processes.insert(
178            HoprProbeProcess::Emit,
179            hopr_async_runtime::spawn_as_abortable!(async move {
180                hopr_async_runtime::prelude::sleep(2 * interval_between_rounds).await;   // delay to allow network to stabilize
181
182                direct_neighbors
183                    .for_each_concurrent(max_parallel_probes, move |(peer, notifier)| {
184                        let db = db.clone();
185                        let cache_peer_routing = cache_peer_routing.clone();
186                        let active_probes = active_probes.clone();
187                        let push_to_network = Sender { downstream: push_to_network.clone() };
188                        let me = self.me;
189
190                        async move {
191                            let result = cache_peer_routing
192                                .try_get_with(peer, async move {
193                                    // TODO: This is a CPU intensive operation, convert the probing mechanism to use OffchainPublicKey instead of PeerIDs!
194                                    // PeerId -> OffchainPublicKey is a CPU-intensive blocking operation
195                                    let pubkey = hopr_parallelize::cpu::spawn_blocking(move || OffchainPublicKey::from_peerid(&peer))
196                                        .await
197                                        .context(format!("failed to convert {peer} to offchain public key"))?;
198                                    let cp_address = db
199                                        .resolve_chain_key(&pubkey)
200                                        .await?
201                                        .ok_or_else(|| anyhow::anyhow!("Failed to resolve chain key for peer: {peer}"))?;
202
203                                    Ok::<ResolvedTransportRouting, anyhow::Error>(ResolvedTransportRouting::Forward {
204                                        pseudonym: HoprPseudonym::random(),
205                                        forward_path: ValidatedPath::direct(pubkey, cp_address),
206                                        return_paths: vec![ValidatedPath::direct(me.0, me.1)],
207                                    })
208                                })
209                                .await;
210
211                            match result {
212                                Ok(ResolvedTransportRouting::Forward { pseudonym, forward_path, return_paths }) => {
213                                    let nonce = NeighborProbe::random_nonce();
214
215                                    let message = Message::Probe(nonce);
216                                    let path = ResolvedTransportRouting::Forward { pseudonym, forward_path, return_paths };
217                                    if push_to_network.send_message(path, message).await.is_ok() {
218                                        active_probes
219                                            .insert((pseudonym, nonce), (peer, current_time().as_unix_timestamp(), notifier))
220                                            .await;
221                                    }
222                                },
223                                Ok(_) => tracing::error!(%peer, error = "logical error", "resolved transport routing is not forward"),
224                                Err(error) => tracing::error!(%peer, %error, "failed to resolve transport routing"),
225                            };
226                        }
227                    })
228                    .inspect(|_| tracing::warn!(task = "transport (probe - generate outgoing)", "long-running background task finished"))
229                    .await;
230            })
231        );
232
233        // -- Process probes --
234        processes.insert(
235            HoprProbeProcess::Process,
236            hopr_async_runtime::spawn_as_abortable!(api.1.for_each_concurrent(max_parallel_probes, move |(pseudonym, in_data)| {
237                let active_probes = active_probes_rx.clone();
238                let push_to_network = Sender { downstream: api.0.clone() };
239                let db = db_rx.clone();
240                let store = store.clone();
241                let move_up = move_up.clone();
242
243                async move {
244                    // TODO(v3.1): compare not only against ping tag, but also against telemetry that will be occurring on random tags
245                    if in_data.data.application_tag == ReservedTag::Ping.into() {
246                        let message: anyhow::Result<Message> = in_data.data.try_into().map_err(|e| anyhow::anyhow!("failed to convert data into message: {e}"));
247
248                        match message {
249                            Ok(message) => {
250                                match message {
251                                    Message::Telemetry(_path_telemetry) => {
252                                        tracing::warn!(%pseudonym, reason = "feature not implemented", "this node could not originate the telemetry");
253                                    },
254                                    Message::Probe(NeighborProbe::Ping(ping)) => {
255                                        tracing::debug!(%pseudonym, nonce = hex::encode(ping), "received ping");
256                                        match db.find_surb(pseudonym.into()).await.map(|found_surb| ResolvedTransportRouting::Return(found_surb.sender_id, found_surb.surb)) {
257                                            Ok(path) => {
258                                                tracing::trace!(%pseudonym, nonce = hex::encode(ping), "wrapping a pong in the found SURB");
259                                            let message = Message::Probe(NeighborProbe::Pong(ping));
260                                                if let Err(error) = push_to_network.send_message(path, message).await {
261                                                    tracing::error!(%pseudonym, %error, "failed to send back a pong");
262                                                }
263                                            },
264                                            Err(error) => tracing::error!(%pseudonym, %error, "failed to get a SURB, cannot send back a pong"),
265                                        }
266                                    },
267                                    Message::Probe(NeighborProbe::Pong(pong)) => {
268                                        tracing::debug!(%pseudonym, nonce = hex::encode(pong), "received pong");
269                                        if let Some((peer, start, replier)) = active_probes.remove(&(pseudonym, NeighborProbe::Ping(pong))).await {
270                                            let latency = current_time()
271                                                .as_unix_timestamp()
272                                                .saturating_sub(start);
273                                            store.on_finished(&peer, &Ok(latency)).await;
274
275                                            if let Some(replier) = replier {
276                                                replier.notify(Ok(latency))
277                                            };
278                                        } else {
279                                            tracing::warn!(%pseudonym, nonce = hex::encode(pong), possible_reasons = "[timeout, adversary]", "received pong for unknown probe");
280                                        };
281                                    },
282                                }
283                            },
284                            Err(error) => tracing::error!(%pseudonym, %error, "cannot deserialize message"),
285                        }
286                    } else {
287                        // If the message is not a probing message, forward it up
288                        pin_mut!(move_up);
289                        if move_up.send((pseudonym, in_data)).await.is_err() {
290                            tracing::error!(%pseudonym, error = "receiver error", "failed to send message up");
291                        }
292                    }
293                }
294            }).inspect(|_| tracing::warn!(task = "transport (probe - processing incoming)", "long-running background task finished")))
295        );
296
297        processes
298    }
299}

439    const NO_PROBE_PASSES: f64 = 0.0;
440    const ALL_PROBES_PASS: f64 = 1.0;
441
442    /// Channel that can drop any probes and concurrently replies to a probe correctly
443    fn concurrent_channel(
444        delay: Option<std::time::Duration>,
445        pass_rate: f64,
446        from_network_to_probing_tx: futures::channel::mpsc::Sender<(HoprPseudonym, ApplicationDataIn)>,
447    ) -> impl Fn((ApplicationDataOut, ResolvedTransportRouting, PacketSendFinalizer)) -> BoxFuture<'static, ()> {
448        debug_assert!(
449            (NO_PROBE_PASSES..=ALL_PROBES_PASS).contains(&pass_rate),
450            "Pass rate must be between {NO_PROBE_PASSES} and {ALL_PROBES_PASS}"
451        );
452
453        move |(data_out, path, finalizer): (ApplicationDataOut, ResolvedTransportRouting, PacketSendFinalizer)| -> BoxFuture<'static, ()> {
454            let mut from_network_to_probing_tx = from_network_to_probing_tx.clone();
455
456            Box::pin(async move {
457                if let ResolvedTransportRouting::Forward { pseudonym, .. } = path {
458                    finalizer.finalize(Ok(()));
459
460                    let message: Message = data_out.data.try_into().expect("failed to convert data into message");
461                    if let Message::Probe(NeighborProbe::Ping(ping)) = message {
462                        let pong_message = Message::Probe(NeighborProbe::Pong(ping));
463
464                        if let Some(delay) = delay {
465                            // Simulate a delay if specified
466                            tokio::time::sleep(delay).await;
467                        }
468
469                        if rand::Rng::gen_range(&mut rand::thread_rng(), NO_PROBE_PASSES..=ALL_PROBES_PASS) < pass_rate {
470                            from_network_to_probing_tx
471                                .send((pseudonym, ApplicationDataIn { data: pong_message.try_into().expect("failed to convert pong message into data"), packet_info: Default::default()}))
472                                .await.expect("failed to send pong message");
473                        }
474                    }
475                };
476            })
477        }
478    }
