14/// Implementation of [`hopr_lib::HoprSessionReactor`] that facilitates
15/// bridging of TCP or UDP sockets from the Session Exit node to a destination.
16#[derive(Debug, Clone)]
17pub struct HoprServerIpForwardingReactor {
18    keypair: HoprOffchainKeypair,
19    cfg: SessionIpForwardingConfig,
20}
21
22impl HoprServerIpForwardingReactor {
23    pub fn new(keypair: HoprOffchainKeypair, cfg: SessionIpForwardingConfig) -> Self {
24        Self { keypair, cfg }
25    }
26
27    fn all_ips_allowed(&self, addrs: &[SocketAddr]) -> bool {
28        if self.cfg.use_target_allow_list {
29            for addr in addrs {
30                if !self.cfg.target_allow_list.contains(addr) {
31                    tracing::error!(%addr, "address not allowed by the target allow list, denying the target");
32                    return false;
33                }
34                tracing::debug!(%addr, "address allowed by the target allow list, accepting the target");
35            }
36        }
37        true
38    }
39}
40
41pub const SERVICE_ID_LOOPBACK: ServiceId = 0;
42
43#[hopr_lib::async_trait]
44impl hopr_lib::traits::session::HoprSessionServer for HoprServerIpForwardingReactor {
45    #[tracing::instrument(level = "debug", skip(self, session))]
46    async fn process(
47        &self,
48        mut session: hopr_lib::exports::transport::session::IncomingSession,
49    ) -> hopr_lib::errors::Result<()> {
50        let session_id = *session.session.id();
51        match session.target {
52            hopr_lib::SessionTarget::UdpStream(udp_target) => {
53                let kp = self.keypair.clone();
54                let udp_target = hopr_lib::utils::parallelize::cpu::spawn_blocking(move || udp_target.unseal(&kp))
55                    .await
56                    .map_err(|e| HoprLibError::GeneralError(format!("cannot unseal target: {e}")))?;
57
58                tracing::debug!(
59                    session_id = ?session_id,
60                    %udp_target,
61                    "binding socket to the UDP server"
62                );
63
64                // In UDP, it is impossible to determine if the target is viable,
65                // so we just take the first resolved address.
66                let resolved_udp_target = udp_target
67                    .clone()
68                    .resolve_tokio()
69                    .await
70                    .map_err(|e| HoprLibError::GeneralError(format!("failed to resolve DNS name {udp_target}: {e}")))?
71                    .first()
72                    .ok_or(HoprLibError::GeneralError(format!(
73                        "failed to resolve DNS name {udp_target}"
74                    )))?
75                    .to_owned();
76                tracing::debug!(
77                    ?session_id,
78                    %udp_target,
79                    resolution = ?resolved_udp_target,
80                    "UDP target resolved"
81                );
82
83                if !self.all_ips_allowed(&[resolved_udp_target]) {
84                    return Err(HoprLibError::GeneralError(format!(
85                        "denied target address {resolved_udp_target}"
86                    )));
87                }
88
89                let mut udp_bridge = ConnectedUdpStream::builder()
90                    .with_buffer_size(HOPR_UDP_BUFFER_SIZE)
91                    .with_counterparty(resolved_udp_target)
92                    .with_foreign_data_mode(ForeignDataMode::Error)
93                    .with_queue_size(HOPR_UDP_QUEUE_SIZE)
94                    .with_receiver_parallelism(
95                        std::env::var("HOPRD_SESSION_EXIT_UDP_RX_PARALLELISM")
96                            .ok()
97                            .and_then(|s| s.parse::<NonZeroUsize>().ok())
98                            .map(UdpStreamParallelism::Specific)
99                            .unwrap_or(UdpStreamParallelism::Auto),
100                    )
101                    .build(("0.0.0.0", 0))
102                    .map_err(|e| {
103                        HoprLibError::GeneralError(format!(
104                            "could not bridge the incoming session to {udp_target}: {e}"
105                        ))
106                    })?;
107
108                tracing::debug!(
109                    ?session_id,
110                    %udp_target,
111                    "bridging the session to the UDP server"
112                );
113
114                tokio::task::spawn(async move {
115                    let _metric_counter_guard = MetricSessionCounterGuard::new("udp");
116
117                    // The Session forwards the termination to the udp_bridge, terminating
118                    // the UDP socket.
119                    match transfer_session(&mut session.session, &mut udp_bridge, HOPR_UDP_BUFFER_SIZE, None).await {
120                        Ok((session_to_stream_bytes, stream_to_session_bytes)) => tracing::info!(
121                            ?session_id,
122                            session_to_stream_bytes,
123                            stream_to_session_bytes,
124                            %udp_target,
125                            "server bridged session to UDP ended"
126                        ),
127                        Err(e) => tracing::error!(
128                            ?session_id,
129                            %udp_target,
130                            error = %e,
131                            "UDP server stream is closed"
132                        ),
133                    }
134                });
135
136                Ok(())
137            }
138            hopr_lib::SessionTarget::TcpStream(tcp_target) => {
139                let kp = self.keypair.clone();
140                let tcp_target = hopr_lib::utils::parallelize::cpu::spawn_blocking(move || tcp_target.unseal(&kp))
141                    .await
142                    .map_err(|e| HoprLibError::GeneralError(format!("cannot unseal target: {e}")))?;
143
144                tracing::debug!(?session_id, %tcp_target, "creating a connection to the TCP server");
145
146                // TCP is able to determine which of the resolved multiple addresses is viable,
147                // and therefore we can pass all of them.
148                let resolved_tcp_targets =
149                    tcp_target.clone().resolve_tokio().await.map_err(|e| {
150                        HoprLibError::GeneralError(format!("failed to resolve DNS name {tcp_target}: {e}"))
151                    })?;
152                tracing::debug!(
153                    ?session_id,
154                    %tcp_target,
155                    resolution = ?resolved_tcp_targets,
156                    "TCP target resolved"
157                );
158
159                if !self.all_ips_allowed(&resolved_tcp_targets) {
160                    return Err(HoprLibError::GeneralError(format!(
161                        "denied target address {resolved_tcp_targets:?}"
162                    )));
163                }
164
165                let strategy = tokio_retry::strategy::FixedInterval::new(self.cfg.tcp_target_retry_delay)
166                    .take(self.cfg.max_tcp_target_retries as usize);
167
168                let mut tcp_bridge = tokio_retry::Retry::spawn(strategy, || {
169                    tokio::net::TcpStream::connect(resolved_tcp_targets.as_slice())
170                })
171                .await
172                .map_err(|e| {
173                    HoprLibError::GeneralError(format!("could not bridge the incoming session to {tcp_target}: {e}"))
174                })?;
175
176                tcp_bridge.set_nodelay(true).map_err(|e| {
177                    HoprLibError::GeneralError(format!(
178                        "could not set the TCP_NODELAY option for the bridged session to {tcp_target}: {e}",
179                    ))
180                })?;
181
182                tracing::debug!(
183                    ?session_id,
184                    %tcp_target,
185                    "bridging the session to the TCP server"
186                );
187
188                tokio::task::spawn(async move {
189                    let _metric_counter_guard = MetricSessionCounterGuard::new("tcp");
190
191                    match transfer_session(&mut session.session, &mut tcp_bridge, HOPR_TCP_BUFFER_SIZE, None).await {
192                        Ok((session_to_stream_bytes, stream_to_session_bytes)) => tracing::info!(
193                            ?session_id,
194                            session_to_stream_bytes,
195                            stream_to_session_bytes,
196                            %tcp_target,
197                            "server bridged session to TCP ended"
198                        ),
199                        Err(error) => tracing::error!(
200                            ?session_id,
201                            %tcp_target,
202                            %error,
203                            "TCP server stream is closed"
204                        ),
205                    }
206                });
207
208                Ok(())
209            }
210            hopr_lib::SessionTarget::ExitNode(SERVICE_ID_LOOPBACK) => {
211                tracing::debug!(?session_id, "bridging the session to the loopback service");
212                let (mut reader, mut writer) = tokio::io::split(session.session);
213
214                let _metric_counter_guard = MetricSessionCounterGuard::new("loopback");
215
216                // Uses 4 kB buffer for copying
217                match tokio::io::copy(&mut reader, &mut writer).await {
218                    Ok(copied) => tracing::info!(?session_id, copied, "server loopback session service ended"),
219                    Err(error) => tracing::error!(
220                        ?session_id,
221                        %error,
222                        "server loopback session service ended with an error"
223                    ),
224                }
225
226                Ok(())
227            }
228            hopr_lib::SessionTarget::ExitNode(_) => Err(HoprLibError::GeneralError(
229                "server does not support internal session processing".into(),
230            )),
231        }
232    }
233}
