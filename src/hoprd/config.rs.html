103/// The main configuration object of the entire node.
104///
105/// The configuration is composed of individual configuration of corresponding
106/// component configuration objects.
107///
108/// An always up-to-date config YAML example can be found in [example_cfg.yaml](https://github.com/hoprnet/hoprnet/tree/master/hoprd/hoprd/example_cfg.yaml)
109/// which is always in the root of this crate.
110#[derive(Debug, Serialize, Deserialize, Validate, Clone, PartialEq, smart_default::SmartDefault)]
111#[serde(deny_unknown_fields)]
112pub struct HoprdConfig {
113    /// Configuration related to hopr functionality
114    #[validate(nested)]
115    #[serde(default)]
116    pub hopr: HoprLibConfig,
117    /// Configuration regarding the identity of the node
118    #[validate(nested)]
119    #[serde(default)]
120    pub identity: Identity,
121    /// Configuration of the underlying database engine
122    #[validate(nested)]
123    #[serde(default)]
124    pub db: Db,
125    /// Configuration relevant for the API of the node
126    #[validate(nested)]
127    #[serde(default)]
128    pub api: Api,
129    /// Configuration of the Session entry/exit node IP protocol forwarding.
130    #[validate(nested)]
131    #[serde(default)]
132    pub session_ip_forwarding: SessionIpForwardingConfig,
133    /// Blokli provider to connect to.
134    #[validate(url)]
135    pub provider: Option<String>,
136    /// Configuration of underlying node behavior in the form strategies
137    ///
138    /// Strategies represent automatically executable behavior performed by
139    /// the node given pre-configured triggers.
140    #[validate(nested)]
141    #[serde(default = "hopr_strategy::hopr_default_strategies")]
142    #[default(hopr_strategy::hopr_default_strategies())]
143    pub strategy: hopr_strategy::StrategyConfig,
144}

152impl HoprdConfig {
153    pub fn from_cli_args(cli_args: crate::cli::CliArgs, skip_validation: bool) -> crate::errors::Result<HoprdConfig> {
154        let mut cfg: HoprdConfig = if let Some(cfg_path) = cli_args.configuration_file_path {
155            debug!(cfg_path, "fetching configuration from file");
156            let yaml_configuration = std::fs::read_to_string(cfg_path.as_str())
157                .map_err(|e| crate::errors::HoprdError::ConfigError(e.to_string()))?;
158            serde_yaml::from_str(&yaml_configuration)
159                .map_err(|e| crate::errors::HoprdError::SerializationError(e.to_string()))?
160        } else {
161            debug!("loading default configuration");
162            HoprdConfig::default()
163        };
164
165        // host
166        if let Some(x) = cli_args.host {
167            cfg.hopr.host = x
168        };
169
170        // hopr.transport
171        if cli_args.test_announce_local_addresses > 0 {
172            cfg.hopr.transport.announce_local_addresses = true;
173        }
174        if cli_args.test_prefer_local_addresses > 0 {
175            cfg.hopr.transport.prefer_local_addresses = true;
176        }
177
178        if let Some(host) = cli_args.default_session_listen_host {
179            cfg.session_ip_forwarding.default_entry_listen_host = match host.address {
180                HostType::IPv4(addr) => IpAddr::from_str(&addr)
181                    .map(|ip| std::net::SocketAddr::new(ip, host.port))
182                    .map_err(|_| HoprdError::ConfigError("invalid default session listen IP address".into())),
183                HostType::Domain(_) => Err(HoprdError::ConfigError("default session listen must be an IP".into())),
184            }?;
185        }
186
187        // db
188        if let Some(data) = cli_args.data {
189            cfg.db.data = data
190        }
191        if cli_args.init > 0 {
192            cfg.db.initialize = true;
193        }
194        if cli_args.force_init > 0 {
195            cfg.db.force_initialize = true;
196        }
197
198        // api
199        if cli_args.api > 0 {
200            cfg.api.enable = true;
201        }
202        if cli_args.disable_api_authentication > 0 && cfg.api.auth != Auth::None {
203            cfg.api.auth = Auth::None;
204        };
205        if let Some(x) = cli_args.api_token {
206            cfg.api.auth = Auth::Token(x);
207        };
208        if let Some(x) = cli_args.api_host {
209            cfg.api.host =
210                HostConfig::from_str(format!("{}:{}", x.as_str(), hoprd_api::config::DEFAULT_API_PORT).as_str())
211                    .map_err(crate::errors::HoprdError::ValidationError)?;
212        }
213        if let Some(x) = cli_args.api_port {
214            cfg.api.host.port = x
215        };
216
217        // probe
218        if let Some(x) = cli_args.probe_recheck_threshold {
219            cfg.hopr.probe.recheck_threshold = std::time::Duration::from_secs(x)
220        };
221
222        // network options
223        if let Some(x) = cli_args.network_quality_threshold {
224            cfg.hopr.network_options.quality_offline_threshold = x
225        };
226
227        // identity
228        if let Some(identity) = cli_args.identity {
229            cfg.identity.file = identity;
230        }
231        if let Some(x) = cli_args.password {
232            cfg.identity.password = x
233        };
234        if let Some(x) = cli_args.private_key {
235            cfg.identity.private_key = Some(x)
236        };
237
238        // chain
239        if cli_args.announce > 0 {
240            cfg.hopr.publish = true;
241        }
242
243        if let Some(x) = cli_args.provider {
244            cfg.provider = Some(x);
245        }
246
247        if let Some(x) = cli_args.safe_address {
248            cfg.hopr.safe_module.safe_address =
249                Address::from_str(&x).map_err(|e| HoprdError::ValidationError(e.to_string()))?
250        };
251        if let Some(x) = cli_args.module_address {
252            cfg.hopr.safe_module.module_address =
253                Address::from_str(&x).map_err(|e| HoprdError::ValidationError(e.to_string()))?
254        };
255
256        // additional updates
257        let home_symbol = '~';
258        if cfg.db.data.starts_with(home_symbol) {
259            cfg.db.data = home::home_dir()
260                .map(|h| h.as_path().display().to_string())
261                .expect("home dir for a user must be specified")
262                + &cfg.db.data[1..];
263        }
264        if cfg.identity.file.starts_with(home_symbol) {
265            cfg.identity.file = home::home_dir()
266                .map(|h| h.as_path().display().to_string())
267                .expect("home dir for a user must be specified")
268                + &cfg.identity.file[1..];
269        }
270
271        if skip_validation {
272            return Ok(cfg);
273        }
274
275        match cfg.validate() {
276            Ok(_) => Ok(cfg),
277            Err(e) => Err(crate::errors::HoprdError::ValidationError(e.to_string())),
278        }
279    }

281    pub fn as_redacted(&self) -> Self {
282        let mut ret = self.clone();
283        // redacting sensitive information
284        match ret.api.auth {
285            Auth::None => {}
286            Auth::Token(_) => ret.api.auth = Auth::Token("<REDACTED>".to_owned()),
287        }
288
289        if ret.identity.private_key.is_some() {
290            ret.identity.private_key = Some("<REDACTED>".to_owned());
291        }
292
293        "<REDACTED>".clone_into(&mut ret.identity.password);
294
295        ret
296    }
297
298    pub fn as_redacted_string(&self) -> crate::errors::Result<String> {
299        let redacted_cfg = self.as_redacted();
300        serde_json::to_string(&redacted_cfg).map_err(|e| crate::errors::HoprdError::SerializationError(e.to_string()))
301    }

320/// Configuration of the Exit node (see [`HoprServerIpForwardingReactor`](crate::exit::HoprServerIpForwardingReactor))
321/// and the Entry node.
322#[serde_as]
323#[derive(
324    Clone, Debug, Eq, PartialEq, smart_default::SmartDefault, serde::Deserialize, serde::Serialize, validator::Validate,
325)]
326pub struct SessionIpForwardingConfig {
327    /// Controls whether allowlisting should be done via `target_allow_list`.
328    /// If set to `false`, the node will act as an Exit node for any target.
329    ///
330    /// Defaults to `true`.
331    #[serde(default = "just_true")]
332    #[default(true)]
333    pub use_target_allow_list: bool,
334
335    /// Enforces only the given target addresses (after DNS resolution).
336    ///
337    /// This is used only if `use_target_allow_list` is set to `true`.
338    /// If left empty (and `use_target_allow_list` is `true`), the node will not act as an Exit node.
339    ///
340    /// Defaults to empty.
341    #[serde(default)]
342    #[serde_as(as = "HashSet<serde_with::DisplayFromStr>")]
343    pub target_allow_list: HashSet<SocketAddr>,
344
345    /// Delay between retries in seconds to reach a TCP target.
346    ///
347    /// Defaults to 2 seconds.
348    #[serde(default = "default_target_retry_delay")]
349    #[default(default_target_retry_delay())]
350    #[serde_as(as = "serde_with::DurationSeconds<u64>")]
351    pub tcp_target_retry_delay: Duration,
352
353    /// Maximum number of retries to reach a TCP target before giving up.
354    ///
355    /// Default is 10.
356    #[serde(default = "default_max_tcp_target_retries")]
357    #[default(default_max_tcp_target_retries())]
358    #[validate(range(min = 1))]
359    pub max_tcp_target_retries: u32,
360
361    /// Specifies the default `listen_host` for Session listening sockets
362    /// at an Entry node.
363    #[serde(default = "default_entry_listen_host")]
364    #[default(default_entry_listen_host())]
365    #[serde_as(as = "serde_with::DisplayFromStr")]
366    pub default_entry_listen_host: SocketAddr,
367}
