1use axum::{http::StatusCode, response::IntoResponse};
2
3use crate::{ApiError, ApiErrorStatus};
4
5#[cfg(all(feature = "prometheus", not(test)))]
6fn collect_hopr_metrics() -> Result<String, ApiErrorStatus> {
7    hopr_metrics::metrics::gather_all_metrics()
8        .map_err(|_| ApiErrorStatus::UnknownFailure("Failed to gather metrics".into()))
9}
10
11#[cfg(any(not(feature = "prometheus"), test))]
12fn collect_hopr_metrics() -> Result<String, ApiErrorStatus> {
13    Err(ApiErrorStatus::UnknownFailure("BUILT WITHOUT METRICS SUPPORT".into()))
14}
15
16/// Retrieve Prometheus metrics from the running node.
17#[utoipa::path(
18        get,
19        path = const_format::formatcp!("/metrics"),
20        description = "Retrieve Prometheus metrics from the running node",
21        responses(
22            (status = 200, description = "Fetched node metrics", body = String),
23            (status = 401, description = "Invalid authorization token.", body = ApiError),
24            (status = 422, description = "Unknown failure", body = ApiError)
25        ),
26        security(
27            ("api_token" = []),
28            ("bearer_token" = [])
29        ),
30        tag = "Metrics"
31    )]
32pub(super) async fn metrics() -> impl IntoResponse {
33    match collect_hopr_metrics() {
34        Ok(metrics) => (StatusCode::OK, metrics).into_response(),
35        Err(error) => (StatusCode::UNPROCESSABLE_ENTITY, error).into_response(),
36    }
37}
