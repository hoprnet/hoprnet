6fn collect_hopr_metrics() -> Result<String, ApiErrorStatus> {

15/// Retrieve Prometheus metrics from the running node.
16#[utoipa::path(
17        get,
18        path = const_format::formatcp!("/metrics"),
19        description = "Retrieve Prometheus metrics from the running node",
20        responses(
21            (status = 200, description = "Fetched node metrics", body = String),
22            (status = 401, description = "Invalid authorization token.", body = ApiError),
23            (status = 422, description = "Unknown failure", body = ApiError)
24        ),
25        security(
26            ("api_token" = []),
27            ("bearer_token" = [])
28        ),
29        tag = "Metrics"
30    )]
31pub(super) async fn metrics() -> impl IntoResponse {
32    match collect_hopr_metrics() {
33        Ok(metrics) => (StatusCode::OK, metrics).into_response(),
34        Err(error) => (StatusCode::UNPROCESSABLE_ENTITY, error).into_response(),
35    }
36}
