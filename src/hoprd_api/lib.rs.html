147impl Modify for SecurityAddon {
148    fn modify(&self, openapi: &mut utoipa::openapi::OpenApi) {
149        let components = openapi
150            .components
151            .as_mut()
152            .expect("components should be registered at this point");
153
154        components.add_security_scheme(
155            "bearer_token",
156            SecurityScheme::Http(
157                HttpBuilder::new()
158                    .scheme(HttpAuthScheme::Bearer)
159                    .bearer_format("token")
160                    .description(Some("Bearer token authentication".to_string()))
161                    .build(),
162            ),
163        );
164        components.add_security_scheme(
165            "api_token",
166            SecurityScheme::ApiKey(ApiKey::Header(ApiKeyValue::with_description(
167                "X-Auth-Token",
168                "API Token",
169            ))),
170        );
171    }
172}

184/// Starts the Rest API listener and router.
185pub async fn serve_api(params: RestApiParameters) -> Result<(), std::io::Error> {
186    let RestApiParameters {
187        listener,
188        hoprd_cfg,
189        cfg,
190        hopr,
191        session_listener_sockets,
192        default_session_listen_host,
193    } = params;
194
195    let router = build_api(
196        hoprd_cfg,
197        cfg,
198        hopr,
199        session_listener_sockets,
200        default_session_listen_host,
201    )
202    .await;
203    axum::serve(listener, router).await
204}

206#[allow(clippy::too_many_arguments)]
207async fn build_api(
208    hoprd_cfg: serde_json::Value,
209    cfg: crate::config::Api,
210    hopr: Arc<hopr_lib::Hopr>,
211    open_listeners: ListenerJoinHandles,
212    default_listen_host: std::net::SocketAddr,
213) -> Router {
214    let state = AppState { hopr };
215    let inner_state = InternalState {
216        auth: Arc::new(cfg.auth.clone()),
217        hoprd_cfg,
218        hopr: state.hopr.clone(),
219        open_listeners,
220        default_listen_host,
221        websocket_active_count: Arc::new(AtomicU16::new(0)),
222    };
223
224    Router::new()
225        .merge(
226            Router::new()
227                .merge(SwaggerUi::new("/swagger-ui").url("/api-docs/openapi.json", ApiDoc::openapi()))
228                .merge(Scalar::with_url("/scalar", ApiDoc::openapi())),
229        )
230        .merge(
231            Router::new()
232                .route("/startedz", get(checks::startedz))
233                .route("/readyz", get(checks::readyz))
234                .route("/healthyz", get(checks::healthyz))
235                .route("/eligiblez", get(checks::eligiblez))
236                .layer(
237                    ServiceBuilder::new().layer(
238                        CorsLayer::new()
239                            .allow_methods([Method::GET])
240                            .allow_origin(Any)
241                            .allow_headers(Any)
242                            .max_age(std::time::Duration::from_secs(86400)),
243                    ),
244                )
245                .with_state(state.into()),
246        )
247        .merge(
248            Router::new()
249                .route("/metrics", get(root::metrics))
250                .layer(axum::middleware::from_fn_with_state(
251                    inner_state.clone(),
252                    middleware::preconditions::authenticate,
253                ))
254                .layer(axum::middleware::from_fn_with_state(
255                    inner_state.clone(),
256                    middleware::preconditions::cap_websockets,
257                ))
258                .layer(
259                    ServiceBuilder::new()
260                        .layer(TraceLayer::new_for_http())
261                        .layer(
262                            CorsLayer::new()
263                                .allow_methods([Method::GET])
264                                .allow_origin(Any)
265                                .allow_headers(Any)
266                                .max_age(std::time::Duration::from_secs(86400)),
267                        )
268                        .layer(axum::middleware::from_fn(middleware::prometheus::record))
269                        .layer(CompressionLayer::new())
270                        .layer(ValidateRequestHeaderLayer::accept("text/plain"))
271                        .layer(SetSensitiveRequestHeadersLayer::new(once(AUTHORIZATION))),
272                ),
273        )
274        .nest(
275            BASE_PATH,
276            Router::new()
277                .route("/account/addresses", get(account::addresses))
278                .route("/account/balances", get(account::balances))
279                .route("/account/withdraw", post(account::withdraw))
280                .route("/peers/{destination}", get(peers::show_peer_info))
281                .route("/channels", get(channels::list_channels))
282                .route("/channels", post(channels::open_channel))
283                .route("/channels/corrupted", get(channels::corrupted_channels))
284                .route("/channels/{channelId}", get(channels::show_channel))
285                .route("/channels/{channelId}/tickets", get(tickets::show_channel_tickets))
286                .route("/channels/{channelId}", delete(channels::close_channel))
287                .route("/channels/{channelId}/fund", post(channels::fund_channel))
288                .route(
289                    "/channels/{channelId}/tickets/redeem",
290                    post(tickets::redeem_tickets_in_channel),
291                )
292                .route(
293                    "/channels/{channelId}/tickets/aggregate",
294                    post(tickets::aggregate_tickets_in_channel),
295                )
296                .route("/tickets", get(tickets::show_all_tickets))
297                .route("/tickets/redeem", post(tickets::redeem_all_tickets))
298                .route("/tickets/statistics", get(tickets::show_ticket_statistics))
299                .route("/tickets/statistics", delete(tickets::reset_ticket_statistics))
300                .route("/network/price", get(network::price))
301                .route("/network/probability", get(network::probability))
302                .route("/node/version", get(node::version))
303                .route("/node/configuration", get(node::configuration))
304                .route("/node/info", get(node::info))
305                .route("/node/peers", get(node::peers))
306                .route("/node/entry-nodes", get(node::entry_nodes))
307                .route("/node/graph", get(node::channel_graph))
308                .route("/peers/{destination}/ping", post(peers::ping_peer))
309                .route("/session/config/{id}", get(session::session_config))
310                .route("/session/config/{id}", post(session::adjust_session))
311                .route("/session/websocket", get(session::websocket))
312                .route("/session/{protocol}", post(session::create_client))
313                .route("/session/{protocol}", get(session::list_clients))
314                .route("/session/{protocol}/{ip}/{port}", delete(session::close_client))
315                .with_state(inner_state.clone().into())
316                .layer(axum::middleware::from_fn_with_state(
317                    inner_state.clone(),
318                    middleware::preconditions::authenticate,
319                ))
320                .layer(axum::middleware::from_fn_with_state(
321                    inner_state.clone(),
322                    middleware::preconditions::cap_websockets,
323                ))
324                .layer(
325                    ServiceBuilder::new()
326                        .layer(TraceLayer::new_for_http())
327                        .layer(
328                            CorsLayer::new()
329                                .allow_methods([Method::GET, Method::POST, Method::OPTIONS, Method::DELETE])
330                                .allow_origin(Any)
331                                .allow_headers(Any)
332                                .max_age(std::time::Duration::from_secs(86400)),
333                        )
334                        .layer(axum::middleware::from_fn(middleware::prometheus::record))
335                        .layer(CompressionLayer::new())
336                        .layer(ValidateRequestHeaderLayer::accept("application/json"))
337                        .layer(SetSensitiveRequestHeadersLayer::new(once(AUTHORIZATION))),
338                ),
339        )
340}

342fn checksum_address_serializer<S: serde::Serializer>(a: &Address, s: S) -> Result<S::Ok, S::Error> {
343    s.serialize_str(&a.to_checksum())
344}
345
346fn option_checksum_address_serializer<S: serde::Serializer>(a: &Option<Address>, s: S) -> Result<S::Ok, S::Error> {
347    if let Some(addr) = a {
348        s.serialize_some(&addr.to_checksum())
349    } else {
350        s.serialize_none()
351    }
352}

354#[derive(Debug, Clone, Serialize, utoipa::ToSchema)]
355#[schema(example = json!({
356    "status": "INVALID_INPUT",
357    "error": "Invalid value passed in parameter 'XYZ'"
358}))]
359/// Standardized error response for the API
360pub(crate) struct ApiError {
361    #[schema(example = "INVALID_INPUT")]
362    pub status: String,
363    #[serde(skip_serializing_if = "Option::is_none")]
364    #[schema(example = "Invalid value passed in parameter 'XYZ'")]
365    pub error: Option<String>,
366}
367
368/// Enumerates all API request errors
369/// Note that `ApiError` should not be instantiated directly, but always rather through the `ApiErrorStatus`.
370#[derive(Debug, Clone, PartialEq, Eq, strum::Display)]
371#[strum(serialize_all = "SCREAMING_SNAKE_CASE")]
372enum ApiErrorStatus {
373    InvalidInput,
374    InvalidChannelId,
375    PeerNotFound,
376    ChannelNotFound,
377    TicketsNotFound,
378    NotEnoughBalance,
379    NotEnoughAllowance,
380    ChannelAlreadyOpen,
381    ChannelNotOpen,
382    UnsupportedFeature,
383    Timeout,
384    PingError(String),
385    Unauthorized,
386    TooManyOpenWebsocketConnections,
387    InvalidQuality,
388    NotReady,
389    ListenHostAlreadyUsed,
390    SessionNotFound,
391    InvalidSessionId,
392    #[strum(serialize = "UNKNOWN_FAILURE")]
393    UnknownFailure(String),
394}
395
396impl From<ApiErrorStatus> for ApiError {
397    fn from(value: ApiErrorStatus) -> Self {
398        Self {
399            status: value.to_string(),
400            error: if let ApiErrorStatus::UnknownFailure(e) = value {
401                Some(e)
402            } else {
403                None
404            },
405        }
406    }
407}
408
409impl IntoResponse for ApiErrorStatus {
410    fn into_response(self) -> Response {
411        Json(ApiError::from(self)).into_response()
412    }
413}
414
415impl IntoResponse for ApiError {
416    fn into_response(self) -> Response {
417        (StatusCode::INTERNAL_SERVER_ERROR, Json(self)).into_response()
418    }
419}
420
421// Errors lead to `UnknownFailure` per default
422impl<T: Error> From<T> for ApiErrorStatus {
423    fn from(value: T) -> Self {
424        Self::UnknownFailure(value.to_string())
425    }
426}
427
428// Errors lead to `UnknownFailure` per default
429impl<T> From<T> for ApiError
430where
431    T: Error + Into<HoprLibError>,
432{
433    fn from(value: T) -> Self {
434        Self {
435            status: ApiErrorStatus::UnknownFailure("unknown error".to_string()).to_string(),
436            error: Some(value.to_string()),
437        }
438    }
439}
