17fn is_a_websocket_uri(uri: &OriginalUri) -> bool {
18    const SESSION_PATH: &str = const_format::formatcp!("{BASE_PATH}/session/websocket");
19
20    uri.path().starts_with(SESSION_PATH)
21}

23pub(crate) async fn cap_websockets(
24    State(state): State<InternalState>,
25    uri: OriginalUri,
26    _headers: HeaderMap,
27    request: Request,
28    next: Next,
29) -> impl IntoResponse {
30    let max_websocket_count = std::env::var("HOPR_INTERNAL_REST_API_MAX_CONCURRENT_WEBSOCKET_COUNT")
31        .and_then(|v| v.parse::<u16>().map_err(|_e| std::env::VarError::NotPresent))
32        .unwrap_or(10);
33
34    if is_a_websocket_uri(&uri) {
35        let ws_count = state.websocket_active_count;
36
37        if ws_count.fetch_add(1, Relaxed) > max_websocket_count {
38            ws_count.fetch_sub(1, Relaxed);
39
40            return (
41                StatusCode::TOO_MANY_REQUESTS,
42                ApiErrorStatus::TooManyOpenWebsocketConnections,
43            )
44                .into_response();
45        }
46    }
47
48    // Go forward to the next middleware or request handler
49    next.run(request).await
50}

52pub(crate) async fn authenticate(
53    State(state): State<InternalState>,
54    uri: OriginalUri,
55    headers: HeaderMap,
56    request: Request,
57    next: Next,
58) -> impl IntoResponse {
59    let auth = state.auth.clone();
60
61    let x_auth_header = HeaderName::from_str("x-auth-token").expect("Invalid header name: x-auth-token");
62    let websocket_proto_header =
63        HeaderName::from_str("Sec-Websocket-Protocol").expect("Invalid header name: Sec-Websocket-Protocol");
64
65    let is_authorized = match auth.as_ref() {
66        Auth::Token(expected_token) => {
67            let auth_headers = headers
68                .iter()
69                .filter_map(|(n, v)| {
70                    (AUTHORIZATION.eq(n) || x_auth_header.eq(n) || websocket_proto_header.eq(n))
71                        .then_some((n, v.to_str().expect("Invalid header value")))
72                })
73                .collect::<Vec<_>>();
74
75            let is_ws_auth = if is_a_websocket_uri(&uri) {
76                uri.query()
77                    .map(|q| {
78                        // Reasonable limit for query string
79                        if q.len() > 2048 {
80                            return false;
81                        }
82                        match decode(q) {
83                            Ok(decoded) => decoded.into_owned().contains(&format!("apiToken={expected_token}")),
84                            Err(_) => false,
85                        }
86                    })
87                    .unwrap_or(false)
88            } else {
89                false
90            };
91            // Use "Authorization Bearer <token>" and "X-Auth-Token <token>" headers and "Sec-Websocket-Protocol"
92            (!auth_headers.is_empty()
93                    && (auth_headers.contains(&(&AUTHORIZATION, &format!("Bearer {expected_token}")))
94                        || auth_headers.contains(&(&x_auth_header, expected_token)))
95                )
96                // The following line would never be needed, if the JavaScript browser was able to properly
97                // pass the x-auth-token or Bearer headers.
98                || is_ws_auth
99        }
100        Auth::None => true,
101    };
102
103    if !is_authorized {
104        return (StatusCode::UNAUTHORIZED, ApiErrorStatus::Unauthorized).into_response();
105    }
106
107    // Go forward to the next middleware or request handler
108    next.run(request).await
109}
