139/// Lists information for `connected peers` and `announced peers`.
140///
141/// Connected peers are nodes which are connected to the node while announced peers are
142/// nodes which have announced to the network.
143///
144/// Optionally pass `quality` parameter to get only peers with higher or equal quality
145/// to the specified value.
146#[utoipa::path(
147        get,
148        path = const_format::formatcp!("{BASE_PATH}/node/peers"),
149        params(NodePeersQueryRequest),
150        responses(
151            (status = 200, description = "Successfully returned observed peers", body = NodePeersResponse),
152            (status = 400, description = "Failed to extract a valid quality parameter", body = ApiError),
153            (status = 401, description = "Invalid authorization token.", body = ApiError),
154        ),
155        security(
156            ("api_token" = []),
157            ("bearer_token" = [])
158        ),
159        tag = "Node"
160    )]
161pub(super) async fn peers(
162    Query(NodePeersQueryRequest { quality }): Query<NodePeersQueryRequest>,
163    State(state): State<Arc<InternalState>>,
164) -> Result<impl IntoResponse, ApiError> {
165    if !(0.0f64..=1.0f64).contains(&quality) {
166        return Ok((StatusCode::BAD_REQUEST, ApiErrorStatus::InvalidQuality).into_response());
167    }
168
169    let hopr = state.hopr.clone();
170
171    let all_network_peers = futures::stream::iter(hopr.network_connected_peers().await?)
172        .filter_map(|peer| {
173            let hopr = hopr.clone();
174
175            async move {
176                if let Ok(Some(info)) = hopr.network_peer_info(&peer).await {
177                    let avg_quality = info.get_average_quality();
178                    if avg_quality >= quality {
179                        Some((peer, info))
180                    } else {
181                        None
182                    }
183                } else {
184                    None
185                }
186            }
187        })
188        .filter_map(|(peer_id, info)| {
189            let hopr = hopr.clone();
190
191            async move {
192                let address = hopr.peerid_to_chain_key(&peer_id).await.ok().flatten();
193
194                // WARNING: Only in Providence and Saint-Louis are all peers public
195                let multiaddresses = hopr.network_observed_multiaddresses(&peer_id).await;
196
197                Some((address, peer_id, multiaddresses, info))
198            }
199        })
200        .map(|(address, peer_id, mas, info)| PeerInfo {
201            peer_id,
202            peer_address: address,
203            multiaddr: mas.first().cloned(),
204            heartbeats: HeartbeatInfo {
205                sent: info.heartbeats_sent,
206                success: info.heartbeats_succeeded,
207            },
208            last_seen: info.last_seen.as_unix_timestamp().as_millis(),
209            last_seen_latency: info.last_seen_latency.as_millis() / 2,
210            quality: info.get_average_quality(),
211            backoff: info.backoff,
212            is_new: info.heartbeats_sent == 0u64,
213            reported_version: info.peer_version.unwrap_or("UNKNOWN".to_string()),
214        })
215        .collect::<Vec<_>>()
216        .await;
217
218    let announced_peers = hopr
219        .accounts_announced_on_chain()
220        .await?
221        .into_iter()
222        .map(|announced| async move {
223            AnnouncedPeer {
224                peer_id: announced.public_key.into(),
225                peer_address: announced.chain_addr,
226                multiaddr: announced.get_multiaddr(),
227            }
228        })
229        .collect::<FuturesUnordered<_>>()
230        .collect()
231        .await;
232
233    let body = NodePeersResponse {
234        connected: all_network_peers,
235        announced: announced_peers,
236    };
237
238    Ok((StatusCode::OK, Json(body)).into_response())
239}

404/// Get information about this HOPR Node.
405#[utoipa::path(
406        get,
407        path = const_format::formatcp!("{BASE_PATH}/node/info"),
408        responses(
409            (status = 200, description = "Fetched node version", body = NodeInfoResponse),
410            (status = 422, description = "Unknown failure", body = ApiError)

418pub(super) async fn info(State(state): State<Arc<InternalState>>) -> Result<impl IntoResponse, ApiError> {
419    let hopr = state.hopr.clone();
420
421    let chain_config = hopr.chain_config();
422    let safe_config = hopr.get_safe_config();
423    let network = hopr.network();
424
425    let indexer_state_info = match hopr.get_indexer_state().await {
426        Ok(info) => info,
427        Err(error) => return Ok((StatusCode::UNPROCESSABLE_ENTITY, ApiErrorStatus::from(error)).into_response()),
428    };
429
430    match hopr.get_channel_closure_notice_period().await {
431        Ok(channel_closure_notice_period) => {
432            let body = NodeInfoResponse {
433                network,
434                announced_address: hopr.local_multiaddresses(),
435                listening_address: hopr.local_multiaddresses(),
436                chain: chain_config.id,
437                provider: hopr.get_provider(),
438                hopr_token: chain_config.token,
439                hopr_channels: chain_config.channels,
440                hopr_network_registry: chain_config.network_registry,
441                hopr_node_safe_registry: chain_config.node_safe_registry,
442                hopr_management_module: chain_config.module_implementation,
443                hopr_node_safe: safe_config.safe_address,
444                is_eligible: hopr.is_allowed_to_access_network(&hopr.me_peer_id()).await?,
445                connectivity_status: hopr.network_health().await,
446                channel_closure_period: channel_closure_notice_period.as_secs(),
447                indexer_block: indexer_state_info.latest_block_number,
448                indexer_last_log_block: indexer_state_info.latest_log_block_number,
449                indexer_last_log_checksum: indexer_state_info.latest_log_checksum,
450            };
451
452            Ok((StatusCode::OK, Json(body)).into_response())
453        }
454        Err(error) => Ok((StatusCode::UNPROCESSABLE_ENTITY, ApiErrorStatus::from(error)).into_response()),
455    }
456}

488pub(super) async fn entry_nodes(State(state): State<Arc<InternalState>>) -> Result<impl IntoResponse, ApiError> {

490
491    match hopr.get_public_nodes().await {
492        Ok(nodes) => {
493            let mut body = HashMap::new();
494            for (peer_id, address, mas) in nodes.into_iter() {
495                body.insert(
496                    address.to_string(),
497                    EntryNode {
498                        multiaddrs: mas,
499                        is_eligible: hopr.is_allowed_to_access_network(&peer_id).await?,
500                    },
501                );
502            }
503
504            Ok((StatusCode::OK, Json(body)).into_response())
505        }
506        Err(error) => Ok((StatusCode::UNPROCESSABLE_ENTITY, ApiErrorStatus::from(error)).into_response()),
507    }
