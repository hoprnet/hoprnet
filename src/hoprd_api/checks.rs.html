53fn is_running(state: Arc<AppState>) -> impl IntoResponse {
54    match state.hopr.status() {
55        HoprState::Running => (StatusCode::OK, "").into_response(),
56        _ => (StatusCode::PRECONDITION_FAILED, "").into_response(),
57    }
58}
