50impl std::fmt::Display for SessionTargetSpec {
51    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
52        match self {
53            SessionTargetSpec::Plain(t) => write!(f, "{t}"),
54            SessionTargetSpec::Sealed(t) => write!(f, "$${}", base64::prelude::BASE64_URL_SAFE.encode(t)),
55            SessionTargetSpec::Service(t) => write!(f, "#{t}"),
56        }
57    }
58}

60impl FromStr for SessionTargetSpec {
61    type Err = HoprLibError;
62
63    fn from_str(s: &str) -> Result<Self, Self::Err> {
64        Ok(if let Some(stripped) = s.strip_prefix("$$") {
65            Self::Sealed(
66                base64::prelude::BASE64_URL_SAFE
67                    .decode(stripped)
68                    .map_err(|e| HoprLibError::GeneralError(e.to_string()))?,
69            )
70        } else if let Some(stripped) = s.strip_prefix("#") {
71            Self::Service(
72                stripped
73                    .parse()
74                    .map_err(|_| HoprLibError::GeneralError("cannot parse service id".into()))?,
75            )
76        } else {
77            Self::Plain(s.to_owned())
78        })
79    }
80}

188/// Request body for creating a new client session.
189pub(crate) struct SessionClientRequest {
190    /// Address of the Exit node.
191    #[serde_as(as = "DisplayFromStr")]
192    #[schema(value_type = String)]
193    pub destination: Address,
194    /// The forward path for the Session.
195    pub forward_path: RoutingOptions,
196    /// The return path for the Session.
197    pub return_path: RoutingOptions,
198    /// Target for the Session.
199    pub target: SessionTargetSpec,
200    /// Listen host (`ip:port`) for the Session socket at the Entry node.
201    ///
202    /// Supports also partial specification (only `ip` or only `:port`) with the
203    /// respective part replaced by the node's configured default.
204    pub listen_host: Option<String>,
205    #[serde_as(as = "Option<Vec<DisplayFromStr>>")]
206    /// Capabilities for the Session protocol.
207    ///
208    /// Defaults to `Segmentation` and `Retransmission` for TCP and nothing for UDP.
209    pub capabilities: Option<Vec<SessionCapability>>,
210    /// The amount of response data the Session counterparty can deliver back to us,
211    /// without us sending any SURBs to them.
212    ///
213    /// In other words, this size is recalculated to a number of SURBs delivered
214    /// to the counterparty upfront and then maintained.
215    /// The maintenance is dynamic, based on the number of responses we receive.
216    ///
217    /// All syntaxes like "2 MB", "128 kiB", "3MiB" are supported. The value must be
218    /// at least the size of 2 Session packet payloads.
219    #[serde_as(as = "Option<DisplayFromStr>")]
220    #[schema(value_type = Option<String>)]
221    pub response_buffer: Option<bytesize::ByteSize>,
222    /// The maximum throughput at which artificial SURBs might be generated and sent
223    /// to the recipient of the Session.
224    ///
225    /// On Sessions that rarely send data but receive a lot (= Exit node has high SURB consumption),
226    /// this should roughly match the maximum retrieval throughput.
227    ///
228    /// All syntaxes like "2 MBps", "1.2Mbps", "300 kb/s", "1.23 Mb/s" are supported.
229    #[serde(default)]
230    #[serde(with = "human_bandwidth::option")]
231    #[schema(value_type = Option<String>)]
232    pub max_surb_upstream: Option<human_bandwidth::re::bandwidth::Bandwidth>,
233    /// How many Sessions to pool for clients.
234    ///
235    /// If no sessions are pooled, they will be opened ad-hoc when a client connects.
236    /// It has no effect on UDP sessions in the current implementation.
237    ///
238    /// Currently, the maximum value is 5.
239    pub session_pool: Option<usize>,
240    /// The maximum number of client sessions that the listener can spawn.
241    ///
242    /// This currently applies only to the TCP sessions, as UDP sessions cannot
243    /// handle multiple clients (and spawn therefore always only a single session).
244    ///
245    /// If this value is smaller than the value specified in `session_pool`, it will
246    /// be set to that value.
247    ///
248    /// The default value is 5.
249    pub max_client_sessions: Option<usize>,
250}

252impl SessionClientRequest {
253    /// Converts the API client session request into protocol-level session configuration.
254    pub(crate) async fn into_protocol_session_config(
255        self,
256        target_protocol: IpProtocol,
257    ) -> Result<(hopr_lib::Address, SessionTarget, SessionClientConfig), ApiErrorStatus> {
258        let target_spec: hopr_utils_session::SessionTargetSpec = self.target.clone().into();
259        Ok((
260            self.destination,
261            target_spec.into_target(target_protocol.into())?,
262            SessionClientConfig {
263                forward_path_options: self.forward_path.resolve().await?,
264                return_path_options: self.return_path.resolve().await?,
265                capabilities: self
266                    .capabilities
267                    .map(|vs| {
268                        let mut caps = SessionCapabilities::empty();
269                        caps.extend(vs.into_iter().map(SessionCapabilities::from));
270                        caps
271                    })
272                    .unwrap_or_else(|| match target_protocol {
273                        IpProtocol::TCP => {
274                            hopr_lib::SessionCapability::RetransmissionAck
275                                | hopr_lib::SessionCapability::RetransmissionNack
276                                | hopr_lib::SessionCapability::Segmentation
277                        }
278                        // Only Segmentation capability for UDP per default
279                        _ => SessionCapability::Segmentation.into(),
280                    }),
281                surb_management: SessionConfig {
282                    response_buffer: self.response_buffer,
283                    max_surb_upstream: self.max_surb_upstream,
284                }
285                .into(),
286                ..Default::default()
287            },
288        ))
289    }
290}

365/// Creates a new client session returning the given session listening host and port over TCP or UDP.
366/// If no listening port is given in the request, the socket will be bound to a random free
367/// port and returned in the response.
368/// Different capabilities can be configured for the session, such as data segmentation or
369/// retransmission.
370///
371/// Once the host and port are bound, it is possible to use the socket for bidirectional read/write
372/// communication over the selected IP protocol and HOPR network routing with the given destination.
373/// The destination HOPR node forwards all the data to the given target over the selected IP protocol.
374///
375/// Various services require different types of socket communications:
376/// - services running over UDP usually do not require data retransmission, as it is already expected
377/// that UDP does not provide these and is therefore handled at the application layer.
378/// - On the contrary, services running over TCP *almost always* expect data segmentation and
379/// retransmission capabilities, so these should be configured while creating a session that passes
380/// TCP data.
381#[utoipa::path(
382        post,
383        path = const_format::formatcp!("{BASE_PATH}/session/{{protocol}}"),
384        description = "Creates a new client HOPR session that will start listening on a dedicated port. Once the port is bound, it is possible to use the socket for bidirectional read and write communication.",
385        params(
386            ("protocol" = String, Path, description = "IP transport protocol", example = "tcp"),
387        ),
388        request_body(
389            content = SessionClientRequest,
390            description = "Creates a new client HOPR session that will start listening on a dedicated port. Once the port is bound, it is possible to use the socket for bidirectional read and write communication.",
391            content_type = "application/json"),
392        responses(
393            (status = 200, description = "Successfully created a new client session.", body = SessionClientResponse),
394            (status = 400, description = "Invalid IP protocol.", body = ApiError),
395            (status = 401, description = "Invalid authorization token.", body = ApiError),
396            (status = 409, description = "Listening address and port already in use.", body = ApiError),
397            (status = 422, description = "Unknown failure", body = ApiError),
398        ),
399        security(
400            ("api_token" = []),
401            ("bearer_token" = [])
402        ),
403        tag = "Session"
404    )]
405pub(crate) async fn create_client(
406    State(state): State<Arc<InternalState>>,
407    Path(protocol): Path<IpProtocol>,
408    Json(args): Json<SessionClientRequest>,
409) -> Result<impl IntoResponse, impl IntoResponse> {
410    let bind_host: std::net::SocketAddr = build_binding_host(args.listen_host.as_deref(), state.default_listen_host);
411
412    let listener_id = ListenerId(protocol.into(), bind_host);
413    if bind_host.port() > 0 && state.open_listeners.0.contains_key(&listener_id) {
414        return Err((StatusCode::CONFLICT, ApiErrorStatus::ListenHostAlreadyUsed));
415    }
416
417    let port_range = std::env::var(crate::env::HOPRD_SESSION_PORT_RANGE).ok();
418    tracing::debug!(%protocol, %bind_host, ?port_range, "binding session listening socket");
419
420    let (bound_host, udp_session_id, max_clients) = match protocol {
421        IpProtocol::TCP => {
422            let session_pool = args.session_pool;
423            let max_client_sessions = args.max_client_sessions;
424            let target_spec: hopr_utils_session::SessionTargetSpec = args.target.clone().into();
425            let (destination, _target, config) = args
426                .clone()
427                .into_protocol_session_config(IpProtocol::TCP)
428                .await
429                .map_err(|e| (StatusCode::UNPROCESSABLE_ENTITY, e))?;
430
431            create_tcp_client_binding(
432                bind_host,
433                port_range,
434                state.hopr.clone(),
435                state.open_listeners.clone(),
436                destination,
437                target_spec,
438                config,
439                session_pool,
440                max_client_sessions,
441            )
442            .await
443            .map_err(|e| match e {
444                hopr_utils_session::BindError::ListenHostAlreadyUsed => {
445                    (StatusCode::CONFLICT, ApiErrorStatus::ListenHostAlreadyUsed)
446                }
447                hopr_utils_session::BindError::UnknownFailure(_) => (
448                    StatusCode::UNPROCESSABLE_ENTITY,
449                    ApiErrorStatus::UnknownFailure(format!("failed to start TCP listener on {bind_host}: {e}")),
450                ),
451            })?
452        }
453        IpProtocol::UDP => {
454            let target_spec: hopr_utils_session::SessionTargetSpec = args.target.clone().into();
455            let (destination, _target, config) = args
456                .clone()
457                .into_protocol_session_config(IpProtocol::UDP)
458                .await
459                .map_err(|e| (StatusCode::UNPROCESSABLE_ENTITY, e))?;
460
461            create_udp_client_binding(
462                bind_host,
463                port_range,
464                state.hopr.clone(),
465                state.open_listeners.clone(),
466                destination,
467                target_spec,
468                config,
469            )
470            .await
471            .map_err(|e| match e {
472                hopr_utils_session::BindError::ListenHostAlreadyUsed => {
473                    (StatusCode::CONFLICT, ApiErrorStatus::ListenHostAlreadyUsed)
474                }
475                hopr_utils_session::BindError::UnknownFailure(_) => (
476                    StatusCode::UNPROCESSABLE_ENTITY,
477                    ApiErrorStatus::UnknownFailure(format!("failed to start UDP listener on {bind_host}: {e}")),
478                ),
479            })?
480        }
481    };
482
483    Ok::<_, (StatusCode, ApiErrorStatus)>(
484        (
485            StatusCode::OK,
486            Json(SessionClientResponse {
487                protocol,
488                ip: bound_host.ip().to_string(),
489                port: bound_host.port(),
490                target: args.target.to_string(),
491                destination: args.destination,
492                forward_path: args.forward_path.clone(),
493                return_path: args.return_path.clone(),
494                hopr_mtu: SESSION_MTU,
495                surb_len: SURB_SIZE,
496                active_clients: udp_session_id.into_iter().map(|s| s.to_string()).collect(),
497                max_client_sessions: max_clients,
498                max_surb_upstream: args.max_surb_upstream,
499                response_buffer: args.response_buffer,
500                session_pool: args.session_pool,
501            }),
502        )
503            .into_response(),
504    )
505}

544pub(crate) async fn list_clients(
545    State(state): State<Arc<InternalState>>,
546    Path(protocol): Path<IpProtocol>,
547) -> Result<impl IntoResponse, impl IntoResponse> {
548    let response = state
549        .open_listeners
550        .0
551        .iter()
552        .filter(|v| v.key().0 == protocol.into())
553        .map(|v| {
554            let ListenerId(_, addr) = *v.key();
555            let entry = v.value();
556            SessionClientResponse {
557                protocol,
558                ip: addr.ip().to_string(),
559                port: addr.port(),
560                target: entry.target.to_string(),
561                forward_path: entry.forward_path.clone().into(),
562                return_path: entry.return_path.clone().into(),
563                destination: entry.destination,
564                hopr_mtu: SESSION_MTU,
565                surb_len: SURB_SIZE,
566                active_clients: entry.get_clients().iter().map(|e| e.key().to_string()).collect(),
567                max_client_sessions: entry.max_client_sessions,
568                max_surb_upstream: entry.max_surb_upstream,
569                response_buffer: entry.response_buffer,
570                session_pool: entry.session_pool,
571            }
572        })
573        .collect::<Vec<_>>();
574
575    Ok::<_, (StatusCode, ApiErrorStatus)>((StatusCode::OK, Json(response)).into_response())
576}

596    #[serde_as(as = "Option<DisplayFromStr>")]
597    #[schema(value_type = String)]
598    pub response_buffer: Option<bytesize::ByteSize>,
599    /// The maximum throughput at which artificial SURBs might be generated and sent
600    /// to the recipient of the Session.
601    ///
602    /// On Sessions that rarely send data but receive a lot (= Exit node has high SURB consumption),
603    /// this should roughly match the maximum retrieval throughput.
604    ///
605    /// All syntaxes like "2 MBps", "1.2Mbps", "300 kb/s", "1.23 Mb/s" are supported.
606    #[serde(default)]
607    #[serde(with = "human_bandwidth::option")]
608    #[schema(value_type = String)]
609    pub max_surb_upstream: Option<human_bandwidth::re::bandwidth::Bandwidth>,
610}
611
612impl From<SessionConfig> for Option<SurbBalancerConfig> {
613    /// Converts the API session config into protocol-level SURB balancer config.
614    fn from(value: SessionConfig) -> Self {
615        match value.response_buffer {
616            // Buffer worth at least 2 reply packets
617            Some(buffer_size) if buffer_size.as_u64() >= 2 * SESSION_MTU as u64 => Some(SurbBalancerConfig {
618                target_surb_buffer_size: buffer_size.as_u64() / SESSION_MTU as u64,
619                max_surbs_per_sec: value
620                    .max_surb_upstream
621                    .map(|b| (b.as_bps() as usize / (8 * SURB_SIZE)) as u64)
622                    .unwrap_or_else(|| SurbBalancerConfig::default().max_surbs_per_sec),
623                ..Default::default()
624            }),
625            // No additional SURBs are set up and maintained, useful for high-send low-reply sessions
626            Some(_) => None,
627            // Use defaults otherwise
628            None => Some(SurbBalancerConfig::default()),
629        }
630    }
631}

1046/// Closes an existing Session listener.
1047/// The listener must've been previously created and bound for the given IP protocol.
1048/// Once a listener is closed, no more socket connections can be made to it.
1049/// If the passed port number is 0, listeners on all ports of the given listening IP and protocol
1050/// will be closed.
1051#[utoipa::path(
1052    delete,
1053    path = const_format::formatcp!("{BASE_PATH}/session/{{protocol}}/{{ip}}/{{port}}"),
1054    description = "Closes an existing Session listener.",
1055    params(SessionCloseClientQuery),
1056    responses(
1057            (status = 204, description = "Listener closed successfully"),
1058            (status = 400, description = "Invalid IP protocol or port.", body = ApiError),
1059            (status = 401, description = "Invalid authorization token.", body = ApiError),
1060            (status = 404, description = "Listener not found.", body = ApiError),
1061            (status = 422, description = "Unknown failure", body = ApiError)
1062    ),
1063    security(
1064            ("api_token" = []),
1065            ("bearer_token" = [])
1066    ),
1067    tag = "Session",
1068)]
1069pub(crate) async fn close_client(
1070    State(state): State<Arc<InternalState>>,
1071    Path(SessionCloseClientQuery { protocol, ip, port }): Path<SessionCloseClientQuery>,
1072) -> Result<impl IntoResponse, impl IntoResponse> {
1073    let listening_ip: IpAddr = ip
1074        .parse()
1075        .map_err(|_| (StatusCode::BAD_REQUEST, ApiErrorStatus::InvalidInput))?;
1076
1077    {
1078        let open_listeners = &state.open_listeners.0;
1079
1080        let mut to_remove = Vec::new();
1081        let protocol: hopr_lib::IpProtocol = protocol.into();
1082
1083        // Find all listeners with protocol, listening IP and optionally port number (if > 0)
1084        open_listeners
1085            .iter()
1086            .filter(|v| {
1087                let ListenerId(proto, addr) = v.key();
1088                protocol == *proto && addr.ip() == listening_ip && (addr.port() == port || port == 0)
1089            })
1090            .for_each(|v| to_remove.push(*v.key()));
1091
1092        if to_remove.is_empty() {
1093            return Err((StatusCode::NOT_FOUND, ApiErrorStatus::InvalidInput));
1094        }
1095
1096        for bound_addr in to_remove {
1097            let (_, entry) = open_listeners
1098                .remove(&bound_addr)
1099                .ok_or((StatusCode::NOT_FOUND, ApiErrorStatus::InvalidInput))?;
1100
1101            entry.abort_handle.abort();
1102        }
1103    }
1104
1105    Ok::<_, (StatusCode, ApiErrorStatus)>((StatusCode::NO_CONTENT, "").into_response())
