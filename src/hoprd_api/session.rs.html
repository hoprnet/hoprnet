40impl std::fmt::Display for SessionTargetSpec {
41    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
42        match self {
43            SessionTargetSpec::Plain(t) => write!(f, "{t}"),
44            SessionTargetSpec::Sealed(t) => write!(f, "$${}", base64::prelude::BASE64_URL_SAFE.encode(t)),
45            SessionTargetSpec::Service(t) => write!(f, "#{t}"),
46        }
47    }
48}

50impl FromStr for SessionTargetSpec {
51    type Err = HoprLibError;
52
53    fn from_str(s: &str) -> Result<Self, Self::Err> {
54        Ok(if let Some(stripped) = s.strip_prefix("$$") {
55            Self::Sealed(
56                base64::prelude::BASE64_URL_SAFE
57                    .decode(stripped)
58                    .map_err(|e| HoprLibError::GeneralError(e.to_string()))?,
59            )
60        } else if let Some(stripped) = s.strip_prefix("#") {
61            Self::Service(
62                stripped
63                    .parse()
64                    .map_err(|_| HoprLibError::GeneralError("cannot parse service id".into()))?,
65            )
66        } else {
67            Self::Plain(s.to_owned())
68        })
69    }
70}

95/// Session capabilities that can be negotiated with the target peer.
96pub enum SessionCapability {
97    /// Frame segmentation
98    Segmentation,
99    /// Frame retransmission (ACK and NACK-based)
100    Retransmission,
101    /// Frame retransmission (only ACK-based)
102    RetransmissionAckOnly,
103    /// Disable packet buffering
104    NoDelay,
105    /// Disable SURB-based egress rate control at the Exit.
106    NoRateControl,
107}

109impl From<SessionCapability> for hopr_lib::SessionCapabilities {
110    fn from(cap: SessionCapability) -> hopr_lib::SessionCapabilities {
111        match cap {
112            SessionCapability::Segmentation => hopr_lib::SessionCapability::Segmentation.into(),

114                hopr_lib::SessionCapability::RetransmissionNack | hopr_lib::SessionCapability::RetransmissionAck
115            }
116            SessionCapability::RetransmissionAckOnly => hopr_lib::SessionCapability::RetransmissionAck.into(),
117            SessionCapability::NoDelay => hopr_lib::SessionCapability::NoDelay.into(),
118            SessionCapability::NoRateControl => hopr_lib::SessionCapability::NoRateControl.into(),
119        }
120    }
121}

162/// Request body for creating a new client session.
163pub(crate) struct SessionClientRequest {
164    /// Address of the Exit node.
165    #[serde_as(as = "DisplayFromStr")]
166    #[schema(value_type = String)]
167    pub destination: Address,
168    /// The forward path for the Session.
169    pub forward_path: RoutingOptions,
170    /// The return path for the Session.
171    pub return_path: RoutingOptions,
172    /// Target for the Session.
173    pub target: SessionTargetSpec,
174    /// Listen host (`ip:port`) for the Session socket at the Entry node.
175    ///
176    /// Supports also partial specification (only `ip` or only `:port`) with the
177    /// respective part replaced by the node's configured default.
178    pub listen_host: Option<String>,
179    #[serde_as(as = "Option<Vec<DisplayFromStr>>")]
180    /// Capabilities for the Session protocol.
181    ///
182    /// Defaults to `Segmentation` and `Retransmission` for TCP and nothing for UDP.
183    pub capabilities: Option<Vec<SessionCapability>>,
184    /// The amount of response data the Session counterparty can deliver back to us,
185    /// without us sending any SURBs to them.
186    ///
187    /// In other words, this size is recalculated to a number of SURBs delivered
188    /// to the counterparty upfront and then maintained.
189    /// The maintenance is dynamic, based on the number of responses we receive.
190    ///
191    /// All syntaxes like "2 MB", "128 kiB", "3MiB" are supported. The value must be
192    /// at least the size of 2 Session packet payloads.
193    #[serde_as(as = "Option<DisplayFromStr>")]
194    #[schema(value_type = Option<String>)]
195    pub response_buffer: Option<bytesize::ByteSize>,
196    /// The maximum throughput at which artificial SURBs might be generated and sent
197    /// to the recipient of the Session.
198    ///
199    /// On Sessions that rarely send data but receive a lot (= Exit node has high SURB consumption),
200    /// this should roughly match the maximum retrieval throughput.
201    ///
202    /// All syntaxes like "2 MBps", "1.2Mbps", "300 kb/s", "1.23 Mb/s" are supported.
203    #[serde(default)]
204    #[serde(with = "human_bandwidth::option")]
205    #[schema(value_type = Option<String>)]
206    pub max_surb_upstream: Option<human_bandwidth::re::bandwidth::Bandwidth>,
207    /// How many Sessions to pool for clients.
208    ///
209    /// If no sessions are pooled, they will be opened ad-hoc when a client connects.
210    /// It has no effect on UDP sessions in the current implementation.
211    ///
212    /// Currently, the maximum value is 5.
213    pub session_pool: Option<usize>,
214    /// The maximum number of client sessions that the listener can spawn.
215    ///
216    /// This currently applies only to the TCP sessions, as UDP sessions cannot
217    /// handle multiple clients (and spawn therefore always only a single session).
218    ///
219    /// If this value is smaller than the value specified in `session_pool`, it will
220    /// be set to that value.
221    ///
222    /// The default value is 5.
223    pub max_client_sessions: Option<usize>,
224}

226impl SessionClientRequest {
227    /// Converts the API client session request into protocol-level session configuration.
228    pub(crate) async fn into_protocol_session_config(
229        self,
230        target_protocol: IpProtocol,
231    ) -> Result<(hopr_lib::Address, SessionTarget, HoprSessionClientConfig), ApiErrorStatus> {
232        let target_spec: hopr_utils_session::SessionTargetSpec = self.target.clone().into();
233        Ok((
234            self.destination,
235            target_spec.into_target(target_protocol.into())?,
236            HoprSessionClientConfig {
237                forward_path: self.forward_path.try_into()?,
238                return_path: self.return_path.try_into()?,
239                capabilities: self
240                    .capabilities
241                    .map(|vs| {
242                        let mut caps = SessionCapabilities::empty();
243                        caps.extend(vs.into_iter().map(SessionCapabilities::from));
244                        caps
245                    })
246                    .unwrap_or_else(|| match target_protocol {
247                        IpProtocol::TCP => {
248                            hopr_lib::SessionCapability::RetransmissionAck
249                                | hopr_lib::SessionCapability::RetransmissionNack
250                                | hopr_lib::SessionCapability::Segmentation
251                        }
252                        // Only Segmentation capability for UDP per default
253                        _ => SessionCapability::Segmentation.into(),
254                    }),
255                surb_management: SessionConfig {
256                    response_buffer: self.response_buffer,
257                    max_surb_upstream: self.max_surb_upstream,
258                }
259                .into(),
260                ..Default::default()
261            },
262        ))
263    }
264}

339/// Creates a new client session returning the given session listening host and port over TCP or UDP.
340/// If no listening port is given in the request, the socket will be bound to a random free
341/// port and returned in the response.
342/// Different capabilities can be configured for the session, such as data segmentation or
343/// retransmission.
344///
345/// Once the host and port are bound, it is possible to use the socket for bidirectional read/write
346/// communication over the selected IP protocol and HOPR network routing with the given destination.
347/// The destination HOPR node forwards all the data to the given target over the selected IP protocol.
348///
349/// Various services require different types of socket communications:
350/// - services running over UDP usually do not require data retransmission, as it is already expected
351/// that UDP does not provide these and is therefore handled at the application layer.
352/// - On the contrary, services running over TCP *almost always* expect data segmentation and
353/// retransmission capabilities, so these should be configured while creating a session that passes
354/// TCP data.
355#[utoipa::path(
356        post,
357        path = const_format::formatcp!("{BASE_PATH}/session/{{protocol}}"),
358        description = "Creates a new client HOPR session that will start listening on a dedicated port. Once the port is bound, it is possible to use the socket for bidirectional read and write communication.",
359        params(
360            ("protocol" = String, Path, description = "IP transport protocol", example = "tcp"),
361        ),
362        request_body(
363            content = SessionClientRequest,
364            description = "Creates a new client HOPR session that will start listening on a dedicated port. Once the port is bound, it is possible to use the socket for bidirectional read and write communication.",
365            content_type = "application/json"),
366        responses(
367            (status = 200, description = "Successfully created a new client session.", body = SessionClientResponse),
368            (status = 400, description = "Invalid IP protocol.", body = ApiError),
369            (status = 401, description = "Invalid authorization token.", body = ApiError),
370            (status = 409, description = "Listening address and port already in use.", body = ApiError),
371            (status = 422, description = "Unknown failure", body = ApiError),
372        ),
373        security(
374            ("api_token" = []),
375            ("bearer_token" = [])
376        ),
377        tag = "Session"
378    )]
379pub(crate) async fn create_client(
380    State(state): State<Arc<InternalState>>,
381    Path(protocol): Path<IpProtocol>,
382    Json(args): Json<SessionClientRequest>,
383) -> Result<impl IntoResponse, impl IntoResponse> {
384    let bind_host: std::net::SocketAddr = build_binding_host(args.listen_host.as_deref(), state.default_listen_host);
385
386    let listener_id = ListenerId(protocol.into(), bind_host);
387    if bind_host.port() > 0 && state.open_listeners.0.contains_key(&listener_id) {
388        return Err((StatusCode::CONFLICT, ApiErrorStatus::ListenHostAlreadyUsed));
389    }
390
391    let port_range = std::env::var(crate::env::HOPRD_SESSION_PORT_RANGE).ok();
392    tracing::debug!(%protocol, %bind_host, ?port_range, "binding session listening socket");
393
394    let (bound_host, udp_session_id, max_clients) = match protocol {
395        IpProtocol::TCP => {
396            let session_pool = args.session_pool;
397            let max_client_sessions = args.max_client_sessions;
398            let target_spec: hopr_utils_session::SessionTargetSpec = args.target.clone().into();
399            let (destination, _target, config) = args
400                .clone()
401                .into_protocol_session_config(IpProtocol::TCP)
402                .await
403                .map_err(|e| (StatusCode::UNPROCESSABLE_ENTITY, e))?;
404
405            create_tcp_client_binding(
406                bind_host,
407                port_range,
408                state.hopr.clone(),
409                state.open_listeners.clone(),
410                destination,
411                target_spec,
412                config,
413                session_pool,
414                max_client_sessions,
415            )
416            .await
417            .map_err(|e| match e {
418                hopr_utils_session::BindError::ListenHostAlreadyUsed => {
419                    (StatusCode::CONFLICT, ApiErrorStatus::ListenHostAlreadyUsed)
420                }
421                hopr_utils_session::BindError::UnknownFailure(_) => (
422                    StatusCode::UNPROCESSABLE_ENTITY,
423                    ApiErrorStatus::UnknownFailure(format!("failed to start TCP listener on {bind_host}: {e}")),
424                ),
425            })?
426        }
427        IpProtocol::UDP => {
428            let target_spec: hopr_utils_session::SessionTargetSpec = args.target.clone().into();
429            let (destination, _target, config) = args
430                .clone()
431                .into_protocol_session_config(IpProtocol::UDP)
432                .await
433                .map_err(|e| (StatusCode::UNPROCESSABLE_ENTITY, e))?;
434
435            create_udp_client_binding(
436                bind_host,
437                port_range,
438                state.hopr.clone(),
439                state.open_listeners.clone(),
440                destination,
441                target_spec,
442                config,
443            )
444            .await
445            .map_err(|e| match e {
446                hopr_utils_session::BindError::ListenHostAlreadyUsed => {
447                    (StatusCode::CONFLICT, ApiErrorStatus::ListenHostAlreadyUsed)
448                }
449                hopr_utils_session::BindError::UnknownFailure(_) => (
450                    StatusCode::UNPROCESSABLE_ENTITY,
451                    ApiErrorStatus::UnknownFailure(format!("failed to start UDP listener on {bind_host}: {e}")),
452                ),
453            })?
454        }
455    };
456
457    Ok::<_, (StatusCode, ApiErrorStatus)>(
458        (
459            StatusCode::OK,
460            Json(SessionClientResponse {
461                protocol,
462                ip: bound_host.ip().to_string(),
463                port: bound_host.port(),
464                target: args.target.to_string(),
465                destination: args.destination,
466                forward_path: args.forward_path.clone(),
467                return_path: args.return_path.clone(),
468                hopr_mtu: SESSION_MTU,
469                surb_len: SURB_SIZE,
470                active_clients: udp_session_id.into_iter().map(|s| s.to_string()).collect(),
471                max_client_sessions: max_clients,
472                max_surb_upstream: args.max_surb_upstream,
473                response_buffer: args.response_buffer,
474                session_pool: args.session_pool,
475            }),
476        )
477            .into_response(),
478    )
479}

518pub(crate) async fn list_clients(
519    State(state): State<Arc<InternalState>>,
520    Path(protocol): Path<IpProtocol>,
521) -> Result<impl IntoResponse, impl IntoResponse> {
522    let response = state
523        .open_listeners
524        .0
525        .iter()
526        .filter(|v| v.key().0 == protocol.into())
527        .map(|v| {
528            let ListenerId(_, addr) = *v.key();
529            let entry = v.value();
530            let forward_path = hopr_lib::HopRouting::try_from(entry.forward_path.count_hops())
531                .expect("stored routing options always have bounded hop count");
532            let return_path = hopr_lib::HopRouting::try_from(entry.return_path.count_hops())
533                .expect("stored routing options always have bounded hop count");
534            SessionClientResponse {
535                protocol,
536                ip: addr.ip().to_string(),
537                port: addr.port(),
538                target: entry.target.to_string(),
539                forward_path: forward_path.into(),
540                return_path: return_path.into(),
541                destination: entry.destination,
542                hopr_mtu: SESSION_MTU,
543                surb_len: SURB_SIZE,
544                active_clients: entry.get_clients().iter().map(|e| e.key().to_string()).collect(),
545                max_client_sessions: entry.max_client_sessions,
546                max_surb_upstream: entry.max_surb_upstream,
547                response_buffer: entry.response_buffer,
548                session_pool: entry.session_pool,
549            }
550        })
551        .collect::<Vec<_>>();
552
553    Ok::<_, (StatusCode, ApiErrorStatus)>((StatusCode::OK, Json(response)).into_response())
554}

590impl From<SessionConfig> for Option<SurbBalancerConfig> {
591    /// Converts the API session config into protocol-level SURB balancer config.
592    fn from(value: SessionConfig) -> Self {
593        match value.response_buffer {
594            // Buffer worth at least 2 reply packets
595            Some(buffer_size) if buffer_size.as_u64() >= 2 * SESSION_MTU as u64 => Some(SurbBalancerConfig {
596                target_surb_buffer_size: buffer_size.as_u64() / SESSION_MTU as u64,
597                max_surbs_per_sec: value
598                    .max_surb_upstream
599                    .map(|b| (b.as_bps() as usize / (8 * SURB_SIZE)) as u64)
600                    .unwrap_or_else(|| SurbBalancerConfig::default().max_surbs_per_sec),
601                ..Default::default()
602            }),
603            // No additional SURBs are set up and maintained, useful for high-send low-reply sessions
604            Some(_) => None,
605            // Use defaults otherwise
606            None => Some(SurbBalancerConfig::default()),
607        }
608    }
609}

1024/// Closes an existing Session listener.
1025/// The listener must've been previously created and bound for the given IP protocol.
1026/// Once a listener is closed, no more socket connections can be made to it.
1027/// If the passed port number is 0, listeners on all ports of the given listening IP and protocol
1028/// will be closed.
1029#[utoipa::path(
1030    delete,
1031    path = const_format::formatcp!("{BASE_PATH}/session/{{protocol}}/{{ip}}/{{port}}"),
1032    description = "Closes an existing Session listener.",
1033    params(SessionCloseClientQuery),
1034    responses(
1035            (status = 204, description = "Listener closed successfully"),
1036            (status = 400, description = "Invalid IP protocol or port.", body = ApiError),
1037            (status = 401, description = "Invalid authorization token.", body = ApiError),
1038            (status = 404, description = "Listener not found.", body = ApiError),
1039            (status = 422, description = "Unknown failure", body = ApiError)
1040    ),
1041    security(
1042            ("api_token" = []),
1043            ("bearer_token" = [])
1044    ),
1045    tag = "Session",
1046)]
1047pub(crate) async fn close_client(
1048    State(state): State<Arc<InternalState>>,
1049    Path(SessionCloseClientQuery { protocol, ip, port }): Path<SessionCloseClientQuery>,
1050) -> Result<impl IntoResponse, impl IntoResponse> {
1051    let listening_ip: IpAddr = ip
1052        .parse()
1053        .map_err(|_| (StatusCode::BAD_REQUEST, ApiErrorStatus::InvalidInput))?;
1054
1055    {
1056        let open_listeners = &state.open_listeners.0;
1057
1058        let mut to_remove = Vec::new();
1059        let protocol: hopr_lib::IpProtocol = protocol.into();
1060
1061        // Find all listeners with protocol, listening IP and optionally port number (if > 0)
1062        open_listeners
1063            .iter()
1064            .filter(|v| {
1065                let ListenerId(proto, addr) = v.key();
1066                protocol == *proto && addr.ip() == listening_ip && (addr.port() == port || port == 0)
1067            })
1068            .for_each(|v| to_remove.push(*v.key()));
1069
1070        if to_remove.is_empty() {
1071            return Err((StatusCode::NOT_FOUND, ApiErrorStatus::InvalidInput));
1072        }
1073
1074        for bound_addr in to_remove {
1075            let (_, entry) = open_listeners
1076                .remove(&bound_addr)
1077                .ok_or((StatusCode::NOT_FOUND, ApiErrorStatus::InvalidInput))?;
1078
1079            entry.abort_handle.abort();
1080        }
1081    }
1082
1083    Ok::<_, (StatusCode, ApiErrorStatus)>((StatusCode::NO_CONTENT, "").into_response())
