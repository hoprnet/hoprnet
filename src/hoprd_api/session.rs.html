74impl std::fmt::Display for SessionTargetSpec {
75    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
76        match self {
77            SessionTargetSpec::Plain(t) => write!(f, "{t}"),
78            SessionTargetSpec::Sealed(t) => write!(f, "$${}", base64::prelude::BASE64_URL_SAFE.encode(t)),
79            SessionTargetSpec::Service(t) => write!(f, "#{t}"),
80        }
81    }

84impl FromStr for SessionTargetSpec {
85    type Err = HoprLibError;
86
87    fn from_str(s: &str) -> Result<Self, Self::Err> {
88        Ok(if let Some(stripped) = s.strip_prefix("$$") {
89            Self::Sealed(
90                base64::prelude::BASE64_URL_SAFE
91                    .decode(stripped)
92                    .map_err(|e| HoprLibError::GeneralError(e.to_string()))?,
93            )
94        } else if let Some(stripped) = s.strip_prefix("#") {
95            Self::Service(
96                stripped
97                    .parse()
98                    .map_err(|_| HoprLibError::GeneralError("cannot parse service id".into()))?,
99            )
100        } else {
101            Self::Plain(s.to_owned())
102        })
103    }
104}

106impl SessionTargetSpec {
107    pub fn into_target(self, protocol: IpProtocol) -> Result<SessionTarget, HoprLibError> {
108        Ok(match (protocol, self) {
109            (IpProtocol::TCP, SessionTargetSpec::Plain(plain)) => SessionTarget::TcpStream(
110                IpOrHost::from_str(&plain)
111                    .map(SealedHost::from)
112                    .map_err(|e| HoprLibError::GeneralError(e.to_string()))?,
113            ),
114            (IpProtocol::UDP, SessionTargetSpec::Plain(plain)) => SessionTarget::UdpStream(
115                IpOrHost::from_str(&plain)
116                    .map(SealedHost::from)
117                    .map_err(|e| HoprLibError::GeneralError(e.to_string()))?,
118            ),
119            (IpProtocol::TCP, SessionTargetSpec::Sealed(enc)) => {
120                SessionTarget::TcpStream(SealedHost::Sealed(enc.into_boxed_slice()))
121            }
122            (IpProtocol::UDP, SessionTargetSpec::Sealed(enc)) => {
123                SessionTarget::UdpStream(SealedHost::Sealed(enc.into_boxed_slice()))
124            }
125            (_, SessionTargetSpec::Service(id)) => SessionTarget::ExitNode(id),
126        })
127    }

280pub(crate) async fn websocket(
281    ws: WebSocketUpgrade,
282    Query(query): Query<SessionWebsocketClientQueryRequest>,
283    State(state): State<Arc<InternalState>>,
284) -> Result<impl IntoResponse, impl IntoResponse> {
285    let (dst, target, data) = query
286        .into_protocol_session_config()
287        .await
288        .map_err(|e| (StatusCode::UNPROCESSABLE_ENTITY, e))?;
289
290    let hopr = state.hopr.clone();
291    let session: HoprSession = hopr.connect_to(dst, target, data).await.map_err(|e| {
292        error!(error = %e, "Failed to establish session");
293        (
294            StatusCode::UNPROCESSABLE_ENTITY,
295            ApiErrorStatus::UnknownFailure(e.to_string()),
296        )
297    })?;
298
299    Ok::<_, (StatusCode, ApiErrorStatus)>(ws.on_upgrade(move |socket| websocket_connection(socket, session)))
300}

302enum WebSocketInput {
303    Network(Result<Box<[u8]>, std::io::Error>),
304    WsInput(Result<Message, Error>),
305}

307/// The maximum number of bytes read from a Session that WS can transfer within a single message.
308const WS_MAX_SESSION_READ_SIZE: usize = 4096;
309
310#[tracing::instrument(level = "debug", skip(socket, session))]
311async fn websocket_connection(socket: WebSocket, session: HoprSession) {
312    let session_id = *session.id();
313
314    let (rx, mut tx) = session.split();
315    let (mut sender, receiver) = socket.split();
316
317    let mut queue = (
318        receiver.map(WebSocketInput::WsInput),
319        AsyncReadStreamer::<WS_MAX_SESSION_READ_SIZE, _>(rx).map(WebSocketInput::Network),
320    )
321        .merge();
322
323    let (mut bytes_to_session, mut bytes_from_session) = (0, 0);
324
325    while let Some(v) = queue.next().await {
326        match v {
327            WebSocketInput::Network(bytes) => match bytes {
328                Ok(bytes) => {
329                    let len = bytes.len();
330                    if let Err(e) = sender.send(Message::Binary(bytes.into())).await {
331                        error!(
332                            error = %e,
333                            "Failed to emit read data onto the websocket, closing connection"
334                        );
335                        break;
336                    };
337                    bytes_from_session += len;
338                }
339                Err(e) => {
340                    error!(
341                        error = %e,
342                        "Failed to push data from network to socket, closing connection"
343                    );
344                    break;
345                }
346            },
347            WebSocketInput::WsInput(ws_in) => match ws_in {
348                Ok(Message::Binary(data)) => {
349                    let len = data.len();
350                    if let Err(e) = tx.write(data.as_ref()).await {
351                        error!(error = %e, "Failed to write data to the session, closing connection");
352                        break;
353                    }
354                    bytes_to_session += len;
355                }
356                Ok(Message::Text(_)) => {
357                    error!("Received string instead of binary data, closing connection");
358                    break;
359                }
360                Ok(Message::Close(_)) => {
361                    debug!("Received close frame, closing connection");
362                    break;
363                }
364                Ok(m) => trace!(message = ?m, "skipping an unsupported websocket message"),
365                Err(e) => {
366                    error!(error = %e, "Failed to get a valid websocket message, closing connection");
367                    break;
368                }
369            },
370        }
371    }
372
373    info!(%session_id, bytes_from_session, bytes_to_session, "WS session connection ended");
374}

453impl SessionClientRequest {
454    pub(crate) async fn into_protocol_session_config(
455        self,
456        target_protocol: IpProtocol,
457    ) -> Result<(Address, SessionTarget, SessionClientConfig), ApiErrorStatus> {
458        Ok((
459            self.destination,
460            self.target.into_target(target_protocol)?,
461            SessionClientConfig {
462                forward_path_options: self.forward_path.resolve().await?,
463                return_path_options: self.return_path.resolve().await?,
464                capabilities: self
465                    .capabilities
466                    .map(|vs| {
467                        let mut caps = SessionCapabilities::empty();
468                        caps.extend(vs.into_iter().map(SessionCapabilities::from));
469                        caps
470                    })
471                    .unwrap_or_else(|| match target_protocol {
472                        IpProtocol::TCP => {
473                            hopr_lib::SessionCapability::RetransmissionAck
474                                | hopr_lib::SessionCapability::RetransmissionNack
475                                | hopr_lib::SessionCapability::Segmentation
476                        }
477                        // Only Segmentation capability for UDP per default
478                        _ => SessionCapability::Segmentation.into(),
479                    }),
480                surb_management: match self.response_buffer {
481                    // Buffer worth at least 2 reply packets
482                    Some(buffer_size) if buffer_size.as_u64() >= 2 * SESSION_MTU as u64 => Some(SurbBalancerConfig {
483                        target_surb_buffer_size: buffer_size.as_u64() / SESSION_MTU as u64,
484                        ..Default::default()
485                    }),
486                    // No SURBs are set up and maintained, useful for high-send low-reply sessions
487                    Some(_) => None,
488                    // Use defaults otherwise
489                    None => Some(SurbBalancerConfig::default()),
490                },
491                ..Default::default()
492            },
493        ))
494    }
495}

537/// This function first tries to parse `requested` as the `ip:port` host pair.
538/// If that does not work, it tries to parse `requested` as a single IP address
539/// and as a `:` prefixed port number. Whichever of those fails, is replaced by the corresponding
540/// part from the given `default`.
541fn build_binding_host(requested: Option<&str>, default: std::net::SocketAddr) -> std::net::SocketAddr {
542    match requested.map(|r| std::net::SocketAddr::from_str(r).map_err(|_| r)) {
543        Some(Err(requested)) => {
544            // If the requested host is not parseable as a whole as `SocketAddr`, try only its parts
545            debug!(requested, %default, "using partially default listen host");
546            std::net::SocketAddr::new(
547                requested.parse().unwrap_or(default.ip()),
548                requested
549                    .strip_prefix(":")
550                    .and_then(|p| u16::from_str(p).ok())
551                    .unwrap_or(default.port()),
552            )
553        }
554        Some(Ok(requested)) => {
555            debug!(%requested, "using requested listen host");
556            requested
557        }
558        None => {
559            debug!(%default, "using default listen host");
560            default
561        }
562    }

570impl SessionPool {
571    pub const MAX_SESSION_POOL_SIZE: usize = 5;
572
573    async fn new(
574        size: usize,
575        dst: Address,
576        target: SessionTarget,
577        cfg: SessionClientConfig,
578        hopr: Arc<Hopr>,
579    ) -> Result<Self, (StatusCode, ApiErrorStatus)> {
580        let pool = Arc::new(std::sync::Mutex::new(VecDeque::with_capacity(size)));
581        let hopr_clone = hopr.clone();
582        let pool_clone = pool.clone();
583        futures::stream::iter(0..size.min(Self::MAX_SESSION_POOL_SIZE))
584            .map(Ok)
585            .try_for_each_concurrent(Self::MAX_SESSION_POOL_SIZE, move |i| {
586                let pool = pool_clone.clone();
587                let hopr = hopr_clone.clone();
588                let target = target.clone();
589                let cfg = cfg.clone();
590                async move {
591                    match hopr.connect_to(dst, target.clone(), cfg.clone()).await {
592                        Ok(s) => {
593                            debug!(session_id = %s.id(), num_session = i, "created a new session in pool");
594                            pool.lock()
595                                .map_err(|_| {
596                                    (
597                                        StatusCode::INTERNAL_SERVER_ERROR,
598                                        ApiErrorStatus::UnknownFailure("lock failed".into()),
599                                    )
600                                })?
601                                .push_back(s);
602                            Ok(())
603                        }
604                        Err(error) => {
605                            error!(%error, num_session = i, "failed to establish session for pool");
606                            Err((
607                                StatusCode::INTERNAL_SERVER_ERROR,
608                                ApiErrorStatus::UnknownFailure(format!(
609                                    "failed to establish session #{i} in pool to {dst}: {error}"
610                                )),
611                            ))
612                        }
613                    }
614                }
615            })
616            .await?;
617
618        // Spawn a task that periodically sends keep alive messages to the Session in the pool.
619        if !pool.lock().map(|p| p.is_empty()).unwrap_or(true) {
620            let pool_clone_1 = pool.clone();
621            let pool_clone_2 = pool.clone();
622            let pool_clone_3 = pool.clone();
623            Ok(Self {
624                pool: Some(pool),
625                ah: Some(hopr_async_runtime::spawn_as_abortable!(
626                    futures_time::stream::interval(futures_time::time::Duration::from(
627                        std::time::Duration::from_secs(1).max(hopr.config().session.idle_timeout / 2)
628                    ))
629                    .take_while(move |_| {
630                        // Continue the infinite interval stream until there are sessions in the pool
631                        futures::future::ready(pool_clone_1.lock().is_ok_and(|p| !p.is_empty()))
632                    })
633                    .flat_map(move |_| {
634                        // Get all SessionIds of the remaining Sessions in the pool
635                        let ids = pool_clone_2.lock().ok().map(|v| v.iter().map(|s| *s.id()).collect::<Vec<_>>());
636                        futures::stream::iter(ids.into_iter().flatten())
637                    })
638                    .for_each(move |id| {
639                        let hopr = hopr.clone();
640                        let pool = pool_clone_3.clone();
641                        async move {
642                            // Make sure the Session is still alive, otherwise remove it from the pool
643                            if let Err(error) = hopr.keep_alive_session(&id).await {
644                                error!(%error, %dst, session_id = %id, "session in pool is not alive, removing from pool");
645                                if let Ok(mut pool) = pool.lock() {
646                                    pool.retain(|s| *s.id() != id);
647                                }
648                            }
649                        }
650                    })
651                ))
652            })
653        } else {
654            Ok(Self { pool: None, ah: None })
655        }
656    }
657
658    fn pop(&mut self) -> Option<HoprSession> {
659        self.pool.as_ref().and_then(|pool| pool.lock().ok()?.pop_front())
660    }
661}
662
663impl Drop for SessionPool {
664    fn drop(&mut self) {
665        if let Some(ah) = self.ah.take() {
666            ah.abort();
667        }
668    }
669}

671async fn create_tcp_client_binding(
672    bind_host: std::net::SocketAddr,
673    state: Arc<InternalState>,
674    args: SessionClientRequest,
675) -> Result<std::net::SocketAddr, (StatusCode, ApiErrorStatus)> {
676    let target_spec = args.target.clone();
677    let (dst, target, data) = args
678        .clone()
679        .into_protocol_session_config(IpProtocol::TCP)
680        .await
681        .map_err(|e| (StatusCode::UNPROCESSABLE_ENTITY, e))?;
682
683    // Bind the TCP socket first
684    let (bound_host, tcp_listener) = tcp_listen_on(bind_host).await.map_err(|e| {
685        if e.kind() == std::io::ErrorKind::AddrInUse {
686            (StatusCode::CONFLICT, ApiErrorStatus::ListenHostAlreadyUsed)
687        } else {
688            (
689                StatusCode::UNPROCESSABLE_ENTITY,
690                ApiErrorStatus::UnknownFailure(format!("failed to start TCP listener on {bind_host}: {e}")),
691            )
692        }
693    })?;
694    info!(%bound_host, "TCP session listener bound");
695
696    // For each new TCP connection coming to the listener,
697    // open a Session with the same parameters
698    let hopr = state.hopr.clone();
699
700    // Create a session pool if requested
701    let mut session_pool = SessionPool::new(
702        args.session_pool.unwrap_or(0),
703        dst,
704        target.clone(),
705        data.clone(),
706        hopr.clone(),
707    )
708    .await?;
709
710    // Create an abort handler for the listener
711    let (abort_handle, abort_reg) = AbortHandle::new_pair();
712    hopr_async_runtime::prelude::spawn(async move {
713        let active_sessions = Arc::new(std::sync::Mutex::new(HashMap::new()));
714        let active_sessions_clone = active_sessions.clone();
715
716        futures::stream::Abortable::new(tokio_stream::wrappers::TcpListenerStream::new(tcp_listener), abort_reg)
717            .and_then(|sock| async { Ok((sock.peer_addr()?, sock)) })
718            .for_each(move |accepted_client| {
719                let data = data.clone();
720                let target = target.clone();
721                let hopr = hopr.clone();
722                let active_sessions = active_sessions_clone.clone();
723
724                // Try to pop from the pool only if a client was accepted
725                let maybe_pooled_session = accepted_client.is_ok().then(|| session_pool.pop()).flatten();
726                async move {
727                    match accepted_client {
728                        Ok((sock_addr, stream)) => {
729                            debug!(?sock_addr, "incoming TCP connection");
730                            let session = match maybe_pooled_session {
731                                Some(s) => {
732                                    debug!(session_id = %s.id(), "using pooled session");
733                                    s
734                                }
735                                None => {
736                                    debug!("no more active sessions in the pool, creating a new one");
737                                    match hopr.connect_to(dst, target, data).await {
738                                        Ok(s) => s,
739                                        Err(error) => {
740                                            error!(%error, "failed to establish session");
741                                            return;
742                                        }
743                                    }
744                                }
745                            };
746
747                            let session_id = *session.id();
748                            debug!(?sock_addr, %session_id, "new session for incoming TCP connection");
749
750                            let (abort_handle, abort_reg) = AbortHandle::new_pair();
751                            if let Ok(mut active_sessions) = active_sessions.lock() {
752                                active_sessions.insert(session_id, abort_handle);
753                            }
754
755                            #[cfg(all(feature = "prometheus", not(test)))]
756                            METRIC_ACTIVE_CLIENTS.increment(&["tcp"], 1.0);
757
758                            hopr_async_runtime::prelude::spawn(
759                                // The stream either terminates naturally (by the client closing the TCP connection)
760                                // or is terminated via the abort handle.
761                                bind_session_to_stream(session, stream, HOPR_TCP_BUFFER_SIZE, Some(abort_reg)).map(
762                                    move |_| async move {
763                                        // Regardless how the session ended, remove the abort handle
764                                        // from the map
765                                        if let Ok(mut active_sessions) = active_sessions.lock() {
766                                            active_sessions.remove(&session_id);
767                                        }
768
769                                        debug!(%session_id, "tcp session has ended");
770
771                                        #[cfg(all(feature = "prometheus", not(test)))]
772                                        METRIC_ACTIVE_CLIENTS.decrement(&["tcp"], 1.0);
773                                    },
774                                ),
775                            );
776                        }
777                        Err(error) => error!(%error, "failed to accept connection"),
778                    }
779                }
780            })
781            .await;
782
783        // Once the listener is done, abort all active sessions created by the listener
784        if let Ok(active_sessions) = active_sessions.lock() {
785            active_sessions.iter().for_each(|(session_id, handle)| {
786                debug!(%session_id, "aborting opened TCP session after listener has been closed");
787                handle.abort()
788            });
789        };
790    });
791
792    state.open_listeners.write_arc().await.insert(
793        ListenerId(hopr_network_types::types::IpProtocol::TCP, bound_host),
794        StoredSessionEntry {
795            destination: dst,
796            target: target_spec.clone(),
797            forward_path: args.forward_path.clone(),
798            return_path: args.return_path.clone(),
799            abort_handle,
800        },
801    );
802    Ok(bound_host)
803}

805async fn create_udp_client_binding(
806    bind_host: std::net::SocketAddr,
807    state: Arc<InternalState>,
808    args: SessionClientRequest,
809) -> Result<std::net::SocketAddr, (StatusCode, ApiErrorStatus)> {
810    let target_spec = args.target.clone();
811    let (dst, target, data) = args
812        .clone()
813        .into_protocol_session_config(IpProtocol::UDP)
814        .await
815        .map_err(|e| (StatusCode::UNPROCESSABLE_ENTITY, e))?;
816
817    // Bind the UDP socket first
818    let (bound_host, udp_socket) = udp_bind_to(bind_host).await.map_err(|e| {
819        if e.kind() == std::io::ErrorKind::AddrInUse {
820            (StatusCode::CONFLICT, ApiErrorStatus::ListenHostAlreadyUsed)
821        } else {
822            (
823                StatusCode::UNPROCESSABLE_ENTITY,
824                ApiErrorStatus::UnknownFailure(format!("failed to start UDP listener on {bind_host}: {e}")),
825            )
826        }
827    })?;
828
829    info!(%bound_host, "UDP session listener bound");
830
831    let hopr = state.hopr.clone();
832
833    // Create a single session for the UDP socket
834    let session = hopr.connect_to(dst, target, data).await.map_err(|e| {
835        (
836            StatusCode::UNPROCESSABLE_ENTITY,
837            ApiErrorStatus::UnknownFailure(e.to_string()),
838        )
839    })?;
840
841    let open_listeners_clone = state.open_listeners.clone();
842    let listener_id = ListenerId(hopr_network_types::types::IpProtocol::UDP, bound_host);
843
844    // Create an abort handle so that the Session can be terminated by aborting
845    // the UDP stream first. Because under the hood, the bind_session_to_stream uses
846    // `transfer_session` which in turn uses `copy_duplex_abortable`, aborting the
847    // `udp_socket` will:
848    //
849    // 1. Initiate graceful shutdown of `udp_socket`
850    // 2. Once done, initiate a graceful shutdown of `session`
851    // 3. Finally, return from the `bind_session_to_stream` which will terminate the spawned task
852    //
853    // This is needed because the `udp_socket` cannot terminate by itself.
854    let (abort_handle, abort_reg) = AbortHandle::new_pair();
855    hopr_async_runtime::prelude::spawn(async move {
856        #[cfg(all(feature = "prometheus", not(test)))]
857        METRIC_ACTIVE_CLIENTS.increment(&["udp"], 1.0);
858
859        bind_session_to_stream(session, udp_socket, HOPR_UDP_BUFFER_SIZE, Some(abort_reg)).await;
860
861        #[cfg(all(feature = "prometheus", not(test)))]
862        METRIC_ACTIVE_CLIENTS.decrement(&["udp"], 1.0);
863
864        // Once the Session closes, remove it from the list
865        open_listeners_clone.write_arc().await.remove(&listener_id);
866    });
867
868    state.open_listeners.write_arc().await.insert(
869        listener_id,
870        StoredSessionEntry {
871            destination: dst,
872            target: target_spec.clone(),
873            forward_path: args.forward_path.clone(),
874            return_path: args.return_path.clone(),
875            abort_handle,
876        },
877    );
878    Ok(bound_host)
879}

1081pub(crate) async fn close_client(
1082    State(state): State<Arc<InternalState>>,
1083    Path(SessionCloseClientQuery { protocol, ip, port }): Path<SessionCloseClientQuery>,
1084) -> Result<impl IntoResponse, impl IntoResponse> {
1085    let listening_ip: IpAddr = ip
1086        .parse()
1087        .map_err(|_| (StatusCode::BAD_REQUEST, ApiErrorStatus::InvalidInput))?;
1088
1089    {
1090        let mut open_listeners = state.open_listeners.write_arc().await;
1091
1092        let mut to_remove = Vec::new();
1093
1094        // Find all listeners with protocol, listening IP and optionally port number (if > 0)
1095        open_listeners
1096            .iter()
1097            .filter(|(ListenerId(proto, addr), _)| {
1098                let protocol: hopr_lib::IpProtocol = protocol.into();
1099                protocol == *proto && addr.ip() == listening_ip && (addr.port() == port || port == 0)
1100            })
1101            .for_each(|(id, _)| to_remove.push(*id));
1102
1103        if to_remove.is_empty() {
1104            return Err((StatusCode::NOT_FOUND, ApiErrorStatus::InvalidInput));
1105        }
1106
1107        for bound_addr in to_remove {
1108            let entry = open_listeners
1109                .remove(&bound_addr)
1110                .ok_or((StatusCode::NOT_FOUND, ApiErrorStatus::InvalidInput))?;
1111
1112            entry.abort_handle.abort();
1113        }
1114    }
1115
1116    Ok::<_, (StatusCode, ApiErrorStatus)>((StatusCode::NO_CONTENT, "").into_response())
1117}

1119async fn try_restricted_bind<F, S, Fut>(
1120    addrs: Vec<std::net::SocketAddr>,
1121    range_str: &str,
1122    binder: F,
1123) -> std::io::Result<S>
1124where
1125    F: Fn(Vec<std::net::SocketAddr>) -> Fut,
1126    Fut: Future<Output = std::io::Result<S>>,
1127{
1128    if addrs.is_empty() {
1129        return Err(std::io::Error::other("no valid socket addresses found"));
1130    }
1131
1132    let range = range_str
1133        .split_once(":")
1134        .and_then(
1135            |(a, b)| match u16::from_str(a).and_then(|a| Ok((a, u16::from_str(b)?))) {
1136                Ok((a, b)) if a <= b => Some(a..=b),
1137                _ => None,
1138            },
1139        )
1140        .ok_or(std::io::Error::other(format!("invalid port range {range_str}")))?;
1141
1142    for port in range {
1143        let addrs = addrs
1144            .iter()
1145            .map(|addr| std::net::SocketAddr::new(addr.ip(), port))
1146            .collect::<Vec<_>>();
1147        match binder(addrs).await {
1148            Ok(listener) => return Ok(listener),
1149            Err(error) => debug!(%error, "listen address not usable"),
1150        }
1151    }
1152
1153    Err(std::io::Error::new(
1154        std::io::ErrorKind::AddrNotAvailable,
1155        format!("no valid socket addresses found within range: {range_str}"),
1156    ))
1157}

1159async fn tcp_listen_on<A: std::net::ToSocketAddrs>(address: A) -> std::io::Result<(std::net::SocketAddr, TcpListener)> {
1160    let addrs = address.to_socket_addrs()?.collect::<Vec<_>>();
1161
1162    // If automatic port allocation is requested and there's a restriction on the port range
1163    // (via HOPRD_SESSION_PORT_RANGE), try to find an address within that range.
1164    if addrs.iter().all(|a| a.port() == 0) {
1165        if let Ok(range_str) = std::env::var(crate::env::HOPRD_SESSION_PORT_RANGE) {
1166            let tcp_listener =
1167                try_restricted_bind(
1168                    addrs,
1169                    &range_str,
1170                    |a| async move { TcpListener::bind(a.as_slice()).await },
1171                )
1172                .await?;
1173            return Ok((tcp_listener.local_addr()?, tcp_listener));
1174        }
1175    }
1176
1177    let tcp_listener = TcpListener::bind(addrs.as_slice()).await?;
1178    Ok((tcp_listener.local_addr()?, tcp_listener))
1179}

1181async fn udp_bind_to<A: std::net::ToSocketAddrs>(
1182    address: A,
1183) -> std::io::Result<(std::net::SocketAddr, ConnectedUdpStream)> {
1184    let addrs = address.to_socket_addrs()?.collect::<Vec<_>>();
1185
1186    let builder = ConnectedUdpStream::builder()
1187        .with_buffer_size(HOPR_UDP_BUFFER_SIZE)
1188        .with_foreign_data_mode(ForeignDataMode::Discard) // discard data from UDP clients other than the first one served
1189        .with_queue_size(HOPR_UDP_QUEUE_SIZE)
1190        .with_receiver_parallelism(UdpStreamParallelism::Auto);
1191
1192    // If automatic port allocation is requested and there's a restriction on the port range
1193    // (via HOPRD_SESSION_PORT_RANGE), try to find an address within that range.
1194    if addrs.iter().all(|a| a.port() == 0) {
1195        if let Ok(range_str) = std::env::var(crate::env::HOPRD_SESSION_PORT_RANGE) {
1196            let udp_listener = try_restricted_bind(addrs, &range_str, |addrs| {
1197                futures::future::ready(builder.clone().build(addrs.as_slice()))
1198            })
1199            .await?;
1200
1201            return Ok((*udp_listener.bound_address(), udp_listener));
1202        }
1203    }
1204
1205    let udp_socket = builder.build(address)?;
1206    Ok((*udp_socket.bound_address(), udp_socket))
1207}

1209async fn bind_session_to_stream<T>(
1210    mut session: HoprSession,
1211    mut stream: T,
1212    max_buf: usize,
1213    abort_reg: Option<AbortRegistration>,
1214) where
1215    T: tokio::io::AsyncRead + tokio::io::AsyncWrite + Unpin,
1216{
1217    let session_id = *session.id();
1218    match transfer_session(&mut session, &mut stream, max_buf, abort_reg).await {
1219        Ok((session_to_stream_bytes, stream_to_session_bytes)) => info!(
1220            session_id = ?session_id,
1221            session_to_stream_bytes, stream_to_session_bytes, "client session ended",
1222        ),
1223        Err(error) => error!(
1224            session_id = ?session_id,
1225            %error,
1226            "error during data transfer"
1227        ),
1228    }
1229}

1246    fn loopback_transport() -> (
1247        UnboundedSender<(DestinationRouting, ApplicationData)>,
1248        UnboundedReceiver<Box<[u8]>>,
1249    ) {
1250        let (input_tx, input_rx) = futures::channel::mpsc::unbounded::<(DestinationRouting, ApplicationData)>();
1251        let (output_tx, output_rx) = futures::channel::mpsc::unbounded::<Box<[u8]>>();
1252        tokio::task::spawn(
1253            input_rx
1254                .map(|(_, data)| Ok(data.plain_text))
1255                .forward(output_tx)
1256                .map(|e| tracing::debug!(?e, "loopback transport completed")),
1257        );
1258
1259        (input_tx, output_rx)
1260    }
