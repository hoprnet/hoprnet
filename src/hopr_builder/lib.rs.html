48#[cfg(feature = "runtime-tokio")]
49pub async fn build_reference(
50    identity: (&ChainKeypair, &OffchainKeypair),
51    config: HoprLibConfig,
52    blokli_url: String,
53    db_data_path: String,
54    #[cfg(feature = "session-server")] server_config: SessionIpForwardingConfig,
55) -> anyhow::Result<(
56    Arc<ReferenceHopr>,
57    impl Future<Output = std::result::Result<HoprTransportIO, HoprLibError>>,
58)> {
59    let (chain_key, packet_key) = identity;
60    let node_db = init_hopr_node_db(&db_data_path, true, false).await?;
61
62    let mut chain_connector = create_trustful_hopr_blokli_connector(
63        chain_key,
64        BlockchainConnectorConfig {
65            connection_sync_timeout: std::time::Duration::from_mins(1),
66            sync_tolerance: 90,
67            tx_timeout_multiplier: std::env::var("HOPR_TX_TIMEOUT_MULTIPLIER")
68                .ok()
69                .and_then(|p| {
70                    p.parse()
71                        .inspect_err(|error| tracing::warn!(%error, "failed to parse HOPR_TX_TIMEOUT_MULTIPLIER"))
72                        .ok()
73                })
74                .unwrap_or_else(|| BlockchainConnectorConfig::default().tx_timeout_multiplier),
75        },
76        BlokliClient::new(
77            blokli_url.parse()?,
78            BlokliClientConfig {
79                timeout: std::time::Duration::from_secs(30),
80                stream_reconnect_timeout: std::time::Duration::from_secs(30),
81                subscription_stream_restart_delay: Some(std::time::Duration::from_secs(1)),
82            },
83        ),
84        config.safe_module.module_address,
85    )
86    .await?;
87    chain_connector.connect().await?;
88    let chain_connector = Arc::new(chain_connector);
89
90    #[cfg(feature = "session-server")]
91    let session_server = HoprServerIpForwardingReactor::new(packet_key.clone(), server_config);
92
93    build_from_chain_and_db(
94        chain_key,
95        packet_key,
96        config,
97        chain_connector,
98        node_db,
99        #[cfg(feature = "session-server")]
100        session_server,
101    )
102    .await
103}

105#[cfg(feature = "runtime-tokio")]
106pub async fn build_from_chain_and_db<
107    Chain,
108    Db,
109    #[cfg(feature = "session-server")] Srv: HoprSessionServer + Clone + Send + 'static,
110>(
111    chain_key: &ChainKeypair,
112    packet_key: &OffchainKeypair,
113    config: HoprLibConfig,
114    chain_connector: Chain,
115    db: Db,
116    #[cfg(feature = "session-server")] server: Srv,
117) -> anyhow::Result<(
118    Arc<Hopr<Chain, Db, SharedChannelGraph, HoprNetwork>>,
119    impl Future<Output = std::result::Result<HoprTransportIO, HoprLibError>>,
120)>
121where
122    Chain: HoprChainApi + Clone + Send + Sync + 'static,
123    Db: HoprNodeDbApi + Clone + Send + Sync + 'static,
124{
125    // Calculate the minimum capacity based on accounts (each account can generate 2 messages),
126    // plus 100 as an additional buffer
127    let minimum_capacity = chain_connector
128        .count_accounts(AccountSelector {
129            public_only: true,
130            ..Default::default()
131        })
132        .await
133        .map_err(hopr_lib::HoprLibError::chain)?
134        .saturating_mul(2)
135        .saturating_add(100);
136
137    let chain_discovery_events_capacity = std::env::var("HOPR_INTERNAL_CHAIN_DISCOVERY_CHANNEL_CAPACITY")
138        .ok()
139        .and_then(|s| s.trim().parse::<usize>().ok())
140        .filter(|&c| c > 0)
141        .unwrap_or(2048)
142        .max(minimum_capacity);
143
144    tracing::debug!(
145        capacity = chain_discovery_events_capacity,
146        minimum_required = minimum_capacity,
147        "creating chain discovery events channel"
148    );
149    let (indexer_peer_update_tx, indexer_peer_update_rx) =
150        futures::channel::mpsc::channel::<PeerDiscovery>(chain_discovery_events_capacity);
151
152    // create network
153    let network_builder = HoprLibp2pNetworkBuilder::new(indexer_peer_update_rx);
154    // create graph
155    let graph = std::sync::Arc::new(ChannelGraph::new(*packet_key.public()));
156
157    // END = implementation definitions
158
159    // START = process chain and network events into graph updates
160    let chain_events = chain_connector
161        .subscribe_with_state_sync([StateSyncOptions::PublicAccounts, StateSyncOptions::OpenedChannels])?;
162    let network_events = network_builder.subscribe_network_events();
163    let graph_updater = graph.clone();
164    let chain_reader = chain_connector.clone();
165    let indexer_peer_update_tx = indexer_peer_update_tx.clone();
166
167    let proc =
168        async move {
169            enum Event {
170                Chain(ChainEvent),
171                Network(NetworkEvent),
172            }
173
174            let ticket_price = std::sync::Arc::new(parking_lot::RwLock::new(chain_reader.minimum_ticket_price().await.unwrap_or_default()));
175            let win_probability = std::sync::Arc::new(parking_lot::RwLock::new(chain_reader.minimum_incoming_ticket_win_prob().await.unwrap_or_default()));
176
177            network_events
178                .map(Event::Network)
179                .merge(chain_events.map(Event::Chain))
180                .for_each(|event| {
181                    let mut indexer_peer_update_tx = indexer_peer_update_tx.clone();
182                    let chain_reader = chain_reader.clone();
183                    let graph_updater = graph_updater.clone();
184                    let ticket_price = ticket_price.clone();
185                    let win_probability = win_probability.clone();
186
187                    async move {
188                        match event {
189                            Event::Chain(chain_event) => {
190                                match chain_event {
191                                    ChainEvent::Announcement(account) =>{
192                                        graph_updater.record_node(account.public_key);
193                                        let peer_id: PeerId = account.public_key.into();
194                                        if let Err(error) = indexer_peer_update_tx.send(PeerDiscovery::Announce(peer_id, account.get_multiaddrs().to_vec())).await {
195                                            tracing::error!(peer = %peer_id, %error, reason = "failed to propagate the record", "ignoring announced peer")
196                                        }
197                                },
198                                ChainEvent::ChannelOpened(channel) |
199                                ChainEvent::ChannelClosed(channel) |
200                                ChainEvent::ChannelBalanceIncreased(channel, _) |
201                                ChainEvent::ChannelBalanceDecreased(channel, _) => {
202                                    let from = chain_reader.chain_key_to_packet_key(&channel.source).await;
203                                    let to = chain_reader.chain_key_to_packet_key(&channel.destination).await;
204
205                                    match (from, to) {
206                                        (Ok(Some(from)), Ok(Some(to))) => {
207                                            let capacity =  if matches!(channel.status, ChannelStatus::Closed) {
208                                                None
209                                            } else if let Ok(ticket_value) = ticket_price.read().div_f64(win_probability.read().as_f64()) {
210                                                Some(
211                                                    channel.balance
212                                                        .amount()
213                                                        .checked_div(ticket_value.amount())
214                                                        .map(|v| v.low_u128())
215                                                        .unwrap_or(u128::MAX)
216                                                )
217                                            } else {
218                                                None
219                                            };
220
221                                            graph_updater.record_edge(hopr_lib::api::graph::MeasurableEdge::<NeighborTelemetry, PathTelemetry>::Capacity(Box::new(EdgeCapacityUpdate{
222                                                capacity,
223                                                src: from,
224                                                dest: to
225                                        })));
226                                        },
227                                        (Ok(_), Ok(_)) => {
228                                            tracing::error!(%channel, "could not find packet keys for the channel endpoints");
229                                        },
230                                        (Err(e), _) | (_, Err(e)) => {
231                                            tracing::error!(%e, %channel, "failed to convert chain keys to packet keys for graph update");
232                                        }
233                                    }
234                                },
235                                ChainEvent::ChannelClosureInitiated(_channel) => {},
236                                ChainEvent::WinningProbabilityIncreased(probability) |
237                                ChainEvent::WinningProbabilityDecreased(probability) => {
238                                    *win_probability.write() = probability;
239                                }
240                                ChainEvent::TicketPriceChanged(price) => {
241                                    *ticket_price.write() = price;
242                                },
243                                _ => {}
244                            }
245                        }
246                        Event::Network(network_event) => {
247                            match network_event {
248                                NetworkEvent::PeerConnected(peer_id) =>
249                                    if let Ok(opk) = hopr_lib::peer_id_to_public_key(&peer_id).await {
250                                        graph_updater.record_edge(hopr_lib::api::graph::MeasurableEdge::<NeighborTelemetry, PathTelemetry>::ConnectionStatus {
251                                            peer: opk,
252                                            connected: true
253                                        });
254                                    } else {
255                                        tracing::error!(%peer_id, "failed to convert peer ID to public key for graph update");
256                                    },
257                                NetworkEvent::PeerDisconnected(peer_id) =>
258                                    if let Ok(opk) = hopr_lib::peer_id_to_public_key(&peer_id).await {
259                                        graph_updater.record_edge(hopr_lib::api::graph::MeasurableEdge::<NeighborTelemetry, PathTelemetry>::ConnectionStatus {
260                                            peer: opk,
261                                            connected: false
262                                        });
263                                    } else {
264                                        tracing::error!(%peer_id, "failed to convert peer ID to public key for graph update");
265                                    },
266                            };
267                        }
268                    }
269                    }
270                })
271                .await;
272        }
273        .inspect(|_| tracing::warn!(task = "Interconnecting chain, graph and network", "long-running background task finished"));
274    let _jh = tokio::spawn(proc);
275    // END = process chain and network events into graph updates
276
277    // create the node
278    let node = Arc::new(
279        hopr_lib::Hopr::new(
280            (chain_key, packet_key),
281            chain_connector.clone(),
282            db,
283            graph.clone(),
284            config,
285        )
286        .await?,
287    );
288
289    let node_for_run = node.clone();
290    let start = async move {
291        node_for_run
292            .run(
293                hopr_ct_immediate::ImmediateNeighborProber::new(Default::default(), graph),
294                network_builder,
295                #[cfg(feature = "session-server")]
296                server,
297            )
298            .await
299    };
300
301    Ok((node, start))
302}
