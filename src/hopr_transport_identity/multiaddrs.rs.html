3/// Remove the `p2p/<PeerId>` component from a multiaddress
4pub fn strip_p2p_protocol(ma: &Multiaddr) -> Multiaddr {
5    Multiaddr::from_iter(ma.iter().filter(|v| !matches!(v, multiaddr::Protocol::P2p(_))))
6}

8/// Check whether the first multiaddress protocol component is a `dns*` component
9pub fn is_dns(ma: &Multiaddr) -> bool {
10    ma.iter()
11        .next()
12        .map(|proto| matches!(proto, multiaddr::Protocol::Dns(_)))
13        .unwrap_or(false)
14}
15
16/// Check whether the first multiaddress protocol component represents a private address
17pub fn is_private(ma: &Multiaddr) -> bool {
18    ma.iter()
19        .next()
20        .map(|proto| match proto {
21            multiaddr::Protocol::Ip4(ip) => ip.is_private(),
22            multiaddr::Protocol::Dns(domain) => domain.to_ascii_lowercase() == "localhost",
23            _ => false,
24        })
25        .unwrap_or(false)
26}
27
28/// Check whether the multiaddr protocol component is supported by this library
29pub fn is_supported(ma: &Multiaddr) -> bool {
30    ma.iter()
31        .next()
32        .map(|proto| {
33            matches!(
34                proto,
35                multiaddr::Protocol::Ip4(_)
36                    | multiaddr::Protocol::Ip6(_)
37                    | multiaddr::Protocol::Dns(_)
38                    | multiaddr::Protocol::Dns4(_)
39                    | multiaddr::Protocol::Dns6(_)
40            )
41        })
42        .unwrap_or(false)
43}
