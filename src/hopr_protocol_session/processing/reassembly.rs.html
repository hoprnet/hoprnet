33/// Reassembler is a stream adaptor that reads [`Segments`](Segment) from the underlying
34/// stream and tries to put them into correct order so they form a [`Frame`].
35///
36/// This is essentially the inverse of [`Segmenter`](super::segmenter_old::Segmenter).
37///
38/// Reassembler takes two parameters: `max_age` and `capacity`:
39///
40/// The `max_age` specifies how long an incomplete Frame (with a missing segment(s)) is to be kept
41/// in the internal buffer until it is considered definitely lost.
42/// In other words, it specifies how long the reassembler is allowed to wait for all segments
43/// of a Frame to arrive from the underlying stream.
44///
45/// The `capacity` specifies the maximum number of incomplete frames to keep in
46/// the internal buffer. If the reassembler is at maximum capacity, the underlying stream is not
47/// polled for new segments, leaving the oldest incomplete frames in the reassembler to expire and
48/// be definitely lost.
49///
50/// By definition, Reassembler is a fallible stream, yielding either `Ok(Some(`[`Frame`]`))`,
51/// `Err(`[`SessionError::FrameDiscarded`]`)` when a frame is lost due to expiry, or `Ok(None)` when
52/// there are no more elements in the underlying stream.
53///
54/// The reassemblers internal buffer is stored in a [`FrameMap`] and can be constructed using
55/// different implementations of it, suitable for different use-cases.
56///
57/// Use [`ReassemblerExt`] methods to turn a ` Segment ` stream into a fallible `Frame` stream using the `Reassembler`.
58#[must_use = "streams do nothing unless polled"]
59#[pin_project::pin_project]
60pub struct Reassembler<S, M> {
61    #[pin]
62    inner: S,
63    #[pin]
64    timer: futures_time::task::Sleep,
65    incomplete_frames: M,
66    expired_frames: Vec<FrameId>,
67    max_age: Duration,
68    capacity: usize,
69    last_expiration: Option<Instant>,
70}
71
72impl<S: futures::Stream<Item = Segment>, M: FrameMap> Reassembler<S, M> {
73    fn new(inner: S, incomplete_frames: M, max_age: Duration, capacity: usize) -> Self {
74        Self {
75            inner,
76            timer: futures_time::task::sleep(max_age.max(Duration::from_millis(1)).into()),
77            incomplete_frames,
78            expired_frames: Vec::with_capacity(capacity),
79            last_expiration: None,
80            max_age,
81            capacity,
82        }
83    }
84
85    fn expire_frames(incomplete_frames: &mut M, expired_frames: &mut Vec<FrameId>, max_age: Duration) {
86        incomplete_frames.retain(|id, builder| {
87            if builder.last_recv.elapsed() >= max_age {
88                expired_frames.push(*id);
89                false
90            } else {
91                true
92            }
93        });
94    }
95}

97impl<S: futures::Stream<Item = Segment>, M: FrameMap> futures::Stream for Reassembler<S, M> {
98    type Item = Result<Frame, SessionError>;
99
100    #[instrument(name = "Reassembler::poll_next", level = "trace", skip(self, cx), fields(num_incomplete = self.incomplete_frames.len()), ret)]
101    fn poll_next(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
102        let mut this = self.project();
103        loop {
104            if let Some(frame_id) = this.expired_frames.pop() {
105                tracing::trace!(frame_id, "emit discarded frame");
106                return Poll::Ready(Some(Err(SessionError::FrameDiscarded(frame_id))));
107            }
108
109            // Poll the timer only if there are incomplete frames
110            let timer_poll = if this.incomplete_frames.len() > 0 {
111                this.timer.as_mut().poll(cx)
112            } else {
113                Poll::Pending
114            };
115
116            // Poll the inner stream only if there's space in the reassembler
117            let inner_poll = if this.incomplete_frames.len() <= *this.capacity {
118                this.inner.as_mut().poll_next(cx)
119            } else {
120                // This essentially forces the incomplete frames to be expired
121                tracing::warn!("reassembler has reached its capacity");
122                Poll::Pending
123            };
124
125            tracing::trace!("polling next");
126            match (inner_poll, timer_poll) {
127                (Poll::Ready(Some(item)), timer) => {
128                    if timer.is_ready() {
129                        this.timer.as_mut().reset_timer();
130                    }
131
132                    tracing::trace!(
133                        frame_id = item.frame_id,
134                        seq_idx = item.seq_idx,
135                        seq_len = %item.seq_flags,
136                        "received segment"
137                    );
138
139                    match this.incomplete_frames.entry(item.frame_id) {
140                        FrameMapEntry::Occupied(mut e) => {
141                            let builder = e.get_builder_mut();
142                            let seg_id = item.id();
143                            match builder.add_segment(item) {
144                                Ok(_) => {
145                                    tracing::trace!(frame_id = builder.frame_id(), %seg_id, "added segment");
146                                    if builder.is_complete() {
147                                        #[cfg(all(not(test), feature = "prometheus"))]
148                                        METRIC_TIME_TO_FRAME_FINISH
149                                            .observe(builder.created.elapsed().as_millis() as f64);
150
151                                        tracing::trace!(frame_id = builder.frame_id(), "frame is complete");
152                                        return Poll::Ready(Some(e.finalize().try_into()));
153                                    }
154                                }
155                                Err(error) => {
156                                    tracing::error!(%error, %seg_id, "encountered invalid segment");
157                                }
158                            }
159                        }
160                        FrameMapEntry::Vacant(e) => {
161                            let builder = FrameBuilder::from(item);
162                            if builder.is_complete() {
163                                #[cfg(all(not(test), feature = "prometheus"))]
164                                METRIC_TIME_TO_FRAME_FINISH.observe(builder.created.elapsed().as_millis() as f64);
165
166                                tracing::trace!(frame_id = builder.frame_id(), "segment frame is complete");
167                                return Poll::Ready(Some(builder.try_into()));
168                            } else {
169                                tracing::trace!(frame_id = builder.frame_id(), "added segment for new frame");
170                                e.insert_builder(builder);
171                            }
172                        }
173                    };
174
175                    // Since the retaining operation is potentially expensive,
176                    // we do it actually only if there's a real chance that a frame is expired
177                    if this.last_expiration.is_none_or(|e| e.elapsed() >= *this.max_age) {
178                        Self::expire_frames(this.incomplete_frames, this.expired_frames, *this.max_age);
179                        *this.last_expiration = Some(Instant::now());
180                    }
181                }
182                (Poll::Ready(None), _) => {
183                    // Inner stream closed, dump all incomplete frames
184                    tracing::trace!("inner stream closed, dumping incomplete frames");
185                    if this.incomplete_frames.len() > 0 {
186                        this.incomplete_frames.retain(|id, _| {
187                            this.expired_frames.push(*id);
188                            false
189                        });
190                    } else {
191                        tracing::trace!("done");
192                        return Poll::Ready(None);
193                    }
194                }
195                (Poll::Pending, Poll::Ready(_)) => {
196                    // Check if some frames are expired
197                    Self::expire_frames(this.incomplete_frames, this.expired_frames, *this.max_age);
198                    *this.last_expiration = Some(Instant::now());
199                    this.timer.as_mut().reset_timer();
200                }
201                (Poll::Pending, Poll::Pending) => return Poll::Pending,
202            }
203        }
204    }
205}

207/// Stream extension methods for frame reassembly.
208pub trait ReassemblerExt: futures::Stream<Item = Segment> {
209    /// Attaches a [`Reassembler`] with the given `timeout` for frame completion and `capacity`
210    /// to this stream.
211    fn reassembler(self, timeout: Duration, capacity: usize) -> Reassembler<Self, FrameHashMap>
212    where
213        Self: Sized,
214    {
215        // FrameHashMap is much faster than a FrameDashMap used in a FrameInspector
216        Reassembler::new(
217            self,
218            FrameHashMap::with_capacity(FrameInspector::INCOMPLETE_FRAME_RATIO * capacity + 1),
219            timeout,
220            capacity,
221        )
222    }
223
224    /// Attaches a [`Reassembler`] with the given `timeout` for frame completion, `capacity`
225    /// to this stream and [`FrameInspector`].
226    ///
227    /// Use only in situations where the [`FrameInspector`] is really needed, as such Reassembler
228    /// is slower than a Reassembler without a `FrameInspector`.
229    fn reassembler_with_inspector(
230        self,
231        timeout: Duration,
232        capacity: usize,
233        inspector: FrameInspector,
234    ) -> Reassembler<Self, FrameDashMap>
235    where
236        Self: Sized,
237    {
238        Reassembler::new(self, inspector.0.clone(), timeout, capacity)
239    }
240}

259    fn result_comparator(a: &Result<Frame, SessionError>, b: &Result<Frame, SessionError>) -> Ordering {
260        match (a, b) {
261            (Ok(a), Ok(b)) => a.frame_id.cmp(&b.frame_id),
262            (Err(SessionError::FrameDiscarded(a)), Ok(b)) => a.cmp(&b.frame_id),
263            (Ok(a), Err(SessionError::FrameDiscarded(b))) => a.frame_id.cmp(b),
264            (Err(SessionError::FrameDiscarded(a)), Err(SessionError::FrameDiscarded(b))) => a.cmp(b),
265            _ => panic!("unexpected result"),
266        }
267    }

405    #[test_log::test(tokio::test)]
406    pub async fn reassembler_should_wait_and_discard_if_full() -> anyhow::Result<()> {
407        let expected = (1u32..=5)
408            .map(|frame_id| Frame {
409                frame_id,
410                data: hopr_crypto_random::random_bytes::<30>().into(),
411                is_terminating: false,
412            })
413            .collect::<Vec<_>>();
414
415        let (r_sink, r_stream) = futures::channel::mpsc::unbounded();
416        let r_stream = r_stream.reassembler(Duration::from_millis(200), 3);
417
418        pin_mut!(r_sink);
419        pin_mut!(r_stream);
420
421        // This creates 5 frames with 2 segments each
422        let segments = expected
423            .iter()
424            .cloned()
425            .flat_map(|f| segment(f.data, 20, f.frame_id).unwrap())
426            .collect::<Vec<_>>();
427
428        let to_send = [
429            // Frame 1: Segment 2, Segment 1 missing
430            segments[1].clone(),
431            // Frame 2: Segment 1, Segment 2 missing
432            segments[2].clone(),
433            // Frame 3: Segment 2, Segment 1 missing
434            segments[5].clone(),
435        ];
436
437        let start = Instant::now();
438
439        // Reassembler now contains 3 incomplete frames
440        r_sink.send_all(&mut futures::stream::iter(to_send).map(Ok)).await?;
441
442        // It must not yield anything
443        assert!(
444            r_stream
445                .next()
446                .timeout(futures_time::time::Duration::from_millis(20))
447                .await
448                .is_err()
449        );
450
451        // Entire Frames 4 & 5
452        r_sink
453            .send_all(
454                &mut futures::stream::iter([
455                    segments[6].clone(),
456                    segments[7].clone(),
457                    segments[8].clone(),
458                    segments[9].clone(),
459                ])
460                .map(Ok),
461            )
462            .await?;
463
464        let mut reassembled = Vec::new();
465        for _ in 0..5 {
466            reassembled.push(r_stream.next().await.ok_or(anyhow!("missing frame"))?);
467        }
468        reassembled.sort_by(result_comparator);
469
470        assert!(
471            matches!(reassembled[0], Err(SessionError::FrameDiscarded(1))),
472            "{:?} must be discarded ID 1",
473            reassembled[0]
474        );
475        assert!(
476            matches!(reassembled[1], Err(SessionError::FrameDiscarded(2))),
477            "{:?} must be discarded ID 2",
478            reassembled[1]
479        );
480        assert!(
481            matches!(reassembled[2], Err(SessionError::FrameDiscarded(3))),
482            "{:?} must be discarded ID 3",
483            reassembled[2]
484        );
485        assert!(
486            matches!(&reassembled[3], Ok(f) if f == &expected[3].clone()),
487            "{:?} (idx 3) must be {:?}",
488            reassembled[3],
489            expected[3]
490        );
491        assert!(
492            matches!(&reassembled[4], Ok(f) if f == &expected[4].clone()),
493            "{:?} (idx 3) must be {:?}",
494            reassembled[4],
495            expected[4]
496        );
497
498        r_sink.close().await?;
499        assert_eq!(None, r_stream.try_next().await?);
500
501        assert!(start.elapsed() >= Duration::from_millis(200));
502
503        Ok(())
504    }
