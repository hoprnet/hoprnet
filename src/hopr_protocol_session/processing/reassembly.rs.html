23/// Reassembler is a stream adaptor that reads [`Segments`](Segment) from the underlying
24/// stream and tries to put them into correct order so they form a [`Frame`].
25///
26/// This is essentially the inverse of [`Segmenter`](super::segmenter_old::Segmenter).
27///
28/// Reassembler takes two parameters: `max_age` and `capacity`:
29///
30/// The `max_age` specifies how long an incomplete Frame (with a missing segment(s)) is to be kept
31/// in the internal buffer until it is considered definitely lost.
32/// In other words, it specifies how long the reassembler is allowed to wait for all segments
33/// of a Frame to arrive from the underlying stream.
34///
35/// The `capacity` specifies the maximum number of incomplete frames to keep in
36/// the internal buffer. If the reassembler is at maximum capacity, the underlying stream is not
37/// polled for new segments, leaving the oldest incomplete frames in the reassembler to expire and
38/// be definitely lost.
39///
40/// By definition, Reassembler is a fallible stream, yielding either `Ok(Some(`[`Frame`]`))`,
41/// `Err(`[`SessionError::FrameDiscarded`]`)` when a frame is lost due to expiry, or `Ok(None)` when
42/// there are no more elements in the underlying stream.
43///
44/// The reassemblers internal buffer is stored in a [`FrameMap`] and can be constructed using
45/// different implementations of it, suitable for different use-cases.
46///
47/// Use [`ReassemblerExt`] methods to turn a ` Segment ` stream into a fallible `Frame` stream using the `Reassembler`.
48#[must_use = "streams do nothing unless polled"]
49#[pin_project::pin_project]
50pub struct Reassembler<S, M> {
51    #[pin]
52    inner: S,
53    #[pin]
54    timer: futures_time::task::Sleep,
55    incomplete_frames: M,
56    expired_frames: Vec<FrameId>,
57    max_age: Duration,
58    capacity: usize,
59    last_expiration: Option<Instant>,
60}
61
62impl<S: futures::Stream<Item = Segment>, M: FrameMap> Reassembler<S, M> {
63    fn new(inner: S, incomplete_frames: M, max_age: Duration, capacity: usize) -> Self {
64        Self {
65            inner,
66            timer: futures_time::task::sleep(max_age.into()),
67            incomplete_frames,
68            expired_frames: Vec::with_capacity(capacity),
69            last_expiration: None,
70            max_age,
71            capacity,
72        }
73    }
74
75    fn expire_frames(incomplete_frames: &mut M, expired_frames: &mut Vec<FrameId>, max_age: Duration) {
76        incomplete_frames.retain(|id, builder| {
77            if builder.last_recv.elapsed() >= max_age {
78                expired_frames.push(*id);
79                false
80            } else {
81                true
82            }
83        });
84    }
85}

87impl<S: futures::Stream<Item = Segment>, M: FrameMap> futures::Stream for Reassembler<S, M> {
88    type Item = Result<Frame, SessionError>;
89
90    #[instrument(name = "Reassembler::poll_next", level = "trace", skip(self, cx), fields(num_incomplete = self.incomplete_frames.len()), ret)]
91    fn poll_next(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
92        let mut this = self.project();
93        loop {
94            if let Some(frame_id) = this.expired_frames.pop() {
95                tracing::trace!(frame_id, "emit discarded frame");
96                return Poll::Ready(Some(Err(SessionError::FrameDiscarded(frame_id))));
97            }
98
99            // Poll the timer only if there are incomplete frames
100            let timer_poll = if this.incomplete_frames.len() > 0 {
101                this.timer.as_mut().poll(cx)
102            } else {
103                Poll::Pending
104            };
105
106            // Poll the inner stream only if there's space in the reassembler
107            let inner_poll = if this.incomplete_frames.len() <= *this.capacity {
108                this.inner.as_mut().poll_next(cx)
109            } else {
110                // This essentially forces the incomplete frames to be expired
111                tracing::warn!("reassembler has reached its capacity");
112                Poll::Pending
113            };
114
115            tracing::trace!("polling next");
116            match (inner_poll, timer_poll) {
117                (Poll::Ready(Some(item)), timer) => {
118                    if timer.is_ready() {
119                        this.timer.as_mut().reset_timer();
120                    }
121
122                    tracing::trace!(
123                        frame_id = item.frame_id,
124                        seq_idx = item.seq_idx,
125                        seq_len = %item.seq_flags,
126                        "received segment"
127                    );
128
129                    match this.incomplete_frames.entry(item.frame_id) {
130                        FrameMapEntry::Occupied(mut e) => {
131                            let builder = e.get_builder_mut();
132                            let seg_id = item.id();
133                            match builder.add_segment(item) {
134                                Ok(_) => {
135                                    tracing::trace!(frame_id = builder.frame_id(), %seg_id, "added segment");
136                                    if builder.is_complete() {
137                                        tracing::trace!(frame_id = builder.frame_id(), "frame is complete");
138                                        return Poll::Ready(Some(e.finalize().try_into()));
139                                    }
140                                }
141                                Err(error) => {
142                                    tracing::error!(%error, %seg_id, "encountered invalid segment");
143                                }
144                            }
145                        }
146                        FrameMapEntry::Vacant(e) => {
147                            let builder = FrameBuilder::from(item);
148                            if builder.is_complete() {
149                                tracing::trace!(frame_id = builder.frame_id(), "segment frame is complete");
150                                return Poll::Ready(Some(builder.try_into()));
151                            } else {
152                                tracing::trace!(frame_id = builder.frame_id(), "added segment for new frame");
153                                e.insert_builder(builder);
154                            }
155                        }
156                    };
157
158                    // Since the retaining operation is potentially expensive,
159                    // we do it actually only if there's a real chance that a frame is expired
160                    if this.last_expiration.is_none_or(|e| e.elapsed() >= *this.max_age) {
161                        Self::expire_frames(this.incomplete_frames, this.expired_frames, *this.max_age);
162                        *this.last_expiration = Some(Instant::now());
163                    }
164                }
165                (Poll::Ready(None), _) => {
166                    // Inner stream closed, dump all incomplete frames
167                    tracing::trace!("inner stream closed, dumping incomplete frames");
168                    if this.incomplete_frames.len() > 0 {
169                        this.incomplete_frames.retain(|id, _| {
170                            this.expired_frames.push(*id);
171                            false
172                        });
173                    } else {
174                        tracing::trace!("done");
175                        return Poll::Ready(None);
176                    }
177                }
178                (Poll::Pending, Poll::Ready(_)) => {
179                    // Check if some frames are expired
180                    Self::expire_frames(this.incomplete_frames, this.expired_frames, *this.max_age);
181                    *this.last_expiration = Some(Instant::now());
182                    this.timer.as_mut().reset_timer();
183                }
184                (Poll::Pending, Poll::Pending) => return Poll::Pending,
185            }
186        }
187    }
188}

190/// Stream extension methods for frame reassembly.
191pub trait ReassemblerExt: futures::Stream<Item = Segment> {
192    /// Attaches a [`Reassembler`] with the given `timeout` for frame completion and `capacity`
193    /// to this stream.
194    fn reassembler(self, timeout: Duration, capacity: usize) -> Reassembler<Self, FrameHashMap>
195    where
196        Self: Sized,
197    {
198        // FrameHashMap is much faster than a FrameDashMap used in a FrameInspector
199        Reassembler::new(
200            self,
201            FrameHashMap::with_capacity(FrameInspector::INCOMPLETE_FRAME_RATIO * capacity + 1),
202            timeout,
203            capacity,
204        )
205    }
206
207    /// Attaches a [`Reassembler`] with the given `timeout` for frame completion, `capacity`
208    /// to this stream and [`FrameInspector`].
209    ///
210    /// Use only in situations where the [`FrameInspector`] is really needed, as such Reassembler
211    /// is slower than a Reassembler without a `FrameInspector`.
212    fn reassembler_with_inspector(
213        self,
214        timeout: Duration,
215        capacity: usize,
216        inspector: FrameInspector,
217    ) -> Reassembler<Self, FrameDashMap>
218    where
219        Self: Sized,
220    {
221        Reassembler::new(self, inspector.0.clone(), timeout, capacity)
222    }
223}

225impl<T: ?Sized> ReassemblerExt for T where T: futures::Stream<Item = Segment> {}

242    fn result_comparator(a: &Result<Frame, SessionError>, b: &Result<Frame, SessionError>) -> Ordering {
243        match (a, b) {
244            (Ok(a), Ok(b)) => a.frame_id.cmp(&b.frame_id),
245            (Err(SessionError::FrameDiscarded(a)), Ok(b)) => a.cmp(&b.frame_id),
246            (Ok(a), Err(SessionError::FrameDiscarded(b))) => a.frame_id.cmp(b),
247            (Err(SessionError::FrameDiscarded(a)), Err(SessionError::FrameDiscarded(b))) => a.cmp(b),
248            _ => panic!("unexpected result"),
249        }
250    }

388    #[test_log::test(tokio::test)]
389    pub async fn reassembler_should_wait_and_discard_if_full() -> anyhow::Result<()> {
390        let expected = (1u32..=5)
391            .map(|frame_id| Frame {
392                frame_id,
393                data: hopr_crypto_random::random_bytes::<30>().into(),
394                is_terminating: false,
395            })
396            .collect::<Vec<_>>();
397
398        let (r_sink, r_stream) = futures::channel::mpsc::unbounded();
399        let r_stream = r_stream.reassembler(Duration::from_millis(200), 3);
400
401        pin_mut!(r_sink);
402        pin_mut!(r_stream);
403
404        // This creates 5 frames with 2 segments each
405        let segments = expected
406            .iter()
407            .cloned()
408            .flat_map(|f| segment(f.data, 20, f.frame_id).unwrap())
409            .collect::<Vec<_>>();
410
411        let to_send = [
412            // Frame 1: Segment 2, Segment 1 missing
413            segments[1].clone(),
414            // Frame 2: Segment 1, Segment 2 missing
415            segments[2].clone(),
416            // Frame 3: Segment 2, Segment 1 missing
417            segments[5].clone(),
418        ];
419
420        let start = Instant::now();
421
422        // Reassembler now contains 3 incomplete frames
423        r_sink.send_all(&mut futures::stream::iter(to_send).map(Ok)).await?;
424
425        // It must not yield anything
426        assert!(
427            r_stream
428                .next()
429                .timeout(futures_time::time::Duration::from_millis(20))
430                .await
431                .is_err()
432        );
433
434        // Entire Frames 4 & 5
435        r_sink
436            .send_all(
437                &mut futures::stream::iter([
438                    segments[6].clone(),
439                    segments[7].clone(),
440                    segments[8].clone(),
441                    segments[9].clone(),
442                ])
443                .map(Ok),
444            )
445            .await?;
446
447        let mut reassembled = Vec::new();
448        for _ in 0..5 {
449            reassembled.push(r_stream.next().await.ok_or(anyhow!("missing frame"))?);
450        }
451        reassembled.sort_by(result_comparator);
452
453        assert!(matches!(reassembled[0], Err(SessionError::FrameDiscarded(1))));
454        assert!(matches!(reassembled[1], Err(SessionError::FrameDiscarded(2))));
455        assert!(matches!(reassembled[2], Err(SessionError::FrameDiscarded(3))));
456        assert!(matches!(&reassembled[3], Ok(f) if f == &expected[3].clone()));
457        assert!(matches!(&reassembled[4], Ok(f) if f == &expected[4].clone()));
458
459        r_sink.close().await?;
460        assert_eq!(None, r_stream.try_next().await?);
461
462        assert!(start.elapsed() >= Duration::from_millis(200));
463
464        Ok(())
465    }
