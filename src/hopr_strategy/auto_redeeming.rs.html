1//! ## Auto Redeeming Strategy
2//! This strategy listens for newly added acknowledged tickets and automatically issues a redeem transaction on that
3//! ticket. It can be configured to automatically redeem all tickets or only aggregated tickets (which results in far
4//! fewer on-chain transactions being issued).
5//!
6//! For details on default parameters, see [AutoRedeemingStrategyConfig].

43/// Configuration object for the `AutoRedeemingStrategy`
44#[serde_as]
45#[derive(Debug, Clone, Copy, PartialEq, Eq, smart_default::SmartDefault, Validate, Serialize, Deserialize)]
46pub struct AutoRedeemingStrategyConfig {
47    /// If set, the strategy will redeem only aggregated tickets.
48    /// Otherwise, it redeems all acknowledged winning tickets.
49    ///
50    /// Default is `false`.
51    #[serde(default = "just_false")]
52    #[default = false]
53    pub redeem_only_aggregated: bool,
54
55    /// If set to true, will redeem all tickets in the channel (which are over the
56    /// `minimum_redeem_ticket_value` threshold) once it transitions to `PendingToClose`.
57    ///
58    /// Default is `true`.
59    #[serde(default = "just_true")]
60    #[default = true]
61    pub redeem_all_on_close: bool,
62
63    /// The strategy will only redeem an acknowledged winning ticket if it has at least this value of HOPR.
64    /// If 0 is given, the strategy will redeem tickets regardless of their value.
65    ///
66    /// Default is `1 wxHOPR`.
67    #[serde(default = "min_redeem_hopr")]
68    #[serde_as(as = "DisplayFromStr")]
69    #[default(min_redeem_hopr())]
70    pub minimum_redeem_ticket_value: HoprBalance,
71
72    /// If set, the strategy will redeem each incoming winning ticket.
73    /// Otherwise, it will try to redeem tickets in all channels periodically.
74    ///
75    /// Set this to `true` when winning tickets are not happening too often (e.g., when winning probability
76    /// is below 1%).
77    /// Set this to `false` when winning tickets are happening very often (e.g., when winning probability
78    /// is above 1%).
79    ///
80    /// Default is `true`
81    #[serde(default = "just_false")]
82    #[default = false]
83    pub redeem_on_winning: bool,
84}

86/// The `AutoRedeemingStrategy` automatically sends an acknowledged ticket
87/// for redemption once encountered.
88/// The strategy does not await the result of the redemption.
89pub struct AutoRedeemingStrategy<A: TicketRedeemActions> {
90    hopr_chain_actions: A,
91    cfg: AutoRedeemingStrategyConfig,
92}

94impl<A: TicketRedeemActions> Debug for AutoRedeemingStrategy<A> {

115#[async_trait]
116impl<A> SingularStrategy for AutoRedeemingStrategy<A>
117where
118    A: TicketRedeemActions + Send + Sync,
119{
120    async fn on_tick(&self) -> crate::errors::Result<()> {
121        if !self.cfg.redeem_on_winning {
122            debug!("trying to redeem all tickets in all channels");
123
124            let count = self
125                .hopr_chain_actions
126                .redeem_all_tickets(self.cfg.minimum_redeem_ticket_value, self.cfg.redeem_only_aggregated)
127                .await?
128                .len();
129            if count > 0 {
130                #[cfg(all(feature = "prometheus", not(test)))]
131                METRIC_COUNT_AUTO_REDEEMS.increment_by(count as u64);
132
133                info!(count, "strategy issued ticket redemptions");
134            } else {
135                debug!(count, "strategy issued no ticket redemptions");
136            }
137
138            Ok(())
139        } else {
140            Err(CriteriaNotSatisfied)
141        }
142    }
143
144    async fn on_acknowledged_winning_ticket(&self, ack: &AcknowledgedTicket) -> crate::errors::Result<()> {
145        if self.cfg.redeem_on_winning
146            && ((!self.cfg.redeem_only_aggregated || ack.verified_ticket().is_aggregated())
147                && ack.verified_ticket().amount.ge(&self.cfg.minimum_redeem_ticket_value))
148        {
149            info!(%ack, "redeeming");
150
151            #[cfg(all(feature = "prometheus", not(test)))]
152            METRIC_COUNT_AUTO_REDEEMS.increment();
153
154            let rxs = self
155                .hopr_chain_actions
156                .redeem_tickets_with_counterparty(
157                    ack.ticket.verified_issuer(),
158                    self.cfg.minimum_redeem_ticket_value,
159                    self.cfg.redeem_only_aggregated,
160                )
161                .await?;
162
163            std::mem::drop(rxs); // The Receiver is not intentionally awaited here and the oneshot Sender can fail safely
164
165            Ok(())
166        } else {
167            Err(CriteriaNotSatisfied)
168        }
169    }
170
171    async fn on_own_channel_changed(
172        &self,
173        channel: &ChannelEntry,
174        direction: ChannelDirection,
175        change: ChannelChange,
176    ) -> crate::errors::Result<()> {
177        if direction != ChannelDirection::Incoming || !self.cfg.redeem_all_on_close {
178            return Ok(());
179        }
180
181        if let ChannelChange::Status { left: old, right: new } = change {
182            if old != ChannelStatus::Open || !matches!(new, ChannelStatus::PendingToClose(_)) {
183                debug!(?channel, "ignoring channel state change that's not in PendingToClose");
184                return Ok(());
185            }
186            info!(%channel, "channel transitioned to PendingToClose, checking if it has tickets to redeem");
187
188            let count = self
189                .hopr_chain_actions
190                .redeem_tickets_in_channel(
191                    channel,
192                    self.cfg.minimum_redeem_ticket_value,
193                    self.cfg.redeem_only_aggregated,
194                )
195                .await?
196                .len();
197
198            #[cfg(all(feature = "prometheus", not(test)))]
199            METRIC_COUNT_AUTO_REDEEMS.increment_by(count as u64);
200
201            if count > 0 {
202                info!(count, %channel, "tickets in channel being closed sent for redemption");
203                Ok(())
204            } else {
205                info!(%channel, "no redeemable tickets with minimum amount in channel being closed");
206                Err(CriteriaNotSatisfied)
207            }
208        } else {
209            Err(CriteriaNotSatisfied)
210        }
211    }
212}

246    fn generate_random_ack_ticket(
247        index: u64,
248        idx_offset: u32,
249        worth_packets: u32,
250    ) -> anyhow::Result<AcknowledgedTicket> {
251        let hk1 = HalfKey::random();
252        let hk2 = HalfKey::random();
253
254        let challenge = Response::from_half_keys(&hk1, &hk2)?.to_challenge()?;
255
256        Ok(TicketBuilder::default()
257            .addresses(&*BOB, &*ALICE)
258            .amount(PRICE_PER_PACKET.div_f64(1.0f64)? * worth_packets)
259            .index(index)
260            .index_offset(idx_offset)
261            .win_prob(WinningProbability::ALWAYS)
262            .channel_epoch(4)
263            .challenge(challenge)
264            .build_signed(&BOB, &Hash::default())?
265            .into_acknowledged(Response::from_half_keys(&hk1, &hk2)?))
266    }
