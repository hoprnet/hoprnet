11/// Enumerates different errors produced by this crate.
12#[derive(Error, Debug)]
13pub enum HelperErrors {
14    /// Error propagated by IO operations
15    #[error(transparent)]
16    UnableToReadFromPath(#[from] std::io::Error),
17
18    /// Error in parsing provided comma-separated addresses
19    #[error("error parsing address: {0:?}")]
20    UnableToParseAddress(String),
21
22    /// System time rrror
23    #[error(transparent)]
24    SystemTime(#[from] std::time::SystemTimeError),
25
26    /// Error when identity cannot be created
27    #[error("unable to create identity")]
28    UnableToCreateIdentity,
29
30    #[error("unable to update identity password")]
31    UnableToUpdateIdentityPassword,
32
33    /// Error due to supplying a non-existing file name
34    #[error("incorrect filename: {0}")]
35    IncorrectFilename(String),
36
37    /// Error when identity existed
38    #[error("identity file exists: {0}")]
39    IdentityFileExists(String),
40
41    /// Fail to read identity
42    #[error("unable to read identity")]
43    UnableToReadIdentity,
44
45    /// Fail to find the identity directory
46    #[error("unable to read identity directory")]
47    MissingIdentityDirectory,
48
49    /// Fail to delete an identity
50    #[error("unable to delete identity")]
51    UnableToDeleteIdentity,
52
53    /// Provided environement does not match with that in the `ethereum/contracts/contracts-addresses.json`
54    #[error("environment info mismatch")]
55    EnvironmentInfoMismatch,
56
57    /// Wrong foundry contract root is provided
58    #[error("unable to set foundry root")]
59    UnableToSetFoundryRoot,
60
61    /// Fail to run foundry
62    #[error("unable to run foundry")]
63    ErrorInRunningFoundry,
64
65    /// Fail to read password
66    #[error("unable read password")]
67    UnableToReadPassword,
68
69    /// Fail to read private key
70    #[error("cannot read private key error: {0}")]
71    UnableToReadPrivateKey(String),
72
73    /// Paramters are missing
74    #[error("missing parameter: {0}")]
75    MissingParameter(String),
76
77    /// Error with the keystore file
78    #[error(transparent)]
79    KeyStoreError(#[from] KeyPairError),
80
81    #[error("deserialization Error: {0}")]
82    /// Serde JSON Error
83    SerdeJson(#[from] serde_json::Error),
84
85    /// Cannot find network details from the given network name
86    #[error("unable to find network details from the given network name ")]
87    UnknownNetwork,
88
89    /// Error with HTTP Json RPC provider
90    #[error(transparent)]
91    RpcError(#[from] hopr_chain_rpc::errors::RpcError),
92
93    /// Error with signer wallet error
94    #[error(transparent)]
95    WalletError(#[from] ethers::signers::WalletError),
96
97    /// Fail to make a multicall
98    #[error("multicall Error: {0}")]
99    MulticallError(String),
100
101    /// Fail to make a multisend call
102    #[error("internal transaction failure in multisend")]
103    MultiSendError,
104
105    /// Txn caller does not have the minter role
106    #[error("caller does not have the privilege to mint tokens")]
107    NotAMinter,
108
109    /// Error with middleware
110    #[error("middleware Error: {0}")]
111    MiddlewareError(String),
112
113    /// A required smart contract (Safe or module proxy instance) is not deployed
114    #[error("contract not deployed: {0}")]
115    ContractNotDeployed(String),
116
117    // encode packed error
118    #[error(transparent)]
119    EncodePackedError(#[from] ethers::abi::EncodePackedError),
120
121    // error of parsing addresses
122    #[error("Cannot parse address: {0}")]
123    InvalidAddress(String),
124
125    // general error of parsing
126    #[error("Cannot parse: {0}")]
127    ParseError(String),
128}
