221/// Implement ethers-rs `get_create2_address` function
222/// Returns the CREATE2 address of a smart contract as specified in
223/// [EIP1014](https://github.com/ethereum/EIPs/blob/master/EIPS/eip-1014.md)
224///
225/// keccak256( 0xff ++ senderAddress ++ salt ++ keccak256(init_code))[12..]
226pub fn get_create2_address(from: Address, salt: impl AsRef<[u8]>, init_code: impl AsRef<[u8]>) -> Address {
227    let salt = salt.as_ref();
228    let init_code_hash = keccak256(init_code.as_ref());
229
230    let mut bytes = Vec::with_capacity(1 + 20 + salt.len() + init_code_hash.len());
231    bytes.push(0xff);
232    bytes.extend_from_slice(from.as_slice());
233    bytes.extend_from_slice(salt);
234    bytes.extend_from_slice(&init_code_hash.0);
235
236    let hash = keccak256(bytes);
237
238    let mut bytes = [0u8; 20];
239    bytes.copy_from_slice(&hash[12..]);
240    Address::from(bytes)
