40impl<T: KeyIvInit> KeyIvInit for IvKey<T> {
41    fn new(key: &Key<Self>, iv: &Iv<Self>) -> Self {
42        let mut out = Vec::with_capacity(Self::SIZE);
43        out.extend_from_slice(iv.as_ref());
44        out.extend_from_slice(key.as_ref());
45        Self(out.into_boxed_slice(), PhantomData)
46    }
47}
48
49impl<T: KeyIvInit> Default for IvKey<T> {
50    fn default() -> Self {
51        Self(vec![0u8; Self::SIZE].into_boxed_slice(), PhantomData)
52    }
53}

79impl<T: KeyIvInit> IvKey<T> {

83    /// Returns the IV part.
84    #[inline]

87    }
88
89    /// Returns IV as a mutable slice.
90    #[inline]

93    }
94
95    /// Returns the key part.
96    #[inline]
97    pub fn key(&self) -> &Key<T> {
98        Key::<T>::from_slice(&self.0[T::IvSize::USIZE..])
99    }
100
101    /// Returns the key as a mutable slice.
102    #[inline]
103    pub fn key_mut(&mut self) -> &mut [u8] {
104        &mut self.0[T::IvSize::USIZE..]
105    }
106
107    /// Turn this instance into another [`KeyIvInit`] with the same IV and key sizes.
108    #[inline]
109    pub fn into_init<V>(self) -> V
110    where
111        V: KeyIvInit<KeySize = T::KeySize, IvSize = T::IvSize>,
112    {
113        V::new(self.key(), self.iv())
114    }
