24/// Generates a random elliptic curve point on the secp256k1 curve (but not a point in infinity).
25/// Returns the encoded secret scalar and the corresponding point.
26pub(crate) fn random_group_element() -> ([u8; 32], NonIdentity<AffinePoint>) {
27    // Since sep256k1 has a group of prime order, a non-zero scalar cannot result into an identity point.
28    // let scalar = k256::NonZeroScalar::random(&mut hopr_crypto_random::rng());
29
30    // Requires manual rejection sampling implementation due to incompatible rand crates
31    let mut rng = hopr_crypto_random::rng();
32    let mut bytes = k256::FieldBytes::default();
33    use hopr_crypto_random::Rng;
34    loop {
35        rng.fill_bytes(&mut bytes);
36        if let Some::<k256::Scalar>(scalar) = k256::Scalar::from_repr(bytes).into() {
37            let point = PublicKey::from_privkey(&scalar.to_bytes())
38                .expect("non-zero scalar cannot represent an invalid public key");
39            return (scalar.to_bytes().into(), point.into());
40        }
41    }
42}

44/// Creates X25519 secret scalar (also compatible with Ed25519 scalar) from the given bytes.
45/// This function ensures the value is pre-multiplied by the curve's co-factor and already
46/// reduced mod 2^255-19.
47pub fn x25519_scalar_from_bytes(bytes: &[u8]) -> crate::errors::Result<curve25519_dalek::scalar::Scalar> {
48    if bytes.len() == 32 {
49        // Representation of the scalar is little-endian
50        let mut clamped = [0u8; 32];
51        clamped.copy_from_slice(&bytes[..32]);
52        clamped[00] &= 0b1111_1000; // clear the 3 LSB bits (= multiply by Curve25519's co-factor)
53        clamped[31] &= 0b0111_1111; // clear the 256-th bit
54        clamped[31] |= 0b0100_0000; // make it a 255-bit number
55
56        Ok(curve25519_dalek::scalar::Scalar::from_bytes_mod_order(clamped))
57    } else {
58        Err(InvalidInputValue("bytes"))
59    }
60}

62/// Creates secp256k1 secret scalar from the given bytes.
63/// Note that this function allows zero scalars.
64#[allow(deprecated)] // Until the dependency updates to newer versions of `generic-array`
65pub fn k256_scalar_from_bytes(bytes: &[u8]) -> crate::errors::Result<k256::Scalar> {
66    if bytes.len() == k256::elliptic_curve::FieldBytesSize::<Secp256k1>::to_usize() {
67        Option::from(k256::Scalar::from_repr(*k256::FieldBytes::from_slice(bytes))).ok_or(InvalidInputValue("bytes"))
68    } else {
69        Err(InvalidInputValue("bytes"))
70    }

73/// Sample a random secp256k1 field element that can represent a valid secp256k1 point.
74///
75/// The implementation uses the ` hash_to_field ` function as defined in
76/// `<https://www.ietf.org/archive/id/draft-irtf-cfrg-hash-to-curve-13.html#name-hashing-to-a-finite-field>`
77/// The `secret` must be at least `SecretKey::LENGTH` long.
78/// The `tag` parameter will be used as an additional Domain Separation Tag.
79pub fn sample_secp256k1_field_element(secret: &[u8], tag: &str) -> crate::errors::Result<HalfKey> {
80    if secret.len() >= SecretKey::LENGTH {
81        let scalar = Secp256k1::hash_to_scalar::<ExpandMsgXmd<Sha3_256>>(
82            &[secret],
83            &[b"secp256k1_XMD:SHA3-256_SSWU_RO_", tag.as_bytes()],
84        )
85        .map_err(|_| CalculationError)?;
86        Ok(HalfKey::try_from(scalar.to_bytes().as_ref())?)
87    } else {
88        Err(InvalidParameterSize {
89            name: "secret",
90            expected: SecretKey::LENGTH,
91        })
92    }
93}
