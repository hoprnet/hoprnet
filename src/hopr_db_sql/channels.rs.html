19/// API for editing [ChannelEntry] in the DB.
20pub struct ChannelEditor {
21    orig: ChannelEntry,
22    model: channel::ActiveModel,
23    delete: bool,
24}
25
26impl ChannelEditor {
27    /// Original channel entry **before** the edits.
28    pub fn entry(&self) -> &ChannelEntry {
29        &self.orig
30    }
31
32    /// Change the HOPR balance of the channel.
33    pub fn change_balance(mut self, balance: HoprBalance) -> Self {
34        self.model.balance = Set(balance.amount().to_be_bytes().to_vec());
35        self
36    }
37
38    /// Change the channel status.
39    pub fn change_status(mut self, status: ChannelStatus) -> Self {
40        self.model.set_status(status);
41        self
42    }
43
44    /// Change the ticket index.
45    pub fn change_ticket_index(mut self, index: impl Into<U256>) -> Self {
46        self.model.ticket_index = Set(index.into().to_be_bytes().to_vec());
47        self
48    }
49
50    /// Change the channel epoch.
51    pub fn change_epoch(mut self, epoch: impl Into<U256>) -> Self {
52        self.model.epoch = Set(epoch.into().to_be_bytes().to_vec());
53        self
54    }
55
56    /// If set, the channel will be deleted, no other edits will be done.
57    pub fn delete(mut self) -> Self {
58        self.delete = true;
59        self
60    }
61}

63/// Defines DB API for accessing information about HOPR payment channels.
64#[async_trait]
65pub trait HoprDbChannelOperations {
66    /// Retrieves channel by its channel ID hash.
67    ///
68    /// See [generate_channel_id] on how to generate a channel ID hash from source and destination [Addresses](Address).
69    async fn get_channel_by_id<'a>(&'a self, tx: OptTx<'a>, id: &Hash) -> Result<Option<ChannelEntry>>;
70
71    /// Start changes to channel entry.
72    /// If the channel with the given ID exists, the [ChannelEditor] is returned.
73    /// Use [`HoprDbChannelOperations::finish_channel_update`] to commit edits to the DB when done.
74    async fn begin_channel_update<'a>(&'a self, tx: OptTx<'a>, id: &Hash) -> Result<Option<ChannelEditor>>;
75
76    /// Commits changes of the channel to the database.
77    /// Returns the updated channel, or on deletion, the deleted channel entry.
78    async fn finish_channel_update<'a>(&'a self, tx: OptTx<'a>, editor: ChannelEditor) -> Result<Option<ChannelEntry>>;
79
80    /// Retrieves the channel by source and destination.
81    /// This operation should be able to use cache since it can be also called from
82    /// performance-sensitive locations.
83    async fn get_channel_by_parties<'a>(
84        &'a self,
85        tx: OptTx<'a>,
86        src: &Address,
87        dst: &Address,
88        use_cache: bool,
89    ) -> Result<Option<ChannelEntry>>;
90
91    /// Fetches all channels that are `Incoming` to this node.
92    /// Shorthand for `get_channels_via(tx, ChannelDirection::Incoming, my_node)`
93    async fn get_incoming_channels<'a>(&'a self, tx: OptTx<'a>) -> Result<Vec<ChannelEntry>>;
94
95    /// Fetches all channels that are `Outgoing` from this node.
96    /// Shorthand for `get_channels_via(tx, ChannelDirection::Outgoing, my_node)`
97    async fn get_outgoing_channels<'a>(&'a self, tx: OptTx<'a>) -> Result<Vec<ChannelEntry>>;
98
99    /// Retrieves all channels information from the DB.
100    async fn get_all_channels<'a>(&'a self, tx: OptTx<'a>) -> Result<Vec<ChannelEntry>>;
101
102    async fn stream_channels<'a, T: RangeBounds<DateTime<Utc>> + Send>(
103        &'a self,
104        source: Option<Address>,
105        destination: Option<Address>,
106        states: &[ChannelStatusDiscriminants],
107        closure_range: T,
108    ) -> Result<BoxStream<'a, ChannelEntry>>;
109
110    /// Inserts or updates the given channel entry.
111    async fn upsert_channel<'a>(&'a self, tx: OptTx<'a>, channel_entry: ChannelEntry) -> Result<()>;
112}

116    async fn get_channel_by_id<'a>(&'a self, tx: OptTx<'a>, id: &Hash) -> Result<Option<ChannelEntry>> {
117        let id_hex = id.to_hex();
118        self.nest_transaction(tx)
119            .await?
120            .perform(|tx| {
121                Box::pin(async move {
122                    Ok::<_, DbSqlError>(
123                        if let Some(model) = Channel::find()
124                            .filter(channel::Column::ChannelId.eq(id_hex))
125                            .one(tx.as_ref())
126                            .await?
127                        {
128                            Some(model.try_into()?)
129                        } else {
130                            None
131                        },
132                    )
133                })
134            })
135            .await
136    }
137
138    async fn begin_channel_update<'a>(&'a self, tx: OptTx<'a>, id: &Hash) -> Result<Option<ChannelEditor>> {
139        let id_hex = id.to_hex();
140        self.nest_transaction(tx)
141            .await?
142            .perform(|tx| {
143                Box::pin(async move {
144                    match Channel::find()
145                        .filter(channel::Column::ChannelId.eq(id_hex.clone()))
146                        .one(tx.as_ref())
147                        .await?
148                    {
149                        Some(model) => Ok(Some(ChannelEditor {
150                            orig: ChannelEntry::try_from(model.clone())?,
151                            model: model.into_active_model(),
152                            delete: false,
153                        })),
154                        None => Ok(None),
155                    }
156                })
157            })
158            .await
159    }
160
161    async fn finish_channel_update<'a>(&'a self, tx: OptTx<'a>, editor: ChannelEditor) -> Result<Option<ChannelEntry>> {
162        let parties = ChannelParties(editor.orig.source, editor.orig.destination);
163        let ret = self
164            .nest_transaction(tx)
165            .await?
166            .perform(|tx| {
167                Box::pin(async move {
168                    if !editor.delete {
169                        let model = editor.model.update(tx.as_ref()).await?;
170                        match ChannelEntry::try_from(model) {
171                            Ok(channel) => Ok::<_, DbSqlError>(Some(channel)),
172                            Err(e) => Err(DbSqlError::from(e)),
173                        }
174                    } else {
175                        editor.model.delete(tx.as_ref()).await?;
176                        Ok::<_, DbSqlError>(Some(editor.orig))
177                    }
178                })
179            })
180            .await?;
181        self.caches.src_dst_to_channel.invalidate(&parties).await;
182
183        Ok(ret)
184    }

186    #[instrument(level = "trace", skip(self, tx), err)]
187    async fn get_channel_by_parties<'a>(
188        &'a self,
189        tx: OptTx<'a>,
190        src: &Address,
191        dst: &Address,
192        use_cache: bool,
193    ) -> Result<Option<ChannelEntry>> {
194        let fetch_channel = async move {
195            let src_hex = src.to_hex();
196            let dst_hex = dst.to_hex();
197            tracing::warn!(%src, %dst, "cache miss on get_channel_by_parties");
198            self.nest_transaction(tx)
199                .await?
200                .perform(|tx| {
201                    Box::pin(async move {
202                        Ok::<_, DbSqlError>(
203                            if let Some(model) = Channel::find()
204                                .filter(channel::Column::Source.eq(src_hex))
205                                .filter(channel::Column::Destination.eq(dst_hex))
206                                .one(tx.as_ref())
207                                .await?
208                            {
209                                Some(model.try_into()?)
210                            } else {
211                                None
212                            },
213                        )
214                    })
215                })
216                .await
217        };
218
219        if use_cache {
220            Ok(self
221                .caches
222                .src_dst_to_channel
223                .try_get_with(ChannelParties(*src, *dst), fetch_channel)
224                .await?)
225        } else {
226            fetch_channel.await
227        }
228    }

282    async fn stream_channels<'a, T: RangeBounds<DateTime<Utc>> + Send>(
283        &'a self,
284        source: Option<Address>,
285        destination: Option<Address>,
286        states: &[ChannelStatusDiscriminants],
287        closure_range: T,
288    ) -> Result<BoxStream<'a, ChannelEntry>> {
289        let mut incoming_cond = Condition::all();
290        if let Some(source) = source {
291            incoming_cond = incoming_cond.add(channel::Column::Source.eq(source.to_hex()));
292        }
293        if let Some(destination) = destination {
294            incoming_cond = incoming_cond.add(channel::Column::Destination.eq(destination.to_hex()));
295        }
296
297        let mut states_condition = Condition::any();
298        for state in states {
299            // If we're including the pending to close channels in the query, make sure
300            // we include range bounds on closure times
301            if state == &ChannelStatusDiscriminants::PendingToClose {
302                let mut closure_range_condition = Condition::all();
303                closure_range_condition = closure_range_condition
304                    .add(channel::Column::Status.eq(ChannelStatusDiscriminants::PendingToClose as i8));
305                match closure_range.start_bound() {
306                    Bound::Included(closure_start) => {
307                        closure_range_condition =
308                            closure_range_condition.add(channel::Column::ClosureTime.gte(*closure_start))
309                    }
310                    Bound::Excluded(closure_start) => {
311                        closure_range_condition =
312                            closure_range_condition.add(channel::Column::ClosureTime.gt(*closure_start))
313                    }
314                    _ => {}
315                }
316                match closure_range.end_bound() {
317                    Bound::Included(closure_end) => {
318                        closure_range_condition =
319                            closure_range_condition.add(channel::Column::ClosureTime.lte(*closure_end))
320                    }
321                    Bound::Excluded(closure_end) => {
322                        closure_range_condition =
323                            closure_range_condition.add(channel::Column::ClosureTime.lt(*closure_end))
324                    }
325                    _ => {}
326                }
327                states_condition = states_condition.add(closure_range_condition);
328            } else {
329                states_condition = states_condition.add(channel::Column::Status.eq(*state as i8));
330            }
331        }
332
333        Ok(Channel::find()
334            .filter(sea_query::all![incoming_cond, states_condition])
335            .stream(self.index_db.read_only())
336            .await?
337            .filter_map(|maybe_channel| {
338                futures::future::ready(
339                    maybe_channel
340                        .and_then(|c| {
341                            ChannelEntry::try_from(c)
342                                .map_err(|e| sea_orm::error::DbErr::Custom(format!("cannot decode entity: {e}")))
343                        })
344                        .inspect_err(|error| tracing::error!(%error, "invalid channel entry"))
345                        .ok(),
346                )
347            })
348            .boxed())
349    }

351    async fn upsert_channel<'a>(&'a self, tx: OptTx<'a>, channel_entry: ChannelEntry) -> Result<()> {
352        let parties = ChannelParties(channel_entry.source, channel_entry.destination);
353        self.nest_transaction(tx)
354            .await?
355            .perform(|tx| {
356                Box::pin(async move {
357                    let mut model = channel::ActiveModel::from(channel_entry);
358                    if let Some(channel) = channel::Entity::find()
359                        .filter(channel::Column::ChannelId.eq(channel_entry.get_id().to_hex()))
360                        .one(tx.as_ref())
361                        .await?
362                    {
363                        model.id = Set(channel.id);
364                    }
365
366                    Ok::<_, DbSqlError>(model.save(tx.as_ref()).await?)
367                })
368            })
369            .await?;
370
371        self.caches.src_dst_to_channel.invalidate(&parties).await;
372
373        // The invalidation of the unrealized value is done in the finish_channel_update function.
374        // Has no effect if the channel has been inserted.
375
376        Ok(())
377    }
