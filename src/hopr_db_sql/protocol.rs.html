50    /// Validates a ticket from a forwarded packet and replaces it with a new ticket for the next hop.
51    async fn validate_and_replace_ticket(
52        &self,
53        mut fwd: HoprForwardedPacket,
54        me: &ChainKeypair,
55        outgoing_ticket_win_prob: f64,
56        outgoing_ticket_price: Balance,
57    ) -> std::result::Result<HoprForwardedPacket, DbSqlError> {
58        let previous_hop_addr = self.resolve_chain_key(&fwd.previous_hop).await?.ok_or_else(|| {
59            DbSqlError::LogicalError(format!(
60                "failed to find channel key for packet key {} on previous hop",
61                fwd.previous_hop.to_peerid_str()
62            ))
63        })?;
64
65        let next_hop_addr = self.resolve_chain_key(&fwd.outgoing.next_hop).await?.ok_or_else(|| {
66            DbSqlError::LogicalError(format!(
67                "failed to find channel key for packet key {} on next hop",
68                fwd.outgoing.next_hop.to_peerid_str()
69            ))
70        })?;
71
72        let incoming_channel = self
73            .get_channel_by_parties(None, &previous_hop_addr, &self.me_onchain, true)
74            .await?
75            .ok_or_else(|| {
76                DbSqlError::LogicalError(format!(
77                    "no channel found for previous hop address '{previous_hop_addr}'"
78                ))
79            })?;
80
81        // Check: is the ticket in the packet really for the given channel?
82        if !fwd.outgoing.ticket.channel_id.eq(&incoming_channel.get_id()) {
83            return Err(DbSqlError::LogicalError("invalid ticket for channel".into()));
84        }
85
86        let chain_data = self.get_indexer_data(None).await?;
87        let domain_separator = chain_data
88            .channels_dst
89            .ok_or_else(|| DbSqlError::LogicalError("failed to fetch the domain separator".into()))?;
90
91        // The ticket price from the oracle times my node's position on the
92        // path is the acceptable minimum
93        let minimum_ticket_price = chain_data
94            .ticket_price
95            .ok_or_else(|| DbSqlError::LogicalError("failed to fetch the ticket price".into()))?
96            .mul(U256::from(fwd.path_pos));
97
98        #[cfg(all(feature = "prometheus", not(test)))]
99        METRIC_INCOMING_WIN_PROB.observe(fwd.outgoing.ticket.win_prob());
100
101        let remaining_balance = incoming_channel
102            .balance
103            .sub(self.ticket_manager.unrealized_value((&incoming_channel).into()).await?);
104
105        // Here also the signature on the ticket gets validated,
106        // so afterward we are sure the source of the `channel`
107        // (which is equal to `previous_hop_addr`) has issued this
108        // ticket.
109        let verified_incoming_ticket = spawn_fifo_blocking(move || {
110            validate_unacknowledged_ticket(
111                fwd.outgoing.ticket,
112                &incoming_channel,
113                minimum_ticket_price,
114                chain_data.minimum_incoming_ticket_winning_prob,
115                remaining_balance,
116                &domain_separator,
117            )
118        })
119        .await?;
120
121        // We currently take the maximum of the win prob from the incoming ticket
122        // and the one configured on this node.
123        // Therefore, the winning probability can only increase along the path.
124        let outgoing_ticket_win_prob = outgoing_ticket_win_prob.max(verified_incoming_ticket.win_prob());
125
126        // The ticket is now validated, let's place it into acknowledgement waiting queue
127        self.caches
128            .unacked_tickets
129            .insert(
130                fwd.outgoing.ack_challenge,
131                PendingAcknowledgement::WaitingAsRelayer(verified_incoming_ticket.into_unacknowledged(fwd.own_key)),
132            )
133            .await;
134
135        // NOTE: that the path position according to the ticket value
136        // may no longer match the path position from the packet header,
137        // because the price of the ticket may be set higher be the ticket
138        // issuer.
139
140        // Create the new ticket for the new packet
141        let ticket_builder = if fwd.path_pos > 1 {
142            // There must be a channel to the next node if it's not the final hop
143            let outgoing_channel = self
144                .get_channel_by_parties(None, &self.me_onchain, &next_hop_addr, true)
145                .await?
146                .ok_or(DbSqlError::LogicalError(format!(
147                    "channel to '{next_hop_addr}' not found",
148                )))?;
149
150            self.create_multihop_ticket(
151                outgoing_channel,
152                self.me_onchain,
153                next_hop_addr,
154                fwd.path_pos,
155                outgoing_ticket_win_prob,
156                outgoing_ticket_price,
157            )
158            .await?
159        } else {
160            TicketBuilder::zero_hop().direction(&self.me_onchain, &next_hop_addr)
161        };
162
163        // Finally, replace the ticket in the outgoing packet with a new one
164        fwd.outgoing.ticket = ticket_builder
165            .challenge(fwd.next_challenge)
166            .build_signed(me, &domain_separator)?
167            .leak();
168
169        Ok(fwd)
170    }

172    async fn validate_acknowledgement(
173        &self,
174        ack: &Acknowledgement,
175    ) -> std::result::Result<ResolvedAcknowledgement, DbSqlError> {
176        let pending_ack = self
177            .caches
178            .unacked_tickets
179            .remove(&ack.ack_challenge()?)
180            .await
181            .ok_or_else(|| {
182                DbSqlError::AcknowledgementValidationError(format!(
183                    "received unexpected acknowledgement for half key challenge {}",
184                    ack.to_hex()
185                ))
186            })?;
187
188        match pending_ack {
189            PendingAcknowledgement::WaitingAsSender => {
190                trace!("received acknowledgement as sender: first relayer has processed the packet");
191                Ok(ResolvedAcknowledgement::Sending(*ack))
192            }
193
194            PendingAcknowledgement::WaitingAsRelayer(unacknowledged) => {
195                let maybe_channel_with_issuer = self
196                    .get_channel_by_parties(None, unacknowledged.ticket.verified_issuer(), &self.me_onchain, true)
197                    .await?;
198
199                // Issuer's channel must have an epoch matching with the unacknowledged ticket
200                if maybe_channel_with_issuer
201                    .is_some_and(|c| c.channel_epoch.as_u32() == unacknowledged.verified_ticket().channel_epoch)
202                {
203                    let domain_separator = self
204                        .get_indexer_data(None)
205                        .await?
206                        .channels_dst
207                        .ok_or_else(|| DbSqlError::LogicalError("domain separator missing".into()))?;
208
209                    let myself = self.clone();
210                    let ack = *ack;
211                    hopr_parallelize::cpu::spawn_blocking(move || {
212                        // This explicitly checks whether the acknowledgement
213                        // solves the challenge on the ticket. It must be done before we
214                        // check that the ticket is winning, which is a lengthy operation
215                        // and should not be done for bogus unacknowledged tickets
216                        let ack_ticket = unacknowledged.acknowledge(&ack.ack_key_share()?)?;
217
218                        if ack_ticket.is_winning(&myself.chain_key, &domain_separator) {
219                            trace!("Found a winning ticket");
220                            Ok(ResolvedAcknowledgement::RelayingWin(ack_ticket))
221                        } else {
222                            trace!("Found a losing ticket");
223                            Ok(ResolvedAcknowledgement::RelayingLoss(
224                                ack_ticket.ticket.verified_ticket().channel_id,
225                            ))
226                        }
227                    })
228                    .await
229                } else {
230                    Err(DbSqlError::LogicalError(format!(
231                        "no channel found for  address '{}' with a matching epoch",
232                        unacknowledged.ticket.verified_issuer()
233                    )))
234                }
235            }
236        }
237    }

242    #[instrument(level = "trace", skip(self, ack))]
243    async fn handle_acknowledgement(&self, ack: Acknowledgement) -> Result<()> {
244        let result = self.validate_acknowledgement(&ack).await?;
245        match &result {
246            ResolvedAcknowledgement::RelayingWin(ack_ticket) => {
247                // If the ticket was a win, store it
248                self.ticket_manager.insert_ticket(ack_ticket.clone()).await?;
249
250                #[cfg(all(feature = "prometheus", not(test)))]
251                {
252                    METRIC_RECEIVED_ACKS.increment(&["true"]);
253                    METRIC_TICKETS_COUNT.increment(&["winning"]);
254
255                    let verified_ticket = ack_ticket.ticket.verified_ticket();
256                    let channel = verified_ticket.channel_id.to_string();
257                    crate::tickets::METRIC_HOPR_TICKETS_INCOMING_STATISTICS.set(
258                        &[&channel, "unredeemed"],
259                        self.ticket_manager
260                            .unrealized_value(hopr_db_api::tickets::TicketSelector::new(
261                                verified_ticket.channel_id,
262                                verified_ticket.channel_epoch,
263                            ))
264                            .await?
265                            .amount()
266                            .as_u128() as f64,
267                    );
268                    crate::tickets::METRIC_HOPR_TICKETS_INCOMING_STATISTICS
269                        .increment(&[&channel, "winning_count"], 1.0f64);
270                }
271            }
272            ResolvedAcknowledgement::RelayingLoss(_channel) => {
273                #[cfg(all(feature = "prometheus", not(test)))]
274                {
275                    METRIC_RECEIVED_ACKS.increment(&["true"]);
276                    METRIC_TICKETS_COUNT.increment(&["losing"]);
277                    crate::tickets::METRIC_HOPR_TICKETS_INCOMING_STATISTICS
278                        .increment(&[&_channel.to_string(), "losing_count"], 1.0f64);
279                }
280            }
281            ResolvedAcknowledgement::Sending(_) => {
282                #[cfg(all(feature = "prometheus", not(test)))]
283                METRIC_RECEIVED_ACKS.increment(&["true"]);
284            }
285        };
286
287        Ok(())
288    }

304    #[tracing::instrument(level = "trace", skip(self, data))]
305    async fn to_send_no_ack(&self, data: Box<[u8]>, destination: OffchainPublicKey) -> Result<OutgoingPacket> {
306        let next_peer = self.resolve_chain_key(&destination).await?.ok_or_else(|| {
307            DbSqlError::LogicalError(format!(
308                "failed to find chain key for packet key {} on previous hop",
309                destination.to_peerid_str()
310            ))
311        })?;
312
313        // No-ack packets are always sent as zero-hops with a random pseudonym
314        let pseudonym = HoprPseudonym::random();
315        let next_ticket = TicketBuilder::zero_hop().direction(&self.me_onchain, &next_peer);
316
317        let domain_separator = self
318            .get_indexer_data(None)
319            .await?
320            .channels_dst
321            .ok_or_else(|| DbSqlError::LogicalError("failed to fetch the domain separator".into()))?;
322
323        // Construct the outgoing packet
324        let myself = self.clone();
325        let (packet, _) = spawn_fifo_blocking(move || {
326            HoprPacket::into_outgoing(
327                &data,
328                &pseudonym,
329                PacketRouting::NoAck::<ValidatedPath>(destination),
330                &myself.chain_key,
331                next_ticket,
332                &myself.caches.key_id_mapper,
333                &domain_separator,
334            )
335            .map_err(|e| DbSqlError::LogicalError(format!("failed to construct chain components for a packet: {e}")))
336        })
337        .await?;
338
339        if let Some(out) = packet.try_as_outgoing() {
340            let mut transport_payload = Vec::with_capacity(HoprPacket::SIZE);
341            transport_payload.extend_from_slice(out.packet.as_ref());
342            transport_payload.extend_from_slice(&out.ticket.into_encoded());
343
344            #[cfg(all(feature = "prometheus", not(test)))]
345            METRIC_SENT_ACKS.increment();
346
347            Ok(OutgoingPacket {
348                next_hop: out.next_hop,
349                ack_challenge: out.ack_challenge,
350                data: transport_payload.into_boxed_slice(),
351            })
352        } else {
353            Err(DbSqlError::LogicalError("must be an outgoing packet".into()).into())
354        }
355    }

357    #[tracing::instrument(level = "trace", skip(self, data, routing))]
358    async fn to_send(
359        &self,
360        data: Box<[u8]>,
361        routing: ResolvedTransportRouting,
362        outgoing_ticket_win_prob: f64,
363        outgoing_ticket_price: Balance,
364    ) -> Result<OutgoingPacket> {
365        // Get necessary packet routing values
366        let (next_peer, num_hops, pseudonym, routing) = match routing {
367            ResolvedTransportRouting::Forward {
368                pseudonym,
369                forward_path,
370                return_paths,
371            } => (
372                forward_path[0],
373                forward_path.num_hops(),
374                pseudonym,
375                PacketRouting::ForwardPath {
376                    forward_path,
377                    return_paths,
378                },
379            ),
380            ResolvedTransportRouting::Return(pseudonym) => {
381                // Try to retrieve the SURB from the pseudonym
382                let surb = self
383                    .caches
384                    .surbs_per_pseudonym
385                    .get(&pseudonym)
386                    .await
387                    .ok_or(DbSqlError::LogicalError("unknown pseudonym".into()))?
388                    .pop_one()?;
389
390                let next = self
391                    .caches
392                    .key_id_mapper
393                    .map_id_to_public(&surb.first_relayer)
394                    .ok_or(DbSqlError::MissingAccount)?;
395
396                (
397                    next,
398                    surb.additional_data_receiver.chain_length() as usize,
399                    pseudonym,
400                    PacketRouting::Surb(surb),
401                )
402            }
403        };
404
405        let next_peer = self.resolve_chain_key(&next_peer).await?.ok_or_else(|| {
406            DbSqlError::LogicalError(format!(
407                "failed to find chain key for packet key {} on previous hop",
408                next_peer.to_peerid_str()
409            ))
410        })?;
411
412        // Decide whether to create a multi-hop or a zero-hop ticket
413        let next_ticket = if num_hops > 1 {
414            let channel = self
415                .get_channel_by_parties(None, &self.me_onchain, &next_peer, true)
416                .await?
417                .ok_or(DbSqlError::LogicalError(format!("channel to '{next_peer}' not found")))?;
418
419            self.create_multihop_ticket(
420                channel,
421                self.me_onchain,
422                next_peer,
423                num_hops as u8,
424                outgoing_ticket_win_prob,
425                outgoing_ticket_price,
426            )
427            .await?
428        } else {
429            TicketBuilder::zero_hop().direction(&self.me_onchain, &next_peer)
430        };
431
432        let domain_separator = self
433            .get_indexer_data(None)
434            .await?
435            .channels_dst
436            .ok_or_else(|| DbSqlError::LogicalError("failed to fetch the domain separator".into()))?;
437
438        // Construct the outgoing packet
439        let myself = self.clone();
440        let (packet, openers) = spawn_fifo_blocking(move || {
441            HoprPacket::into_outgoing(
442                &data,
443                &pseudonym,
444                routing,
445                &myself.chain_key,
446                next_ticket,
447                &myself.caches.key_id_mapper,
448                &domain_separator,
449            )
450            .map_err(|e| DbSqlError::LogicalError(format!("failed to construct chain components for a packet: {e}")))
451        })
452        .await?;
453
454        // Store the reply openers under the given pseudonym
455        // This is a no-op for reply packets
456        openers
457            .into_iter()
458            .for_each(|p| self.caches.pseudonym_openers.insert(pseudonym, p));
459
460        if let Some(out) = packet.try_as_outgoing() {
461            self.caches
462                .unacked_tickets
463                .insert(out.ack_challenge, PendingAcknowledgement::WaitingAsSender)
464                .await;
465
466            let mut transport_payload = Vec::with_capacity(HoprPacket::SIZE);
467            transport_payload.extend_from_slice(out.packet.as_ref());
468            transport_payload.extend_from_slice(&out.ticket.into_encoded());
469
470            Ok(OutgoingPacket {
471                next_hop: out.next_hop,
472                ack_challenge: out.ack_challenge,
473                data: transport_payload.into_boxed_slice(),
474            })
475        } else {
476            Err(DbSqlError::LogicalError("must be an outgoing packet".into()).into())
477        }
478    }

480    #[tracing::instrument(level = "trace", skip(self, data, pkt_keypair, sender), fields(sender = %sender))]
481    async fn from_recv(
482        &self,
483        data: Box<[u8]>,
484        pkt_keypair: &OffchainKeypair,
485        sender: OffchainPublicKey,
486        outgoing_ticket_win_prob: f64,
487        outgoing_ticket_price: Balance,
488    ) -> Result<Option<IncomingPacket>> {
489        let offchain_keypair = pkt_keypair.clone();
490        let myself = self.clone();
491
492        let packet = spawn_fifo_blocking(move || {
493            HoprPacket::from_incoming(&data, &offchain_keypair, sender, &myself.caches.key_id_mapper, |p| {
494                myself.caches.pseudonym_openers.remove(p)
495            })
496            .map_err(|e| DbSqlError::LogicalError(format!("failed to construct an incoming packet: {e}")))
497        })
498        .await?;
499
500        match packet {
501            HoprPacket::Final(incoming) => {
502                // Store all incoming SURBs if any
503                if !incoming.surbs.is_empty() {
504                    let num_surbs = incoming.surbs.len();
505
506                    self.caches
507                        .surbs_per_pseudonym
508                        .entry_by_ref(&incoming.sender)
509                        .or_default() // Default SurbRingBuffer
510                        .await
511                        .value()
512                        .push(incoming.surbs)?;
513
514                    tracing::trace!(pseudonym = %incoming.sender, num_surbs, "stored incoming surbs for pseudonym");
515                }
516
517                // The contained payload represents an Acknowledgement
518                if incoming.ack_key.is_none() {
519                    let ack: Acknowledgement = incoming.plain_text.as_ref().try_into().map_err(|error| {
520                        tracing::error!(%error, "failed to decode the acknowledgement");
521
522                        #[cfg(all(feature = "prometheus", not(test)))]
523                        METRIC_RECEIVED_ACKS.increment(&["false"]);
524
525                        DbSqlError::DecodingError
526                    })?;
527
528                    let ack = ack
529                        .validate(&incoming.previous_hop)
530                        .map_err(|e| crate::errors::DbSqlError::AcknowledgementValidationError(e.to_string()))?;
531                    self.handle_acknowledgement(ack).await?;
532
533                    Ok(None)
534                } else {
535                    Ok(Some(IncomingPacket::Final {
536                        packet_tag: incoming.packet_tag,
537                        previous_hop: incoming.previous_hop,
538                        plain_text: incoming.plain_text,
539                        ack_key: incoming.ack_key.expect("should have contained an ack key"),
540                    }))
541                }
542            }
543            HoprPacket::Forwarded(fwd) => {
544                match self
545                    .validate_and_replace_ticket(*fwd, &self.chain_key, outgoing_ticket_win_prob, outgoing_ticket_price)
546                    .await
547                {
548                    Ok(fwd) => {
549                        let mut payload = Vec::with_capacity(HoprPacket::SIZE);
550                        payload.extend_from_slice(fwd.outgoing.packet.as_ref());
551                        payload.extend_from_slice(&fwd.outgoing.ticket.into_encoded());
552
553                        Ok(Some(IncomingPacket::Forwarded {
554                            packet_tag: fwd.packet_tag,
555                            previous_hop: fwd.previous_hop,
556                            next_hop: fwd.outgoing.next_hop,
557                            data: payload.into_boxed_slice(),
558                            ack: Acknowledgement::new(fwd.ack_key, pkt_keypair),
559                        }))
560                    }
561                    Err(DbSqlError::TicketValidationError(boxed_error)) => {
562                        let (rejected_ticket, error) = *boxed_error;
563                        let rejected_value = rejected_ticket.amount;
564                        warn!(?rejected_ticket, %rejected_value, erorr = ?error, "failure to validate during forwarding");
565
566                        self.mark_unsaved_ticket_rejected(&rejected_ticket).await.map_err(|e| {
567                            DbSqlError::TicketValidationError(Box::new((
568                                rejected_ticket.clone(),
569                                format!("during validation error '{error}' update another error occurred: {e}"),
570                            )))
571                        })?;
572
573                        Err(DbSqlError::TicketValidationError(Box::new((rejected_ticket, error))).into())
574                    }
575                    Err(e) => Err(e.into()),
576                }
577            }
578            HoprPacket::Outgoing(_) => Err(DbSqlError::LogicalError("cannot receive an outgoing packet".into()).into()),
579        }
580    }

583impl HoprDb {
584    async fn create_multihop_ticket(
585        &self,
586        channel: ChannelEntry,
587        me_onchain: Address,
588        destination: Address,
589        current_path_pos: u8,
590        winning_prob: f64,
591        ticket_price: Balance,
592    ) -> crate::errors::Result<TicketBuilder> {
593        // The next ticket is worth: price * remaining hop count / winning probability
594        let amount = Balance::new(
595            ticket_price
596                .amount()
597                .mul(U256::from(current_path_pos - 1))
598                .div_f64(winning_prob)
599                .map_err(|e| {
600                    DbSqlError::LogicalError(format!(
601                        "winning probability outside of the allowed interval (0.0, 1.0]: {e}"
602                    ))
603                })?,
604            BalanceType::HOPR,
605        );
606
607        if channel.balance.lt(&amount) {
608            return Err(DbSqlError::LogicalError(format!(
609                "out of funds: {} with counterparty {destination} has balance {} < {amount}",
610                channel.get_id(),
611                channel.balance
612            )));
613        }
614
615        let ticket_builder = TicketBuilder::default()
616            .direction(&me_onchain, &destination)
617            .balance(amount)
618            .index(self.increment_outgoing_ticket_index(channel.get_id()).await?)
619            .index_offset(1) // unaggregated always have index_offset == 1
620            .win_prob(winning_prob)
621            .channel_epoch(channel.channel_epoch.as_u32());
622
623        Ok(ticket_builder)
624    }
625}
