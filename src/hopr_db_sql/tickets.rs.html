68impl IntoCondition for WrappedTicketSelector {
69    fn into_condition(self) -> Condition {
70        let expr = self
71            .0
72            .channel_identifiers
73            .into_iter()
74            .map(|(channel_id, epoch)| {
75                ticket::Column::ChannelId
76                    .eq(channel_id.to_hex())
77                    .and(ticket::Column::ChannelEpoch.eq(epoch.to_be_bytes().to_vec()))
78            })
79            .reduce(SimpleExpr::or);
80
81        // This cannot happen, but instead of panicking, return an impossible condition object
82        if expr.is_none() {
83            return Condition::any().not();
84        }
85
86        let mut expr = expr.unwrap();
87
88        match self.0.index {
89            TicketIndexSelector::None => {
90                // This will always be the case if there were multiple channel identifiers
91            }
92            TicketIndexSelector::Single(idx) => expr = expr.and(ticket::Column::Index.eq(idx.to_be_bytes().to_vec())),
93            TicketIndexSelector::Multiple(idxs) => {
94                expr = expr.and(ticket::Column::Index.is_in(idxs.into_iter().map(|i| i.to_be_bytes().to_vec())));
95            }
96            TicketIndexSelector::Range((lb, ub)) => {
97                expr = match lb {
98                    Bound::Included(gte) => expr.and(ticket::Column::Index.gte(gte.to_be_bytes().to_vec())),
99                    Bound::Excluded(gt) => expr.and(ticket::Column::Index.gt(gt.to_be_bytes().to_vec())),
100                    Bound::Unbounded => expr,
101                };
102                expr = match ub {
103                    Bound::Included(lte) => expr.and(ticket::Column::Index.lte(lte.to_be_bytes().to_vec())),
104                    Bound::Excluded(lt) => expr.and(ticket::Column::Index.lt(lt.to_be_bytes().to_vec())),
105                    Bound::Unbounded => expr,
106                };
107            }
108        }
109
110        if let Some(state) = self.0.state {
111            expr = expr.and(ticket::Column::State.eq(state as u8))
112        }
113
114        if self.0.only_aggregated {
115            expr = expr.and(ticket::Column::IndexOffset.gt(1));
116        }
117
118        // Win prob lower bound
119        expr = match self.0.win_prob.0 {
120            Bound::Included(gte) => expr.and(ticket::Column::WinningProbability.gte(gte.as_encoded().to_vec())),
121            Bound::Excluded(gt) => expr.and(ticket::Column::WinningProbability.gt(gt.as_encoded().to_vec())),
122            Bound::Unbounded => expr,
123        };
124
125        // Win prob upper bound
126        expr = match self.0.win_prob.1 {
127            Bound::Included(lte) => expr.and(ticket::Column::WinningProbability.lte(lte.as_encoded().to_vec())),
128            Bound::Excluded(lt) => expr.and(ticket::Column::WinningProbability.lt(lt.as_encoded().to_vec())),
129            Bound::Unbounded => expr,
130        };
131
132        // Amount lower bound
133        expr = match self.0.amount.0 {
134            Bound::Included(gte) => expr.and(ticket::Column::Amount.gte(gte.amount().to_be_bytes().to_vec())),
135            Bound::Excluded(gt) => expr.and(ticket::Column::Amount.gt(gt.amount().to_be_bytes().to_vec())),
136            Bound::Unbounded => expr,
137        };
138
139        // Amount upper bound
140        expr = match self.0.amount.1 {
141            Bound::Included(lte) => expr.and(ticket::Column::Amount.lte(lte.amount().to_be_bytes().to_vec())),
142            Bound::Excluded(lt) => expr.and(ticket::Column::Amount.lt(lt.amount().to_be_bytes().to_vec())),
143            Bound::Unbounded => expr,
144        };
145
146        expr.into_condition()
147    }
148}

150/// Filters the list of ticket models according to the prerequisites.
151/// **NOTE:** the input list is assumed to be sorted by ticket index in ascending order.
152///
153/// The following is applied:
154/// - the list of tickets is reduced so that the total amount on the tickets does not exceed the channel balance
155/// - it is checked whether the list size is greater than `min_unaggregated_ratio`
156/// - it is checked whether the ratio of total amount on the unaggregated tickets on the list and the channel balance
157///   ratio is greater than `min_unaggregated_ratio`
158pub(crate) fn filter_satisfying_ticket_models(
159    prerequisites: AggregationPrerequisites,
160    models: Vec<ticket::Model>,
161    channel_entry: &ChannelEntry,
162    min_win_prob: WinningProbability,
163) -> crate::errors::Result<Vec<ticket::Model>> {
164    let channel_id = channel_entry.get_id();
165
166    let mut to_be_aggregated = Vec::with_capacity(models.len());
167    let mut total_balance = HoprBalance::zero();
168
169    for m in models {
170        let ticket_wp: WinningProbability = m
171            .winning_probability
172            .as_slice()
173            .try_into()
174            .map_err(|_| DbSqlError::DecodingError)?;
175
176        if ticket_wp.approx_cmp(&min_win_prob).is_lt() {
177            warn!(
178                channel_id = %channel_entry.get_id(),
179                %ticket_wp, %min_win_prob, "encountered ticket with winning probability lower than the minimum threshold"
180            );
181            continue;
182        }
183
184        let to_add = HoprBalance::from_be_bytes(&m.amount);
185
186        // Do a balance check to be sure not to aggregate more than the current channel stake
187        total_balance += to_add;
188        if total_balance.gt(&channel_entry.balance) {
189            // Remove the last sub-balance which led to the overflow before breaking out of the loop.
190            total_balance -= to_add;
191            break;
192        }
193
194        to_be_aggregated.push(m);
195    }
196
197    // If there are no criteria, just send everything for aggregation
198    if prerequisites.min_ticket_count.is_none() && prerequisites.min_unaggregated_ratio.is_none() {
199        info!(channel = %channel_id, "Aggregation check OK, no aggregation prerequisites were given");
200        return Ok(to_be_aggregated);
201    }
202
203    let to_be_agg_count = to_be_aggregated.len();
204
205    // Check the aggregation threshold
206    if let Some(agg_threshold) = prerequisites.min_ticket_count {
207        if to_be_agg_count >= agg_threshold {
208            info!(channel = %channel_id, count = to_be_agg_count, threshold = agg_threshold, "Aggregation check OK aggregated value greater than threshold");
209            return Ok(to_be_aggregated);
210        } else {
211            debug!(channel = %channel_id, count = to_be_agg_count, threshold = agg_threshold,"Aggregation check FAIL not enough resources to aggregate");
212        }
213    }
214
215    if let Some(unrealized_threshold) = prerequisites.min_unaggregated_ratio {
216        let diminished_balance = channel_entry.balance.mul_f64(unrealized_threshold)?;
217
218        // Trigger aggregation if unrealized balance greater or equal to X percentage of the current balance
219        // and there are at least two tickets
220        if total_balance.ge(&diminished_balance) {
221            if to_be_agg_count > 1 {
222                info!(channel = %channel_id, count = to_be_agg_count, balance = ?total_balance, ?diminished_balance, "Aggregation check OK: more unrealized than diminished balance");
223                return Ok(to_be_aggregated);
224            } else {
225                debug!(channel = %channel_id, count = to_be_agg_count, balance = ?total_balance, ?diminished_balance, "Aggregation check FAIL: more unrealized than diminished balance but only 1 ticket");
226            }
227        } else {
228            debug!(channel = %channel_id, count = to_be_agg_count, balance = ?total_balance, ?diminished_balance, "Aggregation check FAIL: less unrealized than diminished balance");
229        }
230    }
231
232    debug!(channel = %channel_id,"Aggregation check FAIL: no prerequisites were met");
233    Ok(vec![])
234}

236pub(crate) async fn find_stats_for_channel(
237    tx: &OpenTransaction,
238    channel_id: &Hash,
239) -> crate::errors::Result<ticket_statistics::Model> {
240    if let Some(model) = ticket_statistics::Entity::find()
241        .filter(ticket_statistics::Column::ChannelId.eq(channel_id.to_hex()))
242        .one(tx.as_ref())
243        .await?
244    {
245        Ok(model)
246    } else {
247        let new_stats = ticket_statistics::ActiveModel {
248            channel_id: Set(channel_id.to_hex()),
249            ..Default::default()
250        }
251        .insert(tx.as_ref())
252        .await?;
253
254        Ok(new_stats)
255    }

328    async fn mark_tickets_as(&self, selector: TicketSelector, mark_as: TicketMarker) -> Result<usize> {
329        let myself = self.clone();
330        Ok(self
331            .ticket_manager
332            .with_write_locked_db(|tx| {
333                Box::pin(async move {
334                    let mut total_marked_count = 0;
335                    for (channel_id, epoch) in selector.channel_identifiers.iter() {
336                        let channel_selector = selector.clone().just_on_channel(*channel_id, epoch);
337
338                        // Get the number of tickets and their value just for this channel
339                        let (marked_count, marked_value) =
340                            myself.get_tickets_value_int(Some(tx), channel_selector.clone()).await?;
341                        trace!(marked_count, ?marked_value, ?mark_as, "ticket marking");
342
343                        if marked_count > 0 {
344                            // Delete the redeemed tickets first
345                            let deleted = ticket::Entity::delete_many()
346                                .filter(WrappedTicketSelector::from(channel_selector.clone()))
347                                .exec(tx.as_ref())
348                                .await?;
349
350                            // Update the stats if successful
351                            if deleted.rows_affected == marked_count as u64 {
352                                let mut new_stats = find_stats_for_channel(tx, channel_id).await?.into_active_model();
353                                let _current_value = match mark_as {
354                                    TicketMarker::Redeemed => {
355                                        let current_value = U256::from_be_bytes(new_stats.redeemed_value.as_ref());
356                                        new_stats.redeemed_value =
357                                            Set((current_value + marked_value.amount()).to_be_bytes().into());
358                                        current_value
359                                    }
360                                    TicketMarker::Rejected => {
361                                        let current_value = U256::from_be_bytes(new_stats.rejected_value.as_ref());
362                                        new_stats.rejected_value =
363                                            Set((current_value + marked_value.amount()).to_be_bytes().into());
364                                        current_value
365                                    }
366                                    TicketMarker::Neglected => {
367                                        let current_value = U256::from_be_bytes(new_stats.neglected_value.as_ref());
368                                        new_stats.neglected_value =
369                                            Set((current_value + marked_value.amount()).to_be_bytes().into());
370                                        current_value
371                                    }
372                                };
373                                new_stats.save(tx.as_ref()).await?;
374
375                                #[cfg(all(feature = "prometheus", not(test)))]
376                                {
377                                    METRIC_HOPR_TICKETS_INCOMING_STATISTICS.set(
378                                        &[&mark_as.to_string()],
379                                        (_current_value + marked_value.amount()).as_u128() as f64,
380                                    );
381
382                                    // Tickets that are counted as rejected were never counted as unredeemed,
383                                    // so skip the metric subtraction in that case.
384                                    if mark_as != TicketMarker::Rejected {
385                                        let unredeemed_value = myself
386                                            .caches
387                                            .unrealized_value
388                                            .get(&(*channel_id, *epoch))
389                                            .await
390                                            .unwrap_or_default();
391
392                                        METRIC_HOPR_TICKETS_INCOMING_STATISTICS.set(
393                                            &["unredeemed"],
394                                            (unredeemed_value - marked_value.amount()).amount().as_u128() as f64,
395                                        );
396                                    }
397                                }
398
399                                myself.caches.unrealized_value.invalidate(&(*channel_id, *epoch)).await;
400                            } else {
401                                return Err(DbSqlError::LogicalError(format!(
402                                    "could not mark {marked_count} ticket as {mark_as}"
403                                )));
404                            }
405
406                            trace!(marked_count, ?channel_id, ?mark_as, "removed tickets in channel");
407                            total_marked_count += marked_count;
408                        }
409                    }
410
411                    info!(
412                        count = total_marked_count,
413                        ?mark_as,
414                        channel_count = selector.channel_identifiers.len(),
415                        "removed tickets in channels",
416                    );
417                    Ok(total_marked_count)
418                })
419            })
420            .await?)
421    }

456    async fn update_ticket_states_and_fetch<'a>(
457        &'a self,
458        selector: TicketSelector,
459        new_state: AcknowledgedTicketStatus,
460    ) -> Result<BoxStream<'a, AcknowledgedTicket>> {
461        let selector: WrappedTicketSelector = selector.into();
462        Ok(Box::pin(stream! {
463            match ticket::Entity::find()
464                .filter(selector)
465                .stream(self.conn(TargetDb::Tickets))
466                .await {
467                Ok(mut stream) => {
468                    while let Ok(Some(ticket)) = stream.try_next().await {
469                        let active_ticket = ticket::ActiveModel {
470                            id: Set(ticket.id),
471                            state: Set(new_state as i8),
472                            ..Default::default()
473                        };
474
475                        {
476                            let _g = self.ticket_manager.mutex.lock();
477                            if let Err(e) = active_ticket.update(self.conn(TargetDb::Tickets)).await {
478                                error!(error = %e,"failed to update ticket in the db");
479                            }
480                        }
481
482                        match AcknowledgedTicket::try_from(ticket) {
483                            Ok(mut ticket) => {
484                                // Update the state manually, since we do not want to re-fetch the model after the update
485                                ticket.status = new_state;
486                                yield ticket
487                            },
488                            Err(e) => {
489                                tracing::error!(error = %e, "failed to decode ticket from the db");
490                            }
491                        }
492                    }
493                },
494                Err(e) => tracing::error!(error = %e, "failed open ticket db stream")
495            }
496        }))
497    }

520    async fn get_ticket_statistics(&self, channel_id: Option<Hash>) -> Result<ChannelTicketStatistics> {
521        let res = match channel_id {
522            None => {
523                self.nest_transaction_in_db(None, TargetDb::Tickets)
524                    .await?
525                    .perform(|tx| {
526                        Box::pin(async move {
527                            let unredeemed_value = ticket::Entity::find()
528                                .stream(tx.as_ref())
529                                .await?
530                                .try_fold(U256::zero(), |amount, x| {
531                                    let unredeemed_value = U256::from_be_bytes(x.amount);
532                                    futures::future::ok(amount + unredeemed_value)
533                                })
534                                .await?;
535
536                            #[cfg(all(feature = "prometheus", not(test)))]
537                            METRIC_HOPR_TICKETS_INCOMING_STATISTICS
538                                .set(&["unredeemed"], unredeemed_value.as_u128() as f64);
539
540                            let mut all_stats = ticket_statistics::Entity::find()
541                                .all(tx.as_ref())
542                                .await?
543                                .into_iter()
544                                .fold(ChannelTicketStatistics::default(), |mut acc, stats| {
545                                    let neglected_value = HoprBalance::from_be_bytes(stats.neglected_value);
546                                    acc.neglected_value += neglected_value;
547                                    let redeemed_value = HoprBalance::from_be_bytes(stats.redeemed_value);
548                                    acc.redeemed_value += redeemed_value;
549                                    let rejected_value = HoprBalance::from_be_bytes(stats.rejected_value);
550                                    acc.rejected_value += rejected_value;
551                                    acc.winning_tickets += stats.winning_tickets as u128;
552                                    acc
553                                });
554
555                            all_stats.unredeemed_value = unredeemed_value.into();
556
557                            #[cfg(all(feature = "prometheus", not(test)))]
558                            {
559                                METRIC_HOPR_TICKETS_INCOMING_STATISTICS
560                                    .set(&["neglected"], all_stats.neglected_value.amount().as_u128() as f64);
561                                METRIC_HOPR_TICKETS_INCOMING_STATISTICS
562                                    .set(&["redeemed"], all_stats.redeemed_value.amount().as_u128() as f64);
563                                METRIC_HOPR_TICKETS_INCOMING_STATISTICS
564                                    .set(&["rejected"], all_stats.rejected_value.amount().as_u128() as f64);
565                                METRIC_HOPR_TICKETS_INCOMING_STATISTICS
566                                    .set(&["winning_tickets"], all_stats.winning_tickets as f64);
567                            }
568
569                            Ok::<_, DbSqlError>(all_stats)
570                        })
571                    })
572                    .await
573            }
574            Some(channel) => {
575                // We need to make sure the channel exists to avoid creating
576                // statistic entry for a non-existing channel
577                if self.get_channel_by_id(None, &channel).await?.is_none() {
578                    return Err(DbSqlError::ChannelNotFound(channel).into());
579                }
580
581                self.nest_transaction_in_db(None, TargetDb::Tickets)
582                    .await?
583                    .perform(|tx| {
584                        Box::pin(async move {
585                            let stats = find_stats_for_channel(tx, &channel).await?;
586                            let unredeemed_value = ticket::Entity::find()
587                                .filter(ticket::Column::ChannelId.eq(channel.to_hex()))
588                                .stream(tx.as_ref())
589                                .await?
590                                .try_fold(U256::zero(), |amount, x| {
591                                    futures::future::ok(amount + U256::from_be_bytes(x.amount))
592                                })
593                                .await?;
594
595                            Ok::<_, DbSqlError>(ChannelTicketStatistics {
596                                winning_tickets: stats.winning_tickets as u128,
597                                neglected_value: HoprBalance::from_be_bytes(stats.neglected_value),
598                                redeemed_value: HoprBalance::from_be_bytes(stats.redeemed_value),
599                                unredeemed_value: unredeemed_value.into(),
600                                rejected_value: HoprBalance::from_be_bytes(stats.rejected_value),
601                            })

713    async fn persist_outgoing_ticket_indices(&self) -> Result<usize> {
714        let outgoing_indices = outgoing_ticket_index::Entity::find()
715            .all(&self.tickets_db)
716            .await
717            .map_err(DbSqlError::from)?;
718
719        let mut updated = 0;
720        for index_model in outgoing_indices {
721            let channel_id = Hash::from_hex(&index_model.channel_id).map_err(DbSqlError::from)?;
722            let db_index = U256::from_be_bytes(&index_model.index).as_u64();
723            if let Some(cached_index) = self.caches.ticket_index.get(&channel_id).await {
724                // Note that the persisted value is always lagging behind the cache,
725                // so the fact that the cached index can change between this load
726                // storing it in the DB is allowed.
727                let cached_index = cached_index.load(Ordering::SeqCst);
728
729                // Store the ticket index in a separate write transaction
730                if cached_index > db_index {
731                    let mut index_active_model = index_model.into_active_model();
732                    index_active_model.index = Set(cached_index.to_be_bytes().to_vec());
733                    self.ticket_manager
734                        .with_write_locked_db(|wtx| {
735                            Box::pin(async move {
736                                index_active_model.save(wtx.as_ref()).await?;
737                                Ok::<_, DbSqlError>(())
738                            })
739                        })
740                        .await?;
741
742                    debug!("updated ticket index in channel {channel_id} from {db_index} to {cached_index}");
743                    updated += 1;
744                }
745            } else {
746                // The value is not yet in the cache, meaning there's low traffic on this
747                // channel, so the value has not been yet fetched.
748                trace!(?channel_id, "channel not in cache yet");
749            }
750        }
751
752        Ok(Ok::<_, DbSqlError>(updated)?)
753    }

755    async fn prepare_aggregation_in_channel(
756        &self,
757        channel: &Hash,
758        prerequisites: AggregationPrerequisites,
759    ) -> Result<Option<(OffchainPublicKey, Vec<TransferableWinningTicket>, Hash)>> {
760        let myself = self.clone();
761
762        let channel_id = *channel;
763
764        let (channel_entry, peer, domain_separator, min_win_prob) = self
765            .nest_transaction_in_db(None, TargetDb::Index)
766            .await?
767            .perform(|tx| {
768                Box::pin(async move {
769                    let entry = myself
770                        .get_channel_by_id(Some(tx), &channel_id)
771                        .await?
772                        .ok_or(DbSqlError::ChannelNotFound(channel_id))?;
773
774                    if entry.status == ChannelStatus::Closed {
775                        return Err(DbSqlError::LogicalError(format!("channel '{channel_id}' is closed")));
776                    } else if entry.direction(&myself.me_onchain) != Some(ChannelDirection::Incoming) {
777                        return Err(DbSqlError::LogicalError(format!(
778                            "channel '{channel_id}' is not incoming"
779                        )));
780                    }
781
782                    let pk = myself
783                        .resolve_packet_key(&entry.source)
784                        .await?
785                        .ok_or(DbSqlError::LogicalError(format!(
786                            "peer '{}' has no offchain key record",
787                            entry.source
788                        )))?;
789
790                    let indexer_data = myself.get_indexer_data(Some(tx)).await?;
791
792                    let domain_separator = indexer_data
793                        .channels_dst
794                        .ok_or_else(|| crate::errors::DbSqlError::LogicalError("domain separator missing".into()))?;
795
796                    Ok((
797                        entry,
798                        pk,
799                        domain_separator,
800                        indexer_data.minimum_incoming_ticket_winning_prob,
801                    ))
802                })
803            })
804            .await?;
805
806        let myself = self.clone();
807        let tickets = self
808            .ticket_manager
809            .with_write_locked_db(|tx| {
810                Box::pin(async move {
811                    // verify that no aggregation is in progress in the channel
812                    if ticket::Entity::find()
813                        .filter(WrappedTicketSelector::from(
814                            TicketSelector::from(&channel_entry).with_state(AcknowledgedTicketStatus::BeingAggregated),
815                        ))
816                        .one(tx.as_ref())
817                        .await?
818                        .is_some()
819                    {
820                        return Err(DbSqlError::LogicalError(format!(
821                            "'{channel_entry}' is already being aggregated",
822                        )));
823                    }
824
825                    // find the index of the last ticket being redeemed
826                    let first_idx_to_take = ticket::Entity::find()
827                        .filter(WrappedTicketSelector::from(
828                            TicketSelector::from(&channel_entry).with_state(AcknowledgedTicketStatus::BeingRedeemed),
829                        ))
830                        .order_by_desc(ticket::Column::Index)
831                        .one(tx.as_ref())
832                        .await?
833                        .map(|m| U256::from_be_bytes(m.index).as_u64() + 1)
834                        .unwrap_or(0_u64) // go from the lowest possible index of none is found
835                        .max(channel_entry.ticket_index.as_u64()); // but cannot be less than the ticket index on the Channel entry
836
837                    // get the list of all tickets to be aggregated
838                    let to_be_aggregated = ticket::Entity::find()
839                        .filter(WrappedTicketSelector::from(TicketSelector::from(&channel_entry)))
840                        .filter(ticket::Column::Index.gte(first_idx_to_take.to_be_bytes().to_vec()))
841                        .filter(ticket::Column::State.ne(AcknowledgedTicketStatus::BeingAggregated as u8))
842                        .order_by_asc(ticket::Column::Index)// tickets must be sorted by indices in ascending order
843                        .limit(MAX_TICKETS_TO_AGGREGATE_BATCH)
844                        .all(tx.as_ref())
845                        .await?;
846
847                    // Filter the list of tickets according to the prerequisites
848                    let mut to_be_aggregated: Vec<TransferableWinningTicket> =
849                        filter_satisfying_ticket_models(prerequisites, to_be_aggregated, &channel_entry, min_win_prob)?
850                            .into_iter()
851                            .map(|model| {
852                                AcknowledgedTicket::try_from(model)
853                                    .map_err(DbSqlError::from)
854                                    .and_then(|ack| {
855                                        ack.into_transferable(&myself.chain_key, &domain_separator)
856                                            .map_err(DbSqlError::from)
857                                    })
858                            })
859                            .collect::<crate::errors::Result<Vec<_>>>()?;
860
861                    let mut neglected_idxs = Vec::new();
862
863                    if !to_be_aggregated.is_empty() {
864                        // Clean up any tickets in this channel that are already inside an aggregated ticket.
865                        // This situation cannot be avoided 100% as aggregation can be triggered when out-of-order
866                        // tickets arrive and only some of them are necessary to satisfy the aggregation threshold.
867                        // The following code *assumes* that only the first ticket with the lowest index *can be* an aggregate.
868                        let first_ticket = to_be_aggregated[0].ticket.clone();
869                        let mut i = 1;
870                        while i < to_be_aggregated.len() {
871                            let current_idx = to_be_aggregated[i].ticket.index;
872                            if (first_ticket.index..first_ticket.index + first_ticket.index_offset as u64).contains(&current_idx) {
873                                // Cleanup is the only reasonable thing to do at this point,
874                                // since the aggregator will check for index range overlaps and deny
875                                // the aggregation of the entire batch otherwise.
876                                warn!(ticket_id = current_idx, channel = %channel_id, ?first_ticket, "ticket in channel has been already aggregated and will be removed");
877                                neglected_idxs.push(current_idx);
878                                to_be_aggregated.remove(i);
879                            } else {
880                                i += 1;
881                            }
882                        }
883
884                        // The cleanup (neglecting of tickets) is not made directly here but on the next ticket redemption in this channel
885                        // See handler.rs around L402
886                        if !neglected_idxs.is_empty() {
887                            warn!(count = neglected_idxs.len(), channel = %channel_id, "tickets were neglected due to duplication in an aggregated ticket!");
888                        }
889
890                        // mark all tickets with appropriate characteristics as being aggregated
891                        let marked: sea_orm::UpdateResult = ticket::Entity::update_many()
892                            .filter(WrappedTicketSelector::from(TicketSelector::from(&channel_entry)))
893                            .filter(ticket::Column::Index.is_in(to_be_aggregated.iter().map(|t| t.ticket.index.to_be_bytes().to_vec())))
894                            .filter(ticket::Column::State.ne(AcknowledgedTicketStatus::BeingAggregated as u8))
895                            .col_expr(
896                                ticket::Column::State,
897                                Expr::value(AcknowledgedTicketStatus::BeingAggregated as i8),
898                            )
899                            .exec(tx.as_ref())
900                            .await?;
901
902                        if marked.rows_affected as usize != to_be_aggregated.len() {
903                            return Err(DbSqlError::LogicalError(format!(
904                                "expected to mark {}, but was able to mark {}",
905                                to_be_aggregated.len(),
906                                marked.rows_affected,
907                            )));
908                        }
909                    }
910
911                    debug!(
912                        "prepared {} tickets to aggregate in {} ({})",
913                        to_be_aggregated.len(),
914                        channel_entry.get_id(),
915                        channel_entry.channel_epoch,
916                    );
917
918                    Ok(to_be_aggregated)
919                })
920            })
921            .await?;
922
923        Ok((!tickets.is_empty()).then_some((peer, tickets, domain_separator)))
924    }

944    async fn process_received_aggregated_ticket(
945        &self,
946        aggregated_ticket: Ticket,
947        chain_keypair: &ChainKeypair,
948    ) -> Result<AcknowledgedTicket> {
949        if chain_keypair.public().to_address() != self.me_onchain {
950            return Err(DbSqlError::LogicalError(
951                "chain key for ticket aggregation does not match the DB public address".into(),
952            )
953            .into());
954        }
955
956        let myself = self.clone();
957        let channel_id = aggregated_ticket.channel_id;
958
959        let (channel_entry, domain_separator) = self
960            .nest_transaction_in_db(None, TargetDb::Index)
961            .await?
962            .perform(|tx| {
963                Box::pin(async move {
964                    let entry = myself
965                        .get_channel_by_id(Some(tx), &channel_id)
966                        .await?
967                        .ok_or(DbSqlError::ChannelNotFound(channel_id))?;
968
969                    if entry.status == ChannelStatus::Closed {
970                        return Err(DbSqlError::LogicalError(format!("channel '{channel_id}' is closed")));
971                    } else if entry.direction(&myself.me_onchain) != Some(ChannelDirection::Incoming) {
972                        return Err(DbSqlError::LogicalError(format!(
973                            "channel '{channel_id}' is not incoming"
974                        )));
975                    }
976
977                    let domain_separator =
978                        myself.get_indexer_data(Some(tx)).await?.channels_dst.ok_or_else(|| {
979                            crate::errors::DbSqlError::LogicalError("domain separator missing".into())
980                        })?;
981
982                    Ok((entry, domain_separator))
983                })
984            })
985            .await?;
986
987        // Verify the ticket first
988        let aggregated_ticket = aggregated_ticket
989            .verify(&channel_entry.source, &domain_separator)
990            .map_err(|e| {
991                DbSqlError::LogicalError(format!(
992                    "failed to verify received aggregated ticket in {channel_id}: {e}"
993                ))
994            })?;
995
996        // Aggregated tickets always have 100% winning probability
997        if !aggregated_ticket.win_prob().approx_eq(&WinningProbability::ALWAYS) {
998            return Err(DbSqlError::LogicalError("Aggregated tickets must have 100% win probability".into()).into());
999        }
1000
1001        let acknowledged_tickets = self
1002            .nest_transaction_in_db(None, TargetDb::Tickets)
1003            .await?
1004            .perform(|tx| {
1005                Box::pin(async move {
1006                    ticket::Entity::find()
1007                        .filter(WrappedTicketSelector::from(
1008                            TicketSelector::from(&channel_entry).with_state(AcknowledgedTicketStatus::BeingAggregated),
1009                        ))
1010                        .all(tx.as_ref())
1011                        .await
1012                        .map_err(DbSqlError::BackendError)
1013                })
1014            })
1015            .await?;
1016
1017        if acknowledged_tickets.is_empty() {
1018            debug!("Received unexpected aggregated ticket in channel {channel_id}");
1019            return Err(DbSqlError::LogicalError(format!(
1020                "failed insert aggregated ticket, because no tickets seem to be aggregated for '{channel_id}'",
1021            ))
1022            .into());
1023        }
1024
1025        let stored_value = acknowledged_tickets
1026            .iter()
1027            .map(|m| HoprBalance::from_be_bytes(&m.amount))
1028            .sum();
1029
1030        // The value of a received ticket can be higher (profit for us) but not lower
1031        if aggregated_ticket.verified_ticket().amount.lt(&stored_value) {
1032            error!(channel = %channel_id, "Aggregated ticket value in channel is lower than sum of stored tickets");
1033            return Err(DbSqlError::LogicalError("Value of received aggregated ticket is too low".into()).into());
1034        }
1035
1036        let acknowledged_tickets = acknowledged_tickets
1037            .into_iter()
1038            .map(AcknowledgedTicket::try_from)
1039            .collect::<hopr_db_entity::errors::Result<Vec<AcknowledgedTicket>>>()
1040            .map_err(DbSqlError::from)?;
1041
1042        // can be done, because the tickets collection is tested for emptiness before
1043        let first_stored_ticket = acknowledged_tickets.first().unwrap();
1044
1045        // calculate the new current ticket index
1046        #[allow(unused_variables)]
1047        let current_ticket_index_from_aggregated_ticket =
1048            U256::from(aggregated_ticket.verified_ticket().index).add(aggregated_ticket.verified_ticket().index_offset);
1049
1050        let acked_aggregated_ticket = aggregated_ticket.into_acknowledged(first_stored_ticket.response.clone());
1051
1052        let ticket = acked_aggregated_ticket.clone();
1053        self.ticket_manager.replace_tickets(ticket).await?;
1054
1055        info!(%acked_aggregated_ticket, "successfully processed received aggregated ticket");
1056        Ok(acked_aggregated_ticket)
1057    }

1059    async fn aggregate_tickets(
1060        &self,
1061        destination: OffchainPublicKey,
1062        mut acked_tickets: Vec<TransferableWinningTicket>,
1063        me: &ChainKeypair,
1064    ) -> Result<VerifiedTicket> {
1065        if me.public().to_address() != self.me_onchain {
1066            return Err(DbSqlError::LogicalError(
1067                "chain key for ticket aggregation does not match the DB public address".into(),
1068            )
1069            .into());
1070        }
1071
1072        let domain_separator = self
1073            .get_indexer_data(None)
1074            .await?
1075            .domain_separator(DomainSeparator::Channel)
1076            .ok_or_else(|| DbSqlError::LogicalError("domain separator missing".into()))?;
1077
1078        if acked_tickets.is_empty() {
1079            return Err(DbSqlError::LogicalError("at least one ticket required for aggregation".to_owned()).into());
1080        }
1081
1082        if acked_tickets.len() == 1 {
1083            let single = acked_tickets
1084                .pop()
1085                .unwrap()
1086                .into_redeemable(&self.me_onchain, &domain_separator)
1087                .map_err(DbSqlError::from)?;
1088
1089            self.compare_and_set_outgoing_ticket_index(
1090                single.verified_ticket().channel_id,
1091                single.verified_ticket().index + 1,
1092            )
1093            .await?;
1094
1095            return Ok(single.ticket);
1096        }
1097
1098        acked_tickets.sort_by(|a, b| a.partial_cmp(b).unwrap_or(cmp::Ordering::Equal));
1099        acked_tickets.dedup();
1100
1101        let myself = self.clone();
1102        let address = myself
1103            .resolve_chain_key(&destination)
1104            .await?
1105            .ok_or(DbSqlError::LogicalError(format!(
1106                "peer '{}' has no chain key record",
1107                destination.to_peerid_str()
1108            )))?;
1109
1110        let (channel_entry, destination, min_win_prob) = self
1111            .nest_transaction_in_db(None, TargetDb::Index)
1112            .await?
1113            .perform(|tx| {
1114                Box::pin(async move {
1115                    let entry = myself
1116                        .get_channel_by_parties(Some(tx), &myself.me_onchain, &address, false)
1117                        .await?
1118                        .ok_or_else(|| {
1119                            DbSqlError::ChannelNotFound(generate_channel_id(&myself.me_onchain, &address))
1120                        })?;
1121
1122                    if entry.status == ChannelStatus::Closed {
1123                        return Err(DbSqlError::LogicalError(format!("{entry} is closed")));
1124                    } else if entry.direction(&myself.me_onchain) != Some(ChannelDirection::Outgoing) {
1125                        return Err(DbSqlError::LogicalError(format!("{entry} is not outgoing")));
1126                    }
1127
1128                    let min_win_prob = myself
1129                        .get_indexer_data(Some(tx))
1130                        .await?
1131                        .minimum_incoming_ticket_winning_prob;
1132                    Ok((entry, address, min_win_prob))
1133                })
1134            })
1135            .await?;
1136
1137        let channel_balance = channel_entry.balance;
1138        let channel_epoch = channel_entry.channel_epoch.as_u32();
1139        let channel_id = channel_entry.get_id();
1140
1141        let mut final_value = HoprBalance::zero();
1142
1143        // Validate all received tickets and turn them into RedeemableTickets
1144        let verified_tickets = acked_tickets
1145            .into_iter()
1146            .map(|t| t.into_redeemable(&self.me_onchain, &domain_separator))
1147            .collect::<hopr_internal_types::errors::Result<Vec<_>>>()
1148            .map_err(|e| {
1149                DbSqlError::LogicalError(format!("trying to aggregate an invalid or a non-winning ticket: {e}"))
1150            })?;
1151
1152        // Perform additional consistency check on the verified tickets
1153        for (i, acked_ticket) in verified_tickets.iter().enumerate() {
1154            if channel_id != acked_ticket.verified_ticket().channel_id {
1155                return Err(DbSqlError::LogicalError(format!(
1156                    "ticket for aggregation has an invalid channel id {}",
1157                    acked_ticket.verified_ticket().channel_id
1158                ))
1159                .into());
1160            }
1161
1162            if acked_ticket.verified_ticket().channel_epoch != channel_epoch {
1163                return Err(DbSqlError::LogicalError("channel epochs do not match".into()).into());
1164            }
1165
1166            if i + 1 < verified_tickets.len()
1167                && acked_ticket.verified_ticket().index + acked_ticket.verified_ticket().index_offset as u64
1168                    > verified_tickets[i + 1].verified_ticket().index
1169            {
1170                return Err(DbSqlError::LogicalError("tickets with overlapping index intervals".into()).into());
1171            }
1172
1173            if acked_ticket
1174                .verified_ticket()
1175                .win_prob()
1176                .approx_cmp(&min_win_prob)
1177                .is_lt()
1178            {
1179                return Err(DbSqlError::LogicalError(
1180                    "cannot aggregate ticket with lower than minimum winning probability in network".into(),
1181                )
1182                .into());
1183            }
1184
1185            final_value += acked_ticket.verified_ticket().amount;
1186            if final_value.gt(&channel_balance) {
1187                return Err(DbSqlError::LogicalError(format!(
1188                    "ticket amount to aggregate {final_value} is greater than the balance {channel_balance} of \
1189                     channel {channel_id}"
1190                ))
1191                .into());
1192            }
1193        }
1194
1195        info!(
1196            "aggregated {} tickets in channel {channel_id} with total value {final_value}",
1197            verified_tickets.len()
1198        );
1199
1200        let first_acked_ticket = verified_tickets.first().unwrap();
1201        let last_acked_ticket = verified_tickets.last().unwrap();
1202
1203        // calculate the minimum current ticket index as the larger value from the acked ticket index and on-chain
1204        // ticket_index from channel_entry

1207            .await?;
1208
1209        Ok(TicketBuilder::default()
1210            .direction(&self.me_onchain, &destination)
1211            .balance(final_value)
1212            .index(first_acked_ticket.verified_ticket().index)
1213            .index_offset(
1214                (last_acked_ticket.verified_ticket().index - first_acked_ticket.verified_ticket().index + 1) as u32,
1215            )
1216            .win_prob(WinningProbability::ALWAYS) // Aggregated tickets have always 100% winning probability
1217            .channel_epoch(channel_epoch)
1218            .eth_challenge(first_acked_ticket.verified_ticket().challenge)
1219            .build_signed(me, &domain_separator)
1220            .map_err(DbSqlError::from)?)
1221    }

1228                .with_state(AcknowledgedTicketStatus::BeingRedeemed)
