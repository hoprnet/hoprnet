46/// Main database handle for HOPR node operations.
47///
48/// Manages multiple SQLite databases for different data domains to avoid
49/// locking conflicts and improve performance:
50///
51/// - **Index DB**: Blockchain indexing and contract data
52/// - **Tickets DB**: Payment tickets and acknowledgments
53/// - **Peers DB**: Network peer information and metadata
54/// - **Logs DB**: Blockchain logs and processing status
55///
56/// Supports database snapshot imports for fast synchronization via
57/// [`HoprDbGeneralModelOperations::import_logs_db`].
58#[derive(Debug, Clone)]
59pub struct HoprDb {
60    pub(crate) index_db: sea_orm::DatabaseConnection,
61    pub(crate) tickets_db: sea_orm::DatabaseConnection,
62    pub(crate) peers_db: sea_orm::DatabaseConnection,
63    pub(crate) logs_db: sea_orm::DatabaseConnection,
64
65    pub(crate) caches: Arc<HoprDbCaches>,
66    pub(crate) chain_key: ChainKeypair,
67    pub(crate) me_onchain: Address,
68    pub(crate) ticket_manager: Arc<TicketManager>,
69}

71/// Filename for the blockchain indexing database.

73/// Filename for the network peers database.
74pub const SQL_DB_PEERS_FILE_NAME: &str = "hopr_peers.db";
75/// Filename for the payment tickets database.
76pub const SQL_DB_TICKETS_FILE_NAME: &str = "hopr_tickets.db";
77/// Filename for the blockchain logs database (used in snapshots).

81    pub async fn new(directory: &Path, chain_key: ChainKeypair, cfg: HoprDbConfig) -> Result<Self> {
82        #[cfg(all(feature = "prometheus", not(test)))]
83        {
84            lazy_static::initialize(&crate::protocol::METRIC_RECEIVED_ACKS);
85            lazy_static::initialize(&crate::protocol::METRIC_SENT_ACKS);
86            lazy_static::initialize(&crate::protocol::METRIC_TICKETS_COUNT);
87        }
88
89        fs::create_dir_all(directory)
90            .map_err(|_e| DbSqlError::Construction(format!("cannot create main database directory {directory:?}")))?;
91
92        let index = Self::create_pool(
93            cfg.clone(),
94            directory.to_path_buf(),
95            PoolOptions::new(),
96            Some(0),
97            Some(1),
98            SQL_DB_INDEX_FILE_NAME,
99        )
100        .await?;
101
102        let peers_options = PoolOptions::new()
103            .acquire_timeout(Duration::from_secs(60)) // Default is 30
104            .idle_timeout(Some(Duration::from_secs(10 * 60))) // This is the default
105            .max_lifetime(Some(Duration::from_secs(30 * 60))); // This is the default
106
107        let peers = Self::create_pool(
108            cfg.clone(),
109            directory.to_path_buf(),
110            peers_options,
111            Some(0),
112            Some(300),
113            SQL_DB_PEERS_FILE_NAME,
114        )
115        .await?;
116
117        let tickets = Self::create_pool(
118            cfg.clone(),
119            directory.to_path_buf(),
120            PoolOptions::new(),
121            Some(0),
122            Some(50),
123            SQL_DB_TICKETS_FILE_NAME,
124        )
125        .await?;
126
127        let logs = Self::create_pool(
128            cfg.clone(),
129            directory.to_path_buf(),
130            PoolOptions::new(),
131            Some(0),
132            None,
133            SQL_DB_LOGS_FILE_NAME,
134        )
135        .await?;
136
137        Self::new_sqlx_sqlite(chain_key, index, peers, tickets, logs).await
138    }

159    async fn new_sqlx_sqlite(
160        chain_key: ChainKeypair,
161        index_db_pool: SqlitePool,
162        peers_db_pool: SqlitePool,
163        tickets_db_pool: SqlitePool,
164        logs_db_pool: SqlitePool,
165    ) -> Result<Self> {
166        let index_db = SqlxSqliteConnector::from_sqlx_sqlite_pool(index_db_pool.clone());
167
168        MigratorIndex::up(&index_db, None)
169            .await
170            .map_err(|e| DbSqlError::Construction(format!("cannot apply database migration: {e}")))?;
171
172        let tickets_db = SqlxSqliteConnector::from_sqlx_sqlite_pool(tickets_db_pool.clone());
173
174        MigratorTickets::up(&tickets_db, None)
175            .await
176            .map_err(|e| DbSqlError::Construction(format!("cannot apply database migration: {e}")))?;
177
178        let peers_db = SqlxSqliteConnector::from_sqlx_sqlite_pool(peers_db_pool.clone());
179
180        MigratorPeers::up(&peers_db, None)
181            .await
182            .map_err(|e| DbSqlError::Construction(format!("cannot apply database migration: {e}")))?;
183
184        let logs_db = SqlxSqliteConnector::from_sqlx_sqlite_pool(logs_db_pool.clone());
185
186        MigratorChainLogs::up(&logs_db, None)
187            .await
188            .map_err(|e| DbSqlError::Construction(format!("cannot apply database migration: {e}")))?;
189
190        // Reset the peer network information
191        let res = hopr_db_entity::network_peer::Entity::delete_many()
192            .filter(
193                sea_orm::Condition::all().add(
194                    hopr_db_entity::network_peer::Column::LastSeen.lt(chrono::DateTime::<chrono::Utc>::from(
195                        hopr_platform::time::native::current_time()
196                            .checked_sub(std::time::Duration::from_secs(
197                                std::env::var("HOPR_INTERNAL_DB_PEERS_PERSISTENCE_AFTER_RESTART_IN_SECONDS")
198                                    .unwrap_or_else(|_| {
199                                        HOPR_INTERNAL_DB_PEERS_PERSISTENCE_AFTER_RESTART_IN_SECONDS.to_string()
200                                    })
201                                    .parse::<u64>()
202                                    .unwrap_or(HOPR_INTERNAL_DB_PEERS_PERSISTENCE_AFTER_RESTART_IN_SECONDS),
203                            ))
204                            .unwrap_or_else(hopr_platform::time::native::current_time),
205                    )),
206                ),
207            )
208            .exec(&peers_db)
209            .await
210            .map_err(|e| DbSqlError::Construction(format!("must reset peers on init: {e}")))?;
211        debug!(rows = res.rows_affected, "Cleaned up rows from the 'peers' table");
212
213        // Reset all BeingAggregated ticket states to Untouched
214        ticket::Entity::update_many()
215            .filter(ticket::Column::State.eq(AcknowledgedTicketStatus::BeingAggregated as u8))
216            .col_expr(
217                ticket::Column::State,
218                Expr::value(AcknowledgedTicketStatus::Untouched as u8),
219            )
220            .exec(&tickets_db)
221            .await?;
222
223        let caches = Arc::new(HoprDbCaches::default());
224        caches.invalidate_all();
225
226        // Initialize KeyId mapping for accounts
227        Account::find()
228            .find_with_related(Announcement)
229            .all(&index_db)
230            .await?
231            .into_iter()
232            .try_for_each(|(a, b)| match model_to_account_entry(a, b) {
233                Ok(account) => caches.key_id_mapper.update_key_id_binding(&account),
234                Err(error) => {
235                    // Undecodeable accounts are skipped and will be unreachable
236                    tracing::error!(%error, "undecodeable account");
237                    Ok(())
238                }
239            })?;
240
241        Ok(Self {
242            me_onchain: chain_key.public().to_address(),
243            chain_key,
244            ticket_manager: Arc::new(TicketManager::new(tickets_db.clone(), caches.clone())),
245            caches,
246
247            index_db,
248            tickets_db,
249            peers_db,
250            logs_db,
251        })
252    }

254    /// Starts ticket processing by the [TicketManager] with an optional new ticket notifier.
255    /// Without calling this method, tickets will not be persisted into the DB.
256    ///
257    /// If the notifier is given, it will receive notifications once new ticket has been
258    /// persisted into the Tickets DB.
259    pub fn start_ticket_processing(&self, ticket_notifier: Option<UnboundedSender<AcknowledgedTicket>>) -> Result<()> {
260        if let Some(notifier) = ticket_notifier {
261            self.ticket_manager.start_ticket_processing(notifier)
262        } else {
263            self.ticket_manager.start_ticket_processing(futures::sink::drain())
264        }
265    }

267    /// Default SQLite config values for all DBs.
268    fn common_connection_cfg(cfg: HoprDbConfig) -> SqliteConnectOptions {
269        SqliteConnectOptions::default()
270            .create_if_missing(cfg.create_if_missing)
271            .log_slow_statements(LevelFilter::Warn, cfg.log_slow_queries)
272            .log_statements(LevelFilter::Debug)
273            .journal_mode(SqliteJournalMode::Wal)
274            .synchronous(SqliteSynchronous::Normal)
275            .auto_vacuum(SqliteAutoVacuum::Full)
276            //.optimize_on_close(true, None) // Removed, because it causes optimization on each connection, due to min_connections being set to 0
277            .page_size(4096)
278            .pragma("cache_size", "-30000") // 32M
279            .pragma("busy_timeout", "1000") // 1000ms
280    }

282    pub async fn create_pool(
283        cfg: HoprDbConfig,
284        directory: PathBuf,
285        mut options: PoolOptions<sqlx::Sqlite>,
286        min_conn: Option<u32>,
287        max_conn: Option<u32>,
288        path: &str,
289    ) -> Result<SqlitePool> {
290        if let Some(min_conn) = min_conn {
291            options = options.min_connections(min_conn);
292        }
293        if let Some(max_conn) = max_conn {
294            options = options.max_connections(max_conn);
295        }
296
297        let cfg = Self::common_connection_cfg(cfg);
298        let pool = options
299            .connect_with(cfg.filename(directory.join(path)))
300            .await
301            .map_err(|e| DbSqlError::Construction(format!("failed to create {path} database: {e}")))?;
302
303        Ok(pool)
304    }
