64/// Main database handle for HOPR node operations.
65///
66/// Manages multiple SQLite databases for different data domains to avoid
67/// locking conflicts and improve performance:
68///
69/// - **Index DB**: Blockchain indexing and contract data
70/// - **Tickets DB**: Payment tickets and acknowledgments
71/// - **Peers DB**: Network peer information and metadata
72/// - **Logs DB**: Blockchain logs and processing status
73///
74/// Supports database snapshot imports for fast synchronization via
75/// [`HoprDbGeneralModelOperations::import_logs_db`].
76#[derive(Debug, Clone)]
77pub struct HoprDb {
78    pub(crate) index_db: DbConnection,
79    pub(crate) tickets_db: sea_orm::DatabaseConnection,
80    pub(crate) peers_db: sea_orm::DatabaseConnection,
81    pub(crate) logs_db: sea_orm::DatabaseConnection,
82
83    pub(crate) caches: Arc<HoprDbCaches>,
84    pub(crate) chain_key: ChainKeypair,
85    pub(crate) me_onchain: Address,
86    pub(crate) ticket_manager: Arc<TicketManager>,
87}

99    pub async fn new(directory: &Path, chain_key: ChainKeypair, cfg: HoprDbConfig) -> Result<Self> {
100        #[cfg(all(feature = "prometheus", not(test)))]
101        {
102            lazy_static::initialize(&crate::protocol::METRIC_RECEIVED_ACKS);
103            lazy_static::initialize(&crate::protocol::METRIC_SENT_ACKS);
104            lazy_static::initialize(&crate::protocol::METRIC_TICKETS_COUNT);
105        }
106
107        fs::create_dir_all(directory)
108            .map_err(|_e| DbSqlError::Construction(format!("cannot create main database directory {directory:?}")))?;
109
110        let index = Self::create_pool(
111            cfg.clone(),
112            directory.to_path_buf(),
113            PoolOptions::new(),
114            Some(0),
115            Some(1),
116            false,
117            SQL_DB_INDEX_FILE_NAME,
118        )
119        .await?;
120
121        #[cfg(feature = "sqlite")]
122        let index_ro = Self::create_pool(
123            cfg.clone(),
124            directory.to_path_buf(),
125            PoolOptions::new(),
126            Some(0),
127            Some(30),
128            true,
129            SQL_DB_INDEX_FILE_NAME,
130        )
131        .await?;
132
133        let peers_options = PoolOptions::new()
134            .acquire_timeout(Duration::from_secs(60)) // Default is 30
135            .idle_timeout(Some(Duration::from_secs(10 * 60))) // This is the default
136            .max_lifetime(Some(Duration::from_secs(30 * 60))); // This is the default
137
138        let peers = Self::create_pool(
139            cfg.clone(),
140            directory.to_path_buf(),
141            peers_options,
142            Some(0),
143            Some(300),
144            false,
145            SQL_DB_PEERS_FILE_NAME,
146        )
147        .await?;
148
149        let tickets = Self::create_pool(
150            cfg.clone(),
151            directory.to_path_buf(),
152            PoolOptions::new(),
153            Some(0),
154            Some(50),
155            false,
156            SQL_DB_TICKETS_FILE_NAME,
157        )
158        .await?;
159
160        let logs = Self::create_pool(
161            cfg.clone(),
162            directory.to_path_buf(),
163            PoolOptions::new(),
164            Some(0),
165            None,
166            false,
167            SQL_DB_LOGS_FILE_NAME,
168        )
169        .await?;
170
171        #[cfg(feature = "sqlite")]
172        Self::new_sqlx_sqlite(chain_key, index, index_ro, peers, tickets, logs).await
173    }

198    #[cfg(feature = "sqlite")]
199    async fn new_sqlx_sqlite(
200        chain_key: ChainKeypair,
201        index_db_pool: SqlitePool,
202        index_db_ro_pool: SqlitePool,
203        peers_db_pool: SqlitePool,
204        tickets_db_pool: SqlitePool,
205        logs_db_pool: SqlitePool,
206    ) -> Result<Self> {
207        let index_db_rw = SqlxSqliteConnector::from_sqlx_sqlite_pool(index_db_pool);
208        let index_db_ro = SqlxSqliteConnector::from_sqlx_sqlite_pool(index_db_ro_pool);
209
210        MigratorIndex::up(&index_db_rw, None)
211            .await
212            .map_err(|e| DbSqlError::Construction(format!("cannot apply database migration: {e}")))?;
213
214        let tickets_db = SqlxSqliteConnector::from_sqlx_sqlite_pool(tickets_db_pool);
215
216        MigratorTickets::up(&tickets_db, None)
217            .await
218            .map_err(|e| DbSqlError::Construction(format!("cannot apply database migration: {e}")))?;
219
220        let peers_db = SqlxSqliteConnector::from_sqlx_sqlite_pool(peers_db_pool);
221
222        MigratorPeers::up(&peers_db, None)
223            .await
224            .map_err(|e| DbSqlError::Construction(format!("cannot apply database migration: {e}")))?;
225
226        let logs_db = SqlxSqliteConnector::from_sqlx_sqlite_pool(logs_db_pool.clone());
227
228        MigratorChainLogs::up(&logs_db, None)
229            .await
230            .map_err(|e| DbSqlError::Construction(format!("cannot apply database migration: {e}")))?;
231
232        // Reset the peer network information
233        let res = hopr_db_entity::network_peer::Entity::delete_many()
234            .filter(
235                sea_orm::Condition::all().add(
236                    hopr_db_entity::network_peer::Column::LastSeen.lt(chrono::DateTime::<chrono::Utc>::from(
237                        hopr_platform::time::native::current_time()
238                            .checked_sub(std::time::Duration::from_secs(
239                                std::env::var("HOPR_INTERNAL_DB_PEERS_PERSISTENCE_AFTER_RESTART_IN_SECONDS")
240                                    .unwrap_or_else(|_| {
241                                        HOPR_INTERNAL_DB_PEERS_PERSISTENCE_AFTER_RESTART_IN_SECONDS.to_string()
242                                    })
243                                    .parse::<u64>()
244                                    .unwrap_or(HOPR_INTERNAL_DB_PEERS_PERSISTENCE_AFTER_RESTART_IN_SECONDS),
245                            ))
246                            .unwrap_or_else(hopr_platform::time::native::current_time),
247                    )),
248                ),
249            )
250            .exec(&peers_db)
251            .await
252            .map_err(|e| DbSqlError::Construction(format!("must reset peers on init: {e}")))?;
253        debug!(rows = res.rows_affected, "Cleaned up rows from the 'peers' table");
254
255        // Reset all BeingAggregated ticket states to Untouched
256        ticket::Entity::update_many()
257            .filter(ticket::Column::State.eq(AcknowledgedTicketStatus::BeingAggregated as u8))
258            .col_expr(
259                ticket::Column::State,
260                Expr::value(AcknowledgedTicketStatus::Untouched as u8),
261            )
262            .exec(&tickets_db)
263            .await?;
264
265        let caches = Arc::new(HoprDbCaches::default());
266        caches.invalidate_all();
267
268        // Initialize KeyId mapping for accounts
269        Account::find()
270            .find_with_related(Announcement)
271            .all(&index_db_rw)
272            .await?
273            .into_iter()
274            .try_for_each(|(a, b)| match model_to_account_entry(a, b) {
275                Ok(account) => caches.key_id_mapper.update_key_id_binding(&account),
276                Err(error) => {
277                    // Undecodeable accounts are skipped and will be unreachable
278                    tracing::error!(%error, "undecodeable account");
279                    Ok(())
280                }
281            })?;
282
283        Ok(Self {
284            me_onchain: chain_key.public().to_address(),
285            chain_key,
286            ticket_manager: Arc::new(TicketManager::new(tickets_db.clone(), caches.clone())),
287            caches,
288
289            index_db: DbConnection {
290                ro: index_db_ro,
291                rw: index_db_rw,
292            },
293            tickets_db,
294            peers_db,
295            logs_db,
296        })
297    }

299    /// Starts ticket processing by the [TicketManager] with an optional new ticket notifier.
300    /// Without calling this method, tickets will not be persisted into the DB.
301    ///
302    /// If the notifier is given, it will receive notifications once new ticket has been
303    /// persisted into the Tickets DB.
304    pub fn start_ticket_processing(&self, ticket_notifier: Option<UnboundedSender<AcknowledgedTicket>>) -> Result<()> {
305        if let Some(notifier) = ticket_notifier {
306            self.ticket_manager.start_ticket_processing(notifier)
307        } else {
308            self.ticket_manager.start_ticket_processing(futures::sink::drain())
309        }
310    }

312    /// Default SQLite config values for all DBs with RW  (read-write) access.
313    fn common_connection_cfg_rw(cfg: HoprDbConfig) -> SqliteConnectOptions {
314        SqliteConnectOptions::default()
315            .create_if_missing(cfg.create_if_missing)
316            .log_slow_statements(LevelFilter::Warn, cfg.log_slow_queries)
317            .log_statements(LevelFilter::Debug)
318            .journal_mode(SqliteJournalMode::Wal)
319            .synchronous(SqliteSynchronous::Normal)
320            .auto_vacuum(SqliteAutoVacuum::Full)
321            //.optimize_on_close(true, None) // Removed, because it causes optimization on each connection, due to min_connections being set to 0
322            .page_size(4096)
323            .pragma("cache_size", "-30000") // 32M
324            .pragma("busy_timeout", "1000") // 1000ms
325    }

327    /// Default SQLite config values for all DBs with RO (read-only) access.
328    fn common_connection_cfg_ro(cfg: HoprDbConfig) -> SqliteConnectOptions {
329        SqliteConnectOptions::default()
330            .create_if_missing(cfg.create_if_missing)
331            .log_slow_statements(LevelFilter::Warn, cfg.log_slow_queries)
332            .log_statements(LevelFilter::Debug)
333            //.optimize_on_close(true, None) // Removed, because it causes optimization on each connection, due to min_connections being set to 0
334            .page_size(4096)
335            .pragma("cache_size", "-30000") // 32M
336            .pragma("busy_timeout", "1000") // 1000ms

340    pub async fn create_pool(
341        cfg: HoprDbConfig,
342        directory: PathBuf,
343        mut options: PoolOptions<sqlx::Sqlite>,
344        min_conn: Option<u32>,
345        max_conn: Option<u32>,
346        read_only: bool,
347        path: &str,
348    ) -> Result<SqlitePool> {
349        if let Some(min_conn) = min_conn {
350            options = options.min_connections(min_conn);
351        }
352        if let Some(max_conn) = max_conn {
353            options = options.max_connections(max_conn);
354        }
355
356        let cfg = if read_only {
357            Self::common_connection_cfg_ro(cfg)
358        } else {
359            Self::common_connection_cfg_rw(cfg)
360        };
361
362        let pool = options
363            .connect_with(cfg.filename(directory.join(path)))
364            .await
365            .map_err(|e| DbSqlError::Construction(format!("failed to create {path} database: {e}")))?;
366
367        Ok(pool)
368    }
