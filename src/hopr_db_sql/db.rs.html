72/// Main database handle for HOPR node operations.
73///
74/// Manages multiple SQLite databases for different data domains to avoid
75/// locking conflicts and improve performance:
76///
77/// - **Index DB**: Blockchain indexing and contract data
78/// - **Tickets DB**: Payment tickets and acknowledgments
79/// - **Peers DB**: Network peer information and metadata
80/// - **Logs DB**: Blockchain logs and processing status
81///
82/// Supports database snapshot imports for fast synchronization via
83/// [`HoprDbGeneralModelOperations::import_logs_db`].
84#[derive(Debug, Clone)]
85pub struct HoprDb {
86    pub(crate) index_db: DbConnection,
87    pub(crate) tickets_db: sea_orm::DatabaseConnection,
88    pub(crate) peers_db: sea_orm::DatabaseConnection,
89    pub(crate) logs_db: sea_orm::DatabaseConnection,
90
91    pub(crate) caches: Arc<HoprDbCaches>,
92    pub(crate) chain_key: ChainKeypair,
93    pub(crate) me_onchain: Address,
94    pub(crate) ticket_manager: Arc<TicketManager>,
95    pub(crate) cfg: HoprDbConfig,
96}

108    pub async fn new(directory: &Path, chain_key: ChainKeypair, cfg: HoprDbConfig) -> Result<Self> {
109        #[cfg(all(feature = "prometheus", not(test)))]
110        {
111            lazy_static::initialize(&crate::protocol::METRIC_RECEIVED_ACKS);
112            lazy_static::initialize(&crate::protocol::METRIC_SENT_ACKS);
113            lazy_static::initialize(&crate::protocol::METRIC_TICKETS_COUNT);
114        }
115
116        cfg.validate()
117            .map_err(|e| DbSqlError::Construction(format!("failed configuration validation: {e}")))?;
118
119        fs::create_dir_all(directory)
120            .map_err(|_e| DbSqlError::Construction(format!("cannot create main database directory {directory:?}")))?;
121
122        let index = Self::create_pool(
123            cfg.clone(),
124            directory.to_path_buf(),
125            PoolOptions::new(),
126            Some(0),
127            Some(1),
128            false,
129            SQL_DB_INDEX_FILE_NAME,
130        )
131        .await?;
132
133        #[cfg(feature = "sqlite")]
134        let index_ro = Self::create_pool(
135            cfg.clone(),
136            directory.to_path_buf(),
137            PoolOptions::new(),
138            Some(0),
139            Some(30),
140            true,
141            SQL_DB_INDEX_FILE_NAME,
142        )
143        .await?;
144
145        let peers_options = PoolOptions::new()
146            .acquire_timeout(Duration::from_secs(60)) // Default is 30
147            .idle_timeout(Some(Duration::from_secs(10 * 60))) // This is the default
148            .max_lifetime(Some(Duration::from_secs(30 * 60))); // This is the default
149
150        let peers = Self::create_pool(
151            cfg.clone(),
152            directory.to_path_buf(),
153            peers_options,
154            Some(0),
155            Some(300),
156            false,
157            SQL_DB_PEERS_FILE_NAME,
158        )
159        .await?;
160
161        let tickets = Self::create_pool(
162            cfg.clone(),
163            directory.to_path_buf(),
164            PoolOptions::new(),
165            Some(0),
166            Some(50),
167            false,
168            SQL_DB_TICKETS_FILE_NAME,
169        )
170        .await?;
171
172        let logs = Self::create_pool(
173            cfg.clone(),
174            directory.to_path_buf(),
175            PoolOptions::new(),
176            Some(0),
177            None,
178            false,
179            SQL_DB_LOGS_FILE_NAME,
180        )
181        .await?;
182
183        #[cfg(feature = "sqlite")]
184        Self::new_sqlx_sqlite(chain_key, index, index_ro, peers, tickets, logs, cfg).await
185    }

211    #[cfg(feature = "sqlite")]
212    async fn new_sqlx_sqlite(
213        chain_key: ChainKeypair,
214        index_db_pool: SqlitePool,
215        index_db_ro_pool: SqlitePool,
216        peers_db_pool: SqlitePool,
217        tickets_db_pool: SqlitePool,
218        logs_db_pool: SqlitePool,
219        cfg: HoprDbConfig,
220    ) -> Result<Self> {
221        let index_db_rw = SqlxSqliteConnector::from_sqlx_sqlite_pool(index_db_pool);
222        let index_db_ro = SqlxSqliteConnector::from_sqlx_sqlite_pool(index_db_ro_pool);
223
224        MigratorIndex::up(&index_db_rw, None)
225            .await
226            .map_err(|e| DbSqlError::Construction(format!("cannot apply database migration: {e}")))?;
227
228        let tickets_db = SqlxSqliteConnector::from_sqlx_sqlite_pool(tickets_db_pool);
229
230        MigratorTickets::up(&tickets_db, None)
231            .await
232            .map_err(|e| DbSqlError::Construction(format!("cannot apply database migration: {e}")))?;
233
234        let peers_db = SqlxSqliteConnector::from_sqlx_sqlite_pool(peers_db_pool);
235
236        MigratorPeers::up(&peers_db, None)
237            .await
238            .map_err(|e| DbSqlError::Construction(format!("cannot apply database migration: {e}")))?;
239
240        let logs_db = SqlxSqliteConnector::from_sqlx_sqlite_pool(logs_db_pool.clone());
241
242        MigratorChainLogs::up(&logs_db, None)
243            .await
244            .map_err(|e| DbSqlError::Construction(format!("cannot apply database migration: {e}")))?;
245
246        // Reset the peer network information
247        let res = hopr_db_entity::network_peer::Entity::delete_many()
248            .filter(
249                sea_orm::Condition::all().add(
250                    hopr_db_entity::network_peer::Column::LastSeen.lt(chrono::DateTime::<chrono::Utc>::from(
251                        hopr_platform::time::native::current_time()
252                            .checked_sub(std::time::Duration::from_secs(
253                                std::env::var("HOPR_INTERNAL_DB_PEERS_PERSISTENCE_AFTER_RESTART_IN_SECONDS")
254                                    .unwrap_or_else(|_| {
255                                        HOPR_INTERNAL_DB_PEERS_PERSISTENCE_AFTER_RESTART_IN_SECONDS.to_string()
256                                    })
257                                    .parse::<u64>()
258                                    .unwrap_or(HOPR_INTERNAL_DB_PEERS_PERSISTENCE_AFTER_RESTART_IN_SECONDS),
259                            ))
260                            .unwrap_or_else(hopr_platform::time::native::current_time),
261                    )),
262                ),
263            )
264            .exec(&peers_db)
265            .await
266            .map_err(|e| DbSqlError::Construction(format!("must reset peers on init: {e}")))?;
267        debug!(rows = res.rows_affected, "Cleaned up rows from the 'peers' table");
268
269        // Reset all BeingAggregated ticket states to Untouched
270        ticket::Entity::update_many()
271            .filter(ticket::Column::State.eq(AcknowledgedTicketStatus::BeingAggregated as u8))
272            .col_expr(
273                ticket::Column::State,
274                Expr::value(AcknowledgedTicketStatus::Untouched as u8),
275            )
276            .exec(&tickets_db)
277            .await?;
278
279        let caches = Arc::new(HoprDbCaches::default());
280        caches.invalidate_all();
281
282        // Initialize KeyId mapping for accounts
283        Account::find()
284            .find_with_related(Announcement)
285            .all(&index_db_rw)
286            .await?
287            .into_iter()
288            .try_for_each(|(a, b)| match model_to_account_entry(a, b) {
289                Ok(account) => caches.key_id_mapper.update_key_id_binding(&account),
290                Err(error) => {
291                    // Undecodeable accounts are skipped and will be unreachable
292                    tracing::error!(%error, "undecodeable account");
293                    Ok(())
294                }
295            })?;
296
297        Ok(Self {
298            me_onchain: chain_key.public().to_address(),
299            chain_key,
300            ticket_manager: Arc::new(TicketManager::new(tickets_db.clone(), caches.clone())),
301            caches,
302
303            index_db: DbConnection {
304                ro: index_db_ro,
305                rw: index_db_rw,
306            },
307            tickets_db,
308            peers_db,
309            logs_db,
310            cfg,
311        })
312    }

314    /// Starts ticket processing by the `TicketManager` with an optional new ticket notifier.
315    /// Without calling this method, tickets will not be persisted into the DB.
316    ///
317    /// If the notifier is given, it will receive notifications once new ticket has been
318    /// persisted into the Tickets DB.
319    pub fn start_ticket_processing<S>(&self, ticket_notifier: Option<S>) -> Result<()>
320    where
321        S: futures::Sink<AcknowledgedTicket> + Send + 'static,
322        S::Error: std::fmt::Display + std::error::Error,
323    {
324        if let Some(notifier) = ticket_notifier {
325            self.ticket_manager.start_ticket_processing(notifier)
326        } else {
327            self.ticket_manager.start_ticket_processing(futures::sink::drain())
328        }
329    }

331    /// Default SQLite config values for all DBs with RW  (read-write) access.
332    fn common_connection_cfg_rw(cfg: HoprDbConfig) -> SqliteConnectOptions {
333        SqliteConnectOptions::default()
334            .create_if_missing(cfg.create_if_missing)
335            .log_slow_statements(LevelFilter::Warn, cfg.log_slow_queries)
336            .log_statements(LevelFilter::Debug)
337            .journal_mode(SqliteJournalMode::Wal)
338            .synchronous(SqliteSynchronous::Normal)
339            .auto_vacuum(SqliteAutoVacuum::Full)
340            //.optimize_on_close(true, None) // Removed, because it causes optimization on each connection, due to min_connections being set to 0
341            .page_size(4096)
342            .pragma("cache_size", "-30000") // 32M
343            .pragma("busy_timeout", "1000") // 1000ms
344    }
345
346    /// Default SQLite config values for all DBs with RO (read-only) access.
347    fn common_connection_cfg_ro(cfg: HoprDbConfig) -> SqliteConnectOptions {
348        SqliteConnectOptions::default()
349            .create_if_missing(cfg.create_if_missing)
350            .log_slow_statements(LevelFilter::Warn, cfg.log_slow_queries)
351            .log_statements(LevelFilter::Debug)
352            //.optimize_on_close(true, None) // Removed, because it causes optimization on each connection, due to min_connections being set to 0
353            .page_size(4096)
354            .pragma("cache_size", "-30000") // 32M
355            .pragma("busy_timeout", "1000") // 1000ms
356            .read_only(true)
357    }

359    pub async fn create_pool(
360        cfg: HoprDbConfig,
361        directory: PathBuf,
362        mut options: PoolOptions<sqlx::Sqlite>,
363        min_conn: Option<u32>,
364        max_conn: Option<u32>,
365        read_only: bool,
366        path: &str,
367    ) -> Result<SqlitePool> {
368        if let Some(min_conn) = min_conn {
369            options = options.min_connections(min_conn);
370        }
371        if let Some(max_conn) = max_conn {
372            options = options.max_connections(max_conn);
373        }
374
375        let cfg = if read_only {
376            Self::common_connection_cfg_ro(cfg)
377        } else {
378            Self::common_connection_cfg_rw(cfg)
379        };
380
381        let pool = options
382            .connect_with(cfg.filename(directory.join(path)))
383            .await
384            .map_err(|e| DbSqlError::Construction(format!("failed to create {path} database: {e}")))?;
385
386        Ok(pool)
387    }
