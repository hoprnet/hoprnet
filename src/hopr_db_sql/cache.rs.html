15/// Lists all singular data that can be cached and
16/// cannot be represented by a key. These values can be cached for the long term.
17#[derive(Debug, Clone, PartialEq, strum::EnumDiscriminants)]
18#[strum_discriminants(derive(Hash))]
19pub enum CachedValue {
20    /// Cached [IndexerData].
21    IndexerDataCache(IndexerData),
22    /// Cached [SafeInfo].
23    SafeInfoCache(Option<SafeInfo>),
24}
25
26impl TryFrom<CachedValue> for IndexerData {
27    type Error = DbSqlError;
28
29    fn try_from(value: CachedValue) -> Result<Self, Self::Error> {
30        match value {
31            CachedValue::IndexerDataCache(data) => Ok(data),
32            _ => Err(DbSqlError::DecodingError),
33        }
34    }
35}
36
37impl TryFrom<CachedValue> for Option<SafeInfo> {
38    type Error = DbSqlError;
39
40    fn try_from(value: CachedValue) -> Result<Self, Self::Error> {
41        match value {
42            CachedValue::SafeInfoCache(data) => Ok(data),
43            _ => Err(DbSqlError::DecodingError),
44        }
45    }
46}

62impl Default for SurbRingBuffer {
63    fn default() -> Self {
64        // TODO: make the RB capacity configurable in the future ?
65        // With the current packet size, this is roughly for 7 MB of data budget in SURBs
66        Self(Arc::new(Mutex::new(AllocRingBuffer::new(10_000))))

70impl SurbRingBuffer {
71    pub fn push<I: IntoIterator<Item = HoprSurb>>(&self, surbs: I) -> Result<(), DbSqlError> {
72        self.0
73            .lock()
74            .map_err(|_| DbSqlError::LogicalError("failed to lock surbs".into()))?
75            .extend(surbs);
76        Ok(())
77    }
78    pub fn pop_one(&self) -> Result<HoprSurb, DbSqlError> {
79        self.0
80            .lock()
81            .map_err(|_| DbSqlError::LogicalError("failed to lock surbs".into()))?
82            .dequeue()
83            .ok_or(DbSqlError::NoSurbAvailable)
84    }
85}

172pub(crate) struct CacheKeyMapper(
173    DashMap<KeyIdent<4>, OffchainPublicKey>,
174    DashMap<OffchainPublicKey, KeyIdent<4>>,
175);
176
177impl CacheKeyMapper {
178    pub fn with_capacity(capacity: usize) -> Self {
179        Self(DashMap::with_capacity(capacity), DashMap::with_capacity(capacity))
180    }
181
182    /// Creates key id mapping for a public key of an [account](AccountEntry).
183    ///
184    /// Does nothing if the binding already exists. Returns error if an existing binding
185    /// is not consistent.
186    pub fn update_key_id_binding(&self, account: &AccountEntry) -> Result<(), DbSqlError> {
187        let id = account.key_id();
188
189        // Lock entries in the maps to avoid concurrent modifications
190        let id_entry = self.0.entry(id);
191        let key_entry = self.1.entry(account.public_key);
192
193        match (id_entry, key_entry) {
194            (Entry::Vacant(v_id), Entry::Vacant(v_key)) => {
195                v_id.insert(account.public_key);
196                v_key.insert(id);
197                tracing::debug!(%id, %account.public_key, "inserted key-id binding");
198                Ok(())
199            }
200            (Entry::Occupied(v_id), Entry::Occupied(v_key)) => {
201                // Check if the existing binding is consistent with the new one.
202                if v_id.get() != v_key.key() {
203                    Err(DbSqlError::LogicalError(format!(
204                        "attempt to insert key {} with key-id {id} already exists for the key {}",
205                        v_key.key(),
206                        v_id.get()
207                    )))
208                } else {
209                    Ok(())
210                }
211            }
212            // This should never happen.
213            _ => Err(DbSqlError::LogicalError("inconsistent key-id binding".into())),
214        }
215    }
216}
