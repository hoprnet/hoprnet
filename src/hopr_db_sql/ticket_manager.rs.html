16/// Functionality related to locking and structural improvements to the underlying SQLite database
17///
18/// With SQLite, it is only possible to have a single write lock per database, meaning that
19/// high-frequency database access to tickets needed to be split from the rest of the database
20/// operations.
21///
22/// High frequency of locking originating from the ticket processing pipeline could starve the DB
23/// and lock with other concurrent processes. Therefore, a single mutex for write operations exists,
24/// which allows bottle-necking the database write access on the mutex, as well as allowing arbitrary
25/// numbers of concurrent read operations.
26///
27/// The queue-based mechanism also splits the storage of the ticket inside the database from the processing,
28/// effectively allowing the processing pipelines to be independent of a database write access.
29#[derive(Debug, Clone)]
30pub(crate) struct TicketManager {
31    pub(crate) tickets_db: sea_orm::DatabaseConnection,
32    pub(crate) mutex: Arc<async_lock::Mutex<()>>,
33    incoming_ack_tickets_tx: Arc<OnceLock<futures::channel::mpsc::Sender<TicketOperation>>>,
34    caches: Arc<HoprDbCaches>,
35}
36
37enum TicketOperation {
38    /// Inserts a new ticket
39    Insert(AcknowledgedTicket),
40    /// Replaces multiple tickets (in BeingAggregated state) with the given aggregated ticket.
41    Replace(AcknowledgedTicket),
42}
43
44impl TicketOperation {
45    fn ticket(&self) -> &AcknowledgedTicket {
46        match self {
47            TicketOperation::Insert(ticket) => ticket,
48            TicketOperation::Replace(ticket) => ticket,
49        }
50    }
51}

63    /// Must be called to start processing tickets into the DB.
64    pub fn start_ticket_processing<S, E>(&self, ticket_notifier: S) -> Result<()>
65    where
66        S: Sink<AcknowledgedTicket, Error = E> + Send + 'static,
67        E: std::error::Error,
68    {
69        let (tx, mut rx) = futures::channel::mpsc::channel::<TicketOperation>(100_000);
70
71        self.incoming_ack_tickets_tx
72            .set(tx)
73            .map_err(|_| DbSqlError::LogicalError("ticket processing already started".into()))?;
74
75        // Creates a process to desynchronize storing of the ticket into the database
76        // and the processing calls triggering such an operation.
77        let db_clone = self.tickets_db.clone();
78        let mutex_clone = self.mutex.clone();
79
80        // NOTE: This spawned task does not need to be explicitly canceled, since it will
81        // be automatically dropped when the event sender object is dropped.
82        spawn(async move {
83            pin_mut!(ticket_notifier);
84            while let Some(ticket_op) = rx.next().await {
85                let ticket_to_insert = ticket_op.ticket().clone();
86                let ticket_inserted = match db_clone
87                    .begin_with_config(None, None)
88                    .await
89                    .map_err(DbSqlError::BackendError)
90                {
91                    Ok(transaction) => {
92                        let transaction = OpenTransaction(transaction, crate::TargetDb::Tickets);
93
94                        let _quard = mutex_clone.lock().await;
95
96                        if let Err(error) = transaction
97                            .perform(|tx| {
98                                Box::pin(async move {
99                                    match ticket_op {
100                                        // Insertion of a new acknowledged ticket
101                                        TicketOperation::Insert(ack_ticket) => {
102                                            let channel_id = ack_ticket.verified_ticket().channel_id.to_hex();
103
104                                            hopr_db_entity::ticket::ActiveModel::from(ack_ticket)
105                                                .insert(tx.as_ref())
106                                                .await?;
107
108                                            // Update the ticket winning count in the statistics
109                                            if let Some(model) = hopr_db_entity::ticket_statistics::Entity::find()
110                                                .filter(
111                                                    hopr_db_entity::ticket_statistics::Column::ChannelId
112                                                        .eq(channel_id.clone()),
113                                                )
114                                                .one(tx.as_ref())
115                                                .await?
116                                            {
117                                                let winning_tickets = model.winning_tickets + 1;
118                                                let mut active_model = model.into_active_model();
119                                                active_model.winning_tickets = sea_orm::Set(winning_tickets);
120                                                active_model
121                                            } else {
122                                                hopr_db_entity::ticket_statistics::ActiveModel {
123                                                    channel_id: sea_orm::Set(channel_id),
124                                                    winning_tickets: sea_orm::Set(1),
125                                                    ..Default::default()
126                                                }
127                                            }
128                                            .save(tx.as_ref())
129                                            .await?;
130                                        }
131                                        TicketOperation::Replace(ack_ticket) => {
132                                            // Replacement range on the aggregated ticket
133                                            let start_idx = ack_ticket.verified_ticket().index;
134                                            let offset = ack_ticket.verified_ticket().index_offset as u64;
135
136                                            // Replace all BeingAggregated tickets with aggregated index range in this
137                                            // channel
138                                            let selector = TicketSelector::new(
139                                                ack_ticket.verified_ticket().channel_id,
140                                                ack_ticket.verified_ticket().channel_epoch,
141                                            )
142                                            .with_index_range(start_idx..start_idx + offset)
143                                            .with_state(AcknowledgedTicketStatus::BeingAggregated);
144
145                                            let deleted = ticket::Entity::delete_many()
146                                                .filter(WrappedTicketSelector::from(selector))
147                                                .exec(tx.as_ref())
148                                                .await?;
149
150                                            if deleted.rows_affected > offset {
151                                                return Err(DbSqlError::LogicalError(format!(
152                                                    "deleted ticket count ({}) must not be more than the ticket index \
153                                                     offset {offset}",
154                                                    deleted.rows_affected,
155                                                )));
156                                            }
157
158                                            ticket::Entity::insert::<ticket::ActiveModel>(ack_ticket.into())
159                                                .exec(tx.as_ref())
160                                                .await?;
161                                        }
162                                    }
163                                    Ok::<_, DbSqlError>(())
164                                })
165                            })
166                            .await
167                        {
168                            error!(%error, "failed to insert the winning ticket and update the ticket stats");
169                            false
170                        } else {
171                            debug!(acknowledged_ticket = %ticket_to_insert, "ticket persisted into the ticket db");
172                            true
173                        }
174                    }
175                    Err(error) => {
176                        error!(%error, "failed to create a transaction for ticket insertion");
177                        false
178                    }
179                };
180
181                // Notify about the ticket once successfully inserted into the Tickets DB
182                if ticket_inserted {
183                    if let Err(error) = ticket_notifier.send(ticket_to_insert).await {
184                        error!(%error, "failed to notify the ticket notifier about the winning ticket");
185                    }
186                }
187            }
188
189            tracing::warn!(task = "ticket processing", "long-running background task finished")
190        });
191
192        Ok(())
193    }

195    /// Sends a new acknowledged ticket into the FIFO queue.
196    ///
197    /// The [`start_ticket_processing`](TicketManager::start_ticket_processing) method
198    /// must be called before calling this method, or it will fail.
199    pub async fn insert_ticket(&self, ticket: AcknowledgedTicket) -> Result<()> {
200        let channel = ticket.verified_ticket().channel_id;
201        let value = ticket.verified_ticket().amount;
202        let epoch = ticket.verified_ticket().channel_epoch;
203
204        self.incoming_ack_tickets_tx
205            .get()
206            .ok_or(DbSqlError::LogicalError("ticket processing not started".into()))?
207            .clone()
208            .try_send(TicketOperation::Insert(ticket))
209            .map_err(|e| {
210                DbSqlError::LogicalError(format!(
211                    "failed to enqueue acknowledged ticket processing into the DB: {e}"
212                ))
213            })?;
214
215        let unrealized_value = self.unrealized_value(TicketSelector::new(channel, epoch)).await?;
216
217        #[cfg(all(feature = "prometheus", not(test)))]
218        {
219            crate::tickets::METRIC_HOPR_TICKETS_INCOMING_STATISTICS.set(
220                &[&channel.to_string(), "unredeemed"],
221                (unrealized_value + value).amount().as_u128() as f64,
222            );
223        }
224
225        self.caches
226            .unrealized_value
227            .insert((channel, epoch.into()), unrealized_value + value)
228            .await;
229
230        Ok(())
231    }

250    /// Get unrealized value for a channel
251    pub async fn unrealized_value(&self, selector: TicketSelector) -> Result<HoprBalance> {
252        if !selector.is_single_channel() {
253            return Err(crate::DbSqlError::LogicalError(
254                "selector must represent a single channel".into(),
255            ));
256        }
257
258        let channel_id = selector.channel_identifiers[0].0;
259        let channel_epoch = selector.channel_identifiers[0].1;
260        let selector: WrappedTicketSelector = selector.into();
261
262        let tickets_db = self.tickets_db.clone();
263        let selector_clone = selector.clone();
264        Ok(self
265            .caches
266            .unrealized_value
267            .try_get_with_by_ref(&(channel_id, channel_epoch), async move {
268                tracing::warn!(%channel_id, %channel_epoch, "cache miss on unrealized value");
269                OpenTransaction(
270                    tickets_db
271                        .begin_with_config(None, None)
272                        .await
273                        .map_err(DbSqlError::BackendError)?,
274                    crate::TargetDb::Tickets,
275                )
276                .perform(|tx| {
277                    Box::pin(async move {
278                        ticket::Entity::find()
279                            .filter(selector_clone)
280                            .stream(tx.as_ref())
281                            .await
282                            .map_err(crate::errors::DbSqlError::from)?
283                            .map_err(crate::errors::DbSqlError::from)
284                            .try_fold(HoprBalance::zero(), |value, t| async move {
285                                Ok(value + HoprBalance::from_be_bytes(t.amount))
286                            })
287                            .await
288                    })
289                })
290                .await
291            })
292            .await?)
293    }

295    /// Acquires write lock to the Ticket DB and starts a new transaction.
296    pub async fn with_write_locked_db<'a, F, T, E>(&'a self, f: F) -> std::result::Result<T, E>
297    where
298        F: for<'c> FnOnce(&'c OpenTransaction) -> BoxFuture<'c, std::result::Result<T, E>> + Send,
299        T: Send,
300        E: std::error::Error + From<crate::errors::DbSqlError>,
301    {
302        let mutex = self.mutex.clone();
303        let _guard = mutex.lock().await;
304
305        let transaction = OpenTransaction(
306            self.tickets_db
307                .begin_with_config(None, None)
308                .await
309                .map_err(crate::errors::DbSqlError::BackendError)?,
310            crate::TargetDb::Tickets,
311        );
312
313        transaction.perform(f).await
314    }

343    async fn add_peer_mappings(db: &HoprDb, peers: Vec<(OffchainKeypair, ChainKeypair)>) -> crate::errors::Result<()> {
344        for (peer_offchain, peer_onchain) in peers.into_iter() {
345            db.insert_account(
346                None,
347                AccountEntry {
348                    public_key: *peer_offchain.public(),
349                    chain_addr: peer_onchain.public().to_address(),
350                    entry_type: AccountType::NotAnnounced,
351                    published_at: 0,
352                },
353            )
354            .await?
355        }
356
357        Ok(())

360    fn generate_random_ack_ticket(index: u32) -> anyhow::Result<AcknowledgedTicket> {
361        let hk1 = HalfKey::random();
362        let hk2 = HalfKey::random();
363
364        let challenge = Response::from_half_keys(&hk1, &hk2)?.to_challenge()?;
365
366        let ticket = TicketBuilder::default()
367            .direction(&BOB.public().to_address(), &ALICE.public().to_address())
368            .amount(TICKET_VALUE)
369            .index(index as u64)
370            .channel_epoch(4)
371            .challenge(challenge)
372            .build_signed(&BOB, &Hash::default())?;
373
374        Ok(ticket.into_acknowledged(Response::from_half_keys(&hk1, &hk2)?))
375    }
