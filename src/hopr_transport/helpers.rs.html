67impl<T> PathPlanner<T>
68where
69    T: HoprDbAllOperations + std::fmt::Debug + Send + Sync + 'static,
70{
71    pub(crate) fn new(
72        db: T,
73        path_selector_cfg: DfsPathSelectorConfig,
74        channel_graph: Arc<RwLock<hopr_path::channel_graph::ChannelGraph>>,
75    ) -> Self {
76        Self {
77            db,
78            channel_graph,
79            selector: DfsPathSelector::new(path_selector_cfg),
80        }
81    }
82
83    pub(crate) fn channel_graph(&self) -> Arc<RwLock<hopr_path::channel_graph::ChannelGraph>> {
84        self.channel_graph.clone()
85    }
86
87    #[tracing::instrument(level = "trace", skip(self))]
88    pub(crate) async fn resolve_path(
89        &self,
90        destination: PeerId,
91        options: RoutingOptions,
92    ) -> crate::errors::Result<TransportPath> {
93        let path = match options {
94            RoutingOptions::IntermediatePath(path) => {
95                let complete_path = Vec::from_iter(path.into_iter().chain([destination]));
96                trace!(full_path = ?complete_path, "resolved a specific path");
97
98                let cg = self.channel_graph.read().await;
99
100                TransportPath::resolve(complete_path, &self.db, &cg)
101                    .await
102                    .map(|(p, _)| p)?
103            }
104            RoutingOptions::Hops(hops) if u32::from(hops) == 0 => {
105                trace!(hops = 0, %destination, "resolved zero-hop path");
106                TransportPath::direct(destination)
107            }
108            RoutingOptions::Hops(hops) => {
109                trace!(%hops, "resolved path using hop count");
110
111                let pk = OffchainPublicKey::try_from(destination)?;
112
113                if let Some(chain_key) = self
114                    .db
115                    .translate_key(None, pk)
116                    .await
117                    .map_err(hopr_db_sql::api::errors::DbError::from)?
118                {
119                    let target_chain_key: Address = chain_key.try_into()?;
120                    let cp = {
121                        let cg = self.channel_graph.read().await;
122                        self.selector
123                            .select_path(&cg, cg.my_address(), target_chain_key, hops.into(), hops.into())?
124                    };
125
126                    let full_path = cp.into_path(&self.db, target_chain_key).await?;
127                    trace!(%full_path, "resolved automatic path");
128
129                    full_path
130                } else {
131                    return Err(HoprTransportError::Api(
132                        "send msg: unknown destination peer id encountered".to_owned(),
133                    ));
134                }
135            }
136        };
137
138        #[cfg(all(feature = "prometheus", not(test)))]
139        {
140            use hopr_path::path::Path;
141            hopr_metrics::SimpleHistogram::observe(&METRIC_PATH_LENGTH, (path.hops().len() - 1) as f64);
142        }
143
144        Ok(path)
145    }
146}

166#[async_trait::async_trait]
167impl<T> SendMsg for MessageSender<T>
168where
169    T: HoprDbAllOperations + std::fmt::Debug + Send + Sync + 'static,
170{
171    #[tracing::instrument(level = "debug", skip(self, data))]
172    async fn send_message(
173        &self,
174        data: ApplicationData,
175        destination: PeerId,
176        options: RoutingOptions,
177    ) -> std::result::Result<(), TransportSessionError> {
178        data.application_tag
179            .is_some_and(|application_tag| application_tag < RESERVED_SESSION_TAG_UPPER_LIMIT)
180            .then_some(())
181            .ok_or(TransportSessionError::Tag)?;
182
183        let path = self
184            .resolver
185            .resolve_path(destination, options)
186            .await
187            .map_err(|_| TransportSessionError::Path)?;
188
189        self.process_packet_send
190            .get()
191            .ok_or_else(|| SessionManagerError::NotStarted)?
192            .send_packet(data, path)
193            .await
194            .map_err(|_| TransportSessionError::Closed)?
195            .consume_and_wait(crate::constants::PACKET_QUEUE_TIMEOUT_MILLISECONDS)
196            .await
197            .map_err(|_e| TransportSessionError::Timeout)?;
198
199        trace!("Packet sent to the outgoing queue");
200
201        Ok(())
202    }
