1//! The crate aggregates and composes individual transport level objects and functionality
2//! into a unified [`crate::HoprTransport`] object with the goal of isolating the transport layer
3//! and defining a fully specified transport API.
4//!
5//! See also the `hopr_protocol_start` crate for details on Start sub-protocol which initiates a Session.
6//!
7//! As such, the transport layer components should be only those that are directly necessary to:
8//!
9//! 1. send and receive a packet, acknowledgement or ticket aggregation request
10//! 2. send and receive a network telemetry request
11//! 3. automate transport level processes
12//! 4. algorithms associated with the transport layer operational management
13//! 5. interface specifications to allow modular behavioral extensions
14

120/// PeerId -> OffchainPublicKey is a CPU-intensive blocking operation.
121///
122/// This helper uses a cached static object to speed up the lookup and avoid blocking the async
123/// runtime on repeated conversions for the same [`PeerId`]s.
124pub async fn peer_id_to_public_key(peer_id: &PeerId) -> crate::errors::Result<OffchainPublicKey> {
125    PEER_ID_CACHE
126        .try_get_with_by_ref(peer_id, async {
127            OffchainPublicKey::from_peerid(peer_id).map_err(|e| e.into())
128        })
129        .await
130        .map_err(|e: Arc<HoprTransportError>| {
131            crate::errors::HoprTransportError::Other(anyhow::anyhow!(
132                "failed to convert peer_id ({:?}) to an offchain public key: {e}",
133                peer_id
134            ))
135        })
136}

158/// Interface into the physical transport mechanism allowing all off-chain HOPR-related tasks on
159/// the transport.
160pub struct HoprTransport<Chain, Db, Graph, Net>
161where
162    Graph: NetworkGraphView<NodeId = OffchainPublicKey> + NetworkGraphUpdate + Clone + Send + Sync + 'static,
163    Net: NetworkView + NetworkStreamControl + Clone + Send + Sync + 'static,
164{
165    packet_key: OffchainKeypair,
166    chain_key: ChainKeypair,
167    db: Db,
168    chain_api: Chain,
169    ping: Arc<OnceLock<Pinger>>,
170    network: Arc<OnceLock<Net>>,
171    graph: Graph,
172    path_planner: PathPlanner<MemorySurbStore, Chain, HoprGraphPathSelector<Graph>>,
173    my_multiaddresses: Vec<Multiaddr>,
174    smgr: SessionManager<Sender<(DestinationRouting, ApplicationDataOut)>, Sender<IncomingSession>>,
175    session_telemetry_tag_allocator: Arc<dyn hopr_transport_tag_allocator::TagAllocator + Send + Sync>,
176    probing_tag_allocator: Arc<dyn hopr_transport_tag_allocator::TagAllocator + Send + Sync>,
177    cfg: HoprProtocolConfig,
178}

180impl<Chain, Db, Graph, Net> HoprTransport<Chain, Db, Graph, Net>
181where
182    Db: HoprDbTicketOperations + Clone + Send + Sync + 'static,
183    Chain: ChainReadChannelOperations
184        + ChainReadAccountOperations
185        + ChainKeyOperations
186        + ChainValues
187        + Clone
188        + Send
189        + Sync
190        + 'static,
191    Graph: NetworkGraphView<NodeId = OffchainPublicKey>
192        + NetworkGraphUpdate
193        + hopr_api::graph::NetworkGraphTraverse<NodeId = OffchainPublicKey>
194        + Clone
195        + Send
196        + Sync
197        + 'static,
198    <Graph as NetworkGraphView>::Observed: hopr_api::graph::traits::EdgeObservableRead + Send,
199    <Graph as hopr_api::graph::NetworkGraphTraverse>::Observed:
200        hopr_api::graph::traits::EdgeObservableRead + Send + 'static,
201    Net: NetworkView + NetworkStreamControl + Clone + Send + Sync + 'static,
202{
203    pub fn new(
204        identity: (&ChainKeypair, &OffchainKeypair),
205        resolver: Chain,
206        db: Db,
207        graph: Graph,
208        my_multiaddresses: Vec<Multiaddr>,
209        cfg: HoprProtocolConfig,
210    ) -> errors::Result<Self> {
211        let me_offchain = *identity.1.public();
212        let planner_config = cfg.path_planner;
213        let selector = HoprGraphPathSelector::new(me_offchain, graph.clone(), planner_config.max_cached_paths);
214
215        let tag_allocators = hopr_transport_tag_allocator::create_allocators_from_config(&cfg.session.tag_allocator)?;
216
217        let mut session_tag_allocator = None;
218        let mut session_telemetry_tag_allocator = None;
219        let mut probing_tag_allocator = None;
220        for (usage, alloc) in tag_allocators {
221            match usage {
222                hopr_transport_tag_allocator::Usage::Session => session_tag_allocator = Some(alloc),
223                hopr_transport_tag_allocator::Usage::SessionTerminalTelemetry => {
224                    session_telemetry_tag_allocator = Some(alloc)
225                }
226                hopr_transport_tag_allocator::Usage::ProvingTelemetry => probing_tag_allocator = Some(alloc),
227            }
228        }
229        let session_tag_allocator = session_tag_allocator
230            .ok_or_else(|| errors::HoprTransportError::Api("session tag allocator missing".into()))?;
231        let session_telemetry_tag_allocator = session_telemetry_tag_allocator
232            .ok_or_else(|| errors::HoprTransportError::Api("session telemetry tag allocator missing".into()))?;
233        let probing_tag_allocator = probing_tag_allocator
234            .ok_or_else(|| errors::HoprTransportError::Api("probing tag allocator missing".into()))?;
235
236        Ok(Self {
237            packet_key: identity.1.clone(),
238            chain_key: identity.0.clone(),
239            ping: Arc::new(OnceLock::new()),
240            network: Arc::new(OnceLock::new()),
241            graph,
242            path_planner: PathPlanner::new(
243                me_offchain,
244                MemorySurbStore::new(cfg.packet.surb_store),
245                resolver.clone(),
246                selector,
247                planner_config,
248            ),
249            my_multiaddresses,
250            smgr: SessionManager::new(
251                SessionManagerConfig {
252                    frame_mtu: std::env::var("HOPR_SESSION_FRAME_SIZE")
253                        .ok()
254                        .and_then(|s| s.parse::<usize>().ok())
255                        .unwrap_or_else(|| SessionManagerConfig::default().frame_mtu)
256                        .max(ApplicationData::PAYLOAD_SIZE),
257                    max_frame_timeout: std::env::var("HOPR_SESSION_FRAME_TIMEOUT_MS")
258                        .ok()
259                        .and_then(|s| s.parse::<u64>().ok().map(Duration::from_millis))
260                        .unwrap_or_else(|| SessionManagerConfig::default().max_frame_timeout)
261                        .max(Duration::from_millis(100)),
262                    initiation_timeout_base: SESSION_INITIATION_TIMEOUT_BASE,
263                    idle_timeout: cfg.session.idle_timeout,
264                    balancer_sampling_interval: cfg.session.balancer_sampling_interval,
265                    initial_return_session_egress_rate: 10,
266                    minimum_surb_buffer_duration: cfg.session.balancer_minimum_surb_buffer_duration,
267                    maximum_surb_buffer_size: cfg.packet.surb_store.rb_capacity,
268                    surb_balance_notify_period: None,
269                    surb_target_notify: true,
270                },
271                session_tag_allocator,
272            ),
273            db,
274            chain_api: resolver,
275            session_telemetry_tag_allocator,
276            probing_tag_allocator,
277            cfg,
278        })
279    }

281    /// Execute all processes of the [`HoprTransport`] object.
282    ///
283    /// This method will spawn the `HoprTransportProcess::Heartbeat`,
284    /// `HoprTransportProcess::BloomFilterSave`, `HoprTransportProcess::Swarm` and session-related
285    /// processes and return join handles to the calling function. These processes are not started immediately but
286    /// are waiting for a trigger from this piece of code.
287    pub async fn run<T, Ct, NetBuilder>(
288        &self,
289        cover_traffic: Ct,
290        network_builder: NetBuilder,
291        ticket_events: T,
292        on_incoming_session: Sender<IncomingSession>,
293    ) -> errors::Result<(
294        HoprSocket<
295            futures::channel::mpsc::Receiver<ApplicationDataIn>,
296            futures::channel::mpsc::Sender<(DestinationRouting, ApplicationDataOut)>,
297        >,
298        AbortableList<HoprTransportProcess>,
299    )>
300    where
301        T: futures::Sink<TicketEvent> + Clone + Send + Unpin + 'static,
302        T::Error: std::error::Error,
303        Ct: ProbingTrafficGeneration + CoverTrafficGeneration + Send + Sync + 'static,
304        NetBuilder: NetworkBuilder<Network = Net> + Send + Sync + 'static,
305    {
306        let mut processes = AbortableList::<HoprTransportProcess>::default();
307
308        let (unresolved_routing_msg_tx, unresolved_routing_msg_rx) =
309            channel::<(DestinationRouting, ApplicationDataOut)>(MAXIMUM_MSG_OUTGOING_BUFFER_SIZE);
310
311        // -- transport medium
312
313        // NOTE: Private address filtering is implemented at multiple levels for defense-in-depth:
314        // 1. Discovery events are filtered before they reach the transport component
315        // 2. SwarmEvent::NewExternalAddrOfPeer events are filtered using is_public_address()
316        let allow_private_addresses = self.cfg.transport.prefer_local_addresses;
317        let (transport_network, transport_layer_process) = network_builder
318            .build(
319                &self.packet_key,
320                self.my_multiaddresses.clone(),
321                hopr_transport_protocol::CURRENT_HOPR_MSG_PROTOCOL,
322                allow_private_addresses,
323            )
324            .await
325            .map_err(|e| HoprTransportError::Api(e.to_string()))?;
326
327        self.network
328            .clone()
329            .set(transport_network.clone())
330            .map_err(|_| HoprTransportError::Api("transport network viewer already set".into()))?;
331
332        let msg_codec = hopr_transport_protocol::HoprBinaryCodec {};
333        let (wire_msg_tx, wire_msg_rx) =
334            hopr_transport_protocol::stream::process_stream_protocol(msg_codec, transport_network.clone()).await?;
335
336        let (mixing_channel_tx, mixing_channel_rx) =
337            hopr_transport_mixer::channel::<(PeerId, Box<[u8]>)>(build_mixer_cfg_from_env());
338
339        // the process is terminated when the input stream runs out
340        let _mixing_process_before_sending_out = spawn(
341            mixing_channel_rx
342                .inspect(|(peer, _)| tracing::trace!(%peer, "moving message from mixer to p2p stream"))
343                .map(Ok)
344                .forward(wire_msg_tx)
345                .inspect(|_| {
346                    tracing::warn!(
347                        task = "mixer -> egress process",
348                        "long-running background task finished"
349                    )
350                }),
351        );
352
353        // -- path cache background refresh (only when tokio runtime is available)
354        #[cfg(feature = "runtime-tokio")]
355        processes.insert(
356            HoprTransportProcess::PathRefresh,
357            spawn_as_abortable!(self.path_planner.run_background_refresh()),
358        );
359
360        processes.insert(
361            HoprTransportProcess::Medium,
362            spawn_as_abortable!(transport_layer_process().inspect(|_| tracing::warn!(
363                task = %HoprTransportProcess::Medium,
364                "long-running background task finished"
365            ))),
366        );
367
368        let msg_protocol_bidirectional_channel_capacity =
369            std::env::var("HOPR_INTERNAL_PROTOCOL_BIDIRECTIONAL_CHANNEL_CAPACITY")
370                .ok()
371                .and_then(|s| s.trim().parse::<usize>().ok())
372                .filter(|&c| c > 0)
373                .unwrap_or(16_384);
374
375        let (on_incoming_data_tx, on_incoming_data_rx) =
376            channel::<ApplicationDataIn>(msg_protocol_bidirectional_channel_capacity);
377
378        debug!(
379            capacity = msg_protocol_bidirectional_channel_capacity,
380            "creating protocol bidirectional channel"
381        );
382        let (tx_from_protocol, rx_from_protocol) =
383            channel::<(HoprPseudonym, ApplicationDataIn)>(msg_protocol_bidirectional_channel_capacity);
384
385        // === START === cover traffic control
386        // Allocate a cover traffic tag from the session telemetry partition to avoid
387        // collisions with session and probing tags.
388        let cover_traffic_allocated_tag = self
389            .session_telemetry_tag_allocator
390            .allocate()
391            .ok_or_else(|| HoprTransportError::Api("failed to allocate cover traffic tag".into()))?;
392        let cover_traffic_tag: Tag = cover_traffic_allocated_tag.value().into();
393
394        // filter out the known cover traffic not to lose processing time with it
395        // The allocated tag is moved into the closure to keep it alive for the transport lifetime.
396        let rx_from_protocol = rx_from_protocol.filter_map(move |(pseudonym, data)| {
397            let _keep_alive = &cover_traffic_allocated_tag;
398            async move { (data.data.application_tag != cover_traffic_tag).then_some((pseudonym, data)) }
399        });
400
401        // prepare a cover traffic stream
402        let cover_traffic_stream = CoverTrafficGeneration::build(&cover_traffic).filter_map(move |routing| {
403            let start =
404                hopr_api::types::crypto_random::random_integer(0, Some((RANDOM_DATA.len() - 100) as u64)) as usize;
405            let data = &RANDOM_DATA[start..start + 100];
406
407            futures::future::ready(if let Ok(data) = ApplicationData::new(cover_traffic_tag, data) {
408                Some((routing, ApplicationDataOut::with_no_packet_info(data)))
409            } else {
410                tracing::error!("failed to construct cover traffic packet");
411                None
412            })
413        });
414
415        // merge cover traffic with other outgoing data
416        let merged_unresolved_output_data =
417            select_with_strategy(unresolved_routing_msg_rx, cover_traffic_stream, |_: &mut ()| {
418                futures::stream::PollNext::Left
419            });
420
421        // === END === cover traffic control
422
423        // We have to resolve DestinationRouting -> ResolvedTransportRouting before
424        // sending the external packets to the transport pipeline.
425        let path_planner = self.path_planner.clone();
426        let distress_threshold = self.cfg.packet.surb_store.distress_threshold;
427        let all_resolved_external_msg_rx = merged_unresolved_output_data.filter_map(move |(unresolved, mut data)| {
428            let path_planner = path_planner.clone();
429            async move {
430                trace!(?unresolved, "resolving routing for packet");
431                match path_planner
432                    .resolve_routing(data.data.total_len(), data.estimate_surbs_with_msg(), unresolved)
433                    .await
434                {
435                    Ok((resolved, rem_surbs)) => {
436                        // Set the SURB distress/out-of-SURBs flag if applicable.
437                        // These flags are translated into HOPR protocol packet signals and are
438                        // applicable only on the return path.
439                        let mut signals_to_dst = data
440                            .packet_info
441                            .as_ref()
442                            .map(|info| info.signals_to_destination)
443                            .unwrap_or_default();
444
445                        if resolved.is_return() {
446                            signals_to_dst = match rem_surbs {
447                                Some(rem) if (1..distress_threshold.max(2)).contains(&rem) => {
448                                    signals_to_dst | PacketSignal::SurbDistress
449                                }
450                                Some(0) => signals_to_dst | PacketSignal::OutOfSurbs,
451                                _ => signals_to_dst - (PacketSignal::OutOfSurbs | PacketSignal::SurbDistress),
452                            };
453                        } else {
454                            // Unset these flags as they make no sense on the forward path.
455                            signals_to_dst -= PacketSignal::SurbDistress | PacketSignal::OutOfSurbs;
456                        }
457
458                        data.packet_info.get_or_insert_default().signals_to_destination = signals_to_dst;
459                        trace!(?resolved, "resolved routing for packet");
460                        Some((resolved, data))
461                    }
462                    Err(error) => {
463                        error!(%error, "failed to resolve routing");
464                        None
465                    }
466                }
467            }
468            .in_current_span()
469        });
470
471        let channels_dst = self
472            .chain_api
473            .domain_separators()
474            .await
475            .map_err(HoprTransportError::chain)?
476            .channel;
477
478        processes.extend_from(pipeline::run_hopr_packet_pipeline(
479            (self.packet_key.clone(), self.chain_key.clone()),
480            (mixing_channel_tx, wire_msg_rx),
481            (tx_from_protocol, all_resolved_external_msg_rx),
482            HoprPipelineComponents {
483                ticket_events,
484                surb_store: self.path_planner.surb_store.clone(),
485                chain_api: self.chain_api.clone(),
486                db: self.db.clone(),
487            },
488            channels_dst,
489            self.cfg.packet,
490        ));
491
492        // -- network probing
493        debug!(
494            capacity = msg_protocol_bidirectional_channel_capacity,
495            note = "same as protocol bidirectional",
496            "Creating probing channel"
497        );
498
499        let (tx_from_probing, rx_from_probing) =
500            channel::<(HoprPseudonym, ApplicationDataIn)>(msg_protocol_bidirectional_channel_capacity);
501
502        let manual_ping_channel_capacity = std::env::var("HOPR_INTERNAL_MANUAL_PING_CHANNEL_CAPACITY")
503            .ok()
504            .and_then(|s| s.trim().parse::<usize>().ok())
505            .filter(|&c| c > 0)
506            .unwrap_or(128);
507        debug!(capacity = manual_ping_channel_capacity, "Creating manual ping channel");
508        let (manual_ping_tx, manual_ping_rx) =
509            channel::<(OffchainPublicKey, PingQueryReplier)>(manual_ping_channel_capacity);
510
511        let probe = Probe::new(self.cfg.probe, self.probing_tag_allocator.clone());
512
513        let probing_processes = probe
514            .continuously_scan(
515                (unresolved_routing_msg_tx.clone(), rx_from_protocol),
516                manual_ping_rx,
517                tx_from_probing,
518                cover_traffic,
519                self.graph.clone(),
520            )
521            .await;
522
523        processes.flat_map_extend_from(probing_processes, HoprTransportProcess::Probing);
524
525        // manual ping
526        self.ping
527            .clone()
528            .set(Pinger::new(
529                PingConfig {
530                    timeout: self.cfg.probe.timeout,
531                },
532                manual_ping_tx,
533            ))
534            .map_err(|_| HoprTransportError::Api("must set the ticket aggregation writer only once".into()))?;
535
536        // -- session management
537        self.smgr
538            .start(unresolved_routing_msg_tx.clone(), on_incoming_session)
539            .map_err(|_| HoprTransportError::Api("failed to start session manager".into()))?
540            .into_iter()
541            .enumerate()
542            .map(|(i, jh)| (HoprTransportProcess::SessionsManagement(i + 1), jh))
543            .for_each(|(k, v)| {
544                processes.insert(k, v);
545            });
546
547        let smgr = self.smgr.clone();
548        processes.insert(
549            HoprTransportProcess::SessionsManagement(0),
550            spawn_as_abortable!(
551                rx_from_probing
552                    .filter_map(move |(pseudonym, data)| {
553                        let smgr = smgr.clone();
554                        async move {
555                            match smgr.dispatch_message(pseudonym, data).await {
556                                Ok(DispatchResult::Processed) => {
557                                    tracing::trace!("message dispatch completed");
558                                    None
559                                }
560                                Ok(DispatchResult::Unrelated(data)) => {
561                                    tracing::trace!("unrelated message dispatch completed");
562                                    Some(data)
563                                }
564                                Err(error) => {
565                                    tracing::error!(%error, "error while dispatching packet in the session manager");
566                                    None
567                                }
568                            }
569                        }
570                    })
571                    .map(Ok)
572                    .forward(on_incoming_data_tx)
573                    .inspect(|_| tracing::warn!(
574                        task = %HoprTransportProcess::SessionsManagement(0),
575                        "long-running background task finished"
576                    ))
577            ),
578        );
579
580        Ok(((on_incoming_data_rx, unresolved_routing_msg_tx).into(), processes))
581    }

583    #[tracing::instrument(level = "debug", skip(self))]
584    pub async fn ping(
585        &self,
586        peer: &OffchainPublicKey,
587    ) -> errors::Result<(std::time::Duration, <Graph as NetworkGraphView>::Observed)> {
588        let me: &OffchainPublicKey = self.packet_key.public();
589        if peer == me {
590            return Err(HoprTransportError::Api("ping to self does not make sense".into()));
591        }
592
593        let pinger = self
594            .ping
595            .get()
596            .ok_or_else(|| HoprTransportError::Api("ping processing is not yet initialized".into()))?;
597
598        let latency = (*pinger).ping(peer).await?;
599
600        if let Some(observations) = self.graph.edge(me, peer) {
601            Ok((latency, observations))
602        } else {
603            Err(HoprTransportError::Api(format!(
604                "no observations available for peer {peer}",
605            )))
606        }
607    }

650    #[tracing::instrument(level = "debug", skip(self))]
651    pub fn announceable_multiaddresses(&self) -> Vec<Multiaddr> {
652        let mut mas = self
653            .local_multiaddresses()
654            .into_iter()
655            .filter(|ma| {
656                hopr_transport_identity::multiaddrs::is_supported(ma)
657                    && (self.cfg.transport.announce_local_addresses || is_public_address(ma))
658            })
659            .map(|ma| strip_p2p_protocol(&ma))
660            .filter(|v| !v.is_empty())
661            .collect::<Vec<_>>();
662
663        mas.sort_by(|l, r| {
664            let is_left_dns = hopr_transport_identity::multiaddrs::is_dns(l);
665            let is_right_dns = hopr_transport_identity::multiaddrs::is_dns(r);
666
667            if !(is_left_dns ^ is_right_dns) {
668                std::cmp::Ordering::Equal
669            } else if is_left_dns {
670                std::cmp::Ordering::Less
671            } else {
672                std::cmp::Ordering::Greater
673            }
674        });
675
676        mas
677    }

762                    } else {
763                        None
764                    }
765                } else {
766                    None
767                }
768            })
769            .collect::<Vec<_>>())

787fn build_mixer_cfg_from_env() -> MixerConfig {
788    let mixer_cfg = MixerConfig {
789        min_delay: std::time::Duration::from_millis(
790            std::env::var("HOPR_INTERNAL_MIXER_MINIMUM_DELAY_IN_MS")
791                .map(|v| {
792                    v.trim()
793                        .parse::<u64>()
794                        .unwrap_or(hopr_transport_mixer::config::HOPR_MIXER_MINIMUM_DEFAULT_DELAY_IN_MS)
795                })
796                .unwrap_or(hopr_transport_mixer::config::HOPR_MIXER_MINIMUM_DEFAULT_DELAY_IN_MS),
797        ),
798        delay_range: std::time::Duration::from_millis(
799            std::env::var("HOPR_INTERNAL_MIXER_DELAY_RANGE_IN_MS")
800                .map(|v| {
801                    v.trim()
802                        .parse::<u64>()
803                        .unwrap_or(hopr_transport_mixer::config::HOPR_MIXER_DEFAULT_DELAY_RANGE_IN_MS)
804                })
805                .unwrap_or(hopr_transport_mixer::config::HOPR_MIXER_DEFAULT_DELAY_RANGE_IN_MS),
806        ),
807        capacity: {
808            let capacity = std::env::var("HOPR_INTERNAL_MIXER_CAPACITY")
809                .ok()
810                .and_then(|s| s.trim().parse::<usize>().ok())
811                .filter(|&c| c > 0)
812                .unwrap_or(hopr_transport_mixer::config::HOPR_MIXER_CAPACITY);
813            debug!(capacity = capacity, "Setting mixer capacity");
814            capacity
815        },
816        ..MixerConfig::default()
817    };
818    debug!(?mixer_cfg, "Mixer configuration");
819
820    mixer_cfg
