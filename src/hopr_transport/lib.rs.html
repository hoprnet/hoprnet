1//! The crate aggregates and composes individual transport level objects and functionality
2//! into a unified [`crate::HoprTransport`] object with the goal of isolating the transport layer
3//! and defining a fully specified transport API.
4//!
5//! See also the `hopr_protocol_start` crate for details on Start sub-protocol which initiates a Session.
6//!
7//! As such, the transport layer components should be only those that are directly necessary to:
8//!
9//! 1. send and receive a packet, acknowledgement or ticket aggregation request
10//! 2. send and receive a network telemetry request
11//! 3. automate transport level processes
12//! 4. algorithms associated with the transport layer operational management
13//! 5. interface specifications to allow modular behavioral extensions
14

159impl<Db> TicketAggregatorTrait for TicketAggregatorProxy<Db>

162{
163    async fn aggregate_tickets(
164        &self,
165        channel: &Hash,
166        prerequisites: AggregationPrerequisites,
167    ) -> hopr_transport_ticket_aggregation::Result<()> {
168        if let Some(writer) = self.maybe_writer.clone().get() {
169            AwaitingAggregator::new(self.db.clone(), writer.clone(), self.agg_timeout)

174                "Ticket aggregation writer not available, the object was not yet initialized".to_string(),
175            ))
176        }
177    }

183/// Interface into the physical transport mechanism allowing all off-chain HOPR-related tasks on
184/// the transport, as well as off-chain ticket manipulation.
185pub struct HoprTransport<T>
186where
187    T: HoprDbAllOperations + PathAddressResolver + std::fmt::Debug + Clone + Send + Sync + 'static,
188{
189    me: OffchainKeypair,
190    me_peerid: PeerId, // Cache to avoid an expensive conversion: OffchainPublicKey -> PeerId
191    me_address: Address,
192    cfg: HoprTransportConfig,
193    db: T,
194    ping: Arc<OnceLock<Pinger>>,
195    network: Arc<Network<T>>,
196    process_packet_send: Arc<OnceLock<MsgSender<Sender<SendMsgInput>>>>,
197    path_planner: PathPlanner<T, CurrentPathSelector>,
198    my_multiaddresses: Vec<Multiaddr>,
199    process_ticket_aggregate:
200        Arc<OnceLock<TicketAggregationActions<TicketAggregationResponseType, TicketAggregationRequestType>>>,
201    smgr: SessionManager<Sender<(DestinationRouting, ApplicationDataOut)>, Sender<IncomingSession>>,
202}

208    pub fn new(
209        me: &OffchainKeypair,

212        db: T,
213        channel_graph: Arc<RwLock<hopr_path::channel_graph::ChannelGraph>>,

215    ) -> Self {
216        let process_packet_send = Arc::new(OnceLock::new());
217
218        let me_peerid: PeerId = me.into();
219        let me_chain_addr = me_onchain.public().to_address();
220
221        Self {
222            me: me.clone(),
223            me_peerid,
224            me_address: me_chain_addr,

233            path_planner: PathPlanner::new(
234                me_chain_addr,
235                db.clone(),
236                CurrentPathSelector::new(
237                    channel_graph.clone(),
238                    DfsPathSelectorConfig {
239                        node_score_threshold: cfg.network.node_score_auto_path_threshold,
240                        max_first_hop_latency: cfg.network.max_first_hop_latency_threshold,
241                        ..Default::default()
242                    },
243                ),
244                channel_graph.clone(),
245            ),
246            my_multiaddresses,
247            process_ticket_aggregate: Arc::new(OnceLock::new()),
248            smgr: SessionManager::new(SessionManagerConfig {
249                // TODO(v3.1): Use the entire range of tags properly
250                session_tag_range: (16..65535),
251                maximum_sessions: cfg.session.maximum_sessions as usize,

258                    .ok()
259                    .and_then(|s| s.parse::<u64>().ok().map(Duration::from_millis))
260                    .unwrap_or_else(|| SessionManagerConfig::default().max_frame_timeout)
261                    .max(Duration::from_millis(100)),
262                initiation_timeout_base: SESSION_INITIATION_TIMEOUT_BASE,
263                idle_timeout: cfg.session.idle_timeout,
264                balancer_sampling_interval: cfg.session.balancer_sampling_interval,
265                initial_return_session_egress_rate: 10,
266                minimum_surb_buffer_duration: Duration::from_secs(5),
267                maximum_surb_buffer_size: db.get_surb_config().rb_capacity,
268                // Allow a 10% increase of the target SURB buffer on incoming Sessions
269                // if the SURB buffer level has surpassed it by at least 10% in the last 2 minutes.
270                growable_target_surb_buffer: Some((Duration::from_secs(120), 0.10)),
271            }),
272            db,
273            cfg,
274        }
275    }

277    /// Execute all processes of the [`crate::HoprTransport`] object.
278    ///
279    /// This method will spawn the [`crate::HoprTransportProcess::Heartbeat`],
280    /// [`crate::HoprTransportProcess::BloomFilterSave`], [`crate::HoprTransportProcess::Swarm`] and session-related
281    /// processes and return join handles to the calling function. These processes are not started immediately but
282    /// are waiting for a trigger from this piece of code.
283    #[allow(clippy::too_many_arguments)]
284    pub async fn run<S1, S2>(
285        &self,
286        me_onchain: &ChainKeypair,
287        on_incoming_data: S1,
288        discovery_updates: S2,
289        on_incoming_session: Sender<IncomingSession>,
290    ) -> crate::errors::Result<HashMap<HoprTransportProcess, AbortHandle>>
291    where
292        S1: futures::Sink<ApplicationDataIn, Error = SendError> + Send + 'static,
293        S2: futures::Stream<Item = PeerDiscovery> + Send + 'static,
294    {
295        info!("Loading initial peers from the storage");
296        let mut addresses: HashSet<Multiaddr> = HashSet::new();
297        let nodes = self.get_public_nodes().await?;
298
299        // Calculate the minimum capacity based on public nodes (each node can generate 2 messages)
300        // plus 100 as additional buffer
301        let minimum_capacity = nodes.len().saturating_mul(2).saturating_add(100);
302
303        let internal_discovery_updates_capacity = std::env::var("HOPR_INTERNAL_DISCOVERY_UPDATES_CAPACITY")
304            .ok()
305            .and_then(|s| s.trim().parse::<usize>().ok())
306            .filter(|&c| c >= 2048)
307            .unwrap_or(2048)
308            .max(minimum_capacity);
309
310        debug!(
311            capacity = internal_discovery_updates_capacity,
312            minimum_required = minimum_capacity,
313            "Creating internal discovery updates channel"
314        );
315        let (mut internal_discovery_update_tx, internal_discovery_update_rx) =
316            futures::channel::mpsc::channel::<PeerDiscovery>(internal_discovery_updates_capacity);
317
318        let network_clone = self.network.clone();
319        let db_clone = self.db.clone();
320        let me_peerid = self.me_peerid;
321        let discovery_updates =
322            futures_concurrency::stream::StreamExt::merge(discovery_updates, internal_discovery_update_rx)
323                .filter_map(move |event| {
324                    let network = network_clone.clone();
325                    let db = db_clone.clone();
326                    let me = me_peerid;
327
328                    async move {
329                        match event {
330                            PeerDiscovery::Allow(peer_id) => {
331                                debug!(peer = %peer_id, "Processing peer discovery event: Allow");
332
333                                // PeerId -> OffchainPublicKey is a CPU-intensive blocking operation
334                                if let Ok(pubkey) = hopr_parallelize::cpu::spawn_blocking(move || OffchainPublicKey::from_peerid(&peer_id))
335                                    .await {
336                                    if !network.has(&peer_id).await {
337                                        let mas = db
338                                            .get_account(None, hopr_db_sql::accounts::ChainOrPacketKey::PacketKey(pubkey))
339                                            .await
340                                            .map(|entry| {
341                                                entry
342                                                    .map(|v| Vec::from_iter(v.get_multiaddr().into_iter()))
343                                                    .unwrap_or_default()
344                                            })
345                                            .unwrap_or_default();
346
347                                        if let Err(e) = network.add(&peer_id, PeerOrigin::NetworkRegistry, mas).await {
348                                            error!(peer = %peer_id, error = %e, "Failed to allow locally (already allowed on-chain)");
349                                            return None;
350                                        }
351                                    }
352
353                                    return Some(PeerDiscovery::Allow(peer_id))
354                                } else {
355                                    error!(peer = %peer_id, "Failed to allow locally (already allowed on-chain): peer id not convertible to off-chain public key")
356                                }
357                            }
358                            PeerDiscovery::Ban(peer_id) => {
359                                if let Err(e) = network.remove(&peer_id).await {
360                                    error!(peer = %peer_id, error = %e, "Failed to ban locally (already banned on-chain)")
361                                } else {
362                                    return Some(PeerDiscovery::Ban(peer_id))
363                                }
364                            }
365                            PeerDiscovery::Announce(peer, multiaddresses) => {
366                                debug!(peer = %peer, ?multiaddresses, "Processing peer discovery event: Announce");
367                                if peer != me {
368                                    // decapsulate the `p2p/<peer_id>` to remove duplicities
369                                    let mas = multiaddresses
370                                        .into_iter()
371                                        .map(|ma| strip_p2p_protocol(&ma))
372                                        .filter(|v| !v.is_empty())
373                                        .collect::<Vec<_>>();
374
375                                    if ! mas.is_empty() {
376                                        // PeerId -> OffchainPublicKey is a CPU-intensive blocking operation
377                                        if let Ok(pubkey) = hopr_parallelize::cpu::spawn_blocking(move || OffchainPublicKey::from_peerid(&peer))
378                                            .await {
379                                            if let Ok(Some(key)) = db.translate_key(None, hopr_db_sql::accounts::ChainOrPacketKey::PacketKey(pubkey)).await {
380                                                let key: Result<Address, _> = key.try_into();
381
382                                                if let Ok(key) = key {
383                                                    if db
384                                                        .is_allowed_in_network_registry(None, &key)
385                                                        .await
386                                                        .unwrap_or(false)
387                                                    {
388                                                        if let Err(e) = network.add(&peer, PeerOrigin::NetworkRegistry, mas.clone()).await
389                                                        {
390                                                            error!(%peer, error = %e, "failed to record peer from the NetworkRegistry");
391                                                        } else {
392                                                            return Some(PeerDiscovery::Announce(peer, mas))
393                                                        }
394                                                    }
395                                                }
396                                            } else {
397                                                error!(%peer, "Failed to announce peer due to convertibility error");
398                                            }
399                                        }
400                                    }
401                                }
402                            }
403                        }
404
405                        None
406                    }
407                });
408
409        for (peer, _address, multiaddresses) in nodes {
410            if self.is_allowed_to_access_network(either::Left(&peer)).await? {
411                debug!(%peer, ?multiaddresses, "Using initial public node");
412                addresses.extend(multiaddresses.clone());
413
414                internal_discovery_update_tx
415                    .send(PeerDiscovery::Announce(peer, multiaddresses.clone()))
416                    .await
417                    .map_err(|e| HoprTransportError::Api(e.to_string()))?;
418
419                internal_discovery_update_tx
420                    .send(PeerDiscovery::Allow(peer))
421                    .await
422                    .map_err(|e| HoprTransportError::Api(e.to_string()))?;
423            }
424        }
425
426        let mut processes: HashMap<HoprTransportProcess, AbortHandle> = HashMap::new();
427
428        let ticket_agg_proc = TicketAggregationInteraction::new(self.db.clone(), me_onchain);
429        let tkt_agg_writer = ticket_agg_proc.writer();
430
431        let (external_msg_send, external_msg_rx) =
432            channel::<(ApplicationDataOut, ResolvedTransportRouting, PacketSendFinalizer)>(
433                MAXIMUM_MSG_OUTGOING_BUFFER_SIZE,
434            );
435
436        self.process_packet_send
437            .clone()
438            .set(MsgSender::new(external_msg_send.clone()))
439            .expect("must set the packet processing writer only once");
440
441        self.process_ticket_aggregate
442            .clone()
443            .set(tkt_agg_writer.clone())
444            .expect("must set the ticket aggregation writer only once");
445
446        // -- transport medium
447        let mixer_cfg = build_mixer_cfg_from_env();
448
449        #[cfg(feature = "mixer-channel")]
450        let (mixing_channel_tx, mixing_channel_rx) = hopr_transport_mixer::channel::<(PeerId, Box<[u8]>)>(mixer_cfg);
451
452        #[cfg(feature = "mixer-stream")]
453        let (mixing_channel_tx, mixing_channel_rx) = {
454            let (tx, rx) = futures::channel::mpsc::channel::<(PeerId, Box<[u8]>)>(MAXIMUM_MSG_OUTGOING_BUFFER_SIZE);
455            let rx = rx.then_concurrent(move |v| {
456                let cfg = mixer_cfg;
457
458                async move {
459                    let random_delay = cfg.random_delay();
460                    trace!(delay_in_ms = random_delay.as_millis(), "Created random mixer delay",);
461
462                    #[cfg(all(feature = "prometheus", not(test)))]
463                    hopr_transport_mixer::channel::METRIC_QUEUE_SIZE.decrement(1.0f64);
464
465                    sleep(random_delay).await;
466
467                    #[cfg(all(feature = "prometheus", not(test)))]
468                    {
469                        hopr_transport_mixer::channel::METRIC_QUEUE_SIZE.decrement(1.0f64);
470
471                        let weight = 1.0f64 / cfg.metric_delay_window as f64;
472                        hopr_transport_mixer::channel::METRIC_MIXER_AVERAGE_DELAY.set(
473                            (weight * random_delay.as_millis() as f64)
474                                + ((1.0f64 - weight) * hopr_transport_mixer::channel::METRIC_MIXER_AVERAGE_DELAY.get()),
475                        );
476                    }
477
478                    v
479                }
480            });
481
482            (tx, rx)
483        };
484
485        let mut transport_layer =
486            HoprSwarm::new((&self.me).into(), discovery_updates, self.my_multiaddresses.clone()).await;
487
488        if let Some(port) = self.cfg.protocol.autonat_port {
489            transport_layer.run_nat_server(port);
490        }
491
492        if addresses.is_empty() {
493            warn!("No addresses found in the database, not dialing any NAT servers");
494        } else {
495            info!(num_addresses = addresses.len(), "Found addresses from the database");
496            let mut randomized_addresses: Vec<_> = addresses.into_iter().collect();
497            randomized_addresses.shuffle(&mut rand::thread_rng());
498            transport_layer.dial_nat_server(randomized_addresses);
499        }
500
501        let msg_proto_control =
502            transport_layer.build_protocol_control(hopr_transport_protocol::CURRENT_HOPR_MSG_PROTOCOL);
503        let msg_codec = hopr_transport_protocol::HoprBinaryCodec {};
504        let (wire_msg_tx, wire_msg_rx) =
505            hopr_transport_protocol::stream::process_stream_protocol(msg_codec, msg_proto_control).await?;
506
507        let _mixing_process_before_sending_out = hopr_async_runtime::prelude::spawn(
508            mixing_channel_rx
509                .inspect(|(peer, _)| tracing::trace!(%peer, "moving message from mixer to p2p stream"))
510                .map(Ok)
511                .forward(wire_msg_tx)
512                .inspect(|_| {
513                    tracing::warn!(
514                        task = "mixer -> egress process",
515                        "long-running background task finished"
516                    )
517                }),
518        );
519
520        let (transport_events_tx, transport_events_rx) =
521            futures::channel::mpsc::channel::<hopr_transport_p2p::DiscoveryEvent>(2048);
522
523        let network_clone = self.network.clone();
524        spawn(
525            transport_events_rx
526                .for_each(move |event| {
527                    let network = network_clone.clone();
528
529                    async move {
530                        match event {
531                            hopr_transport_p2p::DiscoveryEvent::IncomingConnection(peer, multiaddr) => {
532                                if let Err(error) = network
533                                    .add(&peer, PeerOrigin::IncomingConnection, vec![multiaddr])
534                                    .await
535                                {
536                                    tracing::error!(%peer, %error, "Failed to add incoming connection peer");
537                                }
538                            }
539                            hopr_transport_p2p::DiscoveryEvent::FailedDial(peer) => {
540                                if let Err(error) = network
541                                    .update(&peer, Err(hopr_transport_network::network::UpdateFailure::DialFailure))
542                                    .await
543                                {
544                                    tracing::error!(%peer, %error, "Failed to update peer status after failed dial");
545                                }
546                            }
547                        }
548                    }
549                })
550                .inspect(|_| {
551                    tracing::warn!(
552                        task = "transport events recording",
553                        "long-running background task finished"
554                    )
555                }),
556        );
557
558        processes.insert(
559            HoprTransportProcess::Medium,
560            spawn_as_abortable!(transport_layer.run(transport_events_tx).inspect(|_| tracing::warn!(
561                task = %HoprTransportProcess::Medium,
562                "long-running background task finished"
563            ))),
564        );
565
566        // -- msg-ack protocol over the wire transport
567        let packet_cfg = PacketInteractionConfig {
568            packet_keypair: self.me.clone(),
569            outgoing_ticket_win_prob: self
570                .cfg
571                .protocol
572                .outgoing_ticket_winning_prob
573                .map(WinningProbability::try_from)
574                .transpose()?,
575            outgoing_ticket_price: self.cfg.protocol.outgoing_ticket_price,
576        };
577
578        let msg_protocol_bidirectional_channel_capacity =
579            std::env::var("HOPR_INTERNAL_PROTOCOL_BIDIRECTIONAL_CHANNEL_CAPACITY")
580                .ok()
581                .and_then(|s| s.trim().parse::<usize>().ok())
582                .filter(|&c| c >= 16_384)
583                .unwrap_or(16_384);
584
585        debug!(
586            capacity = msg_protocol_bidirectional_channel_capacity,
587            "Creating protocol bidirectional channel"
588        );
589        let (tx_from_protocol, rx_from_protocol) =
590            channel::<(HoprPseudonym, ApplicationDataIn)>(msg_protocol_bidirectional_channel_capacity);
591        for (k, v) in hopr_transport_protocol::run_msg_ack_protocol(
592            packet_cfg,
593            self.db.clone(),
594            (
595                mixing_channel_tx.with(|(peer, msg): (PeerId, Box<[u8]>)| {
596                    trace!(%peer, "sending message to peer");
597                    futures::future::ok::<_, hopr_transport_mixer::channel::SenderError>((peer, msg))
598                }),
599                wire_msg_rx.inspect(|(peer, _)| trace!(%peer, "received message from peer")),
600            ),
601            (tx_from_protocol, external_msg_rx),
602        )
603        .await
604        .into_iter()
605        {
606            processes.insert(HoprTransportProcess::Protocol(k), v);
607        }
608
609        // -- network probing
610        debug!(
611            capacity = msg_protocol_bidirectional_channel_capacity,
612            note = "same as protocol bidirectional",
613            "Creating probing channel"
614        );
615        let (tx_from_probing, rx_from_probing) =
616            channel::<(HoprPseudonym, ApplicationDataIn)>(msg_protocol_bidirectional_channel_capacity);
617
618        let manual_ping_channel_capacity = std::env::var("HOPR_INTERNAL_MANUAL_PING_CHANNEL_CAPACITY")
619            .ok()
620            .and_then(|s| s.trim().parse::<usize>().ok())
621            .filter(|&c| c >= 128)
622            .unwrap_or(128);
623        debug!(capacity = manual_ping_channel_capacity, "Creating manual ping channel");
624        let (manual_ping_tx, manual_ping_rx) = channel::<(PeerId, PingQueryReplier)>(manual_ping_channel_capacity);
625
626        let probe = Probe::new((*self.me.public(), self.me_address), self.cfg.probe);
627        for (k, v) in probe
628            .continuously_scan(
629                (external_msg_send, rx_from_protocol),
630                manual_ping_rx,
631                network_notifier::ProbeNetworkInteractions::new(
632                    self.network.clone(),
633                    self.db.clone(),
634                    self.path_planner.channel_graph(),
635                ),
636                DbProxy::new(self.db.clone()),
637                tx_from_probing,
638            )
639            .await
640            .into_iter()
641        {
642            processes.insert(HoprTransportProcess::Probing(k), v);
643        }
644
645        // manual ping
646        self.ping
647            .clone()
648            .set(Pinger::new(
649                PingConfig {
650                    timeout: self.cfg.probe.timeout,
651                },
652                manual_ping_tx,
653            ))
654            .expect("must set the ticket aggregation writer only once");
655
656        // -- session management
657        let packet_planner = run_packet_planner(
658            self.path_planner.clone(),
659            self.process_packet_send
660                .get()
661                .cloned()
662                .expect("packet sender must be set"),
663        );
664
665        self.smgr
666            .start(packet_planner, on_incoming_session)
667            .expect("failed to start session manager")
668            .into_iter()
669            .enumerate()
670            .map(|(i, jh)| (HoprTransportProcess::SessionsManagement(i + 1), jh))
671            .for_each(|(k, v)| {
672                processes.insert(k, v);
673            });
674
675        let smgr = self.smgr.clone();
676        processes.insert(
677            HoprTransportProcess::SessionsManagement(0),
678            spawn_as_abortable!(
679                StreamExt::filter_map(rx_from_probing, move |(pseudonym, data)| {
680                    let smgr = smgr.clone();
681                    async move {
682                        match smgr.dispatch_message(pseudonym, data).await {
683                            Ok(DispatchResult::Processed) => {
684                                trace!("message dispatch completed");
685                                None
686                            }
687                            Ok(DispatchResult::Unrelated(data)) => {
688                                trace!("unrelated message dispatch completed");
689                                Some(data)
690                            }
691                            Err(e) => {
692                                error!(error = %e, "error while processing packet");
693                                None
694                            }
695                        }
696                    }
697                })
698                .map(Ok)
699                .forward(on_incoming_data)
700                .inspect(|_| tracing::warn!(
701                    task = %HoprTransportProcess::SessionsManagement(0),
702                    "long-running background task finished"
703                ))
704            ),
705        );
706
707        Ok(processes)
708    }

718    #[tracing::instrument(level = "debug", skip(self))]
719    pub async fn ping(&self, peer: &PeerId) -> errors::Result<(std::time::Duration, PeerStatus)> {
720        if !self.is_allowed_to_access_network(either::Left(peer)).await? {
721            return Err(HoprTransportError::Api(format!(
722                "ping to '{peer}' not allowed due to network registry"
723            )));
724        }
725
726        if peer == &self.me_peerid {
727            return Err(HoprTransportError::Api("ping to self does not make sense".into()));
728        }
729
730        let pinger = self
731            .ping
732            .get()
733            .ok_or_else(|| HoprTransportError::Api("ping processing is not yet initialized".into()))?;
734
735        if let Err(e) = self.network.add(peer, PeerOrigin::ManualPing, vec![]).await {
736            error!(error = %e, "Failed to store the peer observation");
737        }
738
739        let latency = (*pinger).ping(*peer).await?;
740
741        let peer_status = self.network.get(peer).await?.ok_or(HoprTransportError::Probe(
742            hopr_transport_probe::errors::ProbeError::NonExistingPeer,
743        ))?;
744
745        Ok((latency, peer_status))
746    }

775    #[tracing::instrument(level = "info", skip(self, msg), fields(uuid = uuid::Uuid::new_v4().to_string()))]
776    pub async fn send_message(&self, msg: Box<[u8]>, routing: DestinationRouting, tag: Tag) -> errors::Result<()> {
777        if let Tag::Reserved(reserved_tag) = tag {
778            return Err(HoprTransportError::Api(format!(
779                "Application tag must not from range: {:?}, but was {reserved_tag:?}",
780                Tag::APPLICATION_TAG_RANGE
781            )));
782        }
783
784        if msg.len() > HoprPacket::PAYLOAD_SIZE {
785            return Err(HoprTransportError::Api(format!(
786                "Message exceeds the maximum allowed size of {} bytes",
787                HoprPacket::PAYLOAD_SIZE
788            )));
789        }
790
791        let app_data = ApplicationData::new(tag, msg.into_vec())?;
792        let routing = self
793            .path_planner
794            .resolve_routing(app_data.total_len(), usize::MAX, routing)
795            .await?
796            .0;
797
798        // Here we do not use msg_sender directly,
799        // since it internally follows Session-oriented logic
800        let sender = self.process_packet_send.get().ok_or_else(|| {
801            HoprTransportError::Api("send msg: failed because message processing is not yet initialized".into())
802        })?;
803
804        sender
805            .send_packet(ApplicationDataOut::with_no_packet_info(app_data), routing)
806            .await
807            .map_err(|e| HoprTransportError::Api(format!("send msg failed to enqueue msg: {e}")))?
808            .consume_and_wait(crate::constants::PACKET_QUEUE_TIMEOUT_MILLISECONDS)
809            .await
810            .map_err(|e| HoprTransportError::Api(e.to_string()))
811    }

865    pub async fn is_allowed_to_access_network<'a>(
866        &self,
867        address_like: either::Either<&'a PeerId, Address>,
868    ) -> errors::Result<bool>
869    where
870        T: 'a,
871    {
872        let db_clone = self.db.clone();
873        let address_like_noref = address_like.map_left(|peer| *peer);
874
875        Ok(self
876            .db
877            .begin_transaction()
878            .await
879            .map_err(hopr_db_sql::api::errors::DbError::from)?
880            .perform(|tx| {
881                Box::pin(async move {
882                    match address_like_noref {
883                        either::Left(peer) => {
884                            // PeerId -> OffchainPublicKey is a CPU-intensive blocking operation
885                            let pubkey =
886                                hopr_parallelize::cpu::spawn_blocking(move || OffchainPublicKey::from_peerid(&peer))
887                                    .await
888                                    .map_err(|e| hopr_db_sql::api::errors::DbError::General(e.to_string()))?;
889                            if let Some(address) = db_clone.translate_key(Some(tx), pubkey).await? {
890                                db_clone.is_allowed_in_network_registry(Some(tx), &address).await
891                            } else {
892                                Err(hopr_db_sql::errors::DbSqlError::LogicalError(
893                                    "cannot translate off-chain key".into(),
894                                ))
895                            }
896                        }
897                        either::Right(address) => db_clone.is_allowed_in_network_registry(Some(tx), &address).await,
898                    }
899                })
900            })
901            .await
902            .map_err(hopr_db_sql::api::errors::DbError::from)?)
903    }

918    #[tracing::instrument(level = "debug", skip(self))]
919    pub fn announceable_multiaddresses(&self) -> Vec<Multiaddr> {
920        let mut mas = self
921            .local_multiaddresses()
922            .into_iter()
923            .filter(|ma| {
924                hopr_transport_identity::multiaddrs::is_supported(ma)
925                    && (self.cfg.transport.announce_local_addresses
926                        || !hopr_transport_identity::multiaddrs::is_private(ma))
927            })
928            .map(|ma| strip_p2p_protocol(&ma))
929            .filter(|v| !v.is_empty())
930            .collect::<Vec<_>>();
931
932        mas.sort_by(|l, r| {
933            let is_left_dns = hopr_transport_identity::multiaddrs::is_dns(l);
934            let is_right_dns = hopr_transport_identity::multiaddrs::is_dns(r);
935
936            if !(is_left_dns ^ is_right_dns) {
937                std::cmp::Ordering::Equal
938            } else if is_left_dns {
939                std::cmp::Ordering::Less
940            } else {
941                std::cmp::Ordering::Greater
942            }
943        });
944
945        mas
946    }
