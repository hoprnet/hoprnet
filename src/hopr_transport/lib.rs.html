1//! The crate aggregates and composes individual transport level objects and functionality
2//! into a unified [`crate::HoprTransport`] object with the goal of isolating the transport layer
3//! and defining a fully specified transport API.
4//!
5//! See also the `hopr_protocol_start` crate for details on Start sub-protocol which initiates a Session.
6//!
7//! As such, the transport layer components should be only those that are directly necessary to:
8//!
9//! 1. send and receive a packet, acknowledgement or ticket aggregation request
10//! 2. send and receive a network telemetry request
11//! 3. automate transport level processes
12//! 4. algorithms associated with the transport layer operational management
13//! 5. interface specifications to allow modular behavioral extensions
14

116/// Interface into the physical transport mechanism allowing all off-chain HOPR-related tasks on
117/// the transport, as well as off-chain ticket manipulation.
118pub struct HoprTransport<Db, R> {
119    me: OffchainKeypair,
120    me_peerid: PeerId, // Cache to avoid an expensive conversion: OffchainPublicKey -> PeerId
121    me_address: Address,
122    cfg: HoprTransportConfig,
123    db: Db,
124    resolver: R,
125    ping: Arc<OnceLock<Pinger>>,
126    network: Arc<Network<Db>>,
127    path_planner: PathPlanner<Db, R, CurrentPathSelector>,
128    my_multiaddresses: Vec<Multiaddr>,
129    smgr: SessionManager<Sender<(DestinationRouting, ApplicationDataOut)>, Sender<IncomingSession>>,
130}
131
132impl<Db, R> HoprTransport<Db, R>
133where
134    Db: HoprDbTicketOperations + HoprDbPeersOperations + HoprDbProtocolOperations + Clone + Send + Sync + 'static,
135    R: ChainReadChannelOperations
136        + ChainReadAccountOperations
137        + ChainKeyOperations
138        + ChainValues
139        + Clone
140        + Send
141        + Sync
142        + 'static,
143{
144    pub fn new(
145        me: &OffchainKeypair,
146        me_onchain: &ChainKeypair,
147        cfg: HoprTransportConfig,
148        db: Db,
149        resolver: R,
150        my_multiaddresses: Vec<Multiaddr>,
151    ) -> Self {
152        let me_peerid: PeerId = me.into();
153        let me_chain_addr = me_onchain.public().to_address();
154
155        Self {
156            me: me.clone(),
157            me_peerid,
158            me_address: me_chain_addr,
159            ping: Arc::new(OnceLock::new()),
160            network: Arc::new(Network::new(
161                me_peerid,
162                my_multiaddresses.clone(),
163                cfg.network.clone(),
164                db.clone(),
165            )),
166            path_planner: PathPlanner::new(
167                me_chain_addr,
168                db.clone(),
169                resolver.clone(),
170                CurrentPathSelector::default(),
171            ),
172            my_multiaddresses,
173            smgr: SessionManager::new(SessionManagerConfig {
174                // TODO(v3.1): Use the entire range of tags properly
175                session_tag_range: (16..65535),
176                maximum_sessions: cfg.session.maximum_sessions as usize,
177                frame_mtu: std::env::var("HOPR_SESSION_FRAME_SIZE")
178                    .ok()
179                    .and_then(|s| s.parse::<usize>().ok())
180                    .unwrap_or_else(|| SessionManagerConfig::default().frame_mtu)
181                    .max(ApplicationData::PAYLOAD_SIZE),
182                max_frame_timeout: std::env::var("HOPR_SESSION_FRAME_TIMEOUT_MS")
183                    .ok()
184                    .and_then(|s| s.parse::<u64>().ok().map(Duration::from_millis))
185                    .unwrap_or_else(|| SessionManagerConfig::default().max_frame_timeout)
186                    .max(Duration::from_millis(100)),
187                initiation_timeout_base: SESSION_INITIATION_TIMEOUT_BASE,
188                idle_timeout: cfg.session.idle_timeout,
189                balancer_sampling_interval: cfg.session.balancer_sampling_interval,
190                initial_return_session_egress_rate: 10,
191                minimum_surb_buffer_duration: Duration::from_secs(5),
192                maximum_surb_buffer_size: db.get_surb_config().rb_capacity,
193                // Allow a 10% increase of the target SURB buffer on incoming Sessions
194                // if the SURB buffer level has surpassed it by at least 10% in the last 2 minutes.
195                growable_target_surb_buffer: Some((Duration::from_secs(120), 0.10)),
196            }),
197            db,
198            resolver,
199            cfg,
200        }
201    }

203    /// Execute all processes of the [`HoprTransport`] object.
204    ///
205    /// This method will spawn the [`crate::HoprTransportProcess::Heartbeat`],
206    /// [`crate::HoprTransportProcess::BloomFilterSave`], [`crate::HoprTransportProcess::Swarm`] and session-related
207    /// processes and return join handles to the calling function. These processes are not started immediately but
208    /// are waiting for a trigger from this piece of code.
209    #[allow(clippy::too_many_arguments)]
210    pub async fn run<S, Ct>(
211        &self,
212        cover_traffic: Option<Ct>,
213        discovery_updates: S,
214        on_incoming_session: Sender<IncomingSession>,
215    ) -> errors::Result<(
216        HoprSocket<
217            futures::channel::mpsc::Receiver<ApplicationDataIn>,
218            futures::channel::mpsc::Sender<(DestinationRouting, ApplicationDataOut)>,
219        >,
220        AbortableList<HoprTransportProcess>,
221    )>
222    where
223        S: futures::Stream<Item = PeerDiscovery> + Send + 'static,
224        Ct: TrafficGeneration + Send + Sync + 'static,
225    {
226        info!("Loading initial peers from the chain");
227        let public_nodes = self
228            .resolver
229            .stream_accounts(AccountSelector {
230                public_only: true,
231                ..Default::default()
232            })
233            .await
234            .map_err(|e| HoprTransportError::Other(e.into()))?
235            .collect::<Vec<_>>()
236            .await;
237
238        // Calculate the minimum capacity based on public nodes (each node can generate 2 messages)
239        // plus 100 as additional buffer
240        let minimum_capacity = public_nodes.len().saturating_mul(2).saturating_add(100);
241
242        let internal_discovery_updates_capacity = std::env::var("HOPR_INTERNAL_DISCOVERY_UPDATES_CAPACITY")
243            .ok()
244            .and_then(|s| s.trim().parse::<usize>().ok())
245            .filter(|&c| c > 0)
246            .unwrap_or(2048)
247            .max(minimum_capacity);
248
249        debug!(
250            capacity = internal_discovery_updates_capacity,
251            minimum_required = minimum_capacity,
252            "Creating internal discovery updates channel"
253        );
254        let (mut internal_discovery_update_tx, internal_discovery_update_rx) =
255            futures::channel::mpsc::channel::<PeerDiscovery>(internal_discovery_updates_capacity);
256
257        let me_peerid = self.me_peerid;
258        let network = self.network.clone();
259        let discovery_updates = futures_concurrency::stream::StreamExt::merge(
260            discovery_updates,
261            internal_discovery_update_rx,
262        )
263        .filter_map(move |event| {
264            let network = network.clone();
265            async move {
266                match event {
267                    PeerDiscovery::Announce(peer, multiaddresses) => {
268                        debug!(%peer, ?multiaddresses, "processing peer discovery event: Announce");
269                        if peer != me_peerid {
270                            // decapsulate the `p2p/<peer_id>` to remove duplicities
271                            let mas = multiaddresses
272                                .into_iter()
273                                .map(|ma| strip_p2p_protocol(&ma))
274                                .filter(|v| !v.is_empty())
275                                .collect::<Vec<_>>();
276
277                            if !mas.is_empty() {
278                                if let Err(error) = network.add(&peer, PeerOrigin::NetworkRegistry, mas.clone()).await {
279                                    error!(%peer, %error, "failed to add peer to the network");
280                                    None
281                                } else {
282                                    Some(PeerDiscovery::Announce(peer, mas))
283                                }
284                            } else {
285                                None
286                            }
287                        } else {
288                            None
289                        }
290                    }
291                    _ => None,
292                }
293            }
294        });
295
296        info!(
297            public_nodes = public_nodes.len(),
298            "Initializing swarm with peers from chain"
299        );
300
301        for node_entry in public_nodes {
302            if let AccountType::Announced(multiaddresses) = node_entry.entry_type {
303                let peer: PeerId = node_entry.public_key.into();
304
305                debug!(%peer, ?multiaddresses, "Using initial public node");
306
307                internal_discovery_update_tx
308                    .send(PeerDiscovery::Announce(peer, multiaddresses))
309                    .await
310                    .map_err(|e| HoprTransportError::Api(e.to_string()))?;
311            }
312        }
313
314        let mut processes = AbortableList::<HoprTransportProcess>::default();
315
316        let (unresolved_routing_msg_tx, unresolved_routing_msg_rx) =
317            channel::<(DestinationRouting, ApplicationDataOut)>(MAXIMUM_MSG_OUTGOING_BUFFER_SIZE);
318
319        // -- transport medium
320        let mixer_cfg = build_mixer_cfg_from_env();
321
322        #[cfg(feature = "mixer-channel")]
323        let (mixing_channel_tx, mixing_channel_rx) = hopr_transport_mixer::channel::<(PeerId, Box<[u8]>)>(mixer_cfg);
324
325        #[cfg(feature = "mixer-stream")]
326        let (mixing_channel_tx, mixing_channel_rx) = {
327            let (tx, rx) = futures::channel::mpsc::channel::<(PeerId, Box<[u8]>)>(MAXIMUM_MSG_OUTGOING_BUFFER_SIZE);
328            let rx = rx.then_concurrent(move |v| {
329                let cfg = mixer_cfg;
330
331                async move {
332                    let random_delay = cfg.random_delay();
333                    trace!(delay_in_ms = random_delay.as_millis(), "Created random mixer delay",);
334
335                    #[cfg(all(feature = "prometheus", not(test)))]
336                    hopr_transport_mixer::channel::METRIC_QUEUE_SIZE.decrement(1.0f64);
337
338                    sleep(random_delay).await;
339
340                    #[cfg(all(feature = "prometheus", not(test)))]
341                    {
342                        hopr_transport_mixer::channel::METRIC_QUEUE_SIZE.decrement(1.0f64);
343
344                        let weight = 1.0f64 / cfg.metric_delay_window as f64;
345                        hopr_transport_mixer::channel::METRIC_MIXER_AVERAGE_DELAY.set(
346                            (weight * random_delay.as_millis() as f64)
347                                + ((1.0f64 - weight) * hopr_transport_mixer::channel::METRIC_MIXER_AVERAGE_DELAY.get()),
348                        );
349                    }
350
351                    v
352                }
353            });
354
355            (tx, rx)
356        };
357
358        let transport_layer = HoprSwarm::new(
359            (&self.me).into(),
360            discovery_updates,
361            self.my_multiaddresses.clone(),
362            self.cfg.transport.prefer_local_addresses,
363        )
364        .await;
365
366        let msg_proto_control =
367            transport_layer.build_protocol_control(hopr_transport_protocol::CURRENT_HOPR_MSG_PROTOCOL);
368        let msg_codec = hopr_transport_protocol::HoprBinaryCodec {};
369        let (wire_msg_tx, wire_msg_rx) =
370            hopr_transport_protocol::stream::process_stream_protocol(msg_codec, msg_proto_control).await?;
371
372        let _mixing_process_before_sending_out = hopr_async_runtime::prelude::spawn(
373            mixing_channel_rx
374                .inspect(|(peer, _)| tracing::trace!(%peer, "moving message from mixer to p2p stream"))
375                .map(Ok)
376                .forward(wire_msg_tx)
377                .inspect(|_| {
378                    tracing::warn!(
379                        task = "mixer -> egress process",
380                        "long-running background task finished"
381                    )
382                }),
383        );
384
385        let (transport_events_tx, transport_events_rx) =
386            futures::channel::mpsc::channel::<hopr_transport_p2p::DiscoveryEvent>(2048);
387
388        let network_clone = self.network.clone();
389        spawn(
390            transport_events_rx
391                .for_each(move |event| {
392                    let network = network_clone.clone();
393
394                    async move {
395                        match event {
396                            hopr_transport_p2p::DiscoveryEvent::IncomingConnection(peer, multiaddr) => {
397                                if let Err(error) = network
398                                    .add(&peer, PeerOrigin::IncomingConnection, vec![multiaddr])
399                                    .await
400                                {
401                                    tracing::error!(%peer, %error, "Failed to add incoming connection peer");
402                                }
403                            }
404                            hopr_transport_p2p::DiscoveryEvent::FailedDial(peer) => {
405                                if let Err(error) = network
406                                    .update(&peer, Err(hopr_transport_network::network::UpdateFailure::DialFailure))
407                                    .await
408                                {
409                                    tracing::error!(%peer, %error, "Failed to update peer status after failed dial");
410                                }
411                            }
412                        }
413                    }
414                })
415                .inspect(|_| {
416                    tracing::warn!(
417                        task = "transport events recording",
418                        "long-running background task finished"
419                    )
420                }),
421        );
422
423        processes.insert(
424            HoprTransportProcess::Medium,
425            spawn_as_abortable!(transport_layer.run(transport_events_tx).inspect(|_| tracing::warn!(
426                task = %HoprTransportProcess::Medium,
427                "long-running background task finished"
428            ))),
429        );
430
431        // -- msg-ack protocol over the wire transport
432        let packet_cfg = PacketInteractionConfig {
433            packet_keypair: self.me.clone(),
434            outgoing_ticket_win_prob: self
435                .cfg
436                .protocol
437                .outgoing_ticket_winning_prob
438                .map(WinningProbability::try_from)
439                .transpose()?,
440            outgoing_ticket_price: self.cfg.protocol.outgoing_ticket_price,
441        };
442
443        let msg_protocol_bidirectional_channel_capacity =
444            std::env::var("HOPR_INTERNAL_PROTOCOL_BIDIRECTIONAL_CHANNEL_CAPACITY")
445                .ok()
446                .and_then(|s| s.trim().parse::<usize>().ok())
447                .filter(|&c| c > 0)
448                .unwrap_or(16_384);
449
450        let (on_incoming_data_tx, on_incoming_data_rx) =
451            channel::<ApplicationDataIn>(msg_protocol_bidirectional_channel_capacity);
452
453        debug!(
454            capacity = msg_protocol_bidirectional_channel_capacity,
455            "Creating protocol bidirectional channel"
456        );
457        let (tx_from_protocol, rx_from_protocol) =
458            channel::<(HoprPseudonym, ApplicationDataIn)>(msg_protocol_bidirectional_channel_capacity);
459
460        // We have to resolve DestinationRouting -> ResolvedTransportRouting before
461        // sending the external packets to the transport pipeline.
462        let path_planner = self.path_planner.clone();
463        let distress_threshold = self.db.get_surb_config().distress_threshold;
464        let all_resolved_external_msg_rx = unresolved_routing_msg_rx.filter_map(move |(unresolved, mut data)| {
465            let path_planner = path_planner.clone();
466            async move {
467                trace!(?unresolved, "resolving routing for packet");
468                match path_planner
469                    .resolve_routing(data.data.total_len(), data.estimate_surbs_with_msg(), unresolved)
470                    .await
471                {
472                    Ok((resolved, rem_surbs)) => {
473                        // Set the SURB distress/out-of-SURBs flag if applicable.
474                        // These flags are translated into HOPR protocol packet signals and are
475                        // applicable only on the return path.
476                        let mut signals_to_dst = data
477                            .packet_info
478                            .as_ref()
479                            .map(|info| info.signals_to_destination)
480                            .unwrap_or_default();
481
482                        if resolved.is_return() {
483                            signals_to_dst = match rem_surbs {
484                                Some(rem) if (1..distress_threshold.max(2)).contains(&rem) => {
485                                    signals_to_dst | PacketSignal::SurbDistress
486                                }
487                                Some(0) => signals_to_dst | PacketSignal::OutOfSurbs,
488                                _ => signals_to_dst - (PacketSignal::OutOfSurbs | PacketSignal::SurbDistress),
489                            };
490                        } else {
491                            // Unset these flags as they make no sense on the forward path.
492                            signals_to_dst -= PacketSignal::SurbDistress | PacketSignal::OutOfSurbs;
493                        }
494
495                        data.packet_info.get_or_insert_default().signals_to_destination = signals_to_dst;
496                        trace!(?resolved, "resolved routing for packet");
497                        Some((resolved, data))
498                    }
499                    Err(error) => {
500                        error!(%error, "failed to resolve routing");
501                        None
502                    }
503                }
504            }
505            .in_current_span()
506        });
507
508        for (k, v) in hopr_transport_protocol::run_msg_ack_protocol(
509            packet_cfg,
510            self.db.clone(),
511            self.resolver.clone(),
512            (
513                mixing_channel_tx.with(|(peer, msg): (PeerId, Box<[u8]>)| {
514                    trace!(%peer, "sending message to peer");
515                    futures::future::ok::<_, hopr_transport_mixer::channel::SenderError>((peer, msg))
516                }),
517                wire_msg_rx.inspect(|(peer, _)| trace!(%peer, "received message from peer")),
518            ),
519            (tx_from_protocol, all_resolved_external_msg_rx),
520        )
521        .await
522        .into_iter()
523        {
524            processes.insert(HoprTransportProcess::Protocol(k), v);
525        }
526
527        // -- network probing
528        debug!(
529            capacity = msg_protocol_bidirectional_channel_capacity,
530            note = "same as protocol bidirectional",
531            "Creating probing channel"
532        );
533
534        let (tx_from_probing, rx_from_probing) =
535            channel::<(HoprPseudonym, ApplicationDataIn)>(msg_protocol_bidirectional_channel_capacity);
536
537        let manual_ping_channel_capacity = std::env::var("HOPR_INTERNAL_MANUAL_PING_CHANNEL_CAPACITY")
538            .ok()
539            .and_then(|s| s.trim().parse::<usize>().ok())
540            .filter(|&c| c > 0)
541            .unwrap_or(128);
542        debug!(capacity = manual_ping_channel_capacity, "Creating manual ping channel");
543        let (manual_ping_tx, manual_ping_rx) = channel::<(PeerId, PingQueryReplier)>(manual_ping_channel_capacity);
544
545        let probe = Probe::new(self.cfg.probe);
546        let probing_processes = if let Some(ct) = cover_traffic {
547            probe
548                .continuously_scan(
549                    (unresolved_routing_msg_tx.clone(), rx_from_protocol),
550                    manual_ping_rx,
551                    tx_from_probing,
552                    ct,
553                )
554                .await
555        } else {
556            probe
557                .continuously_scan(
558                    (unresolved_routing_msg_tx.clone(), rx_from_protocol),
559                    manual_ping_rx,
560                    tx_from_probing,
561                    ImmediateNeighborProber::new(
562                        self.cfg.probe,
563                        network_notifier::ProbeNetworkInteractions::new(self.network.clone()),
564                    ),
565                )
566                .await
567        };
568
569        for (k, v) in probing_processes.into_iter() {
570            processes.insert(HoprTransportProcess::Probing(k), v);
571        }
572
573        // manual ping
574        self.ping
575            .clone()
576            .set(Pinger::new(
577                PingConfig {
578                    timeout: self.cfg.probe.timeout,
579                },
580                manual_ping_tx,
581            ))
582            .expect("must set the ticket aggregation writer only once");
583
584        // -- session management
585        self.smgr
586            .start(unresolved_routing_msg_tx.clone(), on_incoming_session)
587            .expect("failed to start session manager")
588            .into_iter()
589            .enumerate()
590            .map(|(i, jh)| (HoprTransportProcess::SessionsManagement(i + 1), jh))
591            .for_each(|(k, v)| {
592                processes.insert(k, v);
593            });
594
595        let smgr = self.smgr.clone();
596        processes.insert(
597            HoprTransportProcess::SessionsManagement(0),
598            spawn_as_abortable!(
599                StreamExt::filter_map(rx_from_probing, move |(pseudonym, data)| {
600                    let smgr = smgr.clone();
601                    async move {
602                        match smgr.dispatch_message(pseudonym, data).await {
603                            Ok(DispatchResult::Processed) => {
604                                trace!("message dispatch completed");
605                                None
606                            }
607                            Ok(DispatchResult::Unrelated(data)) => {
608                                trace!("unrelated message dispatch completed");
609                                Some(data)
610                            }
611                            Err(e) => {
612                                error!(error = %e, "error while processing packet");
613                                None
614                            }
615                        }
616                    }
617                })
618                .map(Ok)
619                .forward(on_incoming_data_tx)
620                .inspect(|_| tracing::warn!(
621                    task = %HoprTransportProcess::SessionsManagement(0),
622                    "long-running background task finished"
623                ))
624            ),
625        );
626
627        Ok(((on_incoming_data_rx, unresolved_routing_msg_tx).into(), processes))
628    }
629
630    #[tracing::instrument(level = "debug", skip(self))]
631    pub async fn ping(&self, peer: &PeerId) -> errors::Result<(std::time::Duration, PeerStatus)> {
632        if peer == &self.me_peerid {
633            return Err(HoprTransportError::Api("ping to self does not make sense".into()));
634        }
635
636        let pinger = self
637            .ping
638            .get()
639            .ok_or_else(|| HoprTransportError::Api("ping processing is not yet initialized".into()))?;
640
641        if let Err(e) = self.network.add(peer, PeerOrigin::ManualPing, vec![]).await {
642            error!(error = %e, "Failed to store the peer observation");
643        }
644
645        let latency = (*pinger).ping(*peer).await?;
646
647        let peer_status = self.network.get(peer).await?.ok_or(HoprTransportError::Probe(
648            hopr_transport_probe::errors::ProbeError::NonExistingPeer,
649        ))?;
650
651        Ok((latency, peer_status))
652    }

694    #[tracing::instrument(level = "debug", skip(self))]
695    pub fn announceable_multiaddresses(&self) -> Vec<Multiaddr> {
696        let mut mas = self
697            .local_multiaddresses()
698            .into_iter()
699            .filter(|ma| {
700                hopr_transport_identity::multiaddrs::is_supported(ma)
701                    && (self.cfg.transport.announce_local_addresses || is_public_address(ma))
702            })
703            .map(|ma| strip_p2p_protocol(&ma))
704            .filter(|v| !v.is_empty())
705            .collect::<Vec<_>>();
706
707        mas.sort_by(|l, r| {
708            let is_left_dns = hopr_transport_identity::multiaddrs::is_dns(l);
709            let is_right_dns = hopr_transport_identity::multiaddrs::is_dns(r);
710
711            if !(is_left_dns ^ is_right_dns) {
712                std::cmp::Ordering::Equal
713            } else if is_left_dns {
714                std::cmp::Ordering::Less
715            } else {
716                std::cmp::Ordering::Greater
717            }
718        });
719
720        mas
721    }

807fn build_mixer_cfg_from_env() -> MixerConfig {
808    let mixer_cfg = MixerConfig {
809        min_delay: std::time::Duration::from_millis(
810            std::env::var("HOPR_INTERNAL_MIXER_MINIMUM_DELAY_IN_MS")
811                .map(|v| {
812                    v.trim()
813                        .parse::<u64>()
814                        .unwrap_or(hopr_transport_mixer::config::HOPR_MIXER_MINIMUM_DEFAULT_DELAY_IN_MS)
815                })
816                .unwrap_or(hopr_transport_mixer::config::HOPR_MIXER_MINIMUM_DEFAULT_DELAY_IN_MS),
817        ),
818        delay_range: std::time::Duration::from_millis(
819            std::env::var("HOPR_INTERNAL_MIXER_DELAY_RANGE_IN_MS")
820                .map(|v| {
821                    v.trim()
822                        .parse::<u64>()
823                        .unwrap_or(hopr_transport_mixer::config::HOPR_MIXER_DEFAULT_DELAY_RANGE_IN_MS)
824                })
825                .unwrap_or(hopr_transport_mixer::config::HOPR_MIXER_DEFAULT_DELAY_RANGE_IN_MS),
826        ),
827        capacity: {
828            let capacity = std::env::var("HOPR_INTERNAL_MIXER_CAPACITY")
829                .ok()
830                .and_then(|s| s.trim().parse::<usize>().ok())
831                .filter(|&c| c > 0)
832                .unwrap_or(hopr_transport_mixer::config::HOPR_MIXER_CAPACITY);
833            debug!(capacity = capacity, "Setting mixer capacity");
834            capacity
835        },
836        ..MixerConfig::default()
837    };
838    debug!(?mixer_cfg, "Mixer configuration");
839
840    mixer_cfg
841}
