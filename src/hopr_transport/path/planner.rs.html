34/// Configuration for [`PathPlanner`]'s internal path cache.
35#[derive(Debug, Clone, Copy, PartialEq, smart_default::SmartDefault, Validate)]
36pub struct PathPlannerConfig {
37    /// Maximum number of `(source, destination, options)` entries in the path cache.
38    #[default = 10_000]
39    pub max_cache_capacity: u64,
40    /// Time-to-live for a cached path list.  When an entry expires the next
41    /// [`PathPlanner::resolve_routing`] call transparently recomputes it (lazy refresh).
42    #[default(Duration::from_secs(60))]
43    pub cache_ttl: Duration,
44    /// Period between proactive background cache-refresh sweeps.
45    #[default(Duration::from_secs(30))]
46    pub refresh_period: Duration,
47    /// Maximum number of candidate paths the selector may return per query.
48    /// All returned candidates are validated and cached.
49    #[default = 8]
50    pub max_cached_paths: usize,
51    /// Penalty multiplier for edges lacking probe-based quality observations.
52    /// Applied during path cost evaluation to down-weight unprobed edges.
53    /// Must be finite and in `0.0..=1.0`.
54    #[default = 0.5]
55    #[validate(custom(function = "validate_unit_interval"))]
56    pub edge_penalty: f64,
57    /// Minimum acceptable message acknowledgment rate for path selection.
58    /// Edges with an ack rate below this threshold are excluded from candidate paths.
59    /// Must be finite and in `0.0..=1.0`.
60    #[default = 0.1]
61    #[validate(custom(function = "validate_unit_interval"))]
62    pub min_ack_rate: f64,
63}
64
65fn validate_unit_interval(value: f64) -> std::result::Result<(), ValidationError> {
66    if value.is_finite() && (0.0..=1.0).contains(&value) {
67        Ok(())
68    } else {
69        Err(ValidationError::new("value must be finite and in 0.0..=1.0"))
70    }
71}

73/// Cache key for the path planner: `(source, destination, hops)`.
74///
75/// Only the `Hops` variant of [`RoutingOptions`] is cached (explicit intermediate
76/// paths bypass the cache), so the key stores the hop count as a plain `u32`.
77type PlannerCacheKey = (NodeId, NodeId, u32);
78type PlannerCacheValue = Arc<hopr_utils::statistics::WeightedCollection<ValidatedPath>>;
79
80/// Path planner that resolves [`DestinationRouting`] to [`ResolvedTransportRouting`].
81///
82/// The planner delegates path *discovery* to any [`PathSelector`] implementation and
83/// owns the `moka` cache of fully-validated [`ValidatedPath`] objects paired with
84/// their traversal cost, keyed by `(source: NodeId, destination: NodeId, hops: u32)`.
85///
86/// On a cache miss the planner calls the selector, validates every candidate against
87/// the chain resolver, and stores an `Arc<WeightedCollection<ValidatedPath>>` in the
88/// cache. On a cache hit a candidate is picked via weighted random selection (higher
89/// cost = higher quality = higher probability).
90///
91/// A background sweep (`background_refresh`) can be spawned to
92/// proactively re-warm the cache for all previously-seen keys.
93#[derive(Clone)]
94pub struct PathPlanner<Surb, R, S> {
95    me: OffchainPublicKey,
96    pub surb_store: Surb,
97    resolver: Arc<R>,
98    selector: Arc<S>,
99    cache: moka::future::Cache<PlannerCacheKey, PlannerCacheValue>,
100    refresh_period: Duration,
101}

109    /// Create a new path planner.
110    ///
111    /// `me` is this node's [`OffchainPublicKey`]; it is used as the source in path queries.
112    pub fn new(me: OffchainPublicKey, surb_store: Surb, resolver: R, selector: S, config: PathPlannerConfig) -> Self {
113        let cache = moka::future::Cache::builder()
114            .max_capacity(config.max_cache_capacity)
115            .time_to_live(config.cache_ttl)
116            .build();
117
118        Self {
119            me,
120            surb_store,
121            resolver: Arc::new(resolver),
122            selector: Arc::new(selector),
123            cache,
124            refresh_period: config.refresh_period,
125        }
126    }

128    /// Resolve a [`NodeId`] to an [`OffchainPublicKey`].
129    async fn resolve_node_id_to_offchain_key(&self, node_id: &NodeId) -> Result<OffchainPublicKey> {
130        match node_id {
131            NodeId::Offchain(key) => Ok(*key),
132            NodeId::Chain(addr) => {
133                let resolver = ChainPathResolver::from(&*self.resolver);
134                resolver
135                    .resolve_transport_address(addr)
136                    .await
137                    .map_err(|e| PathPlannerError::Other(anyhow::anyhow!("{e}")))?
138                    .ok_or_else(|| {
139                        PathPlannerError::Other(anyhow::anyhow!("no offchain key found for chain address {addr}"))
140                    })
141            }
142        }
143    }

145    #[tracing::instrument(level = "trace", skip(self))]
146    async fn resolve_path(
147        &self,
148        source: NodeId,
149        destination: NodeId,
150        options: RoutingOptions,
151    ) -> Result<ValidatedPath> {
152        let path = match options {
153            RoutingOptions::IntermediatePath(explicit_path) => {
154                tracing::debug!(
155                    direction = "loopback",
156                    ?source,
157                    ?destination,
158                    ?explicit_path,
159                    "resolving intermediate path"
160                );
161                let resolver = ChainPathResolver::from(&*self.resolver);
162                ValidatedPath::new(
163                    source,
164                    explicit_path
165                        .into_iter()
166                        .chain(std::iter::once(destination))
167                        .collect::<Vec<_>>(),
168                    &resolver,
169                )
170                .await?
171            }
172
173            RoutingOptions::Hops(hops) if u32::from(hops) == 0 => {
174                trace!(hops = 0, "resolving zero-hop direct path");
175                let resolver = ChainPathResolver::from(&*self.resolver);
176                ValidatedPath::new(source, vec![destination], &resolver).await?
177            }
178
179            RoutingOptions::Hops(hops) => {
180                let hops_usize: usize = hops.into();
181                trace!(hops = hops_usize, "resolving path via planner cache");
182
183                let src_key = self.resolve_node_id_to_offchain_key(&source).await?;
184                let dest_key = self.resolve_node_id_to_offchain_key(&destination).await?;
185
186                let cache_key: PlannerCacheKey = (source, destination, u32::from(hops));
187
188                let resolver = self.resolver.clone();
189                let selector = self.selector.clone();
190
191                let paths = self
192                    .cache
193                    .try_get_with(cache_key, async move {
194                        trace!(hops = hops_usize, "path cache miss, querying selector");
195                        let candidates = selector.select_path(src_key, dest_key, hops_usize)?;
196
197                        let chain_resolver = ChainPathResolver::from(&*resolver);
198                        let mut valid_paths: Vec<(ValidatedPath, f64)> = Vec::new();
199                        for pwc in candidates {
200                            let node_ids: Vec<NodeId> = pwc.path.into_iter().map(NodeId::Offchain).collect::<Vec<_>>();
201                            match ValidatedPath::new(source, node_ids, &chain_resolver).await {
202                                Ok(vp) => valid_paths.push((vp, pwc.cost)),
203                                Err(e) => tracing::debug!(error = %e, "path candidate failed validation"),
204                            }
205                        }
206
207                        if valid_paths.is_empty() {
208                            return Err(PathPlannerError::Path(PathError::PathNotFound(
209                                hops_usize,
210                                src_key.to_hex(),
211                                dest_key.to_hex(),
212                            )));
213                        }
214
215                        Ok(Arc::new(hopr_utils::statistics::WeightedCollection::new(valid_paths)))
216                    })
217                    .await
218                    .map_err(PathPlannerError::CacheError)?;
219
220                paths.pick_one().ok_or_else(|| {
221                    PathPlannerError::Path(PathError::PathNotFound(hops_usize, src_key.to_hex(), dest_key.to_hex()))
222                })?
223            }
224        };
225
226        #[cfg(all(feature = "telemetry", not(test)))]
227        {
228            hopr_api::types::telemetry::SimpleHistogram::observe(&METRIC_PATH_LENGTH, (path.num_hops() - 1) as f64);
229        }
230
231        trace!(%path, "validated resolved path");
232        Ok(path)
233    }

235    /// Resolve a [`DestinationRouting`] to a [`ResolvedTransportRouting`].
236    ///
237    /// Returns the resolved routing and, for `Return` variants, the number of remaining SURBs.
238    #[tracing::instrument(level = "trace", skip(self))]
239    pub async fn resolve_routing(
240        &self,
241        size_hint: usize,
242        max_surbs: usize,
243        routing: DestinationRouting,
244    ) -> Result<(ResolvedTransportRouting<HoprSurb>, Option<usize>)> {
245        match routing {
246            DestinationRouting::Forward {
247                destination,
248                pseudonym,
249                forward_options,
250                return_options,
251            } => {
252                tracing::debug!(direction = "forward", %destination, "resolving forward path");
253
254                let forward_path = self
255                    .resolve_path(NodeId::Offchain(self.me), *destination, forward_options)
256                    .await?;
257                tracing::debug!(direction = "forward", %destination, path = %forward_path, "resolved path");
258
259                let return_paths = if let Some(return_options) = return_options {
260                    let num_possible_surbs = HoprPacket::max_surbs_with_message(size_hint).min(max_surbs);
261                    trace!(
262                        %destination,
263                        %num_possible_surbs,
264                        data_len = size_hint,
265                        max_surbs,
266                        "resolving packet return paths"
267                    );
268
269                    (0..num_possible_surbs)
270                        .map(|_| self.resolve_path(*destination, NodeId::Offchain(self.me), return_options.clone()))
271                        .collect::<FuturesUnordered<_>>()
272                        .try_collect::<Vec<ValidatedPath>>()
273                        .await?
274                        .into_iter()
275                        .enumerate()
276                        .inspect(|(i, rp)| {
277                            tracing::debug!(direction = "return", %destination, index = i, path = %rp, "resolved return path");
278                        })
279                        .map(|(_, rp)| rp)
280                        .collect()
281                } else {
282                    vec![]
283                };
284
285                trace!(%destination, num_surbs = return_paths.len(), data_len = size_hint, "resolved packet");
286
287                Ok((
288                    ResolvedTransportRouting::Forward {
289                        pseudonym: pseudonym.unwrap_or_else(HoprPseudonym::random),
290                        forward_path,
291                        return_paths,
292                    },
293                    None,
294                ))
295            }
296
297            DestinationRouting::Return(matcher) => {
298                let FoundSurb {
299                    sender_id,
300                    surb,
301                    remaining,
302                } = self
303                    .surb_store
304                    .find_surb(matcher)
305                    .ok_or_else(|| PathPlannerError::Surb(format!("no surb for pseudonym {}", matcher.pseudonym())))?;
306                Ok((ResolvedTransportRouting::Return(sender_id, surb), Some(remaining)))
307            }
308        }
309    }

312impl<Surb, R, S> BackgroundPathCacheRefreshable for PathPlanner<Surb, R, S>
313where
314    Surb: SurbStore + Send + Sync + 'static,
315    R: ChainKeyOperations + ChainReadChannelOperations + Send + Sync + 'static,
316    S: PathSelector + Send + Sync + 'static,
317{
318    /// Returns a future that runs the background path-cache refresh loop.
319    ///
320    /// The returned future iterates over all keys currently in the planner's cache
321    /// and recomputes their paths on a configurable schedule, so that steady-state
322    /// traffic is always served from cache.
323    fn run_background_refresh(&self) -> impl std::future::Future<Output = ()> + Send + 'static {
324        // Clone only the fields we need — avoids requiring R: Clone + S: Clone.
325        let cache = self.cache.clone();
326        let resolver = self.resolver.clone();
327        let selector = self.selector.clone();
328        let refresh_period = self.refresh_period;
329
330        // run at a non-zero interval
331        futures_time::stream::interval(futures_time::time::Duration::from_millis(
332            refresh_period.as_millis() as u64 + 1u64,
333        ))
334        .for_each(move |_| {
335            let cache = cache.clone();
336            let resolver = resolver.clone();
337            let selector = selector.clone();
338
339            async move {
340                for (key, _) in cache.iter() {
341                    let (src, dest, hops_u32) = {
342                        let k = key.as_ref();
343                        (k.0, k.1, k.2)
344                    };
345
346                    if hops_u32 == 0 {
347                        continue;
348                    }
349                    let hops_usize = hops_u32 as usize;
350
351                    let resolve_key = |node: NodeId| {
352                        let resolver = resolver.clone();
353
354                        async move {
355                            match node {
356                                NodeId::Offchain(k) => Some(k),
357                                NodeId::Chain(addr) => ChainPathResolver::from(&*resolver)
358                                    .resolve_transport_address(&addr)
359                                    .await
360                                    .ok()
361                                    .flatten(),
362                            }
363                        }
364                    };
365
366                    if let (Some(src_key), Some(dest_key)) = (resolve_key(src).await, resolve_key(dest).await)
367                        && let Ok(candidates) = selector.select_path(src_key, dest_key, hops_usize)
368                    {
369                        let chain_resolver = ChainPathResolver::from(&*resolver);
370                        let mut valid_paths: Vec<(ValidatedPath, f64)> = Vec::new();
371                        for pwc in candidates {
372                            let node_ids: Vec<NodeId> = pwc.path.into_iter().map(NodeId::Offchain).collect::<Vec<_>>();
373                            match ValidatedPath::new(src, node_ids, &chain_resolver).await {
374                                Ok(vp) => valid_paths.push((vp, pwc.cost)),
375                                Err(e) => {
376                                    tracing::debug!(error = %e, "background refresh: path candidate failed validation")
377                                }
378                            }
379                        }
380
381                        if !valid_paths.is_empty() {
382                            cache
383                                .insert(
384                                    (src, dest, hops_u32),
385                                    Arc::new(hopr_utils::statistics::WeightedCollection::new(valid_paths)),
386                                )
387                                .await;
388                        }
389                    }
390                }
391            }
392        })
393    }

598        let planner = PathPlanner::new(me, surb_store, chain_api, selector, small_config());

600        let routing = DestinationRouting::Forward {
601            destination: Box::new(NodeId::Offchain(dest)),
602            pseudonym: None,
603            forward_options: RoutingOptions::Hops(0.try_into().expect("valid 0")),
604            return_options: None,

607        let result = planner.resolve_routing(100, 0, routing).await;
