15/// Contains all components required to run the HOPR packet pipeline.
16#[derive(Clone)]
17pub struct HoprPipelineComponents<TEvt, S, Chain, Db> {
18    /// Sink for [`TicketEvents`](TicketEvent).
19    pub ticket_events: TEvt,
20    /// Store for SURBs and Reply Openers.
21    pub surb_store: S,
22    /// Chain API for interacting with the blockchain.
23    pub chain_api: Chain,
24    /// Database for storing tickets and other data.
25    pub db: Db,
26}

28pub fn run_hopr_packet_pipeline<WIn, WOut, Chain, S, Db, TEvt, AppOut, AppIn>(
29    (packet_key, chain_key): (OffchainKeypair, ChainKeypair),
30    wire_msg: (WOut, WIn),
31    api: (AppOut, AppIn),
32    components: HoprPipelineComponents<TEvt, S, Chain, Db>,
33    channels_dst: Hash,
34    cfg: HoprPacketPipelineConfig,
35) -> AbortableList<HoprTransportProcess>
36where
37    WOut: futures::Sink<(PeerId, Box<[u8]>)> + Clone + Unpin + Send + 'static,
38    WOut::Error: std::error::Error,
39    WIn: futures::Stream<Item = (PeerId, Box<[u8]>)> + Send + 'static,
40    Db: HoprDbTicketOperations + Clone + Send + Sync + 'static,
41    Chain: ChainKeyOperations + ChainReadChannelOperations + ChainValues + Clone + Send + Sync + 'static,
42    S: SurbStore + Clone + Send + Sync + 'static,
43    TEvt: futures::Sink<TicketEvent> + Clone + Unpin + Send + 'static,
44    TEvt::Error: std::error::Error,
45    AppOut: futures::Sink<(HoprPseudonym, ApplicationDataIn)> + Send + 'static,
46    AppOut::Error: std::error::Error,
47    AppIn: futures::Stream<Item = (ResolvedTransportRouting, ApplicationDataOut)> + Send + 'static,
48{
49    let HoprPipelineComponents {
50        ticket_events,
51        surb_store,
52        chain_api,
53        db,
54    } = components;
55
56    let ticket_proc = HoprTicketProcessor::new(
57        chain_api.clone(),
58        db.clone(),
59        chain_key.clone(),
60        channels_dst,
61        cfg.ticket_processing,
62    );
63    let encoder = HoprEncoder::new(
64        chain_key.clone(),
65        chain_api.clone(),
66        surb_store.clone(),
67        ticket_proc.clone(),
68        channels_dst,
69        cfg.codec,
70    );
71    let decoder = HoprDecoder::new(
72        (packet_key.clone(), chain_key.clone()),
73        chain_api.clone(),
74        surb_store,
75        ticket_proc.clone(),
76        channels_dst,
77        cfg.codec,
78    );
79
80    let mut processes = AbortableList::default();
81
82    #[cfg(feature = "capture")]
83    let (encoder, decoder) = {
84        use crate::capture;
85
86        let writer: Box<dyn capture::PacketWriter + Send + 'static> =
87            if let Ok(desc) = std::env::var("HOPR_CAPTURE_PACKETS") {
88                if let Ok(pcap_writer) = std::fs::File::create(&desc).and_then(capture::PcapPacketWriter::new) {
89                    tracing::warn!("pcap file packet capture initialized to {desc}");
90                    Box::new(pcap_writer)
91                } else {
92                    tracing::error!(desc, "failed to create packet capture: invalid socket address or file");
93                    Box::new(capture::NullWriter)
94                }
95            } else {
96                tracing::warn!("no packet capture specified");
97                Box::new(capture::NullWriter)
98            };
99
100        let (sender, ah) = capture::packet_capture_channel(writer);
101        processes.insert(HoprTransportProcess::Capture, ah);
102        (
103            capture::CapturePacketCodec::new(encoder, *packet_key.public(), sender.clone()),
104            capture::CapturePacketCodec::new(decoder, *packet_key.public(), sender.clone()),
105        )
106    };
107
108    let (index_sync_handle, index_sync_reg) = futures::future::AbortHandle::new_pair();
109    hopr_async_runtime::prelude::spawn(ticket_proc.outgoing_index_sync_task(index_sync_reg));
110    processes.insert(HoprTransportProcess::OutgoingIndexSync, index_sync_handle);
111
112    processes.flat_map_extend_from(
113        run_packet_pipeline(
114            packet_key.clone(),
115            wire_msg,
116            (encoder, decoder),
117            ticket_proc,
118            ticket_events,
119            cfg.pipeline,
120            api,
121        ),
122        HoprTransportProcess::Pipeline,
123    );
124
125    processes
126}
