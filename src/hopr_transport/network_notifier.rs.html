32/// Implementor of the ping external API.
33///
34/// Ping requires functionality from external components in order to obtain
35/// the triggers for its functionality. This class implements the basic API by
36/// aggregating all necessary ping resources without leaking them into the
37/// `Ping` object and keeping both the adaptor and the ping object OCP and SRP
38/// compliant.
39#[derive(Debug, Clone)]
40pub struct ProbeNetworkInteractions<T>
41where
42    T: HoprDbPeersOperations + HoprDbResolverOperations + Sync + Send + Clone + std::fmt::Debug,
43{
44    network: Arc<Network<T>>,
45    resolver: T,
46    channel_graph: Arc<RwLock<ChannelGraph>>,
47}
48
49impl<T> ProbeNetworkInteractions<T>
50where
51    T: HoprDbPeersOperations + HoprDbResolverOperations + Sync + Send + Clone + std::fmt::Debug,
52{
53    pub fn new(network: Arc<Network<T>>, resolver: T, channel_graph: Arc<RwLock<ChannelGraph>>) -> Self {
54        Self {
55            network,
56            resolver,
57            channel_graph,
58        }
59    }
60}

62#[async_trait]
63impl<T> PeerDiscoveryFetch for ProbeNetworkInteractions<T>
64where
65    T: HoprDbPeersOperations + HoprDbResolverOperations + Sync + Send + Clone + std::fmt::Debug,
66{
67    /// Get all peers considered by the `Network` to be pingable.
68    ///
69    /// After a duration of non-pinging based specified by the configurable threshold.
70    #[tracing::instrument(level = "trace", skip(self))]
71    async fn get_peers(&self, from_timestamp: std::time::SystemTime) -> Vec<hopr_transport_network::PeerId> {
72        self.network
73            .find_peers_to_ping(from_timestamp)
74            .await
75            .unwrap_or_else(|error| {
76                tracing::error!(%error, "failed to generate peers for the heartbeat procedure");
77                vec![]
78            })
79    }
80}

82#[async_trait]
83impl<T> ProbeStatusUpdate for ProbeNetworkInteractions<T>
84where
85    T: HoprDbPeersOperations + HoprDbResolverOperations + Sync + Send + Clone + std::fmt::Debug,
86{
87    #[tracing::instrument(level = "debug", skip(self))]
88    async fn on_finished(
89        &self,
90        peer: &hopr_transport_network::PeerId,
91        result: &hopr_transport_probe::errors::Result<std::time::Duration>,
92    ) {
93        let result = match &result {
94            Ok(duration) => {
95                #[cfg(all(feature = "prometheus", not(test)))]
96                {
97                    METRIC_TIME_TO_PING.observe((duration.as_millis() as f64) / 1000.0); // precision for seconds
98                    METRIC_PROBE_COUNT.increment(&["true"]);
99                }
100
101                Ok(*duration)
102            }
103            Err(error) => {
104                #[cfg(all(feature = "prometheus", not(test)))]
105                METRIC_PROBE_COUNT.increment(&["false"]);
106
107                tracing::trace!(%error, "Encountered timeout on peer ping");
108                Err(UpdateFailure::Timeout)
109            }
110        };
111
112        // Update the channel graph
113        let peer = *peer;
114        // PeerId -> OffchainPublicKey is a CPU-intensive blocking operation
115        if let Ok(pubkey) = hopr_parallelize::cpu::spawn_blocking(move || OffchainPublicKey::from_peerid(&peer)).await {
116            let maybe_chain_key = self.resolver.resolve_chain_key(&pubkey).await;
117            if let Ok(Some(chain_key)) = maybe_chain_key {
118                let mut g = self.channel_graph.write_arc().await;
119                g.update_node_score(&chain_key, result.into());
120                debug!(%chain_key, ?result, "update node score for peer");
121            } else {
122                error!("could not resolve chain key");
123            }
124        } else {
125            error!("encountered invalid peer id");
126        }
127
128        if let Err(error) = self.network.update(&peer, result).await {
129            error!(%error, "Encountered error on on updating the collected ping data")
130        }
131    }
132}
