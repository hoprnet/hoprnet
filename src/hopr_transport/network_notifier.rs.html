32/// Implementor of the ping external API.
33///
34/// Ping requires functionality from external components in order to obtain
35/// the triggers for its functionality. This class implements the basic API by
36/// aggregating all necessary ping resources without leaking them into the
37/// `Ping` object and keeping both the adaptor and the ping object OCP and SRP
38/// compliant.
39#[derive(Debug, Clone)]
40pub struct ProbeNetworkInteractions<T>
41where
42    T: HoprDbPeersOperations + HoprDbResolverOperations + Sync + Send + Clone + std::fmt::Debug,
43{
44    network: Arc<Network<T>>,
45    resolver: T,
46    channel_graph: Arc<RwLock<ChannelGraph>>,
47    /// Implementation of the network interface allowing emitting events
48    /// based on the [hopr_transport_network::network::Network] events into the p2p swarm.
49    emitter: futures::channel::mpsc::Sender<NetworkTriggeredEvent>,
50}
51
52impl<T> ProbeNetworkInteractions<T>
53where
54    T: HoprDbPeersOperations + HoprDbResolverOperations + Sync + Send + Clone + std::fmt::Debug,
55{
56    pub fn new(
57        network: Arc<Network<T>>,
58        resolver: T,
59        channel_graph: Arc<RwLock<ChannelGraph>>,
60        emitter: futures::channel::mpsc::Sender<NetworkTriggeredEvent>,
61    ) -> Self {
62        Self {
63            network,
64            resolver,
65            channel_graph,
66            emitter,
67        }
68    }
69}

71#[async_trait]
72impl<T> PeerDiscoveryFetch for ProbeNetworkInteractions<T>
73where
74    T: HoprDbPeersOperations + HoprDbResolverOperations + Sync + Send + Clone + std::fmt::Debug,
75{
76    /// Get all peers considered by the `Network` to be pingable.
77    ///
78    /// After a duration of non-pinging based specified by the configurable threshold.
79    #[tracing::instrument(level = "trace", skip(self))]
80    async fn get_peers(&self, from_timestamp: std::time::SystemTime) -> Vec<hopr_transport_network::PeerId> {
81        self.network
82            .find_peers_to_ping(from_timestamp)
83            .await
84            .unwrap_or_else(|e| {
85                tracing::error!(error = %e, "Failed to generate peers for the heartbeat procedure");
86                vec![]
87            })
88    }
89}

91#[async_trait]
92impl<T> ProbeStatusUpdate for ProbeNetworkInteractions<T>
93where
94    T: HoprDbPeersOperations + HoprDbResolverOperations + Sync + Send + Clone + std::fmt::Debug,
95{
96    #[tracing::instrument(level = "debug", skip(self))]
97    async fn on_finished(
98        &self,
99        peer: &hopr_transport_network::PeerId,
100        result: &hopr_transport_probe::errors::Result<std::time::Duration>,
101    ) {
102        let result = match &result {
103            Ok(duration) => {
104                #[cfg(all(feature = "prometheus", not(test)))]
105                {
106                    METRIC_TIME_TO_PING.observe((duration.as_millis() as f64) / 1000.0); // precision for seconds
107                    METRIC_PROBE_COUNT.increment(&["true"]);
108                }
109
110                Ok(*duration)
111            }
112            Err(_) => {
113                #[cfg(all(feature = "prometheus", not(test)))]
114                METRIC_PROBE_COUNT.increment(&["false"]);
115
116                Err(())
117            }
118        };
119
120        // Update the channel graph
121        if let Ok(pk) = OffchainPublicKey::try_from(peer) {
122            let maybe_chain_key = self.resolver.resolve_chain_key(&pk).await;
123            if let Ok(Some(chain_key)) = maybe_chain_key {
124                let mut g = self.channel_graph.write().await;
125                g.update_node_score(&chain_key, result.into());
126                debug!(%chain_key, ?result, "update node score for peer");
127            } else {
128                error!(%peer, "could not resolve chain key");
129            }
130        } else {
131            error!(%peer, "encountered invalid peer id");
132        }
133
134        match self.network.update(peer, result, None).await {
135            Ok(Some(updated)) => match updated {
136                NetworkTriggeredEvent::CloseConnection(peer) => {
137                    if let Err(e) = self
138                        .emitter
139                        .clone()
140                        .try_send(NetworkTriggeredEvent::CloseConnection(peer))
141                    {
142                        error!(error = %e, "Failed to emit a network event 'close connection'")
143                    }
144                }
145                NetworkTriggeredEvent::UpdateQuality(peer, quality) => {
146                    tracing::trace!(%peer, %quality, "peer changed quality");
147                }
148            },
149            Ok(None) => trace!("No update necessary"),
150            Err(e) => error!(error = %e, "Encountered error on on updating the collected ping data"),
151        }
152    }
153}
