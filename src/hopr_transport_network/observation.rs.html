1use std::sync::atomic::{AtomicU64, Ordering};
2
3use hopr_api::network::Observable;
4use hopr_statistics::ExponentialMovingAverage;

6/// Observations related to a specific peer in the network.
7#[derive(Debug, Copy, Clone, Default, PartialEq)]
8pub struct Observations {
9    pub msg_sent: u64,
10    pub ack_received: u64,
11    last_update: std::time::Duration,
12    latency_average: ExponentialMovingAverage<3>,
13    probe_success_rate: ExponentialMovingAverage<5>,
14}
15
16impl Observable for Observations {
17    fn record_probe(&mut self, latency: std::result::Result<std::time::Duration, ()>) {
18        self.last_update = std::time::SystemTime::now()
19            .duration_since(std::time::UNIX_EPOCH)
20            .unwrap_or_default();
21
22        if let Ok(latency) = latency {
23            self.latency_average.update(latency.as_millis() as f64);
24            self.probe_success_rate.update(1.0);
25        } else {
26            self.probe_success_rate.update(0.0);
27        }
28    }
29
30    #[inline]
31    fn last_update(&self) -> std::time::Duration {
32        self.last_update
33    }
34
35    fn average_latency(&self) -> Option<std::time::Duration> {
36        if self.latency_average.get() <= 0.0 {
37            None
38        } else {
39            Some(std::time::Duration::from_millis(self.latency_average.get() as u64))
40        }
41    }
42
43    fn average_probe_rate(&self) -> f64 {
44        self.probe_success_rate.get()
45    }
46
47    fn score(&self) -> f64 {
48        self.probe_success_rate.get()
49    }
50}

52/// Atomic counters for tracking packet statistics per peer.
53///
54/// Uses atomic operations for thread-safe, lock-free updates.
55/// Stats are automatically reset when the peer disconnects (the entry is dropped).
56#[derive(Debug, Default)]
57pub struct PeerPacketStats {
58    packets_out: AtomicU64,
59    packets_in: AtomicU64,
60    bytes_out: AtomicU64,
61    bytes_in: AtomicU64,
62}
63
64impl PeerPacketStats {
65    /// Record an outgoing packet with the given size in bytes.
66    #[inline]
67    pub fn record_packet_out(&self, bytes: usize) {
68        self.packets_out.fetch_add(1, Ordering::Relaxed);
69        self.bytes_out.fetch_add(bytes as u64, Ordering::Relaxed);
70    }
71
72    /// Record an incoming packet with the given size in bytes.
73    #[inline]
74    pub fn record_packet_in(&self, bytes: usize) {
75        self.packets_in.fetch_add(1, Ordering::Relaxed);
76        self.bytes_in.fetch_add(bytes as u64, Ordering::Relaxed);
77    }
78
79    /// Create a snapshot of the current stats.
80    pub fn snapshot(&self) -> PeerPacketStatsSnapshot {
81        PeerPacketStatsSnapshot {
82            packets_out: self.packets_out.load(Ordering::Relaxed),
83            packets_in: self.packets_in.load(Ordering::Relaxed),
84            bytes_out: self.bytes_out.load(Ordering::Relaxed),
85            bytes_in: self.bytes_in.load(Ordering::Relaxed),
86        }
87    }
88}

90/// A point-in-time snapshot of peer packet statistics.
91///
92/// This is a non-atomic, serializable copy of the stats for API responses.
93#[derive(Debug, Clone, Copy, Default, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
94#[serde(rename_all = "camelCase")]
95pub struct PeerPacketStatsSnapshot {
96    pub packets_out: u64,
97    pub packets_in: u64,
98    pub bytes_out: u64,
99    pub bytes_in: u64,
100}

109    #[test]
110    fn observations_should_update_the_timestamp_on_latency_update() {
111        let mut observation = Observations::default();
112
113        assert_eq!(observation.last_update, std::time::Duration::default());
114
115        observation.record_probe(Ok(std::time::Duration::from_millis(50)));
116
117        std::thread::sleep(std::time::Duration::from_millis(10));
118
119        let after = std::time::SystemTime::now()
120            .duration_since(std::time::UNIX_EPOCH)
121            .unwrap_or_default();
122
123        assert_gt!(observation.last_update, std::time::Duration::default());
124        assert_lt!(observation.last_update, after);
125    }

127    #[test]
128    fn observations_should_store_an_average_latency_value_after_multiple_updates() -> anyhow::Result<()> {
129        let big_latency = std::time::Duration::from_millis(300);
130        let small_latency = std::time::Duration::from_millis(10);
131
132        let mut observation = Observations::default();
133
134        for _ in 0..10 {
135            observation.record_probe(Ok(small_latency));
136        }
137
138        assert_eq!(
139            observation.average_latency().context("should contain a value")?,
140            small_latency
141        );
142
143        observation.record_probe(Ok(big_latency));
144
145        assert_gt!(
146            observation.average_latency().context("should contain a value")?,
147            small_latency
148        );
149        assert_lt!(
150            observation.average_latency().context("should contain a value")?,
151            big_latency
152        );
153
154        Ok(())
155    }
