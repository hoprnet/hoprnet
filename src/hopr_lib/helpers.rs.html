16/// Creates a pipeline that chains the indexer-generated data, processes them into
17/// the individual components, and creates a filtered output stream that is fed into
18/// the transport layer swarm.
19///
20/// * `event_stream` - represents the events generated by the indexer. If the Indexer is not synced, it will not
21///   generate any events.
22/// * `preloading_event_stream` - a stream used by the components to preload the data from the objects (db, channel
23///   graph...)
24#[allow(clippy::too_many_arguments)]
25pub fn chain_events_to_transport_events<StreamIn>(
26    event_stream: StreamIn,
27    me_onchain: Address,
28    multi_strategy: Arc<MultiStrategy>,
29    channel_graph: Arc<RwLock<hopr_path::channel_graph::ChannelGraph>>,
30    db: HoprNodeDb,
31) -> impl Stream<Item = PeerDiscovery> + Send + 'static
32where
33    StreamIn: Stream<Item = SignificantChainEvent> + Send + 'static,
34{
35    Box::pin(
36        event_stream
37            .filter_map(move |event| {
38                let multi_strategy = multi_strategy.clone();
39                let channel_graph = channel_graph.clone();
40                let db = db.clone();
41
42                async move {
43                    match event.event_type {
44                        ChainEventType::Announcement{peer, multiaddresses, ..} => {
45                            Some(vec![PeerDiscovery::Announce(peer, multiaddresses)])
46                        }
47                        ChainEventType::ChannelOpened(channel) |
48                        ChainEventType::ChannelClosureInitiated(channel) |
49                        ChainEventType::ChannelClosed(channel) |
50                        ChainEventType::ChannelBalanceIncreased(channel, _) | // needed ?
51                        ChainEventType::ChannelBalanceDecreased(channel, _) | // needed ?
52                        ChainEventType::TicketRedeemed(channel, _) => {   // needed ?
53                            let maybe_direction = channel.direction(&me_onchain);
54
55                            let change = channel_graph
56                                .write_arc()
57                                .await
58                                .update_channel(channel);
59
60                            db.invalidate_unrealized_value(&channel).await;
61
62                            // Check if this is our own channel
63                            if let Some(own_channel_direction) = maybe_direction {
64                                if let Some(change_set) = change {
65                                    for channel_change in change_set {
66                                        let _ = hopr_strategy::strategy::SingularStrategy::on_own_channel_changed(
67                                            &*multi_strategy,
68                                            &channel,
69                                            own_channel_direction,
70                                            channel_change,
71                                        )
72                                            .await;
73                                    }
74                                } else if channel.status == ChannelStatus::Open {
75                                    // Emit Opening event if the channel did not exist before in the graph
76                                    let _ = hopr_strategy::strategy::SingularStrategy::on_own_channel_changed(
77                                        &*multi_strategy,
78                                        &channel,
79                                        own_channel_direction,
80                                        ChannelChange::Status {
81                                            left: ChannelStatus::Closed,
82                                            right: ChannelStatus::Open,
83                                        },
84                                    )
85                                        .await;
86                                }
87                            }
88
89                            None
90                        }
91                        ChainEventType::NetworkRegistryUpdate(address, allowed) => {
92                            info!(%address, ?allowed, "network registry update received as a no-op");
93                            None
94                        }
95                        ChainEventType::NodeSafeRegistered(safe_address) =>  {
96                            info!(%safe_address, "Node safe registered");
97                            None
98                        }
99                    }
100                }
101            })
102            .flat_map(stream::iter),
103    )
104}
