11/// Error type for [`CrossfireSink`].
12#[derive(Debug, Clone, thiserror::Error, PartialEq, Eq)]
13pub enum CrossfireSinkError {
14    /// The channel receiver has been dropped.
15    #[error("channel receiver disconnected")]
16    Disconnected,
17}

19/// A [`futures::Sink`] backed by a bounded crossfire MPSC channel sender.
20///
21/// Enforces strict capacity: unlike `futures::channel::mpsc::Sender`, cloning this sink
22/// does **not** add extra slots to the channel. When the channel is full, `poll_ready`
23/// returns `Poll::Pending` and registers a waker, allowing combinators such as
24/// [`crate::network_types::timeout::TimeoutSink`] to arm their timers correctly.
25///
26/// Construct via [`bounded_sink_channel`].
27pub struct CrossfireSink<T: 'static> {
28    tx: crossfire::MAsyncTx<crossfire::mpsc::Array<T>>,
29    /// Item buffered by `start_send`, awaiting forwarding to the channel.
30    buffered: Option<T>,
31    /// In-flight send future created when `try_send` returned `Full` during `poll_ready`.
32    ///
33    /// This is what makes `poll_ready` return `Pending` when the channel is saturated,
34    /// which is necessary for `TimeoutSink::poll_ready` to arm its deadline timer.
35    pending: Option<PendingFut<T>>,
36}
37
38// CrossfireSink does not use structural pinning — no field is accessed exclusively through
39// Pin<&mut CrossfireSink<T>>. Moving a CrossfireSink after pinning is safe.
40impl<T: 'static> Unpin for CrossfireSink<T> {}
41
42// SAFETY: All &self methods (try_send, clone) only access `tx: MAsyncTx<Array<T>>` which
43// is Sync. The `pending` and `buffered` fields are accessed exclusively via &mut self
44// (Sink trait methods), making concurrent &self access free of data races.
45unsafe impl<T: Send + 'static> Sync for CrossfireSink<T> where crossfire::MAsyncTx<crossfire::mpsc::Array<T>>: Sync {}

56impl<T: Send + 'static> Clone for CrossfireSink<T> {
57    fn clone(&self) -> Self {
58        // Each clone starts idle — no buffered item and no in-flight future.
59        // Cloning does not expand channel capacity.
60        Self {
61            tx: self.tx.clone(),
62            buffered: None,
63            pending: None,
64        }
65    }

68impl<T: 'static> CrossfireSink<T> {
69    /// Attempts to send `item` without waiting for channel space.
70    ///
71    /// Returns `Err(TrySendError::Full(item))` when the channel is at capacity,
72    /// or `Err(TrySendError::Disconnected(item))` when all receivers have been dropped.
73    /// Never inflates channel capacity — capacity is strict regardless of clone count.
74    pub fn try_send(&self, item: T) -> Result<(), crossfire::TrySendError<T>> {
75        self.tx.try_send(item)
76    }
77}

79impl<T: Send + Unpin + 'static> futures::Sink<T> for CrossfireSink<T> {
80    type Error = CrossfireSinkError;
81
82    fn poll_ready(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
83        let this = self.get_mut();
84        loop {
85            if let Some(mut fut) = this.pending.take() {
86                // Poll the in-flight send future. While it is Pending the channel is full,
87                // so we return Pending here — this is the waker registration point for
88                // TimeoutSink.
89                match fut.as_mut().poll(cx) {
90                    Poll::Ready(Ok(())) => return Poll::Ready(Ok(())),
91                    Poll::Ready(Err(_)) => return Poll::Ready(Err(CrossfireSinkError::Disconnected)),
92                    Poll::Pending => {
93                        this.pending = Some(fut);
94                        return Poll::Pending;
95                    }
96                }
97            } else if let Some(item) = this.buffered.take() {
98                // An item was stored by `start_send`. Try to place it in the channel.
99                match this.tx.try_send(item) {
100                    Ok(()) => return Poll::Ready(Ok(())),
101                    Err(crossfire::TrySendError::Full(item)) => {
102                        // Channel is full. Create an owned send future so the waker from the
103                        // next poll_ready call is registered with the crossfire scheduler.
104                        let tx = this.tx.clone();
105                        this.pending = Some(Box::pin(async move { tx.send(item).await }));
106                        // Loop once more to immediately poll the new future with the current
107                        // waker, so we don't miss a wake-up that arrived between try_send
108                        // and now.
109                    }
110                    Err(crossfire::TrySendError::Disconnected(_)) => {
111                        return Poll::Ready(Err(CrossfireSinkError::Disconnected));
112                    }
113                }
114            } else {
115                return Poll::Ready(Ok(()));
116            }
117        }
118    }
119
120    fn start_send(self: Pin<&mut Self>, item: T) -> Result<(), Self::Error> {
121        let this = self.get_mut();
122        debug_assert!(
123            this.buffered.is_none() && this.pending.is_none(),
124            "start_send called without a preceding successful poll_ready"
125        );
126        this.buffered = Some(item);
127        Ok(())
128    }
129
130    fn poll_flush(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
131        // Drive buffered/pending to completion.
132        self.poll_ready(cx)
133    }
134
135    fn poll_close(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
136        // Flush the remaining buffered item. The channel closes automatically when
137        // all CrossfireSink clones are dropped.
138        self.poll_flush(cx)
139    }
140}

142/// Creates a strict-capacity bounded MPSC channel returning a [`CrossfireSink`] sender
143/// and a [`futures::Stream`] receiver.
144///
145/// The channel holds at most `capacity` items at any time, independent of how many times
146/// the sender is cloned. This contrasts with `futures::channel::mpsc::channel(N)`, where
147/// each live `Sender` clone silently adds one extra slot.
148pub fn bounded_sink_channel<T: Send + Unpin + 'static>(
149    capacity: usize,
150) -> (CrossfireSink<T>, futures::stream::BoxStream<'static, T>) {
151    let (tx, rx) = crossfire::mpsc::bounded_async::<T>(capacity);
152    let sink = CrossfireSink {
153        tx,
154        buffered: None,
155        pending: None,
156    };
157    (sink, rx.into_stream().boxed())
158}

170    #[test_log::test(tokio::test)]
171    async fn capacity_should_be_buffer_not_buffer_plus_clones() -> anyhow::Result<()> {
172        let capacity = 4usize;
173        let (tx, _rx) = bounded_sink_channel::<u32>(capacity);
174
175        // Clone the sender many times — this inflates capacity in futures::mpsc but not here.
176        let _clones: Vec<_> = (0..100).map(|_| tx.clone()).collect();
177
178        // Fill exactly `capacity` items via a separate clone so `tx` stays idle.
179        let mut filler = tx.clone();
180        for i in 0..capacity as u32 {
181            filler.send(i).await.context("send within capacity")?;
182        }
183
184        // A send beyond capacity must block. We race it against a short sleep: if the
185        // sleep wins, the send correctly blocked; if the send wins first, capacity grew.
186        let mut overflow = tx.clone();
187        tokio::select! {
188            result = overflow.send(999u32) => {
189                anyhow::bail!(
190                    "send on full channel should block regardless of clone count, got: {result:?}"
191                );
192            }
193            _ = tokio::time::sleep(Duration::from_millis(20)) => {
194                // Correct: send blocked because capacity is strictly 4, not 4 + clones.
195            }
196        }
197
198        Ok(())
199    }

215    #[test_log::test(tokio::test)]
216    async fn with_timeout_should_fire_when_channel_is_full() -> anyhow::Result<()> {
217        let capacity = 2usize;
218        let (mut tx, _rx) = bounded_sink_channel::<u32>(capacity);
219
220        // Fill to capacity; _rx is held but not polled, so the channel stays full.
221        for i in 0..capacity as u32 {
222            tx.send(i).await.context("send within capacity")?;
223        }
224
225        // TimeoutSink::poll_ready fires only when inner poll_ready returns Pending.
226        // CrossfireSink::poll_ready returns Pending only when it holds a buffered item
227        // that cannot be sent (channel full).
228        //
229        // `feed` calls poll_ready then start_send but does NOT flush — so the first
230        // feed stores 999 in CrossfireSink's buffer without blocking (poll_ready sees
231        // an empty sink → Ready). The second feed's poll_ready finds the buffered item,
232        // attempts try_send → Full → Pending, allowing TimeoutSink to arm its timer.
233        let mut timed_tx = (&mut tx).with_timeout(Duration::from_millis(10));
234        timed_tx.feed(999u32).await.context("first feed should succeed")?;
235        let result = timed_tx.feed(998u32).await;
236        assert!(
237            matches!(result, Err(SinkTimeoutError::Timeout)),
238            "expected Timeout when channel is full and sink has buffered item, got {result:?}"
239        );
240
241        Ok(())
242    }
