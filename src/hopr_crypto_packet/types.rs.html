15pub type HoprSurbId = [u8; SURB_ID_SIZE];

17/// Identifier of the packet sender.
18///
19/// This consists of two parts:
20/// - [`HoprSenderId::pseudonym`] of the sender
21/// - [`HoprSenderId::surb_id`] is an identifier a single SURB that routes the packet back to the sender.
22///
23/// The `surb_id` always identifies a single SURB. The instance can be turned into a pseudorandom
24/// sequence using [`HoprSenderId::into_sequence`] to create identifiers for more SURBs
25/// with the same pseudonym.
26#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
27#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
28pub struct HoprSenderId(#[cfg_attr(feature = "serde", serde(with = "serde_bytes"))] [u8; Self::SIZE]);
29
30impl HoprSenderId {
31    pub fn new(pseudonym: &HoprPseudonym) -> Self {
32        let mut ret: [u8; Self::SIZE] = hopr_crypto_random::random_bytes();
33        ret[..HoprPseudonym::SIZE].copy_from_slice(pseudonym.as_ref());
34        Self(ret)
35    }
36
37    pub fn from_pseudonym_and_id(pseudonym: &HoprPseudonym, id: HoprSurbId) -> Self {
38        let mut ret = [0u8; Self::SIZE];
39        ret[..HoprPseudonym::SIZE].copy_from_slice(pseudonym.as_ref());
40        ret[HoprPseudonym::SIZE..HoprPseudonym::SIZE + SURB_ID_SIZE].copy_from_slice(&id);
41        Self(ret)
42    }
43
44    pub fn pseudonym(&self) -> HoprPseudonym {
45        HoprPseudonym::try_from(&self.0[..HoprPseudonym::SIZE]).expect("must have valid pseudonym")
46    }
47
48    pub fn surb_id(&self) -> HoprSurbId {
49        self.0[HoprPseudonym::SIZE..HoprPseudonym::SIZE + SURB_ID_SIZE]
50            .try_into()
51            .expect("must have valid nonce")
52    }
53
54    pub fn into_sequence(self) -> impl Iterator<Item = Self> {
55        std::iter::successors(Some((1u32, self)), |&(i, prev)| {
56            let hash = Hash::create(&[&i.to_be_bytes(), prev.as_ref()]);
57            Some((
58                i + 1,
59                Self::from_pseudonym_and_id(&prev.pseudonym(), hash.as_ref()[0..SURB_ID_SIZE].try_into().unwrap()),
60            ))
61        })
62        .map(|(_, v)| v)
63    }
64}
65
66impl AsRef<[u8]> for HoprSenderId {
67    fn as_ref(&self) -> &[u8] {
68        &self.0
69    }
70}

72impl<'a> TryFrom<&'a [u8]> for HoprSenderId {
73    type Error = GeneralError;
74
75    fn try_from(value: &'a [u8]) -> Result<Self, Self::Error> {
76        value
77            .try_into()
78            .map(Self)
79            .map_err(|_| GeneralError::ParseError("HoprPacketReceiverData.size".into()))
80    }
81}

85}

89        Self::new(&HoprPseudonym::random())
90    }

93/// Additional encoding of a packet message that can be preceded by a number of [`SURBs`](SURB).
94pub struct PacketMessage<S: SphinxSuite, H: SphinxHeaderSpec, const P: usize>(PaddedPayload<P>, PhantomData<(S, H)>);
95
96/// Convenience alias for HOPR specific [`PacketMessage`].
97pub type HoprPacketMessage = PacketMessage<HoprSphinxSuite, HoprSphinxHeaderSpec, PAYLOAD_SIZE_INT>;
98
99/// Individual parts of a [`PacketMessage`]: SURBs and the actual message.
100pub type PacketParts<S, H> = (Vec<SURB<S, H>>, Box<[u8]>);
101
102impl<S: SphinxSuite, H: SphinxHeaderSpec, const P: usize> PacketMessage<S, H, P> {
103    /// Size of the message header.
104    ///
105    /// This is currently 1 byte to indicate the number of SURBs, that precede the message.
106    pub const HEADER_LEN: usize = 1;
107
108    /// Converts this instance into [`PacketParts`].
109    pub fn try_into_parts(self) -> Result<PacketParts<S, H>, SphinxError> {
110        let data = self.0.into_unpadded()?;
111        let num_surbs = data[0] as usize;
112
113        if num_surbs > 0 {
114            let surb_end = num_surbs * SURB::<S, H>::SIZE;
115            if surb_end >= data.len() {
116                return Err(GeneralError::ParseError("HoprPacketMessage.num_surbs not valid".into()).into());
117            }
118
119            let mut data = data.into_vec();
120
121            let surb_data = data.drain(0..=surb_end).skip(1).collect::<Vec<_>>();
122
123            let surbs = surb_data
124                .as_slice()
125                .chunks_exact(SURB::<S, H>::SIZE)
126                .map(SURB::<S, H>::try_from)
127                .collect::<Result<Vec<_>, _>>()?;
128
129            Ok((surbs, data.into_boxed_slice()))
130        } else {
131            let mut data = data.into_vec();
132            data.remove(0);
133            Ok((Vec::with_capacity(0), data.into_boxed_slice()))
134        }
135    }
136
137    /// Allocates a new instance from the given parts.
138    pub fn from_parts(surbs: Vec<SURB<S, H>>, payload: &[u8]) -> Result<Self, SphinxError> {
139        if surbs.len() > 255 {
140            return Err(GeneralError::ParseError("HoprPacketMessage.num_surbs not valid".into()).into());
141        }
142
143        // The total size of the packet message must not exceed the maximum packet size.
144        if Self::HEADER_LEN + surbs.len() * SURB::<S, H>::SIZE + payload.len() > P {
145            return Err(GeneralError::ParseError("HoprPacketMessage.size not valid".into()).into());
146        }
147
148        let mut ret = Vec::with_capacity(PaddedPayload::<P>::SIZE);
149        ret.push(surbs.len() as u8);
150        for surb in surbs.into_iter().map(|s| s.into_boxed()) {
151            ret.extend_from_slice(surb.as_ref());
152        }
153        ret.extend_from_slice(payload);
154
155        // Save one reallocation by using the vector that we just created
156        Ok(Self(PaddedPayload::new_from_vec(ret)?, PhantomData))
157    }
158}

160impl<S: SphinxSuite, H: SphinxHeaderSpec, const P: usize> From<PaddedPayload<P>> for PacketMessage<S, H, P> {
161    fn from(value: PaddedPayload<P>) -> Self {
162        Self(value, PhantomData)
163    }
164}

166impl<S: SphinxSuite, H: SphinxHeaderSpec, const P: usize> From<PacketMessage<S, H, P>> for PaddedPayload<P> {
167    fn from(value: PacketMessage<S, H, P>) -> Self {
168        value.0
169    }
170}

204    fn generate_surbs(count: usize) -> anyhow::Result<Vec<SURB<HoprSphinxSuite, HoprSphinxHeaderSpec>>> {
205        let path = PEERS.iter().map(|(_, k)| *k.public()).collect::<Vec<_>>();
206        let path_ids =
207            <BiHashMap<_, _> as KeyIdMapper<HoprSphinxSuite, HoprSphinxHeaderSpec>>::map_keys_to_ids(&*MAPPER, &path)
208                .into_iter()
209                .map(|v| v.ok_or(anyhow!("missing id")))
210                .collect::<Result<Vec<_>, _>>()?;
211        let pseudonym = SimplePseudonym::random();
212        let recv_data = HoprSenderId::new(&pseudonym);
213
214        Ok((0..count)
215            .map(|_| {
216                let shared_keys = HoprSphinxSuite::new_shared_keys(&path)?;
217                let (por_strings, por_values) = generate_proof_of_relay(&shared_keys.secrets)
218                    .map_err(|e| CryptoError::Other(GeneralError::NonSpecificError(e.to_string())))?;
219
220                create_surb::<HoprSphinxSuite, HoprSphinxHeaderSpec>(
221                    shared_keys,
222                    &path_ids,
223                    &por_strings,
224                    recv_data,
225                    SurbReceiverInfo::new(por_values, [0u8; 32]),
226                )
227                .map(|(s, _)| s)
228            })
229            .collect::<Result<Vec<_>, _>>()?)
230    }
