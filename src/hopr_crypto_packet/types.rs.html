14   /// Individual packet signals passed up between the packet sender and destination.
15   #[repr(u8)]
16   #[derive(PartialOrd, Ord, strum::EnumString, strum::Display)]
17   pub enum PacketSignal: u8 {
18        /// The other party is in a "SURB distress" state, potentially running out of SURBs soon.
19        ///
20        /// Has no effect on packets that take the "forward path".
21        SurbDistress = 0b0000_0001,
22        /// The other party has run out of SURBs, and this was potentially the last message they could
23        /// send.
24        ///
25        /// Has no effect on packets that take the "forward path".
26        ///
27        /// Implies [`SurbDistress`].
28        OutOfSurbs = 0b0000_0011,
29   }
30}

41/// An ID that uniquely identifies SURB for a certain pseudonym.
42pub type HoprSurbId = [u8; SURB_ID_SIZE];

44/// Identifier of a single packet sender.
45///
46/// This consists of two parts:
47/// - [`HoprSenderId::pseudonym`] of the sender
48/// - [`HoprSenderId::surb_id`] is an identifier a single SURB that routes the packet back to the sender.
49///
50/// The `surb_id` always identifies a single SURB. The instance can be turned into a pseudorandom
51/// sequence using [`HoprSenderId::into_sequence`] to create identifiers for more SURBs
52/// with the same pseudonym.
53#[derive(Clone, Copy, PartialEq, Eq, Hash)]
54#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
55pub struct HoprSenderId(#[cfg_attr(feature = "serde", serde(with = "serde_bytes"))] [u8; Self::SIZE]);

66impl HoprSenderId {
67    pub fn new(pseudonym: &HoprPseudonym) -> Self {
68        let mut ret: [u8; Self::SIZE] = hopr_crypto_random::random_bytes();
69        ret[..HoprPseudonym::SIZE].copy_from_slice(pseudonym.as_ref());
70        Self(ret)
71    }
72
73    pub fn from_pseudonym_and_id(pseudonym: &HoprPseudonym, id: HoprSurbId) -> Self {
74        let mut ret = [0u8; Self::SIZE];
75        ret[..HoprPseudonym::SIZE].copy_from_slice(pseudonym.as_ref());
76        ret[HoprPseudonym::SIZE..HoprPseudonym::SIZE + SURB_ID_SIZE].copy_from_slice(&id);
77        Self(ret)
78    }
79
80    pub fn pseudonym(&self) -> HoprPseudonym {
81        HoprPseudonym::try_from(&self.0[..HoprPseudonym::SIZE]).expect("must have valid pseudonym")
82    }
83
84    pub fn surb_id(&self) -> HoprSurbId {
85        self.0[HoprPseudonym::SIZE..HoprPseudonym::SIZE + SURB_ID_SIZE]
86            .try_into()
87            .expect("must have valid nonce")
88    }
89
90    /// Creates a pseudorandom sequence of IDs.
91    ///
92    /// Each item has the same [`pseudonym`](HoprSenderId::pseudonym)
93    /// but different [`surb_id`](HoprSenderId::surb_id).
94    ///
95    /// The `surb_id` of the `n`-th item (n > 1) is computed as `Blake3(n || I_prev)`
96    /// where `I_prev` is the whole `n-1`-th ID, the `n` is represented as big-endian and
97    /// `||` denotes byte-array concatenation.
98    /// The first item (n = 1) is always `self`.
99    ///
100    /// The entropy of the whole pseudorandom sequence is completely given by `self` (the first
101    /// item in the sequence). It follows that the next element of the sequence can be computed
102    /// by just knowing any preceding element; therefore, the sequence is fully predictable
103    /// once an element is known.
104    pub fn into_sequence(self) -> impl Iterator<Item = Self> {
105        std::iter::successors(Some((1u32, self)), |&(i, prev)| {
106            let hash = HashFast::create(&[&i.to_be_bytes(), prev.as_ref()]);
107            Some((
108                i + 1,
109                Self::from_pseudonym_and_id(&prev.pseudonym(), hash.as_ref()[0..SURB_ID_SIZE].try_into().unwrap()),
110            ))
111        })
112        .map(|(_, v)| v)
113    }
114}

149/// Individual parts of a [`PacketMessage`]: SURBs, the actual message (payload) and additional signals for the

208// Coerces PacketSignals to only lower 4 bits.
209pub(crate) const S_MASK: u8 = 0b0000_1111;
210
211impl<S: SphinxSuite, H: SphinxHeaderSpec, const P: usize> PacketMessage<S, H, P> {
212    /// Size of the message header.
213    ///
214    /// This is currently 1 byte to indicate the number of SURBs that precede the message.
215    pub const HEADER_LEN: usize = 1;
216    /// The maximum number of SURBs a packet message can hold, according to RFC-0003.
217    ///
218    /// The number of SURBs in a `PacketMessage` is intentionally limited to 15, so that
219    /// the upper 4 bits remain reserved for additional flags.
220    pub const MAX_SURBS_PER_MESSAGE: usize = S_MASK as usize;
221}
222
223impl<S: SphinxSuite, H: SphinxHeaderSpec, const P: usize> TryFrom<PacketParts<'_, S, H>> for PacketMessage<S, H, P> {
224    type Error = SphinxError;
225
226    fn try_from(value: PacketParts<S, H>) -> Result<Self, Self::Error> {
227        if value.surbs.len() > Self::MAX_SURBS_PER_MESSAGE {
228            return Err(GeneralError::ParseError("HoprPacketMessage.num_surbs not valid".into()).into());
229        }
230
231        if value.signals.bits() > S_MASK {
232            return Err(GeneralError::ParseError("HoprPacketMessage.flags not valid".into()).into());
233        }
234
235        // The total size of the packet message must not exceed the maximum packet size.
236        if Self::HEADER_LEN + value.surbs.len() * SURB::<S, H>::SIZE + value.payload.len() > P {
237            return Err(GeneralError::ParseError("HoprPacketMessage.size not valid".into()).into());
238        }
239
240        let mut ret = Vec::with_capacity(PaddedPayload::<P>::SIZE);
241        let flags_and_len = (value.signals.bits() << S_MASK.trailing_ones()) | (value.surbs.len() as u8 & S_MASK);
242        ret.push(flags_and_len);
243        for surb in value.surbs.into_iter().map(|s| s.into_boxed()) {
244            ret.extend(surb);
245        }
246        ret.extend_from_slice(value.payload.as_ref());
247
248        // Save one reallocation by using the vector that we just created
249        Ok(Self(PaddedPayload::new_from_vec(ret)?, PhantomData))
250    }
251}

253impl<S: SphinxSuite, H: SphinxHeaderSpec, const P: usize> TryFrom<PacketMessage<S, H, P>> for PacketParts<'_, S, H> {
254    type Error = SphinxError;
255
256    fn try_from(value: PacketMessage<S, H, P>) -> Result<Self, Self::Error> {
257        let data = value.0.into_unpadded()?;
258        if data.is_empty() {
259            return Err(GeneralError::ParseError("HoprPacketMessage.size".into()).into());
260        }
261
262        let num_surbs = (data[0] & S_MASK) as usize;
263        let signals = PacketSignals::new((data[0] & S_MASK.not()) >> S_MASK.trailing_ones())
264            .map_err(|_| GeneralError::ParseError("HoprPacketMessage.signals".into()))?;
265
266        if num_surbs > 0 {
267            let surb_end = num_surbs * SURB::<S, H>::SIZE;
268            if surb_end >= data.len() {
269                return Err(GeneralError::ParseError("HoprPacketMessage.num_surbs not valid".into()).into());
270            }
271
272            let mut data = data.into_vec();
273
274            let surbs = data[1..=surb_end]
275                .chunks_exact(SURB::<S, H>::SIZE)
276                .map(SURB::<S, H>::try_from)
277                .collect::<Result<Vec<_>, _>>()?;
278
279            // Skip buffer all the way to the end of the SURBs.
280            data.drain(0..=surb_end).for_each(drop);
281
282            Ok(PacketParts {
283                surbs,
284                payload: Cow::Owned(data),
285                signals,
286            })
287        } else {
288            let mut data = data.into_vec();
289            data.remove(0);
290            Ok(PacketParts {
291                surbs: Vec::with_capacity(0),
292                payload: Cow::Owned(data),
293                signals,
294            })
295        }
296    }
297}
