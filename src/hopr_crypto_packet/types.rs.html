8/// Additional encoding of a packet message that can be preceded by a number of [`SURBs`](SURB).
9pub struct PacketMessage<S: SphinxSuite, H: SphinxHeaderSpec, const P: usize>(PaddedPayload<P>, PhantomData<(S, H)>);
10
11/// Convenience alias for HOPR specific [`PacketMessage`].
12pub type HoprPacketMessage = PacketMessage<HoprSphinxSuite, HoprSphinxHeaderSpec, PAYLOAD_SIZE_INT>;
13
14/// Individual parts of a [`PacketMessage`]: SURBs and the actual message.
15pub type PacketParts<S, H> = (Vec<SURB<S, H>>, Box<[u8]>);
16
17impl<S: SphinxSuite, H: SphinxHeaderSpec, const P: usize> PacketMessage<S, H, P> {
18    /// Size of the message header.
19    ///
20    /// This is currently 1 byte to indicate the number of SURBs, that precede the message.
21    pub const HEADER_LEN: usize = 1;
22
23    /// Converts this instance into [`PacketParts`].
24    pub fn try_into_parts(self) -> Result<PacketParts<S, H>, SphinxError> {
25        let data = self.0.into_unpadded()?;
26        let num_surbs = data[0] as usize;
27
28        if num_surbs > 0 {
29            let surb_end = num_surbs * SURB::<S, H>::SIZE;
30            if surb_end >= data.len() {
31                return Err(GeneralError::ParseError("HoprPacketMessage.num_surbs not valid".into()).into());
32            }
33
34            let mut data = data.into_vec();
35
36            let surb_data = data.drain(0..=surb_end).skip(1).collect::<Vec<_>>();
37
38            let surbs = surb_data
39                .as_slice()
40                .chunks_exact(SURB::<S, H>::SIZE)
41                .map(SURB::<S, H>::try_from)
42                .collect::<Result<Vec<_>, _>>()?;
43
44            Ok((surbs, data.into_boxed_slice()))
45        } else {
46            let mut data = data.into_vec();
47            data.remove(0);
48            Ok((Vec::with_capacity(0), data.into_boxed_slice()))
49        }
50    }
51
52    /// Allocates a new instance from the given parts.
53    pub fn from_parts(surbs: Vec<SURB<S, H>>, payload: &[u8]) -> Result<Self, SphinxError> {
54        if surbs.len() > 255 {
55            return Err(GeneralError::ParseError("HoprPacketMessage.num_surbs not valid".into()).into());
56        }
57
58        // The total size of the packet message must not exceed the maximum packet size.
59        if Self::HEADER_LEN + surbs.len() * SURB::<S, H>::SIZE + payload.len() > P {
60            return Err(GeneralError::ParseError("HoprPacketMessage.size not valid".into()).into());
61        }
62
63        let mut ret = Vec::with_capacity(PaddedPayload::<P>::SIZE);
64        ret.push(surbs.len() as u8);
65        for surb in surbs.into_iter().map(|s| s.into_boxed()) {
66            ret.extend_from_slice(surb.as_ref());
67        }
68        ret.extend_from_slice(payload);
69
70        // Save one reallocation by using the vector that we just created
71        Ok(Self(PaddedPayload::new_from_vec(ret)?, PhantomData))
72    }
73}

118    fn generate_surbs(count: usize) -> anyhow::Result<Vec<SURB<HoprSphinxSuite, HoprSphinxHeaderSpec>>> {
119        let path = PEERS.iter().map(|(_, k)| k.public().clone()).collect::<Vec<_>>();
120        let path_ids =
121            <BiHashMap<_, _> as KeyIdMapper<HoprSphinxSuite, HoprSphinxHeaderSpec>>::map_keys_to_ids(&*MAPPER, &path)
122                .into_iter()
123                .map(|v| v.ok_or(anyhow!("missing id")))
124                .collect::<Result<Vec<_>, _>>()?;
125        let pseudonym = SimplePseudonym::random();
126
127        Ok((0..count)
128            .into_iter()
129            .map(|_| {
130                let shared_keys = HoprSphinxSuite::new_shared_keys(&path)?;
131                let (por_strings, por_values) = generate_proof_of_relay(&shared_keys.secrets)
132                    .map_err(|e| CryptoError::Other(GeneralError::NonSpecificError(e.to_string())))?;
133
134                create_surb::<HoprSphinxSuite, HoprSphinxHeaderSpec>(
135                    shared_keys,
136                    &path_ids,
137                    &por_strings,
138                    &pseudonym,
139                    por_values,
140                )
141                .map(|(s, _)| s)
142            })
143            .collect::<Result<Vec<_>, _>>()?)
144    }
