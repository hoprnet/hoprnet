1//! The discovery mechanism uses an external stimulus to trigger the discovery
2//! process on the libp2p side. It is responsible for processing the events
3//! generated by other components and passing them to the libp2p swarm in
4//! an appropriate format.
5use std::collections::{HashMap, HashSet, VecDeque};
6
7use futures::stream::{BoxStream, Stream, StreamExt};
8use hopr_transport_protocol::PeerDiscovery;
9use libp2p::{
10    Multiaddr, PeerId,
11    core::Endpoint,
12    swarm::{
13        CloseConnection, ConnectionDenied, ConnectionId, NetworkBehaviour, ToSwarm, dial_opts::DialOpts,
14        dummy::ConnectionHandler,
15    },
16};
17use tracing::{debug, trace};
18
19#[derive(Debug)]
20pub enum DiscoveryInput {
21    Indexer(PeerDiscovery),
22}
23
24#[derive(Debug)]
25pub enum Event {}
26
27pub struct Behaviour {
28    me: PeerId,
29    events: BoxStream<'static, DiscoveryInput>,
30    pending_events: VecDeque<
31        libp2p::swarm::ToSwarm<
32            <Self as NetworkBehaviour>::ToSwarm,
33            <<Self as NetworkBehaviour>::ConnectionHandler as libp2p::swarm::ConnectionHandler>::FromBehaviour,
34        >,
35    >,
36    all_peers: HashMap<PeerId, Vec<Multiaddr>>,
37    allowed_peers: HashSet<PeerId>,
38    connected_peers: HashMap<PeerId, usize>,
39}
40
41impl Behaviour {
42    pub fn new<T>(me: PeerId, onchain_events: T) -> Self
43    where
44        T: Stream<Item = PeerDiscovery> + Send + 'static,
45    {
46        Self {
47            me,
48            events: Box::pin(onchain_events.map(DiscoveryInput::Indexer)),
49            all_peers: HashMap::new(),
50            pending_events: VecDeque::new(),
51            allowed_peers: HashSet::new(),
52            connected_peers: HashMap::new(),
53        }
54    }
55
56    fn is_peer_connected(&self, peer: &PeerId) -> bool {
57        self.connected_peers.get(peer).map(|v| *v > 0).unwrap_or(false)
58    }
59}
60
61impl NetworkBehaviour for Behaviour {
62    type ConnectionHandler = ConnectionHandler;
63    type ToSwarm = Event;
64
65    #[tracing::instrument(level = "debug", skip(self), fields(transport = "p2p discovery"), err(Display))]
66    fn handle_established_inbound_connection(
67        &mut self,
68        connection_id: libp2p::swarm::ConnectionId,
69        peer: libp2p::PeerId,
70        local_addr: &libp2p::Multiaddr,
71        remote_addr: &libp2p::Multiaddr,
72    ) -> Result<libp2p::swarm::THandler<Self>, libp2p::swarm::ConnectionDenied> {
73        if self.allowed_peers.contains(&peer) {
74            trace!(%peer, "Discovery::handle_established_inbound_connection in allowed_peers");
75            Ok(Self::ConnectionHandler {})
76        } else {
77            trace!(%peer, "Discovery::handle_established_inbound_connection NOT in allowed_peers");
78            Err(libp2p::swarm::ConnectionDenied::new(crate::errors::P2PError::Logic(
79                format!("Connection from '{peer}' is not allowed"),
80            )))
81        }
82    }
83
84    #[tracing::instrument(
85        level = "debug",
86        skip(self),
87        fields(transport = "p2p discovery"),
88        ret(Debug),
89        err(Display)
90    )]
91    fn handle_pending_outbound_connection(
92        &mut self,
93        connection_id: ConnectionId,
94        maybe_peer: Option<PeerId>,
95        addresses: &[Multiaddr],
96        effective_role: Endpoint,
97    ) -> Result<Vec<Multiaddr>, ConnectionDenied> {
98        if let Some(peer) = maybe_peer {
99            if self.allowed_peers.contains(&peer) {
100                trace!(%peer, "Discovery::handle_pending_outbound_connection in allowed_peers");
101                // inject the multiaddress of the peer for possible dial usage by stream protocols
102                return Ok(self.all_peers.get(&peer).map_or_else(
103                    || {
104                        tracing::debug!(%peer, "No multiaddress found for peer");
105                        vec![]
106                    },
107                    |addresses| addresses.clone(),
108                ));
109            } else {
110                trace!(%peer, "Discovery::handle_pending_outbound_connection NOT in allowed_peers");
111                return Err(libp2p::swarm::ConnectionDenied::new(crate::errors::P2PError::Logic(
112                    format!("Connection to '{peer}' is not allowed"),
113                )));
114            }
115        }
116
117        Ok(vec![])
118    }
119
120    #[tracing::instrument(level = "debug", skip(self), fields(transport = "p2p discovery"), err(Display))]
121    fn handle_established_outbound_connection(
122        &mut self,
123        connection_id: libp2p::swarm::ConnectionId,
124        peer: libp2p::PeerId,
125        addr: &libp2p::Multiaddr,
126        role_override: libp2p::core::Endpoint,
127        port_use: libp2p::core::transport::PortUse,
128    ) -> Result<libp2p::swarm::THandler<Self>, libp2p::swarm::ConnectionDenied> {
129        // cannot connect without the handle_ending_outbound_connection being called first
130        Ok(Self::ConnectionHandler {})
131    }
132
133    fn on_swarm_event(&mut self, event: libp2p::swarm::FromSwarm) {
134        match event {
135            libp2p::swarm::FromSwarm::ConnectionEstablished(data) => {
136                *self.connected_peers.entry(data.peer_id).or_insert(0) += 1
137            }
138            libp2p::swarm::FromSwarm::ConnectionClosed(data) => {
139                let v = self.connected_peers.entry(data.peer_id).or_insert(0);
140                if *v > 0 {
141                    *v -= 1;
142                };
143            }
144            _ => {}
145        }
146    }
147
148    fn on_connection_handler_event(
149        &mut self,
150        _peer_id: libp2p::PeerId,
151        _connection_id: libp2p::swarm::ConnectionId,
152        _event: libp2p::swarm::THandlerOutEvent<Self>,
153    ) {
154        // Nothing is necessary here, because no ConnectionHandler events should be generated
155    }
156
157    #[tracing::instrument(
158        level = "debug",
159        name = "poll_on_event",
160        skip(self, cx),
161        fields(transport = "p2p discovery")
162    )]
163    fn poll(
164        &mut self,
165        cx: &mut std::task::Context<'_>,
166    ) -> std::task::Poll<libp2p::swarm::ToSwarm<Self::ToSwarm, libp2p::swarm::THandlerInEvent<Self>>> {
167        if let Some(value) = self.pending_events.pop_front() {
168            return std::task::Poll::Ready(value);
169        };
170
171        let poll_result = self.events.poll_next_unpin(cx).map(|e| match e {
172            Some(DiscoveryInput::Indexer(event)) => match event {
173                PeerDiscovery::Allow(peer) => {
174                    let inserted_into_allow_list = self.allowed_peers.insert(peer);
175                    debug!(%peer, inserted_into_allow_list, "Discovery::Poll Insert allowed_peers");
176
177                    let multiaddresses = self.all_peers.get(&peer);
178                    if let Some(multiaddresses) = self.all_peers.get(&peer) {
179                        for address in multiaddresses {
180                            self.pending_events.push_back(ToSwarm::NewExternalAddrOfPeer {
181                                peer_id: peer,
182                                address: address.clone(),
183                            });
184                        }
185                    }
186
187                    tracing::debug!(%peer, inserted_into_allow_list, emitted_libp2p_address_announce = multiaddresses.is_some_and(|v| !v.is_empty()), "Network registry allow");
188                }
189                PeerDiscovery::Ban(peer) => {
190                    let was_allowed = self.allowed_peers.remove(&peer);
191                    debug!(%peer, was_allowed, "Discovery::Poll Remove allowed_peers");
192                    tracing::debug!(%peer, was_allowed, "Network registry ban");
193
194                    if self.is_peer_connected(&peer) {
195                        self.pending_events.push_back(ToSwarm::CloseConnection {
196                            peer_id: peer,
197                            connection: CloseConnection::default(),
198                        });
199                        tracing::debug!(%peer, "Requesting disconnect due to ban");
200                    }
201                }
202                PeerDiscovery::Announce(peer, multiaddresses) => {
203                    if peer != self.me {
204                        tracing::debug!(%peer, addresses = ?&multiaddresses, "Announcement");
205
206                        for multiaddress in &multiaddresses {
207                            self.pending_events.push_back(ToSwarm::NewExternalAddrOfPeer {
208                                peer_id: peer,
209                                address: multiaddress.clone(),
210                            });
211                        }
212
213                        self.all_peers.insert(peer, multiaddresses.clone());
214
215                        if self.allowed_peers.contains(&peer) {
216                            // the dial is performed to create a first connection some time before the heartbeat mechanism
217                            // kicks in, otherwise the heartbeat is likely to fail on the first try due to dial and protocol
218                            // negotiation taking longer than the request response timeout
219                            self.pending_events.push_back(ToSwarm::Dial { opts: DialOpts::peer_id(peer).addresses(multiaddresses).build()});
220                        }
221                    }
222                }
223            },
224            None => {}
225        });
226
227        if matches!(poll_result, std::task::Poll::Pending) {
228            std::task::Poll::Pending
229        } else if let Some(value) = self.pending_events.pop_front() {
230            std::task::Poll::Ready(value)
231        } else {
232            std::task::Poll::Pending
233        }
234    }
235}
