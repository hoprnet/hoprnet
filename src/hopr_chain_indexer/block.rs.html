57/// Indexer
58///
59/// Accepts the RPC operational functionality [hopr_chain_rpc::HoprIndexerRpcOperations]
60/// and provides the indexing operation resulting in and output of
61/// [hopr_chain_types::chain_events::SignificantChainEvent] streamed outside the indexer by the unbounded channel.
62///
63/// The roles of the indexer:
64/// 1. prime the RPC endpoint
65/// 2. request an RPC stream of changes to process
66/// 3. process block and log stream
67/// 4. ensure finalization by postponing processing until the head is far enough
68/// 5. store relevant data into the DB
69/// 6. pass the processing on to the business logic
70#[derive(Debug, Clone)]
71pub struct Indexer<T, U, Db>
72where
73    T: HoprIndexerRpcOperations + Send + 'static,
74    U: ChainLogHandler + Send + 'static,
75    Db: HoprDbGeneralModelOperations + HoprDbInfoOperations + HoprDbLogOperations + Clone + Send + Sync + 'static,
76{
77    rpc: Option<T>,
78    db_processor: Option<U>,
79    db: Db,
80    cfg: IndexerConfig,
81    egress: async_channel::Sender<SignificantChainEvent>,
82    // If true (default), the indexer will panic if the event stream is terminated.
83    // Setting it to false is useful for testing.
84    panic_on_completion: bool,
85}
86
87impl<T, U, Db> Indexer<T, U, Db>
88where
89    T: HoprIndexerRpcOperations + Sync + Send + 'static,
90    U: ChainLogHandler + Send + Sync + 'static,
91    Db: HoprDbGeneralModelOperations + HoprDbInfoOperations + HoprDbLogOperations + Clone + Send + Sync + 'static,

116    pub async fn start(mut self) -> Result<AbortHandle>
117    where
118        T: HoprIndexerRpcOperations + 'static,
119        U: ChainLogHandler + 'static,
120        Db: HoprDbGeneralModelOperations + HoprDbInfoOperations + HoprDbLogOperations + Clone + Send + Sync + 'static,
121    {
122        if self.rpc.is_none() || self.db_processor.is_none() {
123            return Err(CoreEthereumIndexerError::ProcessError(
124                "indexer cannot start, missing components".into(),
125            ));
126        }
127
128        info!("Starting chain indexing");
129
130        let rpc = self.rpc.take().expect("rpc should be present");
131        let logs_handler = Arc::new(self.db_processor.take().expect("db_processor should be present"));
132        let db = self.db.clone();
133        let tx_significant_events = self.egress.clone();
134        let panic_on_completion = self.panic_on_completion;
135
136        let (log_filters, address_topics) = Self::generate_log_filters(&logs_handler);
137
138        // Check that the contract addresses and topics are consistent with what is in the logs DB,
139        // or if the DB is empty, prime it with the given addresses and topics.
140        db.ensure_logs_origin(address_topics).await?;
141
142        let is_synced = Arc::new(AtomicBool::new(false));
143        let chain_head = Arc::new(AtomicU64::new(0));
144
145        // update the chain head once at startup to get a reference for initial syncing
146        // progress calculation
147        debug!("Updating chain head at indexer startup");
148        Self::update_chain_head(&rpc, chain_head.clone()).await;
149
150        // First, check whether fast sync is enabled and can be performed.
151        // If so:
152        //   1. Download the snapshot if the logs database is empty and the snapshot is enabled
153        //   2. Delete the existing indexed data
154        //   3. Reset fast sync progress
155        //   4. Run the fast sync process until completion
156        //   5. Finally, starting the rpc indexer.
157        let fast_sync_configured = self.cfg.fast_sync;
158        let index_empty = self.db.index_is_empty().await?;
159
160        // Pre-start operations to ensure the indexer is ready, including snapshot fetching
161        self.pre_start().await?;
162
163        #[derive(PartialEq, Eq)]
164        enum FastSyncMode {
165            None,
166            FromScratch,
167            Continue,
168        }
169
170        let will_perform_fast_sync = match (fast_sync_configured, index_empty) {
171            (true, false) => {
172                info!(
173                    "Fast sync is enabled, but the index database is not empty. Fast sync will continue on existing \
174                     unprocessed logs."
175                );
176                FastSyncMode::Continue
177            }
178            (false, true) => {
179                info!("Fast sync is disabled, but the index database is empty. Doing a full re-sync.");
180                // Clean the last processed log from the Log DB, to allow full resync
181                self.db.clear_index_db(None).await?;
182                self.db.set_logs_unprocessed(None, None).await?;
183                FastSyncMode::None
184            }
185            (false, false) => {
186                info!("Fast sync is disabled and the index database is not empty. Continuing normal sync.");
187                FastSyncMode::None
188            }
189            (true, true) => {
190                info!("Fast sync is enabled, starting the fast sync process");
191                // To ensure a proper state, reset any auxiliary data in the database
192                self.db.clear_index_db(None).await?;
193                self.db.set_logs_unprocessed(None, None).await?;
194                FastSyncMode::FromScratch
195            }
196        };
197
198        let (tx, mut rx) = futures::channel::mpsc::channel::<()>(1);
199
200        // Perform the fast-sync if requested
201        if FastSyncMode::None != will_perform_fast_sync {
202            let processed = match will_perform_fast_sync {
203                FastSyncMode::FromScratch => None,
204                FastSyncMode::Continue => Some(false),
205                _ => unreachable!(),
206            };
207
208            #[cfg(all(feature = "prometheus", not(test)))]
209            {
210                METRIC_INDEXER_SYNC_SOURCE.set(&["fast-sync"], 1.0);
211                METRIC_INDEXER_SYNC_SOURCE.set(&["rpc"], 0.0);
212            }
213
214            let log_block_numbers = self.db.get_logs_block_numbers(None, None, processed).await?;
215            let _first_log_block_number = log_block_numbers.first().copied().unwrap_or(0);
216            let _head = chain_head.load(Ordering::Relaxed);
217            for block_number in log_block_numbers {
218                debug!(
219                    block_number,
220                    first_log_block_number = _first_log_block_number,
221                    head = _head,
222                    "computing processed logs"
223                );
224                // Do not pollute the logs with the fast-sync progress
225                Self::process_block_by_id(&db, &logs_handler, block_number, is_synced.load(Ordering::Relaxed)).await?;
226
227                #[cfg(all(feature = "prometheus", not(test)))]
228                {
229                    let progress =
230                        (block_number - _first_log_block_number) as f64 / (_head - _first_log_block_number) as f64;
231                    METRIC_INDEXER_SYNC_PROGRESS.set(progress);
232                }
233            }
234        }
235
236        info!("Building rpc indexer background process");
237
238        let next_block_to_process = if let Some(last_log) = self.db.get_last_checksummed_log().await? {
239            info!(
240                start_block = last_log.block_number,
241                start_checksum = last_log.checksum.unwrap(),
242                "Loaded indexer state",
243            );
244
245            if self.cfg.start_block_number < last_log.block_number {
246                // If some prior indexing took place already, avoid reprocessing
247                last_log.block_number + 1
248            } else {
249                self.cfg.start_block_number
250            }
251        } else {
252            self.cfg.start_block_number
253        };
254
255        info!(next_block_to_process, "Indexer start point");
256
257        let indexing_abort_handle = hopr_async_runtime::spawn_as_abortable!(
258            async move {
259                // Update the chain head once again
260                debug!("Updating chain head at indexer startup");
261                Self::update_chain_head(&rpc, chain_head.clone()).await;
262
263                #[cfg(all(feature = "prometheus", not(test)))]
264                {
265                    METRIC_INDEXER_SYNC_SOURCE.set(&["fast-sync"], 0.0);
266                    METRIC_INDEXER_SYNC_SOURCE.set(&["rpc"], 1.0);
267                }
268
269                let rpc_ref = &rpc;
270
271                let event_stream = rpc
272                    .try_stream_logs(next_block_to_process, log_filters, is_synced.load(Ordering::Relaxed))
273                    .expect("block stream should be constructible")
274                    .then(|block| {
275                        let db = db.clone();
276                        let chain_head = chain_head.clone();
277                        let is_synced = is_synced.clone();
278                        let tx = tx.clone();
279                        let logs_handler = logs_handler.clone();
280
281                        async move {
282                            Self::calculate_sync_process(
283                                block.block_id,
284                                rpc_ref,
285                                db,
286                                chain_head.clone(),
287                                is_synced.clone(),
288                                next_block_to_process,
289                                tx.clone(),
290                                logs_handler.safe_address().into(),
291                                logs_handler.contract_addresses_map().channels.into(),
292                            )
293                            .await;
294
295                            block
296                        }
297                    })
298                    .filter_map(|block| {
299                        let db = db.clone();
300                        let logs_handler = logs_handler.clone();
301
302                        async move {
303                            debug!(%block, "storing logs from block");
304                            let logs = block.logs.clone();
305
306                            // Filter out the token contract logs because we do not need to store these
307                            // in the database.
308                            let logs_vec = logs
309                                .into_iter()
310                                .filter(|log| log.address != logs_handler.contract_addresses_map().token)
311                                .collect();
312
313                            match db.store_logs(logs_vec).await {
314                                Ok(store_results) => {
315                                    if let Some(error) = store_results
316                                        .into_iter()
317                                        .filter(|r| r.is_err())
318                                        .map(|r| r.unwrap_err())
319                                        .next()
320                                    {
321                                        error!(%block, %error, "failed to processed stored logs from block");
322                                        None
323                                    } else {
324                                        Some(block)
325                                    }
326                                }
327                                Err(error) => {
328                                    error!(%block, %error, "failed to store logs from block");
329                                    None
330                                }
331                            }
332                        }
333                    })
334                    .filter_map(|block| {
335                        let db = db.clone();
336                        let logs_handler = logs_handler.clone();
337                        let is_synced = is_synced.clone();
338                        async move {
339                            Self::process_block(&db, &logs_handler, block, false, is_synced.load(Ordering::Relaxed))
340                                .await
341                        }
342                    })
343                    .flat_map(stream::iter);
344
345                futures::pin_mut!(event_stream);
346                while let Some(event) = event_stream.next().await {
347                    trace!(%event, "processing on-chain event");
348                    // Pass the events further only once we're fully synced
349                    if is_synced.load(Ordering::Relaxed) {
350                        if let Err(error) = tx_significant_events.try_send(event) {
351                            error!(%error, "failed to pass a significant chain event further");
352                        }
353                    }
354                }
355
356                if panic_on_completion {
357                    panic!(
358                        "Indexer event stream has been terminated. This error may be caused by a failed RPC \
359                         connection."
360                    );
361                }
362            }
363            .inspect(|_| tracing::warn!(task = "indexer", "long-running background task finished"))
364        );
365
366        if rx.next().await.is_some() {
367            Ok(indexing_abort_handle)
368        } else {
369            Err(crate::errors::CoreEthereumIndexerError::ProcessError(
370                "Error during indexing start".into(),
371            ))
372        }
373    }

375    pub async fn pre_start(&self) -> Result<()> {
376        let fast_sync_configured = self.cfg.fast_sync;
377        let index_empty = self.db.index_is_empty().await?;
378
379        // Check if we need to download snapshot before fast sync
380        let logs_db_has_data = self.has_logs_data().await?;
381
382        if fast_sync_configured && index_empty && !logs_db_has_data && self.cfg.enable_logs_snapshot {
383            info!("Logs database is empty, attempting to download logs snapshot...");
384
385            match self.download_snapshot().await {
386                Ok(snapshot_info) => {
387                    info!("Logs snapshot downloaded successfully: {:?}", snapshot_info);
388                }
389                Err(e) => {
390                    error!("Failed to download logs snapshot: {}. Continuing with regular sync.", e);
391                }
392            }
393        }
394
395        Ok(())
396    }

398    /// Generates specialized log filters for efficient blockchain event processing.
399    ///
400    /// This function creates a comprehensive set of log filters that optimize
401    /// indexer performance by categorizing filters based on contract types and
402    /// event relevance to the specific node.
403    ///
404    /// # Arguments
405    /// * `logs_handler` - Handler containing contract addresses and safe address
406    ///
407    /// # Returns
408    /// * `(FilterSet, Vec<(Address, Hash)>)` - A tuple containing:
409    ///   - `FilterSet`: Categorized filters for blockchain event processing.
410    ///   - `Vec<(Address, Hash)>`: A vector of address-topic pairs for logs origin validation.
411    ///
412    /// # Filter Categories
413    /// * `all` - Complete set of filters for normal operation
414    /// * `token` - Token-specific filters (Transfer, Approval events for safe)
415    /// * `no_token` - Non-token contract filters for initial sync optimization
416    fn generate_log_filters(logs_handler: &U) -> (FilterSet, Vec<(Address, Hash)>) {
417        let safe_address = logs_handler.safe_address();
418        let addresses_no_token = logs_handler
419            .contract_addresses()
420            .into_iter()
421            .filter(|a| *a != logs_handler.contract_addresses_map().token)
422            .collect::<Vec<_>>();
423        let mut filter_base_addresses = vec![];
424        let mut filter_base_topics = vec![];
425        let mut address_topics = vec![];
426
427        addresses_no_token.iter().for_each(|address| {
428            let topics = logs_handler.contract_address_topics(*address);
429            if !topics.is_empty() {
430                filter_base_addresses.push(alloy::primitives::Address::from(*address));
431                filter_base_topics.extend(topics.clone());
432                for topic in topics.iter() {
433                    address_topics.push((*address, Hash::from(topic.0)))
434                }
435            }
436        });
437
438        let filter_base = alloy::rpc::types::Filter::new()
439            .address(filter_base_addresses)
440            .event_signature(filter_base_topics);
441        let filter_token = alloy::rpc::types::Filter::new().address(alloy::primitives::Address::from(
442            logs_handler.contract_addresses_map().token,
443        ));
444
445        let filter_transfer_to = filter_token
446            .clone()
447            .event_signature(Transfer::SIGNATURE_HASH)
448            .topic2(alloy::primitives::B256::from_slice(safe_address.to_bytes32().as_ref()));
449
450        let filter_transfer_from = filter_token
451            .clone()
452            .event_signature(Transfer::SIGNATURE_HASH)
453            .topic1(alloy::primitives::B256::from_slice(safe_address.to_bytes32().as_ref()));
454
455        let filter_approval = filter_token
456            .event_signature(Approval::SIGNATURE_HASH)
457            .topic1(alloy::primitives::B256::from_slice(safe_address.to_bytes32().as_ref()))
458            .topic2(alloy::primitives::B256::from_slice(
459                logs_handler.contract_addresses_map().channels.to_bytes32().as_ref(),
460            ));
461
462        let set = FilterSet {
463            all: vec![
464                filter_base.clone(),
465                filter_transfer_from.clone(),
466                filter_transfer_to.clone(),
467                filter_approval.clone(),
468            ],
469            token: vec![filter_transfer_from, filter_transfer_to, filter_approval],
470            no_token: vec![filter_base],
471        };
472
473        (set, address_topics)
474    }

476    /// Processes a block by its ID.
477    ///
478    /// This function retrieves logs for the given block ID and processes them using the database
479    /// and log handler.
480    ///
481    /// # Arguments
482    ///
483    /// * `db` - The database operations handler.
484    /// * `logs_handler` - The database log handler.
485    /// * `block_id` - The ID of the block to process.
486    ///
487    /// # Returns
488    ///
489    /// A `Result` containing an optional vector of significant chain events if the operation succeeds or an error if it
490    /// fails.
491    async fn process_block_by_id(
492        db: &Db,
493        logs_handler: &U,
494        block_id: u64,
495        is_synced: bool,
496    ) -> crate::errors::Result<Option<Vec<SignificantChainEvent>>>
497    where
498        U: ChainLogHandler + 'static,
499        Db: HoprDbLogOperations + 'static,
500    {
501        let logs = db.get_logs(Some(block_id), Some(0)).await?;
502        let mut block = BlockWithLogs {
503            block_id,
504            ..Default::default()
505        };
506
507        for log in logs {
508            if log.block_number == block_id {
509                block.logs.insert(log);
510            } else {
511                error!(
512                    expected = block_id,
513                    actual = log.block_number,
514                    "block number mismatch in logs from database"
515                );
516                panic!("block number mismatch in logs from database")
517            }
518        }
519
520        Ok(Self::process_block(db, logs_handler, block, true, is_synced).await)
521    }

523    /// Processes a block and its logs.
524    ///
525    /// This function collects events from the block logs and updates the database with the processed logs.
526    ///
527    /// # Arguments
528    ///
529    /// * `db` - The database operations handler.
530    /// * `logs_handler` - The database log handler.
531    /// * `block` - The block with logs to process.
532    /// * `fetch_checksum_from_db` - A boolean indicating whether to fetch the checksum from the database.
533    ///
534    /// # Returns
535    ///
536    /// An optional vector of significant chain events if the operation succeeds.
537    async fn process_block(
538        db: &Db,
539        logs_handler: &U,
540        block: BlockWithLogs,
541        fetch_checksum_from_db: bool,
542        is_synced: bool,
543    ) -> Option<Vec<SignificantChainEvent>>
544    where
545        U: ChainLogHandler + 'static,
546        Db: HoprDbLogOperations + 'static,
547    {
548        let block_id = block.block_id;
549        let log_count = block.logs.len();
550        debug!(block_id, "processing events");
551
552        // FIXME: The block indexing and marking as processed should be done in a single
553        // transaction. This is difficult since currently this would be across databases.
554        let events = stream::iter(block.logs.clone())
555            .filter_map(|log| async move {
556                match logs_handler.collect_log_event(log.clone(), is_synced).await {
557                    Ok(data) => match db.set_log_processed(log).await {
558                        Ok(_) => data,
559                        Err(error) => {
560                            error!(block_id, %error, "failed to mark log as processed, panicking to prevent data loss");
561                            panic!("failed to mark log as processed, panicking to prevent data loss")
562                        }
563                    },
564                    Err(CoreEthereumIndexerError::ProcessError(error)) => {
565                        error!(block_id, %error, "failed to process log into event, continuing indexing");
566                        None
567                    }
568                    Err(error) => {
569                        error!(block_id, %error, "failed to process log into event, panicking to prevent data loss");
570                        panic!("failed to process log into event, panicking to prevent data loss")
571                    }
572                }
573            })
574            .collect::<Vec<SignificantChainEvent>>()
575            .await;
576
577        // if we made it this far, no errors occurred and we can update checksums and indexer state
578        match db.update_logs_checksums().await {
579            Ok(last_log_checksum) => {
580                let checksum = if fetch_checksum_from_db {
581                    let last_log = block.logs.into_iter().next_back()?;
582                    let log = db.get_log(block_id, last_log.tx_index, last_log.log_index).await.ok()?;
583
584                    log.checksum?
585                } else {
586                    last_log_checksum.to_string()
587                };
588
589                if log_count != 0 {
590                    info!(
591                        block_number = block_id,
592                        log_count, last_log_checksum = ?checksum, "Indexer state update",
593                    );
594
595                    #[cfg(all(feature = "prometheus", not(test)))]
596                    {
597                        if let Ok(checksum_hash) = Hash::from_hex(checksum.as_str()) {
598                            let low_4_bytes =
599                                hopr_primitive_types::prelude::U256::from_big_endian(checksum_hash.as_ref()).low_u32();
600                            METRIC_INDEXER_CHECKSUM.set(low_4_bytes.into());
601                        } else {
602                            error!("Invalid checksum generated from logs");
603                        }
604                    }
605                }
606
607                // finally update the block number in the database to the last processed block
608                match db.set_indexer_state_info(None, block_id as u32).await {
609                    Ok(_) => {
610                        trace!(block_id, "updated indexer state info");
611                    }
612                    Err(error) => error!(block_id, %error, "failed to update indexer state info"),
613                }
614            }
615            Err(error) => error!(block_id, %error, "failed to update checksums for logs from block"),
616        }
617
618        debug!(
619            block_id,
620            num_events = events.len(),
621            "processed significant chain events from block",
622        );
623
624        Some(events)
625    }

628    where
629        T: HoprIndexerRpcOperations + 'static,
630    {
631        match rpc.block_number().await {

635                head

637            Err(error) => {

639                panic!("Failed to fetch block number from RPC, cannot continue indexing due to {error}")
640            }
641        }

644    /// Calculates the synchronization progress.
645    ///
646    /// This function processes a block and updates synchronization metrics and state.
647    ///
648    /// # Arguments
649    ///
650    /// * `block` - The block with logs to process.
651    /// * `rpc` - The RPC operations handler.
652    /// * `chain_head` - The current chain head block number.
653    /// * `is_synced` - A boolean indicating whether the indexer is synced.
654    /// * `start_block` - The first block number to process.
655    /// * `tx` - A sender channel for synchronization notifications.
656    ///
657    /// # Returns
658    ///
659    /// The block which was provided as input.
660    #[allow(clippy::too_many_arguments)]
661    async fn calculate_sync_process(
662        current_block: u64,
663        rpc: &T,
664        db: Db,
665        chain_head: Arc<AtomicU64>,
666        is_synced: Arc<AtomicBool>,
667        next_block_to_process: u64,
668        mut tx: futures::channel::mpsc::Sender<()>,
669        safe_address: Option<Address>,
670        channels_address: Option<Address>,
671    ) where
672        T: HoprIndexerRpcOperations + 'static,
673        Db: HoprDbInfoOperations + Clone + Send + Sync + 'static,
674    {
675        #[cfg(all(feature = "prometheus", not(test)))]
676        {
677            METRIC_INDEXER_CURRENT_BLOCK.set(current_block as f64);
678        }
679
680        let mut head = chain_head.load(Ordering::Relaxed);
681
682        // We only print out sync progress if we are not yet synced.
683        // Once synced, we don't print out progress anymore.
684        if !is_synced.load(Ordering::Relaxed) {
685            let mut block_difference = head.saturating_sub(next_block_to_process);
686
687            let progress = if block_difference == 0 {
688                // Before we call the sync complete, we check the chain again.
689                head = Self::update_chain_head(rpc, chain_head.clone()).await;
690                block_difference = head.saturating_sub(next_block_to_process);
691
692                if block_difference == 0 {
693                    1_f64
694                } else {
695                    (current_block - next_block_to_process) as f64 / block_difference as f64
696                }
697            } else {
698                (current_block - next_block_to_process) as f64 / block_difference as f64
699            };
700
701            info!(
702                progress = progress * 100_f64,
703                block = current_block,
704                head,
705                "Sync progress to last known head"
706            );
707
708            #[cfg(all(feature = "prometheus", not(test)))]
709            METRIC_INDEXER_SYNC_PROGRESS.set(progress);
710
711            if current_block >= head {
712                info!("indexer sync completed successfully");
713                is_synced.store(true, Ordering::Relaxed);
714
715                if let Some(safe_address) = safe_address {
716                    info!("updating safe balance from chain after indexer sync completed");
717                    match rpc.get_hopr_balance(safe_address).await {
718                        Ok(balance) => {
719                            if let Err(error) = db.set_safe_hopr_balance(None, balance).await {
720                                error!(%error, "failed to update safe balance from chain after indexer sync completed");
721                            }
722                        }
723                        Err(error) => {
724                            error!(%error, "failed to fetch safe balance from chain after indexer sync completed");
725                        }
726                    }
727                }
728
729                if let Some((channels_address, safe_address)) = channels_address.zip(safe_address) {
730                    info!("updating safe allowance from chain after indexer sync completed");
731                    match rpc.get_hopr_allowance(safe_address, channels_address).await {
732                        Ok(allowance) => {
733                            if let Err(error) = db.set_safe_hopr_allowance(None, allowance).await {
734                                error!(%error, "failed to update safe allowance from chain after indexer sync completed");
735                            }
736                        }
737                        Err(error) => {
738                            error!(%error, "failed to fetch safe allowance from chain after indexer sync completed");
739                        }
740                    }
741                }
742
743                if let Err(error) = tx.try_send(()) {
744                    error!(%error, "failed to notify about achieving indexer synchronization")
745                }
746            }
747        }
748    }

750    /// Checks if the logs database has any existing data.
751    ///
752    /// This method determines whether the database already contains logs, which helps
753    /// decide whether to download a snapshot for faster synchronization. It queries
754    /// the database for the total log count and returns an error if the query fails
755    /// (e.g., when the database doesn't exist yet).
756    ///
757    /// # Returns
758    ///
759    /// - `Ok(true)` if the database contains one or more logs
760    /// - `Ok(false)` if the database is empty
761    /// - `Err(e)` if the database cannot be queried
762    async fn has_logs_data(&self) -> Result<bool> {
763        self.db
764            .get_logs_count(None, None)
765            .await
766            .map(|count| count > 0)
767            .map_err(|e| CoreEthereumIndexerError::SnapshotError(e.to_string()))
768    }

770    /// Downloads and installs a database snapshot for faster initial synchronization.
771    ///
772    /// This method coordinates the snapshot download process by:
773    /// 1. Validating the indexer configuration
774    /// 2. Creating a snapshot manager instance
775    /// 3. Downloading and extracting the snapshot to the data directory
776    ///
777    /// Snapshots allow new nodes to quickly synchronize with the network by downloading
778    /// pre-built database files instead of fetching all historical logs from scratch.
779    ///
780    /// # Returns
781    ///
782    /// - `Ok(SnapshotInfo)` containing details about the downloaded snapshot
783    /// - `Err(CoreEthereumIndexerError::SnapshotError)` if validation or download fails
784    ///
785    /// # Prerequisites
786    ///
787    /// - Configuration must be valid (proper URL format, data directory set)
788    /// - Sufficient disk space must be available
789    /// - Network connectivity to the snapshot URL
790    pub async fn download_snapshot(&self) -> Result<SnapshotInfo> {
791        // Validate config before proceeding
792        if let Err(e) = self.cfg.validate() {
793            return Err(CoreEthereumIndexerError::SnapshotError(e.to_string()));
794        }
795
796        let snapshot_manager = SnapshotManager::with_db(self.db.clone())
797            .map_err(|e| CoreEthereumIndexerError::SnapshotError(e.to_string()))?;
798
799        let data_dir = Path::new(&self.cfg.data_directory);
800
801        // The URL has been verified so we can just use it.
802        if let Some(url) = &self.cfg.logs_snapshot_url {
803            snapshot_manager
804                .download_and_setup_snapshot(url, data_dir)
805                .await
806                .map_err(|e| CoreEthereumIndexerError::SnapshotError(e.to_string()))
807        } else {
808            Err(CoreEthereumIndexerError::SnapshotError(
809                "Logs snapshot URL is not configured".to_string(),
810            ))
811        }
812    }
