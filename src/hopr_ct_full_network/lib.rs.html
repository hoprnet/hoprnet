20/// Configuration for the probing mechanism
21#[derive(Debug, Clone, Copy, PartialEq, smart_default::SmartDefault, Validate)]
22#[cfg_attr(feature = "serde", derive(Serialize, Deserialize), serde(deny_unknown_fields))]
23pub struct ProberConfig {
24    /// The delay between individual probing rounds for neighbor discovery
25    #[cfg_attr(
26        feature = "serde",
27        serde(default = "default_probing_interval", with = "humantime_serde")
28    )]
29    #[default(default_probing_interval())]
30    pub interval: std::time::Duration,
31}
32
33/// Delay before repeating immediate probing rounds, should include enough time to traverse NATs
34const DEFAULT_REPEATED_PROBING_DELAY: std::time::Duration = std::time::Duration::from_secs(30);
35
36#[inline]
37const fn default_probing_interval() -> std::time::Duration {
38    DEFAULT_REPEATED_PROBING_DELAY
39}

53impl<U> CoverTrafficGeneration for FullNetworkDiscovery<U>
54where
55    U: NetworkGraphTraverse<NodeId = OffchainPublicKey> + Clone + Send + Sync + 'static,
56{
57    fn build(&self) -> BoxStream<'static, DestinationRouting> {
58        cfg_if::cfg_if! {
59            if #[cfg(feature = "noise")] {
60                let cfg = self.cfg;
61                let me = self.me;
62                let graph_intermediates = self.graph.clone();
63
64                let junk = futures::stream::repeat(futures::stream::iter([2, 3, 4]))
65                    .flatten()
66                    .filter_map(move |edge_count| async move {
67                        hopr_async_runtime::prelude::sleep(cfg.interval).await;
68                        Some(edge_count)
69                    })
70                    .flat_map(move |edge_count| {
71                        let simple_paths =
72                            graph_intermediates.simple_paths(&me, &me, edge_count, Some(100), HoprCostFn::new(edge_count));
73                        futures::stream::iter(simple_paths)
74                    })
75                    .filter_map(move |(path, _path_id, _cost)| {
76                        let me_node: NodeId = me.into();
77                        let path: Vec<NodeId> = path.into_iter().map(NodeId::from).collect();
78
79                        let routing =
80                            hopr_api::network::BoundedVec::try_from(path)
81                                .ok()
82                                .map(|path| DestinationRouting::Forward {
83                                    destination: Box::new(me_node),
84                                    pseudonym: Some(HoprPseudonym::random()),
85                                    forward_options: RoutingOptions::IntermediatePath(path),
86                                    return_options: None,
87                                });
88
89                        futures::future::ready(routing)
90                    });
91
92                junk.boxed()
93            } else {
94                // Cover traffic not generating any data
95                Box::pin(futures::stream::empty())
96            }
97        }
98    }
99}

101impl<U> ProbingTrafficGeneration for FullNetworkDiscovery<U>
102where
103    U: NetworkGraphView<NodeId = OffchainPublicKey>
104        + NetworkGraphTraverse<NodeId = OffchainPublicKey, Observed = hopr_network_graph::Observations>
105        + Clone
106        + Send
107        + Sync
108        + 'static,
109{
110    fn build(&self) -> BoxStream<'static, ProbeRouting> {
111        // For each probe target a cached version of transport routing is stored
112        let cache_immediate_neighbor_routing: moka::future::Cache<OffchainPublicKey, ProbeRouting> =
113            moka::future::Cache::builder()
114                .time_to_live(std::time::Duration::from_secs(600))
115                .max_capacity(100_000)
116                .build();
117
118        let cfg = self.cfg;
119        let graph = self.graph.clone();
120        let graph_intermediates = self.graph.clone();
121
122        let immediates = futures::stream::repeat(())
123            .filter_map(move |_| {
124                let nodes = graph.nodes();
125
126                async move {
127                    hopr_async_runtime::prelude::sleep(cfg.interval).await;
128                    Some(nodes)
129                }
130            })
131            .flatten()
132            .filter_map(move |peer| {
133                let cache_immediate_neighbor_routing = cache_immediate_neighbor_routing.clone();
134
135                async move {
136                    cache_immediate_neighbor_routing
137                        .try_get_with(peer, async move {
138                            Ok::<ProbeRouting, anyhow::Error>(ProbeRouting::Neighbor(DestinationRouting::Forward {
139                                destination: Box::new(peer.into()),
140                                pseudonym: Some(HoprPseudonym::random()),
141                                forward_options: RoutingOptions::Hops(0.try_into().expect("0 is a valid u8")),
142                                return_options: Some(RoutingOptions::Hops(0.try_into().expect("0 is a valid u8"))),
143                            }))
144                        })
145                        .await
146                        .ok()
147                }
148            });
149
150        let me = self.me;
151
152        // 2, 3 and 4 edges => 1-, 2- and 3-hops in the HOPR protocol
153        let intermediates = futures::stream::repeat(futures::stream::iter([2, 3, 4]))
154            .flatten()
155            .filter_map(move |edge_count| async move {
156                hopr_async_runtime::prelude::sleep(cfg.interval).await;
157                std::num::NonZeroUsize::new(edge_count)
158            })
159            .flat_map(move |edge_count| {
160                let simple_paths = graph_intermediates.simple_paths(
161                    &me,
162                    &me,
163                    edge_count.get(),
164                    Some(100),
165                    HoprCostFn::new(edge_count),
166                );
167                futures::stream::iter(simple_paths)
168            })
169            .filter_map(move |(path, path_id, _cost)| {
170                let me_node: NodeId = me.into();
171                let path: Vec<NodeId> = path.into_iter().map(NodeId::from).collect();
172
173                let routing = hopr_api::network::BoundedVec::try_from(path).ok().map(|path| {
174                    ProbeRouting::Looping((
175                        DestinationRouting::Forward {
176                            destination: Box::new(me_node),
177                            pseudonym: Some(HoprPseudonym::random()),
178                            forward_options: RoutingOptions::IntermediatePath(path),
179                            return_options: None,
180                        },
181                        path_id,
182                    ))
183                });
184
185                futures::future::ready(routing)
186            });
187
188        immediates.merge(intermediates).boxed()
189    }
190}

243    async fn peers_should_have_randomized_order() -> anyhow::Result<()> {
244        let me = OffchainKeypair::random().public().clone();
245        let channel_graph = Arc::new(ChannelGraph::new(me));
246
247        for node in RANDOM_PEERS.iter() {
248            channel_graph.record_node(node.clone());
249        }
250
251        let prober = FullNetworkDiscovery::new(
252            me,
253            ProberConfig {
254                interval: std::time::Duration::from_millis(1),
255                ..Default::default()
256            },
257            channel_graph,
258        );
259
260        let stream = ProbingTrafficGeneration::build(&prober);
261        pin_mut!(stream);
262
263        let actual = timeout(
264            TINY_TIMEOUT * 20,
265            stream
266                .take(RANDOM_PEERS.len())
267                .map(|routing| match routing {
268                    ProbeRouting::Neighbor(DestinationRouting::Forward { destination, .. }) => {
269                        if let NodeId::Offchain(peer_key) = destination.as_ref() {
270                            *peer_key
271                        } else {
272                            panic!("expected offchain destination, got chain address");
273                        }
274                    }
275                    _ => panic!("expected Neighbor Forward routing"),
276                })
277                .collect::<Vec<_>>(),
278        )
279        .await?;
280
281        assert_eq!(actual.len(), RANDOM_PEERS.len());
282        assert!(!actual.iter().zip(RANDOM_PEERS.iter()).all(|(a, b)| a == &b.id));
283
284        Ok(())
285    }

359    #[tokio::test]
360    async fn neighbor_probes_should_cover_all_known_nodes_across_rounds() -> anyhow::Result<()> {
361        let cfg = ProberConfig {
362            interval: std::time::Duration::from_millis(1),
363        };
364        let me = OffchainKeypair::random().public().clone();
365        let channel_graph = Arc::new(ChannelGraph::new(me));
366
367        let peer_a = OffchainKeypair::random().public().clone();
368        let peer_b = OffchainKeypair::random().public().clone();
369        channel_graph.record_node(peer_a);
370        channel_graph.record_node(peer_b);
371
372        let prober = FullNetworkDiscovery::new(me, cfg, channel_graph);
373
374        let stream = ProbingTrafficGeneration::build(&prober);
375        pin_mut!(stream);
376
377        // graph.nodes() returns me + peer_a + peer_b = 3 nodes per round.
378        // Collect enough items for at least 2 full rounds.
379        let destinations: Vec<NodeId> = timeout(
380            TINY_TIMEOUT * 50,
381            stream
382                .filter_map(|r| {
383                    futures::future::ready(match r {
384                        ProbeRouting::Neighbor(DestinationRouting::Forward { destination, .. }) => Some(*destination),
385                        _ => None,
386                    })
387                })
388                .take(6)
389                .collect::<Vec<_>>(),
390        )
391        .await?;
392
393        let unique: HashSet<NodeId> = destinations.iter().cloned().collect();
394        let expected_nodes: HashSet<NodeId> = [me, peer_a, peer_b].into_iter().map(NodeId::from).collect();
395
396        assert_eq!(
397            unique, expected_nodes,
398            "probes should cover all known graph nodes (me + peers)"
399        );
400        // With 3 unique nodes and 6 items, each node appeared in multiple rounds
401        assert_eq!(
402            destinations.len(),
403            6,
404            "should have collected probes across multiple rounds"
405        );
406
407        Ok(())
408    }
