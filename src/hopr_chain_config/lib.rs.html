162/// Check whether the version is allowed
163fn satisfies(version: &str, allowed_versions: &str) -> Result<bool> {
164    let allowed_versions = VersionReq::parse(allowed_versions)
165        .map_err(|e| HoprConfigError::Configuration(format!("failed to deserialize allowed version string: {e}")))?;
166
167    let version = Version::from_str(version).map_err(|e| {
168        HoprConfigError::Configuration(format!("failed to deserialize current lib version string: {e}"))
169    })?;
170
171    Ok(allowed_versions.matches(&version))
172}

174impl ChainNetworkConfig {
175    /// Returns the network details, returns an error if network is not supported
176    pub fn new(
177        id: &str,
178        version: &str,
179        maybe_custom_provider: Option<&str>,
180        max_rpc_requests_per_sec: Option<u32>,
181        protocol_config: &mut ProtocolsConfig,
182    ) -> std::result::Result<Self, String> {
183        let network = protocol_config
184            .networks
185            .get_mut(id)
186            .ok_or(format!("Could not find network {id} in protocol config"))?;
187
188        let chain = protocol_config
189            .chains
190            .get_mut(&network.chain)
191            .ok_or(format!("Invalid chain {} for network {id}", network.chain))?;
192
193        if let Some(custom_provider) = maybe_custom_provider {
194            chain.default_provider = custom_provider.into();
195        }
196
197        match satisfies(version, network.version_range.as_str()) {
198            Ok(true) => Ok(ChainNetworkConfig {
199                announcements: network.addresses.announcements.to_owned(),
200                chain: chain.to_owned(),
201                channel_contract_deploy_block: network.indexer_start_block_number,
202                channels: network.addresses.channels.to_owned(),
203                confirmations: network.confirmations,
204                environment_type: network.environment_type,
205                id: network.chain.to_owned(),
206                module_implementation: network.addresses.module_implementation.to_owned(),
207                network_registry: network.addresses.network_registry.to_owned(),
208                network_registry_proxy: network.addresses.network_registry_proxy.to_owned(),
209                node_safe_registry: network.addresses.node_safe_registry.to_owned(),
210                node_stake_v2_factory: network.addresses.node_stake_v2_factory.to_owned(),
211                ticket_price_oracle: network.addresses.ticket_price_oracle.to_owned(),
212                winning_probability_oracle: network.addresses.winning_probability_oracle.to_owned(),
213                token: network.addresses.token.to_owned(),
214                tx_polling_interval: network.tx_polling_interval,
215                max_block_range: network.max_block_range,
216                max_requests_per_sec: max_rpc_requests_per_sec.or(chain.max_rpc_requests_per_sec),
217            }),
218            Ok(false) => Err(HoprConfigError::UnsupportedNetwork(id.into()).to_string()),
219            Err(e) => Err(e.to_string()),
220        }
221    }
222}

241/// The entire on-chain protocol configuration containing the information about
242/// usable networks and chains.
243#[derive(Debug, Clone, Serialize, Deserialize, Validate)]
244#[serde(deny_unknown_fields)]
245pub struct ProtocolsConfig {
246    pub networks: BTreeMap<String, Network>,
247    pub chains: BTreeMap<String, ChainOptions>,
248}
249
250impl Default for ProtocolsConfig {
251    fn default() -> Self {
252        Self::from_str(include_str!("../data/protocol-config.json"))
253            .expect("bundled protocol config should be always valid")
254    }
255}
256
257impl FromStr for ProtocolsConfig {
258    type Err = HoprConfigError;
259
260    /// Reads the protocol config JSON file and returns it
261    fn from_str(data: &str) -> std::result::Result<Self, Self::Err> {
262        serde_json::from_str::<ProtocolsConfig>(data).map_err(|e| HoprConfigError::Configuration(e.to_string()))
263    }
264}
265
266impl std::cmp::PartialEq for ProtocolsConfig {
267    fn eq(&self, other: &Self) -> bool {
268        Vec::from_iter(self.networks.clone()) == Vec::from_iter(other.networks.clone())
269            && Vec::from_iter(self.chains.clone()) == Vec::from_iter(other.chains.clone())
270    }
271}

273impl ProtocolsConfig {
274    /// Returns a list of environments which the node is able to work with
275    /// TODO: crate::constants::APP_VERSION_COERCED
276    pub fn supported_networks(&self, version: &str) -> Vec<String> {
277        let mut allowed = vec![];
278
279        for (name, env) in self.networks.iter() {
280            let range = env.version_range.to_owned();
281
282            if let Ok(true) = satisfies(version, range.as_str()) {
283                allowed.push(name.clone())
284            }
285        }
286
287        allowed
288    }
