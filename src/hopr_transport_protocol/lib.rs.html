1//! Collection of objects and functionality allowing building of p2p or stream protocols for the higher business logic
2//! layers.
3//!
4//! ## Contents
5//!
6//! Supported protocol configurations:
7//!
8//! - `mix`
9//! - `ack`
10//! - `heartbeat`
11//! - `ticket_aggregation`
12//!
13//! Supported protocol processors:
14//!
15//! - `ticket_aggregation`
16//!
17//! ### `ticket_aggregation`
18//!
19//! Ticket aggregation processing mechanism is responsible for ingesting the ticket aggregation related requests:
20//!
21//! - `Receive(PeerId, U)`,
22//! - `Reply(PeerId, std::result::Result<Ticket, String>, T)`,
23//! - `Send(PeerId, Vec<AcknowledgedTicket>, TicketAggregationFinalizer)`,
24//!
25//! where `U` is the type of an aggregated ticket extractable (`ResponseChannel<Result<Ticket, String>>`) and `T`
26//! represents a network negotiated identifier (`RequestId`).
27//!
28//! In broader context the protocol flow is as follows:
29//!
30//! 1. requesting ticket aggregation
31//!
32//!    - the peer A desires to aggregate tickets, collects the tickets into a data collection and sends a request
33//!      containing the collection to aggregate `Vec<AcknowledgedTicket>` to peer B using the `Send` mechanism
34//!
35//! 2. responding to ticket aggregation
36//!
37//!    - peer B obtains the request from peer A, performs the ticket aggregation and returns a result of that operation
38//!      in the form of `std::result::Result<Ticket, String>` using the `Reply` mechanism
39//!
40//! 3. accepting the aggregated ticket
41//!    - peer A receives the aggregated ticket using the `Receive` mechanism
42//!
43//! Furthermore, apart from the basic positive case scenario, standard mechanics of protocol communication apply:
44//!
45//! - the requesting side can time out, if the responding side takes too long to provide an aggregated ticket, in which
46//!   case the ticket is not considered aggregated, even if eventually an aggregated ticket is delivered
47//! - the responder can fail to aggregate tickets in which case it replies with an error string describing the failure
48//!   reason and it is the requester's responsibility to handle the negative case as well
49//!   - in the absence of response, the requester will time out
50

156#[cfg(feature = "capture")]
157fn inspect_ticket_data_in_packet(raw_packet: &[u8]) -> &[u8] {
158    use hopr_primitive_types::traits::BytesEncodable;
159    if raw_packet.len() >= hopr_internal_types::tickets::Ticket::SIZE {
160        &raw_packet[raw_packet.len() - hopr_internal_types::tickets::Ticket::SIZE..]
161    } else {
162        &[]
163    }
164}

166/// Run all processes responsible for handling the msg and acknowledgment protocols.
167///
168/// The pipeline does not handle the mixing itself, that needs to be injected as a separate process
169/// overlay on top of the `wire_msg` Stream or Sink.
170#[allow(clippy::too_many_arguments)]
171pub async fn run_msg_ack_protocol<Db>(
172    packet_cfg: processor::PacketInteractionConfig,
173    db: Db,
174    wire_msg: (
175        impl futures::Sink<(PeerId, Box<[u8]>)> + Clone + Unpin + Send + Sync + 'static,
176        impl futures::Stream<Item = (PeerId, Box<[u8]>)> + Send + Sync + 'static,
177    ),
178    api: (
179        impl futures::Sink<(HoprPseudonym, ApplicationDataIn)> + Send + Sync + 'static,
180        impl futures::Stream<Item = (ApplicationDataOut, ResolvedTransportRouting, PacketSendFinalizer)>
181        + Send
182        + Sync
183        + 'static,
184    ),
185) -> HashMap<ProtocolProcesses, hopr_async_runtime::AbortHandle>
186where
187    Db: HoprDbProtocolOperations + std::fmt::Debug + Clone + Send + Sync + 'static,
188{
189    let me = packet_cfg.packet_keypair.clone();
190
191    #[cfg(feature = "capture")]
192    let me_pub = *hopr_crypto_types::keypairs::Keypair::public(&me);
193
194    let mut processes = HashMap::new();
195
196    #[cfg(all(feature = "prometheus", not(test)))]
197    {
198        // Initialize the lazy statics here
199        lazy_static::initialize(&METRIC_PACKET_COUNT);
200        lazy_static::initialize(&METRIC_PACKET_COUNT_PER_PEER);
201        lazy_static::initialize(&METRIC_REPLAYED_PACKET_COUNT);
202        lazy_static::initialize(&METRIC_REJECTED_TICKETS_COUNT);
203    }
204
205    #[cfg(feature = "capture")]
206    let capture = {
207        use std::str::FromStr;
208        let writer: Box<dyn capture::PacketWriter + Send + 'static> =
209            if let Ok(desc) = std::env::var("HOPR_CAPTURE_PACKETS") {
210                if let Ok(udp_writer) = std::net::SocketAddr::from_str(&desc)
211                    .map_err(std::io::Error::other)
212                    .and_then(capture::UdpPacketDump::new)
213                {
214                    warn!("udp packet capture initialized to {desc}");
215                    Box::new(udp_writer)
216                } else if let Ok(pcap_writer) = std::fs::File::create(&desc).and_then(capture::PcapPacketWriter::new) {
217                    warn!("pcap file packet capture initialized to {desc}");
218                    Box::new(pcap_writer)
219                } else {
220                    error!(desc, "failed to create packet capture: invalid socket address or file");
221                    Box::new(capture::NullWriter)
222                }
223            } else {
224                warn!("no packet capture specified");
225                Box::new(capture::NullWriter)
226            };
227        let (capture, ah) = capture::packet_capture_channel(writer);
228        processes.insert(ProtocolProcesses::Capture, ah);
229        capture
230    };
231
232    let tbf = std::sync::Arc::new(parking_lot::Mutex::new(TagBloomFilter::default()));
233
234    let (ticket_ack_tx, ticket_ack_rx) =
235        futures::channel::mpsc::channel::<(Acknowledgement, OffchainPublicKey)>(TICKET_ACK_BUFFER_SIZE);
236
237    let db_clone = db.clone();
238    processes.insert(
239        ProtocolProcesses::TicketAck,
240        spawn_as_abortable!(ticket_ack_rx
241            .for_each_concurrent(NUM_CONCURRENT_TICKET_ACK_PROCESSING, move |(ack, sender)| {
242                let db = db_clone.clone();
243                async move {
244                    if let Ok(verified) = hopr_parallelize::cpu::spawn_blocking(move || ack.verify(&sender)).await {
245                        trace!(%sender, "received a valid acknowledgement");
246                            match db.handle_acknowledgement(verified).await {
247                                Ok(_) => trace!(%sender, "successfully processed a known acknowledgement"),
248                                // Eventually, we do not care here if the acknowledgement does not belong to any
249                                // unacknowledged packets.
250                                Err(error) => trace!(%sender, %error, "valid acknowledgement is unknown or error occurred while processing it"),
251                            }
252                    } else {
253                        error!(%sender, "failed to verify signature on acknowledgement");
254                    }
255                }
256            })
257            .inspect(|_| tracing::warn!(task = "transport (protocol - ticket acknowledgement)", "long-running background task finished")))
258    );
259
260    let (ack_out_tx, ack_out_rx) =
261        futures::channel::mpsc::channel::<(Option<HalfKey>, OffchainPublicKey)>(ACK_OUT_BUFFER_SIZE);
262
263    #[cfg(feature = "capture")]
264    let capture_clone = capture.clone();
265
266    let db_clone = db.clone();
267    let me_clone = me.clone();
268    let msg_to_send_tx = wire_msg.0.clone();
269    processes.insert(
270        ProtocolProcesses::AckOut,
271        spawn_as_abortable!(
272            ack_out_rx
273                .for_each_concurrent(
274                    NUM_CONCURRENT_ACK_OUT_PROCESSING,
275                    move |(maybe_ack_key, destination)| {
276                        let db = db_clone.clone();
277                        let me = me_clone.clone();
278                        let mut msg_to_send_tx_clone = msg_to_send_tx.clone();
279
280                        #[cfg(feature = "capture")]
281                        let mut capture = capture_clone.clone();
282                        async move {
283                            #[cfg(feature = "capture")]
284                            let (is_random, me_pub) = (
285                                maybe_ack_key.is_none(),
286                                *hopr_crypto_types::keypairs::Keypair::public(&me),
287                            );
288
289                            // Sign acknowledgement with the given half-key or generate a signed random one
290                            let ack = hopr_parallelize::cpu::spawn_blocking(move || {
291                                maybe_ack_key
292                                    .map(|ack_key| VerifiedAcknowledgement::new(ack_key, &me))
293                                    .unwrap_or_else(|| VerifiedAcknowledgement::random(&me))
294                            })
295                            .await;
296
297                            #[cfg(feature = "capture")]
298                            let captured_packet: capture::CapturedPacket = capture::PacketBeforeTransit::OutgoingAck {
299                                me: me_pub,
300                                ack,
301                                is_random,
302                                next_hop: destination,
303                            }
304                            .into();
305
306                            match db.to_send_no_ack(ack.leak().as_ref().into(), destination).await {
307                                Ok(ack_packet) => {
308                                    let now = std::time::Instant::now();
309                                    if msg_to_send_tx_clone
310                                        .send((ack_packet.next_hop.into(), ack_packet.data))
311                                        .await
312                                        .is_err()
313                                    {
314                                        error!("failed to forward an acknowledgement to the transport layer");
315                                    }
316                                    let elapsed = now.elapsed();
317                                    if elapsed.as_millis() > SLOW_OP_MS {
318                                        warn!(?elapsed, " msg_to_send_tx.send on ack took too long");
319                                    }
320
321                                    #[cfg(feature = "capture")]
322                                    let _ = capture.try_send(captured_packet);
323                                }
324                                Err(error) => tracing::error!(%error, "failed to create ack packet"),
325                            }
326                        }
327                    }
328                )
329                .inspect(|_| tracing::warn!(
330                    task = "transport (protocol - ack outgoing)",
331                    "long-running background task finished"
332                ))
333        ),
334    );
335
336    let msg_processor_read = processor::PacketProcessor::new(db.clone(), packet_cfg);
337    let msg_processor_write = msg_processor_read.clone();
338
339    #[cfg(feature = "capture")]
340    let capture_clone = capture.clone();
341
342    let msg_to_send_tx = wire_msg.0.clone();
343    processes.insert(
344        ProtocolProcesses::MsgOut,
345        spawn_as_abortable!(async move {
346            let _neverending = api
347                .1
348                .then_concurrent(|(data, routing, finalizer)| {
349                    let msg_processor = msg_processor_write.clone();
350
351                    #[cfg(feature = "capture")]
352                    let (mut capture_clone, data_clone, num_surbs) =
353                        (capture_clone.clone(), data.clone(), routing.count_return_paths() as u8);
354
355                    async move {
356                        match PacketWrapping::send(&msg_processor, data, routing).await {
357                            Ok(v) => {
358                                #[cfg(all(feature = "prometheus", not(test)))]
359                                {
360                                    METRIC_PACKET_COUNT_PER_PEER.increment(&[&v.next_hop.to_string(), "out"]);
361                                    METRIC_PACKET_COUNT.increment(&["sent"]);
362                                }
363                                finalizer.finalize(Ok(()));
364
365                                #[cfg(feature = "capture")]
366                                let _ = capture_clone.try_send(
367                                    capture::PacketBeforeTransit::OutgoingPacket {
368                                        me: me_pub,
369                                        next_hop: v.next_hop,
370                                        num_surbs,
371                                        is_forwarded: false,
372                                        data: data_clone.data.to_bytes().into_vec().into(),
373                                        ack_challenge: v.ack_challenge.as_ref().into(),
374                                        signals: data_clone.packet_info.unwrap_or_default().signals_to_destination,
375                                        ticket: inspect_ticket_data_in_packet(&v.data).into(),
376                                    }
377                                    .into(),
378                                );
379
380                                Some((v.next_hop.into(), v.data))
381                            }
382                            Err(e) => {
383                                finalizer.finalize(Err(e));
384                                None
385                            }
386                        }
387                    }
388                })
389                .filter_map(futures::future::ready)
390                .inspect(|(peer, _)| trace!(%peer, "protocol message out"))
391                .map(Ok)
392                .forward(msg_to_send_tx)
393                .instrument(tracing::trace_span!("msg protocol processing - egress"))
394                .inspect(|_| {
395                    tracing::warn!(
396                        task = "transport (protocol - msg egress)",
397                        "long-running background task finished"
398                    )
399                })
400                .await;
401        }),
402    );
403
404    let ack_out_tx_clone_1 = ack_out_tx.clone();
405    let ack_out_tx_clone_2 = ack_out_tx.clone();
406
407    #[cfg(feature = "capture")]
408    let capture_clone = capture.clone();
409
410    // Create a cache for a CPU-intensive conversion PeerId -> OffchainPublicKey
411    let peer_id_cache: moka::future::Cache<PeerId, OffchainPublicKey> = moka::future::Cache::builder()
412        .time_to_idle(Duration::from_secs(600))
413        .max_capacity(100_000)
414        .build();
415
416    processes.insert(
417        ProtocolProcesses::MsgIn,
418        spawn_as_abortable!(async move {
419            let _neverending = wire_msg
420                .1
421                .then_concurrent(move |(peer, data)| {
422                    let msg_processor = msg_processor_read.clone();
423                    let mut ack_out_tx = ack_out_tx_clone_1.clone();
424                    let peer_id_key_cache = peer_id_cache.clone();
425
426                    trace!(%peer, "protocol message in");
427
428                    #[cfg(feature = "capture")]
429                    let (mut capture_clone, ticket_data_clone) = (
430                        capture.clone(),
431                        inspect_ticket_data_in_packet(&data).to_vec()
432                    );
433
434                    async move {
435                        // Try to retrieve the peer's public key from the cache or compute it if it does not exist yet
436                        let peer_key = match peer_id_key_cache
437                                .try_get_with_by_ref(&peer, hopr_parallelize::cpu::spawn_fifo_blocking(move || OffchainPublicKey::from_peerid(&peer)))
438                                .await {
439                            Ok(peer) => peer,
440                            Err(error) => {
441                                // There absolutely nothing we can do when the peer id is unparseable (e.g., non-ed25519 based)
442                                error!(%peer, %error, "dropping packet - cannot convert peer id");
443                                return None;
444                            }
445                        };
446
447                        let now = std::time::Instant::now();
448                        let res = msg_processor.recv(peer_key, data).await.map_err(move |e| (peer, e));
449                        let elapsed = now.elapsed();
450                        if elapsed.as_millis() > SLOW_OP_MS {
451                            warn!(%peer, ?elapsed, "msg_processor.recv took too long");
452                        }
453
454                        // If there was an error caused by interpretation of the packet data,
455                        // we must send a random acknowledgement back.
456                        if let Err((peer, error)) = &res {
457                            #[cfg(all(feature = "prometheus", not(test)))]
458                            if let hopr_crypto_packet::errors::PacketError::TicketValidation(_) = error {
459                                METRIC_REJECTED_TICKETS_COUNT.increment();
460                            }
461
462                            error!(%peer, %error, "failed to process the received message");
463
464                            // Send random signed acknowledgement to give feedback to the sender
465                            if let hopr_crypto_packet::errors::PacketError::PacketDecodingError(_) | hopr_crypto_packet::errors::PacketError::SphinxError(_) = error {
466                                // Do not send an ack back if the packet could not be decoded at all
467                                // 
468                                // Potentially adversarial behavior
469                                tracing::trace!(%peer, "not sending ack back on undecodable packet - possible adversarial behavior");
470                            } else {
471                                let now = std::time::Instant::now();
472
473                                if let Err(error) = ack_out_tx.send((None, peer_key)).await {
474                                    tracing::error!(%error, "failed to send ack to the egress queue");
475                                }
476                                let elapsed = now.elapsed();
477                                if elapsed.as_millis() > SLOW_OP_MS {
478                                    warn!(%peer, ?elapsed, "ack_out.send on failed packet took too long");
479                                }
480                            }
481                        }
482
483                        #[cfg(feature = "capture")]
484                        if let Ok(packet) = &res {
485                            let _ = capture_clone.try_send(capture::PacketBeforeTransit::IncomingPacket {
486                                    me: me_pub,
487                                    packet,
488                                    ticket: ticket_data_clone.into(),
489                                }.into()
490                            );
491                        }
492
493                        res.ok()
494                    }
495                })
496                .filter_map(move |maybe_packet| {
497                    let tbf = tbf.clone();
498
499                    futures::future::ready(
500                        if let Some(packet) = maybe_packet {
501                            match packet {
502                                IncomingPacket::Acknowledgement { packet_tag, previous_hop, .. } |
503                                IncomingPacket::Final { packet_tag, previous_hop,.. } |
504                                IncomingPacket::Forwarded { packet_tag, previous_hop, .. } => {
505                                    // This operation has run-time of ~10 nanoseconds,
506                                    // and therefore does not need to be invoked via spawn_blocking
507                                    if tbf.lock().check_and_set(&packet_tag) {
508                                        warn!(%previous_hop, "replayed packet received");
509
510                                        #[cfg(all(feature = "prometheus", not(test)))]
511                                        METRIC_REPLAYED_PACKET_COUNT.increment();
512
513                                        None
514                                    } else {
515                                        Some(packet)
516                                    }
517                                }
518                            }
519                        } else {
520                            trace!("received empty packet");
521                            None
522                        }
523                    )
524                })
525                .then_concurrent(move |packet| {
526                    let mut msg_to_send_tx = wire_msg.0.clone();
527
528                    #[cfg(feature = "capture")]
529                    let mut capture_clone = capture_clone.clone();
530
531                    let mut ticket_ack_tx_clone = ticket_ack_tx.clone();
532                    let mut ack_out_tx = ack_out_tx_clone_2.clone();
533                    async move {
534
535                    match packet {
536                        IncomingPacket::Acknowledgement {
537                            previous_hop,
538                            ack,
539                            ..
540                        } => {
541                            trace!(%previous_hop, "acknowledging ticket using received ack");
542                            let now = std::time::Instant::now();
543                            if let Err(error) = ticket_ack_tx_clone.send((ack, previous_hop)).await {
544                                tracing::error!(%error, "failed dispatching received acknowledgement to the ticket ack queue");
545                            }
546                            let elapsed = now.elapsed();
547                            if elapsed.as_millis() > SLOW_OP_MS {
548                                warn!(?elapsed," ack_tx.send took too long");
549                            }
550                            let elapsed = now.elapsed();
551                            if elapsed.as_millis() > SLOW_OP_MS {
552                                warn!(%previous_hop, ?elapsed, "ack_processor.handle_acknowledgement took too long");
553                            }
554
555                            // We do not acknowledge back acknowledgements.
556                            None
557                        },
558                        IncomingPacket::Final {
559                            previous_hop,
560                            sender,
561                            plain_text,
562                            ack_key,
563                            info,
564                            ..
565                        } => {
566                            // Send acknowledgement back
567                            trace!(%previous_hop, "acknowledging final packet back");
568                            let now = std::time::Instant::now();
569                            if let Err(error) = ack_out_tx.send((Some(ack_key), previous_hop)).await {
570                                tracing::error!(%error, "failed to send ack to the egress queue");
571                            }
572                            let elapsed = now.elapsed();
573                            if elapsed.as_millis() > SLOW_OP_MS {
574                                warn!(%previous_hop, ?elapsed, "ack_out.send on final packet took too long");
575                            }
576
577                            #[cfg(all(feature = "prometheus", not(test)))]
578                            {
579                                METRIC_PACKET_COUNT_PER_PEER.increment(&[&previous_hop.to_string(), "in"]);
580                                METRIC_PACKET_COUNT.increment(&["received"]);
581                            }
582
583                            Some((sender, plain_text, info))
584                        }
585                        IncomingPacket::Forwarded {
586                            previous_hop,
587                            next_hop,
588                            data,
589                            ack_key,
590                            ..
591                        } => {
592                            // First, relay the packet to the next hop
593                            trace!(%previous_hop, %next_hop, "forwarding packet to the next hop");
594
595                            #[cfg(feature = "capture")]
596                            let captured_packet: capture::CapturedPacket = capture::PacketBeforeTransit::OutgoingPacket {
597                                me: me_pub,
598                                next_hop,
599                                num_surbs: 0,
600                                is_forwarded: true,
601                                data: data.as_ref().into(),
602                                ack_challenge: Default::default(),
603                                signals: None.into(),
604                                ticket: inspect_ticket_data_in_packet(data.as_ref()).into()
605                            }.into();
606
607                            msg_to_send_tx
608                                .send((next_hop.into(), data))
609                                .await
610                                .unwrap_or_else(|_| {
611                                    error!("failed to forward a packet to the transport layer");
612                                });
613
614                            #[cfg(all(feature = "prometheus", not(test)))]
615                            {
616                                METRIC_PACKET_COUNT.increment(&["forwarded"]);
617                            }
618
619                            #[cfg(feature = "capture")]
620                            let _ = capture_clone.try_send(captured_packet);
621
622                             // Send acknowledgement back
623                            trace!(%previous_hop, "acknowledging forwarded packet back");
624                            let now = std::time::Instant::now();
625                            if let Err(error) = ack_out_tx.send((Some(ack_key), previous_hop)).await {
626                                tracing::error!(%error, "failed to send ack to the egress queue");
627                            }
628                            let elapsed = now.elapsed();
629                            if elapsed.as_millis() > SLOW_OP_MS {
630                                warn!(%previous_hop, ?elapsed, "ack_out.send on forwarded packet took too long");
631                            }
632
633                            None
634                        }
635                    }
636                }})
637                .filter_map(|maybe_data| futures::future::ready(
638                    // Create the ApplicationDataIn data structure for incoming data
639                    maybe_data
640                        .and_then(|(sender, data, aux_info)| ApplicationData::try_from(data.as_ref())
641                            .inspect_err(|error| tracing::error!(%sender, %error, "failed to decode application data"))
642                            .ok()
643                            .map(|data| (sender, ApplicationDataIn {
644                                data,
645                                packet_info: IncomingPacketInfo {
646                                    signals_from_sender: aux_info.packet_signals,
647                                    num_saved_surbs: aux_info.num_surbs,
648                                }
649                            })))
650                ))
651                .map(Ok)
652                .forward(api.0)
653                .instrument(tracing::trace_span!("msg protocol processing - ingress"))
654                .inspect(|_| tracing::warn!(task = "transport (protocol - msg ingress)", "long-running background task finished"))
655                .await;
656        }),
657    );
658
659    processes
