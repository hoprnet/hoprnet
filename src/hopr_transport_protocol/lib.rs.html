1//! Collection of objects and functionality allowing building of p2p or stream protocols for the higher business logic
2//! layers.
3//!
4//! ## Contents
5//!
6//! Supported protocol configurations:
7//!
8//! - `mix`
9//! - `ack`
10//! - `heartbeat`
11//! - `ticket_aggregation`
12//!
13//! Supported protocol processors:
14//!
15//! - `ticket_aggregation`
16//!
17//! ### `ticket_aggregation`
18//!
19//! Ticket aggregation processing mechanism is responsible for ingesting the ticket aggregation related requests:
20//!
21//! - `Receive(PeerId, U)`,
22//! - `Reply(PeerId, std::result::Result<Ticket, String>, T)`,
23//! - `Send(PeerId, Vec<AcknowledgedTicket>, TicketAggregationFinalizer)`,
24//!
25//! where `U` is the type of an aggregated ticket extractable (`ResponseChannel<Result<Ticket, String>>`) and `T`
26//! represents a network negotiated identifier (`RequestId`).
27//!
28//! In broader context the protocol flow is as follows:
29//!
30//! 1. requesting ticket aggregation
31//!
32//!    - the peer A desires to aggregate tickets, collects the tickets into a data collection and sends a request
33//!      containing the collection to aggregate `Vec<AcknowledgedTicket>` to peer B using the `Send` mechanism
34//!
35//! 2. responding to ticket aggregation
36//!
37//!    - peer B obtains the request from peer A, performs the ticket aggregation and returns a result of that operation
38//!      in the form of `std::result::Result<Ticket, String>` using the `Reply` mechanism
39//!
40//! 3. accepting the aggregated ticket
41//!    - peer A receives the aggregated ticket using the `Receive` mechanism
42//!
43//! Furthermore, apart from the basic positive case scenario, standard mechanics of protocol communication apply:
44//!
45//! - the requesting side can time out, if the responding side takes too long to provide an aggregated ticket, in which
46//!   case the ticket is not considered aggregated, even if eventually an aggregated ticket is delivered
47//! - the responder can fail to aggregate tickets in which case it replies with an error string describing the failure
48//!   reason and it is the requester's responsibility to handle the negative case as well
49//!   - in the absence of response, the requester will time out

145#[cfg(feature = "capture")]
146fn inspect_ticket_data_in_packet(raw_packet: &[u8]) -> &[u8] {
147    use hopr_primitive_types::traits::BytesEncodable;
148    if raw_packet.len() >= hopr_internal_types::tickets::Ticket::SIZE {
149        &raw_packet[raw_packet.len() - hopr_internal_types::tickets::Ticket::SIZE..]
150    } else {
151        &[]
152    }
153}

155/// Run all processes responsible for handling the msg and acknowledgment protocols.
156///
157/// The pipeline does not handle the mixing itself, that needs to be injected as a separate process
158/// overlayed on top of the `wire_msg` Stream or Sink.
159#[allow(clippy::too_many_arguments)]
160pub async fn run_msg_ack_protocol<Db>(
161    packet_cfg: processor::PacketInteractionConfig,
162    db: Db,
163    bloom_filter_persistent_path: Option<String>,
164    wire_msg: (
165        impl futures::Sink<(PeerId, Box<[u8]>)> + Clone + Unpin + Send + Sync + 'static,
166        impl futures::Stream<Item = (PeerId, Box<[u8]>)> + Send + Sync + 'static,
167    ),
168    api: (
169        impl futures::Sink<(HoprPseudonym, ApplicationData)> + Send + Sync + 'static,
170        impl futures::Stream<Item = (ApplicationData, ResolvedTransportRouting, PacketSendFinalizer)>
171        + Send
172        + Sync
173        + 'static,
174    ),
175) -> HashMap<ProtocolProcesses, hopr_async_runtime::AbortHandle>
176where
177    Db: HoprDbProtocolOperations + std::fmt::Debug + Clone + Send + Sync + 'static,
178{
179    let me = packet_cfg.packet_keypair.clone();
180
181    #[cfg(feature = "capture")]
182    let me_pub = *hopr_crypto_types::keypairs::Keypair::public(&me);
183
184    let mut processes = HashMap::new();
185
186    #[cfg(all(feature = "prometheus", not(test)))]
187    {
188        // Initialize the lazy statics here
189        // lazy_static::initialize(&METRIC_RECEIVED_ACKS);
190        // lazy_static::initialize(&METRIC_SENT_ACKS);
191        // lazy_static::initialize(&METRIC_TICKETS_COUNT);
192        lazy_static::initialize(&METRIC_PACKET_COUNT);
193        lazy_static::initialize(&METRIC_PACKET_COUNT_PER_PEER);
194        lazy_static::initialize(&METRIC_REPLAYED_PACKET_COUNT);
195        lazy_static::initialize(&METRIC_REJECTED_TICKETS_COUNT);
196    }
197
198    #[cfg(feature = "capture")]
199    let capture = {
200        use std::str::FromStr;
201        let writer: Box<dyn capture::PacketWriter + Send + 'static> =
202            if let Ok(desc) = std::env::var("HOPR_CAPTURE_PACKETS") {
203                if let Ok(udp_writer) = std::net::SocketAddr::from_str(&desc)
204                    .map_err(std::io::Error::other)
205                    .and_then(capture::UdpPacketDump::new)
206                {
207                    warn!("udp packet capture initialized to {desc}");
208                    Box::new(udp_writer)
209                } else if let Ok(pcap_writer) = std::fs::File::create(&desc).and_then(capture::PcapPacketWriter::new) {
210                    warn!("pcap file packet capture initialized to {desc}");
211                    Box::new(pcap_writer)
212                } else {
213                    error!(desc, "failed to create packet capture: invalid socket address or file");
214                    Box::new(capture::NullWriter)
215                }
216            } else {
217                warn!("no packet capture specified");
218                Box::new(capture::NullWriter)
219            };
220        let (capture, ah) = capture::packet_capture_channel(writer);
221        processes.insert(ProtocolProcesses::Capture, ah);
222        capture
223    };
224
225    let tbf = if let Some(bloom_filter_persistent_path) = bloom_filter_persistent_path {
226        let tbf = WrappedTagBloomFilter::new(bloom_filter_persistent_path);
227        let tbf_2 = tbf.clone();
228        processes.insert(
229            ProtocolProcesses::BloomPersist,
230            spawn_as_abortable!(Box::pin(execute_on_tick(
231                std::time::Duration::from_secs(90),
232                move || {
233                    let tbf_clone = tbf_2.clone();
234
235                    async move { tbf_clone.save().await }
236                },
237                "persisting the bloom filter to disk".into(),
238            ))),
239        );
240        tbf
241    } else {
242        WrappedTagBloomFilter::new("no_tbf".into())
243    };
244
245    let msg_processor_read = processor::PacketProcessor::new(db.clone(), packet_cfg);
246    let msg_processor_write = msg_processor_read.clone();
247
248    #[cfg(feature = "capture")]
249    let capture_clone = capture.clone();
250
251    let msg_to_send_tx = wire_msg.0.clone();
252    processes.insert(
253        ProtocolProcesses::MsgOut,
254        spawn_as_abortable!(async move {
255            let _neverending = api
256                .1
257                .then_concurrent(|(data, routing, finalizer)| {
258                    let msg_processor = msg_processor_write.clone();
259
260                    #[cfg(feature = "capture")]
261                    let (mut capture_clone, data_clone, num_surbs) =
262                        (capture_clone.clone(), data.clone(), routing.count_return_paths() as u8);
263
264                    async move {
265                        match PacketWrapping::send(&msg_processor, data, routing).await {
266                            Ok(v) => {
267                                #[cfg(all(feature = "prometheus", not(test)))]
268                                {
269                                    METRIC_PACKET_COUNT_PER_PEER.increment(&[&v.next_hop.to_string(), "out"]);
270                                    METRIC_PACKET_COUNT.increment(&["sent"]);
271                                }
272                                finalizer.finalize(Ok(()));
273
274                                #[cfg(feature = "capture")]
275                                let _ = capture_clone.try_send(
276                                    capture::PacketBeforeTransit::OutgoingPacket {
277                                        me: me_pub,
278                                        next_hop: v.next_hop,
279                                        num_surbs,
280                                        is_forwarded: false,
281                                        data: data_clone.to_bytes().into_vec().into(),
282                                        ack_challenge: v.ack_challenge.as_ref().into(),
283                                        ticket: inspect_ticket_data_in_packet(&v.data).into(),
284                                    }
285                                    .into(),
286                                );
287
288                                Some((v.next_hop.into(), v.data))
289                            }
290                            Err(e) => {
291                                finalizer.finalize(Err(e));
292                                None
293                            }
294                        }
295                    }
296                })
297                .filter_map(futures::future::ready)
298                .inspect(|(peer, _)| trace!(%peer, "protocol message out"))
299                .map(Ok)
300                .forward(msg_to_send_tx)
301                .instrument(tracing::trace_span!("msg protocol processing - outgoing"))
302                .await;
303        }),
304    );
305
306    let msg_to_send_tx = wire_msg.0.clone();
307    let db_for_recv = db.clone();
308    let me_for_recv = me.clone();
309
310    #[cfg(feature = "capture")]
311    let capture_clone = capture.clone();
312
313    processes.insert(
314        ProtocolProcesses::MsgIn,
315        spawn_as_abortable!(async move {
316            let _neverending = wire_msg
317                .1
318                .then_concurrent(move |(peer, data)| {
319                    let msg_processor = msg_processor_read.clone();
320                    let db = db_for_recv.clone();
321                    let mut msg_to_send_tx = msg_to_send_tx.clone();
322                    let me = me.clone();
323
324                    trace!(%peer, "protocol message in");
325
326                    #[cfg(feature = "capture")]
327                    let (mut capture_clone, ticket_data_clone) = (
328                        capture.clone(),
329                        inspect_ticket_data_in_packet(&data).to_vec()
330                    );
331
332                    async move {
333                        let now = std::time::Instant::now();
334                        let res = msg_processor.recv(&peer, data).await.map_err(move |e| (peer, e));
335                        let elapsed = now.elapsed();
336                        if elapsed.as_millis() > SLOW_OP_MS {
337                            warn!(%peer, ?elapsed, "msg_processor.recv took too long");
338                        }
339
340                        // If there was an error caused by interpretation of the packet data,
341                        // we must send a random acknowledgement back.
342                        if let Err((peer, error)) = &res {
343                            #[cfg(all(feature = "prometheus", not(test)))]
344                            if let hopr_crypto_packet::errors::PacketError::TicketValidation(_) = error {
345                                METRIC_REJECTED_TICKETS_COUNT.increment();
346                            }
347
348                            error!(%peer, %error, "failed to process the received message");
349
350                            let peer: OffchainPublicKey = match peer.try_into() {
351                                Ok(p) => p,
352                                Err(error) => {
353                                    warn!(%peer, %error, "Dropping packet - cannot convert peer id");
354                                    return None;
355                                }
356                            };
357
358                            // Send random signed acknowledgement to give feedback to the sender
359                            let ack = Acknowledgement::random(&me);
360
361                            #[cfg(feature = "capture")]
362                            let captured_packet: capture::CapturedPacket = capture::PacketBeforeTransit::OutgoingAck {
363                                me: me_pub,
364                                next_hop: peer,
365                                ack,
366                                is_random: true,
367                            }.into();
368
369                            match db
370                                .to_send_no_ack(ack.as_ref().to_vec().into_boxed_slice(), peer)
371                                .await {
372                                    Ok(ack_packet) => {
373                                        let now = std::time::Instant::now();
374                                        if msg_to_send_tx.send((
375                                                ack_packet.next_hop.into(),
376                                                ack_packet.data,
377                                            )).await.is_err() {
378                                            error!("failed to forward an acknowledgement for a failed packet recv to the transport layer");
379                                        }
380                                        let elapsed = now.elapsed();
381                                        if elapsed.as_millis() > SLOW_OP_MS {
382                                            warn!(?elapsed," msg_to_send_tx.send took too long");
383                                        }
384
385                                        #[cfg(feature = "capture")]
386                                        let _ = capture_clone.try_send(captured_packet);
387                                    },
388                                    Err(error) => tracing::error!(%error, "Failed to create random ack packet for a failed receive"),
389                                }
390                        }
391
392                        #[cfg(feature = "capture")]
393                        if let Ok(packet) = &res {
394                            let _ = capture_clone.try_send(capture::PacketBeforeTransit::IncomingPacket {
395                                    me: me_pub,
396                                    packet,
397                                    ticket: ticket_data_clone.into(),
398                                }.into()
399                            );
400                        }
401
402                        res.ok()
403                    }
404                })
405                .filter_map(move |maybe_packet| {
406                    let tbf = tbf.clone();
407
408                    async move {
409                    if let Some(packet) = maybe_packet {
410                        match packet {
411                            IncomingPacket::Acknowledgement { packet_tag, previous_hop, .. } |
412                            IncomingPacket::Final { packet_tag, previous_hop,.. } |
413                            IncomingPacket::Forwarded { packet_tag, previous_hop, .. } => {
414                                if tbf.is_tag_replay(&packet_tag).await {
415                                    warn!(%previous_hop, "replayed packet received");
416
417                                    #[cfg(all(feature = "prometheus", not(test)))]
418                                    METRIC_REPLAYED_PACKET_COUNT.increment();
419
420                                    None
421                                } else {
422                                    Some(packet)
423                                }
424                            }
425                        }
426                    } else {
427                        trace!("received empty packet");
428                        None
429                    }
430                }
431                })
432                .then_concurrent(move |packet| {
433                    let mut msg_to_send_tx = wire_msg.0.clone();
434                    let db = db.clone();
435                    let me = me_for_recv.clone();
436
437                    #[cfg(feature = "capture")]
438                    let mut capture_clone = capture_clone.clone();
439
440                    async move {
441
442                    match packet {
443                        IncomingPacket::Acknowledgement {
444                            previous_hop,
445                            ack,
446                            ..
447                        } => {
448                            trace!(%previous_hop, "received a valid acknowledgement");
449                            match db.handle_acknowledgement(ack).await {
450                                Ok(_) => trace!(%previous_hop, "successfully processed a known acknowledgement"),
451                                // Eventually, we do not care here if the acknowledgement does not belong to any
452                                // unacknowledged packets.
453                                Err(error) => trace!(%previous_hop, %error, "valid acknowledgement is unknown or error occurred while processing it"),
454                            }
455
456                            // We do not acknowledge back acknowledgements.
457
458                            None
459                        },
460                        IncomingPacket::Final {
461                            previous_hop,
462                            sender,
463                            plain_text,
464                            ack_key,
465                            ..
466                        } => {
467                            // Send acknowledgement back
468                            trace!(%previous_hop, "acknowledging final packet back");
469                            let ack = Acknowledgement::new(ack_key, &me);
470
471                            #[cfg(feature = "capture")]
472                            let captured_packet: capture::CapturedPacket = capture::PacketBeforeTransit::OutgoingAck {
473                                me: me_pub,
474                                next_hop: previous_hop,
475                                ack,
476                                is_random: false,
477                            }.into();
478
479                            #[cfg(all(feature = "prometheus", not(test)))]
480                            {
481                                METRIC_PACKET_COUNT_PER_PEER.increment(&[&previous_hop.to_string(), "in"]);
482                                METRIC_PACKET_COUNT.increment(&["received"]);
483                            }
484
485                            if let Ok(ack_packet) = db
486                                .to_send_no_ack(ack.as_ref().to_vec().into_boxed_slice(), previous_hop)
487                                .await
488                                .inspect_err(|error| error!(%error, "failed to create ack packet for a received message"))
489                                {
490                                    if msg_to_send_tx.send((ack_packet.next_hop.into(), ack_packet.data)).await.is_err() {
491                                        error!("failed to send an acknowledgement for a received packet to the transport layer");
492                                    }
493
494                                    #[cfg(feature = "capture")]
495                                    let _ = capture_clone.try_send(captured_packet);
496                                }
497
498                                Some((sender, plain_text))
499                        }
500                        IncomingPacket::Forwarded {
501                            previous_hop,
502                            next_hop,
503                            data,
504                            ack,
505                            ..
506                        } => {
507                            // First, relay the packet to the next hop
508                            trace!(%previous_hop, %next_hop, "forwarding packet to the next hop");
509                            #[cfg(feature = "capture")]
510                            let captured_packet: capture::CapturedPacket = capture::PacketBeforeTransit::OutgoingPacket {
511                                me: me_pub,
512                                next_hop,
513                                num_surbs: 0,
514                                is_forwarded: true,
515                                data: data.as_ref().into(),
516                                ack_challenge: Default::default(),
517                                ticket: inspect_ticket_data_in_packet(data.as_ref()).into()
518                            }.into();
519
520                            msg_to_send_tx
521                                .send((next_hop.into(), data))
522                                .await
523                                .unwrap_or_else(|_| {
524                                    error!("failed to forward a packet to the transport layer");
525                                });
526
527                            #[cfg(all(feature = "prometheus", not(test)))]
528                            {
529                                METRIC_PACKET_COUNT.increment(&["forwarded"]);
530                            }
531
532                            #[cfg(feature = "capture")]
533                            let _ = capture_clone.try_send(captured_packet);
534
535                            // And then send acknowledgement to the previous hop
536                            trace!(%previous_hop, %next_hop, "acknowledging forwarded packet back");
537                            #[cfg(feature = "capture")]
538                            let captured_packet: capture::CapturedPacket = capture::PacketBeforeTransit::OutgoingAck {
539                                me: me_pub,
540                                next_hop: previous_hop,
541                                ack,
542                                is_random: false,
543                            }.into();
544
545                            if let Ok(ack_packet) = db
546                                .to_send_no_ack(ack.as_ref().to_vec().into_boxed_slice(), previous_hop)
547                                .await
548                                .inspect_err(|error| error!(%error, "failed to create ack packet for a relayed message"))
549                            {
550                                msg_to_send_tx
551                                    .send((ack_packet.next_hop.into(), ack_packet.data))
552                                    .await
553                                    .unwrap_or_else(|_| {
554                                        error!("failed to send an acknowledgement for a relayed packet to the transport layer");
555                                    });
556
557                                #[cfg(feature = "capture")]
558                                let _ = capture_clone.try_send(captured_packet);
559                            }
560                            None
561                        }
562                    }
563                }})
564                .filter_map(|maybe_data| async move {
565                    if let Some((sender, data)) = maybe_data {
566                        ApplicationData::from_bytes(data.as_ref())
567                            .inspect_err(|error| tracing::error!(%error, "failed to decode application data"))
568                            .ok()
569                            .map(|data| (sender, data))
570                    } else {
571                        None
572                    }
573                })
574                .map(Ok)
575                .forward(api.0)
576                .instrument(tracing::trace_span!("msg protocol processing - incoming"))
577                .await;
578        }),
579    );
580
581    processes
582}
