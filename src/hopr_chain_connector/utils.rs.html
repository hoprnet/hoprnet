11pub(crate) fn model_to_account_entry(
12    model: blokli_client::api::types::Account,
13) -> Result<AccountEntry, ConnectorError> {
14    let entry_type = if !model.multi_addresses.is_empty() {
15        AccountType::Announced(
16            model
17                .multi_addresses
18                .into_iter()
19                .filter_map(|addr| match Multiaddr::from_str(&addr) {
20                    Ok(addr) => Some(addr),
21                    Err(_) => {
22                        tracing::error!(%addr, "invalid multiaddress");
23                        None
24                    }
25                })
26                .collect(),
27        )
28    } else {
29        AccountType::NotAnnounced
30    };
31
32    Ok(AccountEntry {
33        public_key: model.packet_key.parse()?,
34        chain_addr: model.chain_key.parse()?,
35        key_id: (model.keyid as u32).into(),
36        entry_type,
37        safe_address: model.safe_address.map(|addr| Address::from_hex(&addr)).transpose()?,
38    })
39}

41pub(crate) fn model_to_graph_entry(
42    model: blokli_client::api::types::OpenedChannelsGraphEntry,
43) -> Result<(AccountEntry, AccountEntry, ChannelEntry), ConnectorError> {
44    let src = model_to_account_entry(model.source)?;
45    let dst = model_to_account_entry(model.destination)?;
46    let channel = ChannelEntry::new(
47        src.chain_addr,
48        dst.chain_addr,
49        model.channel.balance.0.parse()?,
50        model
51            .channel
52            .ticket_index
53            .0
54            .parse()
55            .map_err(|e| ConnectorError::TypeConversion(format!("invalid ticket index: {e}")))?,
56        match model.channel.status {
57            blokli_client::api::types::ChannelStatus::Open => ChannelStatus::Open,
58            blokli_client::api::types::ChannelStatus::PendingToClose => ChannelStatus::PendingToClose(
59                model
60                    .channel
61                    .closure_time
62                    .as_ref()
63                    .ok_or(ConnectorError::TypeConversion("invalid closure time".into()))
64                    .and_then(|t| {
65                        hopr_api::chain::DateTime::from_str(&t.0)
66                            .map_err(|e| ConnectorError::TypeConversion(format!("invalid closure time: {e}")))
67                    })?
68                    .into(),
69            ),
70            blokli_client::api::types::ChannelStatus::Closed => ChannelStatus::Closed,
71        },
72        model.channel.epoch as u32,
73    );
74
75    Ok((src, dst, channel))
76}

99pub(crate) fn model_to_chain_info(
100    model: blokli_client::api::types::ChainInfo,
101) -> Result<ParsedChainInfo, ConnectorError> {
102    Ok(ParsedChainInfo {
103        channel_closure_grace_period: model
104            .channel_closure_grace_period
105            .0
106            .parse()
107            .map(Duration::from_secs)
108            .map_err(|e| ConnectorError::TypeConversion(format!("invalid channel grace period: {e}")))?,
109        domain_separators: DomainSeparators {
110            ledger: model
111                .ledger_dst
112                .ok_or(ConnectorError::TypeConversion("missing ledger dst".into()))
113                .and_then(|v| {
114                    Hash::from_hex(&v).map_err(|e| ConnectorError::TypeConversion(format!("invalid ledger dst: {e}")))
115                })?,
116            safe_registry: model
117                .safe_registry_dst
118                .ok_or(ConnectorError::TypeConversion("missing safe registry dst".into()))
119                .and_then(|v| {
120                    Hash::from_hex(&v)
121                        .map_err(|e| ConnectorError::TypeConversion(format!("invalid safe registry dst: {e}")))
122                })?,
123            channel: model
124                .channel_dst
125                .ok_or(ConnectorError::TypeConversion("missing channel dst".into()))
126                .and_then(|v| {
127                    Hash::from_hex(&v).map_err(|e| ConnectorError::TypeConversion(format!("invalid channel dst: {e}")))
128                })?,
129        },
130        key_binding_fee: model
131            .key_binding_fee
132            .0
133            .parse()
134            .map_err(|e| ConnectorError::TypeConversion(format!("invalid key binding fee: {e}")))?,
135        info: ChainInfo {
136            chain_id: model.chain_id as u64,
137            hopr_network_name: model.network,
138            contract_addresses: serde_json::from_str(&model.contract_addresses.0)
139                .map_err(|e| ConnectorError::TypeConversion(format!("invalid contract addresses JSON: {e}")))?,
140        },
141        ticket_win_prob: WinningProbability::try_from_f64(model.min_ticket_winning_probability)
142            .map_err(|e| ConnectorError::TypeConversion(format!("invalid winning probability info: {e}")))?,
143        ticket_price: model
144            .ticket_price
145            .0
146            .parse()
147            .map_err(|e| ConnectorError::TypeConversion(format!("invalid ticket price: {e}")))?,
148        finality: model
149            .finality
150            .0
151            .parse::<u32>()
152            .map_err(|e| ConnectorError::TypeConversion(format!("failed to parse finality: {e}")))?
153            .max(1),
154        expected_block_time: Duration::from_secs(
155            model
156                .expected_block_time
157                .0
158                .parse()
159                .map_err(|e| ConnectorError::TypeConversion(format!("failed to parse expected block time: {e}")))?,
160        )
161        .max(Duration::from_secs(1)),
162    })
163}

165pub(crate) fn model_to_deployed_safe(model: blokli_client::api::types::Safe) -> Result<DeployedSafe, ConnectorError> {

167        address: Address::from_hex(&model.address)?,
168        owner: Address::from_hex(&model.chain_key)?,
169        module: Address::from_hex(&model.module_address)?,
170        registered_nodes: model
171            .registered_nodes
172            .into_iter()
173            .map(|addr| Address::from_hex(&addr))
174            .collect::<Result<Vec<_>, _>>()?,
175    })
176}

178pub(crate) async fn process_channel_changes_into_events(
179    new_channel: ChannelEntry,
180    changes: Vec<ChannelChange>,
181    me: &Address,
182    event_tx: &async_broadcast::Sender<ChainEvent>,
183) {
184    for change in changes {
185        tracing::trace!(id = %new_channel.get_id(), %change, "channel updated");
186        match change {
187            ChannelChange::Status {
188                left: ChannelStatus::Open,
189                right: ChannelStatus::PendingToClose(_),
190            } => {
191                tracing::debug!(id = %new_channel.get_id(), "channel pending to close");
192                let _ = event_tx
193                    .broadcast_direct(ChainEvent::ChannelClosureInitiated(new_channel))
194                    .await;
195            }
196            ChannelChange::Status {
197                left: ChannelStatus::PendingToClose(_),
198                right: ChannelStatus::Closed,
199            } => {
200                tracing::debug!(id = %new_channel.get_id(), "channel closed");
201                let _ = event_tx.broadcast_direct(ChainEvent::ChannelClosed(new_channel)).await;
202            }
203            ChannelChange::Status {
204                left: ChannelStatus::Closed,
205                right: ChannelStatus::Open,
206            } => {
207                tracing::debug!(id = %new_channel.get_id(), "channel reopened");
208                let _ = event_tx.broadcast_direct(ChainEvent::ChannelOpened(new_channel)).await;
209            }
210            ChannelChange::Balance { left, right } => {
211                if left > right {
212                    tracing::debug!(id = %new_channel.get_id(), "channel balance decreased");
213                    let _ = event_tx
214                        .broadcast_direct(ChainEvent::ChannelBalanceDecreased(new_channel, left - right))
215                        .await;
216                } else {
217                    tracing::debug!(id = %new_channel.get_id(), "channel balance increased");
218                    let _ = event_tx
219                        .broadcast_direct(ChainEvent::ChannelBalanceIncreased(new_channel, right - left))
220                        .await;
221                }
222            }
223            // Ticket index can wrap (left > right) on a channel re-open,
224            // but we're not interested in that here
225            ChannelChange::TicketIndex { left, right } if left < right => match new_channel.direction(me) {
226                Some(ChannelDirection::Incoming) => {
227                    // The corresponding event is raised in the ticket redeem tracker,
228                    // as the failure must be tracked there too.
229                    tracing::debug!(id = %new_channel.get_id(), "ticket redemption succeeded");
230                }
231                Some(ChannelDirection::Outgoing) => {
232                    tracing::debug!(id = %new_channel.get_id(), "counterparty has redeemed ticket on our channel");
233                    let _ = event_tx
234                        .broadcast_direct(ChainEvent::TicketRedeemed(new_channel, None))
235                        .await;
236                }
237                None => {
238                    tracing::debug!(id = %new_channel.get_id(), "ticket redeemed on foreign channel");
239                    let _ = event_tx
240                        .broadcast_direct(ChainEvent::TicketRedeemed(new_channel, None))
241                        .await;
242                }
243            },
244            _ => {}
245        }
246    }
