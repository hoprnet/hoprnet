10/// Errors from the temporary database backend.
11#[derive(Debug, thiserror::Error)]
12pub enum TempDbError {
13    #[error("database error: {0}")]
14    Database(#[from] redb::Error),
15    #[error("serialization error: {0}")]
16    Serialization(#[from] postcard::Error),
17    #[error("I/O error: {0}")]
18    Io(#[from] std::io::Error),
19}
20

51/// A backend that is implemented via [`redb`](https://docs.rs/redb/latest/redb/) database stored in a temporary file.
52///
53/// The database file is dropped once the last instance is dropped.
54#[derive(Clone)]
55pub struct TempDbBackend {
56    // `db` is declared before `_tmp` so the database is closed before the temp file is deleted
57    // (Rust drops fields in declaration order).
58    db: std::sync::Arc<redb::Database>,
59    _tmp: std::sync::Arc<tempfile::NamedTempFile>,
60}
61
62impl TempDbBackend {
63    pub fn new() -> Result<Self, TempDbError> {
64        let file = tempfile::NamedTempFile::new()?;
65
66        tracing::info!(path = %file.path().display(), "opened temporary redb database");
67
68        let db = redb::Database::create(file.path())?;
69
70        // Create all tables eagerly so that read-only lookups on a fresh
71        // database return Ok(None) instead of a TableError.
72        {
73            let write_tx = db.begin_write()?;
74            write_tx.open_table(ACCOUNTS_TABLE_DEF)?;
75            write_tx.open_table(CHANNELS_TABLE_DEF)?;
76            write_tx.open_table(ADDRESS_TO_ID)?;
77            write_tx.open_table(KEY_TO_ID)?;
78            write_tx.commit()?;
79        }
80
81        Ok(Self {
82            db: std::sync::Arc::new(db),
83            _tmp: std::sync::Arc::new(file),
84        })
85    }
86}

88const ACCOUNTS_TABLE_DEF: TableDefinition<u32, Vec<u8>> = TableDefinition::new("id_accounts");
89const CHANNELS_TABLE_DEF: TableDefinition<[u8; ChannelId::SIZE], Vec<u8>> = TableDefinition::new("id_channels");
90const ADDRESS_TO_ID: TableDefinition<[u8; Address::SIZE], u32> = TableDefinition::new("address_to_id");
91const KEY_TO_ID: TableDefinition<[u8; OffchainPublicKey::SIZE], u32> = TableDefinition::new("key_to_id");

93impl super::Backend for TempDbBackend {
94    type Error = TempDbError;

96    fn insert_account(&self, account: AccountEntry) -> Result<Option<AccountEntry>, Self::Error> {
97        let write_tx = self.db.begin_write()?;
98        let old_value = {
99            let mut accounts = write_tx.open_table(ACCOUNTS_TABLE_DEF)?;
100            let old_value = accounts
101                .insert(u32::from(account.key_id), postcard::to_allocvec(&account)?)?
102                .map(|v| postcard::from_bytes::<AccountEntry>(&v.value()))
103                .transpose()?;
104
105            let mut address_to_id = write_tx.open_table(ADDRESS_TO_ID)?;
106            let mut key_to_id = write_tx.open_table(KEY_TO_ID)?;
107
108            // Remove old account entry references not to create stale mappings if keys changed
109            if let Some(old_entry) = &old_value {
110                let chain_addr: [u8; Address::SIZE] = old_entry.chain_addr.into();
111                let packet_addr: [u8; OffchainPublicKey::SIZE] = old_entry.public_key.into();
112                address_to_id.remove(&chain_addr)?;
113                key_to_id.remove(&packet_addr)?;
114            }
115
116            let chain_addr: [u8; Address::SIZE] = account.chain_addr.into();
117            address_to_id.insert(chain_addr, u32::from(account.key_id))?;
118
119            let packet_addr: [u8; OffchainPublicKey::SIZE] = account.public_key.into();
120            key_to_id.insert(packet_addr, u32::from(account.key_id))?;
121
122            old_value
123        };
124        write_tx.commit()?;
125
126        tracing::debug!(new = %account, old = ?old_value, "upserted account");
127        Ok(old_value)
128    }
129
130    fn insert_channel(&self, channel: ChannelEntry) -> Result<Option<ChannelEntry>, Self::Error> {
131        let write_tx = self.db.begin_write()?;
132        let old_value = {
133            let mut channels = write_tx.open_table(CHANNELS_TABLE_DEF)?;
134            let channel_id: [u8; ChannelId::SIZE] = channel.get_id().into();
135            channels
136                .insert(channel_id, postcard::to_allocvec(&channel)?)?
137                .map(|v| postcard::from_bytes::<ChannelEntry>(&v.value()))
138                .transpose()?
139        };
140        write_tx.commit()?;
141
142        tracing::debug!(new = %channel, old = ?old_value, "upserted channel");
143        Ok(old_value)
144    }
145
146    fn get_account_by_id(&self, id: &HoprKeyIdent) -> Result<Option<AccountEntry>, Self::Error> {
147        let read_tx = self.db.begin_read()?;
148        let accounts = read_tx.open_table(ACCOUNTS_TABLE_DEF)?;
149        accounts
150            .get(u32::from(*id))?
151            .map(|v| postcard::from_bytes::<AccountEntry>(&v.value()))
152            .transpose()
153            .map_err(TempDbError::from)
154    }
155
156    fn get_account_by_key(&self, key: &OffchainPublicKey) -> Result<Option<AccountEntry>, Self::Error> {
157        let read_tx = self.db.begin_read()?;
158        let keys_to_id = read_tx.open_table(KEY_TO_ID)?;
159        let packet_addr: [u8; OffchainPublicKey::SIZE] = (*key).into();
160        let Some(id) = keys_to_id.get(packet_addr)?.map(|v| v.value()) else {
161            return Ok(None);
162        };
163        let accounts = read_tx.open_table(ACCOUNTS_TABLE_DEF)?;
164        accounts
165            .get(id)?
166            .map(|v| postcard::from_bytes::<AccountEntry>(&v.value()))
167            .transpose()
168            .map_err(TempDbError::from)
169    }
170
171    fn get_account_by_address(&self, chain_key: &Address) -> Result<Option<AccountEntry>, Self::Error> {
172        let read_tx = self.db.begin_read()?;
173        let address_to_id = read_tx.open_table(ADDRESS_TO_ID)?;
174        let chain_key: [u8; Address::SIZE] = (*chain_key).into();
175        let Some(id) = address_to_id.get(chain_key)?.map(|v| v.value()) else {
176            return Ok(None);
177        };
178        let accounts = read_tx.open_table(ACCOUNTS_TABLE_DEF)?;
179        accounts
180            .get(id)?
181            .map(|v| postcard::from_bytes::<AccountEntry>(&v.value()))
182            .transpose()
183            .map_err(TempDbError::from)
184    }
185
186    fn get_channel_by_id(&self, id: &ChannelId) -> Result<Option<ChannelEntry>, Self::Error> {
187        let read_tx = self.db.begin_read()?;
188        let channels = read_tx.open_table(CHANNELS_TABLE_DEF)?;
189        let id: [u8; ChannelId::SIZE] = (*id).into();
190        channels
191            .get(id)?
192            .map(|v| postcard::from_bytes::<ChannelEntry>(&v.value()))
193            .transpose()
194            .map_err(TempDbError::from)
195    }

216    #[test]
217    fn upsert_cleans_up_stale_index_entries() -> anyhow::Result<()> {
218        let backend = TempDbBackend::new()?;
219
220        let kp_a = OffchainKeypair::random();
221        let cp = ChainKeypair::random();
222
223        let account = AccountEntry {
224            public_key: (*kp_a.public()),
225            chain_addr: cp.public().to_address(),
226            entry_type: AccountType::NotAnnounced,
227            safe_address: None,
228            key_id: 1.into(),
229        };
230        backend.insert_account(account)?;
231
232        // Update same ID with a different offchain key
233        let kp_b = OffchainKeypair::random();
234        let updated = AccountEntry {
235            public_key: (*kp_b.public()),
236            chain_addr: cp.public().to_address(),
237            entry_type: AccountType::NotAnnounced,
238            safe_address: None,
239            key_id: 1.into(),
240        };
241        backend.insert_account(updated.clone())?;
242
243        // Old key should no longer resolve
244        assert!(backend.get_account_by_key(kp_a.public())?.is_none());
245        // New key should resolve
246        assert_eq!(backend.get_account_by_key(kp_b.public())?, Some(updated));
247
248        Ok(())
249    }
