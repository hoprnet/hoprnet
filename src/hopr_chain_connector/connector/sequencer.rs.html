13/// Takes care of sequencing transactions, nonce management, and submitting the transactions to the blockchain in-order.
14///
15/// This object is meant to be a singleton and cannot be cloned.
16pub struct TransactionSequencer<C, R> {
17    sender: futures::channel::mpsc::Sender<TxRequest<R>>,
18    client: std::sync::Arc<C>,
19}
20
21const TX_QUEUE_CAPACITY: usize = 2048;
22
23impl<C, R> TransactionSequencer<C, R>
24where
25    C: BlokliQueryClient + BlokliTransactionClient + Send + Sync + 'static,
26    R: SignableTransaction + Send + Sync + 'static,
27{
28    pub fn new(signer: ChainKeypair, client: std::sync::Arc<C>) -> Self {
29        tracing::debug!(signer = %signer.public().to_address(), "starting transaction sequencer");
30
31        let client_clone = client.clone();
32        let (sender, receiver) = futures::channel::mpsc::channel::<TxRequest<R>>(TX_QUEUE_CAPACITY);
33        let chain_info_cache = std::sync::Arc::new(async_lock::OnceCell::new());
34        let current_nonce = std::sync::Arc::new(std::sync::atomic::AtomicU64::new(0));
35        let current_nonce_clone = current_nonce.clone();
36        hopr_async_runtime::prelude::spawn(
37            receiver
38                .then(move |(tx, notifier): (R, _)| {
39                    let client = client_clone.clone();
40                    let signer = signer.clone();
41                    let chain_info_cache = chain_info_cache.clone();
42                    let current_nonce = current_nonce.clone();
43                    async move {
44                        // The chain ID is fetched only once and cached for the lifetime of the sequencer.
45                        let chain_id = match chain_info_cache
46                            .get_or_try_init(|| {
47                                client
48                                    .query_chain_info()
49                                    .map_err(ConnectorError::from)
50                                    .and_then(|ci| futures::future::ok(ci.chain_id as u64))
51                            })
52                            .await
53                        {
54                            Ok(chain_id) => {
55                                tracing::debug!(chain_id, "chain id initialized");
56                                *chain_id
57                            }
58                            Err(e) => return (Err(e), notifier),
59                        };
60
61                        // Currently use fixed gas estimation.
62                        let gas_estimation = GasEstimation::default();
63                        tracing::debug!(?gas_estimation, "gas estimation used for tx");
64
65                        // We always query the transaction count for the signer and use
66                        // the maximum between this value and the local counter
67                        match client
68                            .query_transaction_count(&signer.public().to_address().into())
69                            .map_err(ConnectorError::from)
70                            .await
71                        {
72                            Ok(tx_count) => {
73                                let prev_nonce =
74                                    current_nonce.fetch_max(tx_count, std::sync::atomic::Ordering::Relaxed);
75                                tracing::debug!(prev_nonce, tx_count, "transaction count retrieved");
76                            }
77                            Err(e) => return (Err(e), notifier),
78                        }
79
80                        let nonce = current_nonce.load(std::sync::atomic::Ordering::Relaxed);
81                        tracing::debug!(nonce, "nonce used for the tx");
82                        tx.sign_and_encode_to_eip2718(nonce, chain_id, gas_estimation.into(), &signer)
83                            .map_err(ConnectorError::from)
84                            .and_then(move |tx| {
85                                tracing::debug!(nonce, "submitting transaction");
86                                let client = client.clone();
87                                async move {
88                                    client
89                                        .submit_and_track_transaction(&tx)
90                                        .map_err(ConnectorError::from)
91                                        .await
92                                }
93                            })
94                            .map(|res| (res, notifier))
95                            .await
96                    }
97                })
98                .for_each(move |(res, notifier)| {
99                    // The nonce is incremented when the transaction succeeded or failed due to on-chain
100                    // rejection.
101                    if res.is_ok()
102                        || res
103                            .as_ref()
104                            .is_err_and(|error| error.as_transaction_rejection_error().is_some())
105                    {
106                        let prev_nonce = current_nonce_clone.fetch_add(1, std::sync::atomic::Ordering::SeqCst);
107                        tracing::debug!(prev_nonce, ?res, "nonce incremented due to tx success or rejection");
108                    } else {
109                        tracing::warn!(?res, "nonce not incremented due to tx failure other than rejection");
110                    }
111
112                    if notifier.send(res).is_err() {
113                        tracing::debug!(
114                            "failed to notify transaction result - the caller may not want to await the result \
115                             anymore."
116                        );
117                    }
118                    futures::future::ready(())
119                })
120                .inspect(|_| tracing::warn!("transaction sequencer queue stopped")),
121        );
122
123        Self { sender, client }
124    }
125}

127impl<C, R> TransactionSequencer<C, R>
128where
129    C: BlokliTransactionClient + Send + Sync + 'static,
130{
131    /// Adds the transaction to the [`TransactionSequencer`] queue.
132    ///
133    /// The `timeout_until_finalized` is a total time until the TX is submitted and either confirmed or rejected.
134    pub async fn enqueue_transaction(
135        &self,
136        transaction: R,
137        timeout_until_finalized: std::time::Duration,
138    ) -> errors::Result<impl Future<Output = errors::Result<blokli_client::api::types::Transaction>>> {
139        let (notifier_tx, notifier_rx) = futures::channel::oneshot::channel();
140
141        self.sender
142            .clone()
143            .send((transaction, notifier_tx))
144            .await
145            .map_err(|_| ConnectorError::InvalidState("transaction queue dropped"))?;
146
147        Ok(notifier_rx
148            .inspect_ok(|res| tracing::debug!(?res, "transaction tracking id received"))
149            .map(move |result| {
150                result
151                    .map_err(|_| ConnectorError::InvalidState("transaction notifier dropped"))
152                    .and_then(|tx_res| tx_res.map(|id| (id, timeout_until_finalized)))
153            })
154            .and_then(|(tx_id, timeout)| {
155                self.client
156                    .track_transaction(tx_id, timeout)
157                    .map_err(ConnectorError::from)
158                    .inspect(|res| tracing::debug!(?res, "transaction tracking done"))
159            }))
160    }
161}

163impl<C, R> Drop for TransactionSequencer<C, R> {
164    fn drop(&mut self) {
165        // Causes the internally spawned task that drives the queue to terminate
166        self.sender.close_channel();
167    }
