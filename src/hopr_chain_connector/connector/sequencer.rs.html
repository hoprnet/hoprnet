13/// Takes care of sequencing transactions, nonce management and submitting the transactions to the blockchain in-order.
14pub struct TransactionSequencer<C, R> {
15    sender: Option<futures::channel::mpsc::Sender<TxRequest<R>>>,
16    nonce: std::sync::Arc<std::sync::atomic::AtomicU64>,
17    client: std::sync::Arc<C>,
18    signer: ChainKeypair,
19}
20
21const TX_QUEUE_CAPACITY: usize = 2048;
22
23impl<C, R> TransactionSequencer<C, R>
24where
25    C: BlokliQueryClient + BlokliTransactionClient + Send + Sync + 'static,
26    R: SignableTransaction + Send + Sync + 'static,
27{
28    pub fn new(signer: ChainKeypair, client: std::sync::Arc<C>) -> Self {
29        Self {
30            sender: None,
31            nonce: std::sync::Arc::new(std::sync::atomic::AtomicU64::new(1)),
32            client,
33            signer,
34        }
35    }
36
37    pub async fn start(&mut self) -> errors::Result<()> {
38        let tx_count = self
39            .client
40            .query_transaction_count(&self.signer.public().to_address().into())
41            .await?;
42        self.nonce.store(tx_count, std::sync::atomic::Ordering::SeqCst);
43
44        let chain_id = self.client.query_chain_info().await?.chain_id as u64;
45
46        let (sender, receiver) = futures::channel::mpsc::channel(TX_QUEUE_CAPACITY);
47        self.sender = Some(sender);
48
49        tracing::debug!(tx_count, signer = %self.signer.public().to_address(), "starting transaction sequencer");
50
51        let nonce_load = self.nonce.clone();
52        let nonce_inc = self.nonce.clone();
53        let client = self.client.clone();
54        let signer = self.signer.clone();
55        hopr_async_runtime::prelude::spawn(
56            receiver
57                .then(move |(tx, notifier)| {
58                    let client = client.clone();
59                    let signer = signer.clone();
60                    let nonce = nonce_load.load(std::sync::atomic::Ordering::SeqCst);
61                    async move {
62                        // We currently use fixed gas estimation.
63                        tx.sign_and_encode_to_eip2718(nonce, chain_id, GasEstimation::default().into(), &signer)
64                            .map_err(ConnectorError::from)
65                            .and_then(move |tx| {
66                                tracing::debug!(nonce, "submitting transaction");
67                                let client = client.clone();
68                                async move {
69                                    client
70                                        .submit_and_track_transaction(&tx)
71                                        .map_err(ConnectorError::from)
72                                        .await
73                                }
74                            })
75                            .map(|res| (res, notifier))
76                            .await
77                    }
78                })
79                .for_each(move |(res, notifier)| {
80                    // The nonce is incremented when the transaction succeeded or failed due to on-chain
81                    // rejection.
82                    if res.is_ok()
83                        || res
84                            .as_ref()
85                            .is_err_and(|error| error.as_transaction_rejection_error().is_some())
86                    {
87                        let prev_nonce = nonce_inc.fetch_add(1, std::sync::atomic::Ordering::SeqCst);
88                        tracing::debug!(prev_nonce, ?res, "nonce incremented due to tx success or rejection");
89                    } else {
90                        tracing::debug!(?res, "nonce not incremented due to tx failure other than rejection");
91                    }
92                    if notifier.send(res).is_err() {
93                        tracing::debug!(
94                            "failed to notify transaction result - the caller may not want to await the result \
95                             anymore."
96                        );
97                    }
98                    futures::future::ready(())
99                })
100                .inspect(|_| tracing::warn!("transaction sequencer queue stopped")),
101        );
102
103        Ok(())
104    }
105}

107impl<C, R> TransactionSequencer<C, R>
108where
109    C: BlokliTransactionClient + Send + Sync + 'static,
110{
111    /// Adds the transaction to the [`TransactionSequencer`] queue.
112    ///
113    /// The `timeout_until_finalized` is a total time until the TX is submitted and either confirmed or rejected.
114    pub async fn enqueue_transaction(
115        &self,
116        transaction: R,
117        timeout_until_finalized: std::time::Duration,
118    ) -> errors::Result<impl Future<Output = errors::Result<blokli_client::api::types::Transaction>>> {
119        let mut sender = self
120            .sender
121            .clone()
122            .ok_or(ConnectorError::InvalidState("transaction sender not started"))?;
123
124        let (notifier_tx, notifier_rx) = futures::channel::oneshot::channel();
125
126        sender
127            .send((transaction, notifier_tx))
128            .await
129            .map_err(|_| ConnectorError::InvalidState("transaction queue dropped"))?;
130
131        Ok(notifier_rx
132            .inspect_ok(|res| tracing::debug!(?res, "transaction tracking id received"))
133            .map(move |result| {
134                result
135                    .map_err(|_| ConnectorError::InvalidState("transaction notifier dropped"))
136                    .and_then(|tx_res| tx_res.map(|id| (id, timeout_until_finalized)))
137            })
138            .and_then(|(tx_id, timeout)| {
139                self.client
140                    .track_transaction(tx_id, timeout)
141                    .map_err(ConnectorError::from)
142                    .inspect(|res| tracing::debug!(?res, "transaction tracking done"))
143            }))
144    }
145}

147impl<C, R> Drop for TransactionSequencer<C, R> {
148    fn drop(&mut self) {
149        if let Some(mut sender) = self.sender.take() {
150            // Causes the internally spawned task that drives the queue to terminate
151            sender.close_channel();
152        }
153    }
154}

167    #[tokio::test]
168    async fn test_sequencer_should_increase_nonce_after_successful_tx() -> anyhow::Result<()> {
169        let blokli_client = BlokliTestStateBuilder::default()
170            .with_balances([([1u8; Address::SIZE].into(), HoprBalance::zero())])
171            .with_balances([([1u8; Address::SIZE].into(), XDaiBalance::zero())])
172            .with_balances([(
173                ChainKeypair::from_secret(&PRIVATE_KEY_1)?.public().to_address(),
174                XDaiBalance::new_base(10),
175            )])
176            .with_balances([(
177                ChainKeypair::from_secret(&PRIVATE_KEY_1)?.public().to_address(),
178                HoprBalance::new_base(1000),
179            )])
180            .with_hopr_network_chain_info("rotsee")
181            .build_dynamic_client(MODULE_ADDR.into());
182
183        let blokli_client = std::sync::Arc::new(blokli_client);
184
185        let mut tx_sequencer = TransactionSequencer::<_, <SafePayloadGenerator as PayloadGenerator>::TxRequest>::new(
186            ChainKeypair::from_secret(&PRIVATE_KEY_1)?,
187            blokli_client.clone(),
188        );
189        tx_sequencer.start().await?;
190
191        let safe_tx_gen = SafePayloadGenerator::new(
192            &ChainKeypair::from_secret(&PRIVATE_KEY_1)?,
193            contract_addresses_for_network("rotsee").unwrap().1,
194            MODULE_ADDR.into(),
195        );
196
197        let nonce_before = tx_sequencer.nonce.load(std::sync::atomic::Ordering::SeqCst);
198
199        tx_sequencer
200            .enqueue_transaction(
201                safe_tx_gen.transfer([1u8; Address::SIZE].into(), HoprBalance::new_base(100))?,
202                std::time::Duration::from_secs(2),
203            )
204            .await?
205            .await?;
206
207        assert!(
208            nonce_before < tx_sequencer.nonce.load(std::sync::atomic::Ordering::SeqCst),
209            "nonce should be incremented after successful tx"
210        );
211
212        insta::assert_yaml_snapshot!(*blokli_client.snapshot());
213
214        Ok(())
215    }
