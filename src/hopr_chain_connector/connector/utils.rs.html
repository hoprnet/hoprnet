11pub(crate) fn model_to_account_entry(
12    model: blokli_client::api::types::Account,
13) -> Result<AccountEntry, ConnectorError> {
14    let entry_type = if !model.multi_addresses.is_empty() {
15        AccountType::Announced(
16            model
17                .multi_addresses
18                .into_iter()
19                .filter_map(|addr| match Multiaddr::from_str(&addr) {
20                    Ok(addr) => Some(addr),
21                    Err(_) => {
22                        tracing::error!(%addr, "invalid multiaddress");
23                        None
24                    }
25                })
26                .collect(),
27        )
28    } else {
29        AccountType::NotAnnounced
30    };
31
32    Ok(AccountEntry {
33        public_key: model.packet_key.parse()?,
34        chain_addr: model.chain_key.parse()?,
35        key_id: (model.keyid as u32).into(),
36        entry_type,
37        safe_address: model.safe_address.map(|addr| Address::from_hex(&addr)).transpose()?,
38    })
39}

41pub(crate) fn model_to_graph_entry(
42    model: blokli_client::api::types::OpenedChannelsGraphEntry,
43) -> Result<(AccountEntry, AccountEntry, ChannelEntry), ConnectorError> {
44    let src = model_to_account_entry(model.source)?;
45    let dst = model_to_account_entry(model.destination)?;
46    let channel = ChannelEntry::new(
47        src.chain_addr,
48        dst.chain_addr,
49        model.channel.balance.0.parse()?,
50        model
51            .channel
52            .ticket_index
53            .0
54            .parse()
55            .map_err(|e| ConnectorError::TypeConversion(format!("invalid ticket index: {e}")))?,
56        match model.channel.status {
57            blokli_client::api::types::ChannelStatus::Open => ChannelStatus::Open,
58            blokli_client::api::types::ChannelStatus::PendingToClose => ChannelStatus::PendingToClose(
59                model
60                    .channel
61                    .closure_time
62                    .as_ref()
63                    .ok_or(ConnectorError::TypeConversion("invalid closure time".into()))
64                    .and_then(|t| {
65                        hopr_api::chain::DateTime::from_str(&t.0)
66                            .map_err(|e| ConnectorError::TypeConversion(format!("invalid closure time: {e}")))
67                    })?
68                    .into(),
69            ),
70            blokli_client::api::types::ChannelStatus::Closed => ChannelStatus::Closed,
71        },
72        (model.channel.epoch as u32).into(),
73    );
74    Ok((src, dst, channel))
75}

95pub(crate) fn model_to_chain_info(
96    model: blokli_client::api::types::ChainInfo,
97) -> Result<ParsedChainInfo, ConnectorError> {
98    Ok(ParsedChainInfo {
99        channel_closure_grace_period: model
100            .channel_closure_grace_period
101            .ok_or(ConnectorError::TypeConversion("missing channel grace period".into()))
102            .and_then(|v| {
103                v.0.parse()
104                    .map(Duration::from_secs)
105                    .map_err(|e| ConnectorError::TypeConversion(format!("invalid channel grace period: {e}")))
106            })?,
107        domain_separators: DomainSeparators {
108            ledger: model
109                .ledger_dst
110                .ok_or(ConnectorError::TypeConversion("missing ledger dst".into()))
111                .and_then(|v| {
112                    Hash::from_hex(&v).map_err(|e| ConnectorError::TypeConversion(format!("invalid ledger dst: {e}")))
113                })?,
114            safe_registry: model
115                .safe_registry_dst
116                .ok_or(ConnectorError::TypeConversion("missing safe registry dst".into()))
117                .and_then(|v| {
118                    Hash::from_hex(&v)
119                        .map_err(|e| ConnectorError::TypeConversion(format!("invalid safe registry dst: {e}")))
120                })?,
121            channel: model
122                .channel_dst
123                .ok_or(ConnectorError::TypeConversion("missing channel dst".into()))
124                .and_then(|v| {
125                    Hash::from_hex(&v).map_err(|e| ConnectorError::TypeConversion(format!("invalid channel dst: {e}")))
126                })?,
127        },
128        info: ChainInfo {
129            chain_id: model.chain_id as u64,
130            hopr_network_name: model.network,
131            contract_addresses: serde_json::from_str(&model.contract_addresses.0)
132                .map_err(|e| ConnectorError::TypeConversion(format!("invalid contract addresses JSON: {e}")))?,
133        },
134        ticket_win_prob: WinningProbability::try_from_f64(model.min_ticket_winning_probability)
135            .map_err(|e| ConnectorError::TypeConversion(format!("invalid winning probability info: {e}")))?,
136        ticket_price: model
137            .ticket_price
138            .0
139            .parse()
140            .map_err(|e| ConnectorError::TypeConversion(format!("invalid ticket price: {e}")))?,
141    })
142}

144pub(crate) async fn process_channel_changes_into_events(
145    new_channel: ChannelEntry,
146    changes: Vec<ChannelChange>,
147    me: &Address,
148    event_tx: &async_broadcast::Sender<ChainEvent>,
149) {
150    for change in changes {
151        tracing::trace!(id = %new_channel.get_id(), %change, "channel updated");
152        match change {
153            ChannelChange::Status {
154                left: ChannelStatus::Open,
155                right: ChannelStatus::PendingToClose(_),
156            } => {
157                tracing::debug!(id = %new_channel.get_id(), "channel pending to close");
158                let _ = event_tx
159                    .broadcast_direct(ChainEvent::ChannelClosureInitiated(new_channel))
160                    .await;
161            }
162            ChannelChange::Status {
163                left: ChannelStatus::PendingToClose(_),
164                right: ChannelStatus::Closed,
165            } => {
166                tracing::debug!(id = %new_channel.get_id(), "channel closed");
167                let _ = event_tx.broadcast_direct(ChainEvent::ChannelClosed(new_channel)).await;
168            }
169            ChannelChange::Status {
170                left: ChannelStatus::Closed,
171                right: ChannelStatus::Open,
172            } => {
173                tracing::debug!(id = %new_channel.get_id(), "channel reopened");
174                let _ = event_tx.broadcast_direct(ChainEvent::ChannelOpened(new_channel)).await;
175            }
176            ChannelChange::Balance { left, right } => {
177                if left > right {
178                    tracing::debug!(id = %new_channel.get_id(), "channel balance decreased");
179                    let _ = event_tx
180                        .broadcast_direct(ChainEvent::ChannelBalanceDecreased(new_channel, left - right))
181                        .await;
182                } else {
183                    tracing::debug!(id = %new_channel.get_id(), "channel balance increased");
184                    let _ = event_tx
185                        .broadcast_direct(ChainEvent::ChannelBalanceIncreased(new_channel, right - left))
186                        .await;
187                }
188            }
189            // Ticket index can wrap (left > right) on a channel re-open,
190            // but we're not interested in that here
191            ChannelChange::TicketIndex { left, right } if left < right => match new_channel.direction(me) {
192                Some(ChannelDirection::Incoming) => {
193                    // The corresponding event is raised in the ticket redeem tracker,
194                    // as the failure must be tracked there too.
195                    tracing::debug!(id = %new_channel.get_id(), "ticket redemption succeeded");
196                }
197                Some(ChannelDirection::Outgoing) => {
198                    tracing::debug!(id = %new_channel.get_id(), "counterparty has redeemed ticket on our channel");
199                    let _ = event_tx
200                        .broadcast_direct(ChainEvent::TicketRedeemed(new_channel, None))
201                        .await;
202                }
203                None => {
204                    tracing::debug!(id = %new_channel.get_id(), "ticket redeemed on foreign channel");
205                    let _ = event_tx
206                        .broadcast_direct(ChainEvent::TicketRedeemed(new_channel, None))
207                        .await;
208                }
209            },
210            _ => {}
211        }
212    }
