61/// A connector acting as middleware between the HOPR APIs (see the [`hopr_api`] crate) and the Blokli Client API (see
62/// the [`blokli_client`] crate).
63///
64/// The connector object cannot be cloned and shall be used inside an `Arc` if cloning is needed.
65pub struct HoprBlockchainConnector<C, B, P, R> {
66    payload_generator: P,
67    chain_key: ChainKeypair,
68    client: std::sync::Arc<C>,
69    graph: std::sync::Arc<parking_lot::RwLock<DiGraphMap<HoprKeyIdent, ChannelId, ahash::RandomState>>>,
70    backend: std::sync::Arc<B>,
71    connection_handle: Option<AbortHandle>,
72    sequencer: TransactionSequencer<C, R>,
73    events: EventsChannel,
74    cfg: BlockchainConnectorConfig,
75
76    // KeyId <-> OffchainPublicKey mapping
77    mapper: HoprKeyMapper<B>,
78    // Fast retrieval of chain keys by address
79    chain_to_packet: moka::future::Cache<Address, Option<OffchainPublicKey>, ahash::RandomState>,
80    // Fast retrieval of packet keys by chain key
81    packet_to_chain: moka::future::Cache<OffchainPublicKey, Option<Address>, ahash::RandomState>,
82    // Fast retrieval of channel entries by id
83    channel_by_id: moka::future::Cache<ChannelId, Option<ChannelEntry>, ahash::RandomState>,
84    // Fast retrieval of channel entries by parties
85    channel_by_parties: moka::future::Cache<ChannelParties, Option<ChannelEntry>, ahash::RandomState>,
86    // Contains chain info (no TTL - kept fresh by subscription handler)
87    values: moka::future::Cache<u32, ParsedChainInfo>,
88}

156    async fn do_connect(&self, timeout: Duration) -> Result<AbortHandle, ConnectorError> {
157        let sync_quota = self.cfg.sync_tolerance.clamp(1, 100) as f64 / 100.0;
158        let min_accounts = (self
159            .client
160            .count_accounts(blokli_client::api::AccountSelector::Any)
161            .await? as f64
162            * sync_quota)
163            .round() as u32;
164        let min_channels = (self
165            .client
166            .count_channels(blokli_client::api::ChannelSelector {
167                filter: None,
168                status: Some(blokli_client::api::types::ChannelStatus::Open),
169            })
170            .await? as f64
171            * sync_quota)
172            .round() as u32;
173        tracing::debug!(min_accounts, min_channels, "connection thresholds");
174
175        let server_health = self.client.query_health().await?;
176        if !server_health.eq_ignore_ascii_case("OK") {
177            tracing::error!(server_health, "blokli server not healthy");
178            return Err(ConnectorError::ServerNotHealthy);
179        }
180
181        let (abort_handle, abort_reg) = AbortHandle::new_pair();
182
183        let (connection_ready_tx, connection_ready_rx) = futures::channel::oneshot::channel();
184        let mut connection_ready_tx = Some(connection_ready_tx);
185
186        let client = self.client.clone();
187        let mapper = self.mapper.clone();
188        let backend = self.backend.clone();
189        let graph = self.graph.clone();
190        let event_tx = self.events.0.clone();
191        let me = self.chain_key.public().to_address();
192        let values_cache = self.values.clone();
193
194        let chain_to_packet = self.chain_to_packet.clone();
195        let packet_to_chain = self.packet_to_chain.clone();
196
197        let channel_by_id = self.channel_by_id.clone();
198        let channel_by_parties = self.channel_by_parties.clone();
199
200        // Query chain info to populate the cache
201        self.query_cached_chain_info().await?;
202
203        #[allow(clippy::large_enum_variant)]
204        #[derive(Debug)]
205        enum SubscribedEventType {
206            Account((AccountEntry, Option<AccountEntry>)),
207            Channel((ChannelEntry, Option<Vec<ChannelChange>>)),
208            WinningProbability((WinningProbability, Option<WinningProbability>)),
209            TicketPrice((HoprBalance, Option<HoprBalance>)),
210        }
211
212        hopr_async_runtime::prelude::spawn(async move {
213            let sync_started = std::time::Instant::now();
214
215            let connections = client
216                .subscribe_accounts(blokli_client::api::AccountSelector::Any)
217                .and_then(|accounts| Ok((accounts, client.subscribe_graph()?)))
218                .and_then(|(accounts, channels)| Ok((accounts, channels, client.subscribe_ticket_params()?)));
219
220            if let Err(error) = connections {
221                if let Some(connection_ready_tx) = connection_ready_tx.take() {
222                    let _ = connection_ready_tx.send(Err(error));
223                }
224                return;
225            }
226
227            let (account_stream, channel_stream, ticket_params_stream) = connections.unwrap();
228
229            // Stream of Account events (Announcements)
230            let graph_clone = graph.clone();
231            let account_stream = account_stream
232                .inspect_ok(|entry| tracing::trace!(?entry, "new account event"))
233                .map_err(ConnectorError::from)
234                .try_filter_map(|account| futures::future::ready(model_to_account_entry(account).map(Some)))
235                .and_then(move |account| {
236                    let graph = graph_clone.clone();
237                    let mapper = mapper.clone();
238                    let chain_to_packet = chain_to_packet.clone();
239                    let packet_to_chain = packet_to_chain.clone();
240                    hopr_async_runtime::prelude::spawn_blocking(move || {
241                        mapper.key_to_id.insert(account.public_key, Some(account.key_id));
242                        mapper.id_to_key.insert(account.key_id, Some(account.public_key));
243                        graph.write().add_node(account.key_id);
244                        mapper.backend.insert_account(account.clone()).map(|old| (account, old))
245                    })
246                    .map_err(ConnectorError::backend)
247                    .and_then(move |res| {
248                        let chain_to_packet = chain_to_packet.clone();
249                        let packet_to_chain = packet_to_chain.clone();
250                        async move {
251                            if let Ok((upserted_account, _)) = &res {
252                                // Rather update the cached entry than invalidating it
253                                chain_to_packet
254                                    .insert(upserted_account.chain_addr, Some(upserted_account.public_key))
255                                    .await;
256                                packet_to_chain
257                                    .insert(upserted_account.public_key, Some(upserted_account.chain_addr))
258                                    .await;
259                            }
260                            res.map(SubscribedEventType::Account).map_err(ConnectorError::backend)
261                        }
262                    })
263                })
264                .fuse();
265
266            // Stream of channel graph updates
267            let channel_stream = channel_stream
268                .map_err(ConnectorError::from)
269                .inspect_ok(|entry| tracing::trace!(?entry, "new graph event"))
270                .try_filter_map(|graph_event| futures::future::ready(model_to_graph_entry(graph_event).map(Some)))
271                .and_then(move |(src, dst, channel)| {
272                    let graph = graph.clone();
273                    let backend = backend.clone();
274                    let channel_by_id = channel_by_id.clone();
275                    let channel_by_parties = channel_by_parties.clone();
276                    hopr_async_runtime::prelude::spawn_blocking(move || {
277                        graph.write().add_edge(src.key_id, dst.key_id, *channel.get_id());
278                        backend
279                            .insert_channel(channel)
280                            .map(|old| (channel, old.map(|old| old.diff(&channel))))
281                    })
282                    .map_err(ConnectorError::backend)
283                    .and_then(move |res| {
284                        let channel_by_id = channel_by_id.clone();
285                        let channel_by_parties = channel_by_parties.clone();
286                        async move {
287                            if let Ok((upserted_channel, _)) = &res {
288                                // Rather update the cached entry than invalidating it
289                                channel_by_id
290                                    .insert(*upserted_channel.get_id(), Some(*upserted_channel))
291                                    .await;
292                                channel_by_parties
293                                    .insert(ChannelParties::from(upserted_channel), Some(*upserted_channel))
294                                    .await;
295                            }
296                            res.map(SubscribedEventType::Channel).map_err(ConnectorError::backend)
297                        }
298                    })
299                })
300                .fuse();
301
302            // Stream of ticket parameter updates (ticket price, minimum winning probability)
303            let ticket_params_stream = ticket_params_stream
304                .map_err(ConnectorError::from)
305                .inspect_ok(|entry| tracing::trace!(?entry, "new ticket params"))
306                .try_filter_map(|ticket_value_event| {
307                    futures::future::ready(model_to_ticket_params(ticket_value_event).map(Some))
308                })
309                .and_then(|(new_ticket_price, new_win_prob)| {
310                    let values_cache = values_cache.clone();
311                    async move {
312                        let mut events = Vec::<SubscribedEventType>::new();
313                        values_cache
314                            .entry(CHAIN_INFO_CACHE_KEY)
315                            .and_compute_with(|cached_entry| {
316                                futures::future::ready(match cached_entry {
317                                    Some(chain_info) => {
318                                        let mut chain_info = chain_info.into_value();
319                                        if chain_info.ticket_price != new_ticket_price {
320                                            events.push(SubscribedEventType::TicketPrice((
321                                                new_ticket_price,
322                                                Some(chain_info.ticket_price),
323                                            )));
324                                            chain_info.ticket_price = new_ticket_price;
325                                        }
326                                        if !chain_info.ticket_win_prob.approx_eq(&new_win_prob) {
327                                            events.push(SubscribedEventType::WinningProbability((
328                                                new_win_prob,
329                                                Some(chain_info.ticket_win_prob),
330                                            )));
331                                            chain_info.ticket_win_prob = new_win_prob;
332                                        }
333
334                                        if !events.is_empty() {
335                                            moka::ops::compute::Op::Put(chain_info)
336                                        } else {
337                                            moka::ops::compute::Op::Nop
338                                        }
339                                    }
340                                    None => {
341                                        tracing::warn!(
342                                            "chain info not present in the cache before ticket params update"
343                                        );
344                                        events.push(SubscribedEventType::TicketPrice((new_ticket_price, None)));
345                                        events.push(SubscribedEventType::WinningProbability((new_win_prob, None)));
346                                        moka::ops::compute::Op::Nop
347                                    }
348                                })
349                            })
350                            .await;
351                        Ok(futures::stream::iter(events).map(Ok::<_, ConnectorError>))
352                    }
353                })
354                .try_flatten()
355                .fuse();
356
357            let mut account_counter = 0;
358            let mut channel_counter = 0;
359            if min_accounts == 0 && min_channels == 0 {
360                tracing::info!(account_counter, channel_counter, time = ?sync_started.elapsed(), "on-chain graph has been synced");
361                let _ = connection_ready_tx.take().unwrap().send(Ok(()));
362            }
363
364            futures::stream::Abortable::new(
365                (account_stream, channel_stream, ticket_params_stream).merge(),
366                abort_reg,
367            )
368            .inspect_ok(move |event_type| {
369                if connection_ready_tx.is_some() {
370                    match event_type {
371                        SubscribedEventType::Account(_) => account_counter += 1,
372                        SubscribedEventType::Channel(_) => channel_counter += 1,
373                        _ => {}
374                    }
375
376                    let pct_synced =
377                        ((account_counter + channel_counter) * 100 / (min_accounts + min_channels)).clamp(0, 100);
378                    tracing::debug!(
379                        pct_synced,
380                        sync_quota,
381                        account_counter,
382                        channel_counter,
383                        "percentage of connection quota synced"
384                    );
385
386                    // Send the completion notification
387                    // once we reach the expected number of accounts and channels with
388                    // the given tolerance
389                    if account_counter >= min_accounts && channel_counter >= min_channels {
390                        tracing::info!(account_counter, channel_counter, time = ?sync_started.elapsed(), "on-chain graph has been synced");
391                        let _ = connection_ready_tx.take().unwrap().send(Ok(()));
392                    }
393                }
394            })
395            .for_each(|event_type| {
396                let event_tx = event_tx.clone();
397                async move {
398                    match event_type {
399                        Ok(SubscribedEventType::Account((new_account, old_account))) => {
400                            tracing::debug!(%new_account, "account inserted");
401                            // We only track public accounts as events and also
402                            // broadcast announcements of already existing accounts (old_account == None).
403                            if new_account.has_announced_with_routing_info()
404                                && old_account.is_none_or(|a| !a.has_announced_with_routing_info())
405                            {
406                                tracing::debug!(account = %new_account, "new announcement");
407                                let _ = event_tx
408                                    .broadcast_direct(ChainEvent::Announcement(new_account.clone()))
409                                    .await;
410                            }
411                        }
412                        Ok(SubscribedEventType::Channel((new_channel, Some(changes)))) => {
413                            tracing::debug!(
414                                id = %new_channel.get_id(),
415                                src = %new_channel.source, dst = %new_channel.destination,
416                                num_changes = changes.len(),
417                                "channel updated"
418                            );
419                            process_channel_changes_into_events(new_channel, changes, &me, &event_tx).await;
420                        }
421                        Ok(SubscribedEventType::Channel((new_channel, None))) => {
422                            tracing::debug!(
423                                id = %new_channel.get_id(),
424                                src = %new_channel.source, dst = %new_channel.destination,
425                                "channel opened"
426                            );
427                            let _ = event_tx.broadcast_direct(ChainEvent::ChannelOpened(new_channel)).await;
428                        }
429                        Ok(SubscribedEventType::WinningProbability((new, old))) => {
430                            let old = old.unwrap_or_default();
431                            match new.approx_cmp(&old) {
432                                Ordering::Less => {
433                                    tracing::debug!(%new, %old, "winning probability decreased");
434                                    let _ = event_tx
435                                        .broadcast_direct(ChainEvent::WinningProbabilityDecreased(new))
436                                        .await;
437                                }
438                                Ordering::Greater => {
439                                    tracing::debug!(%new, %old, "winning probability increased");
440                                    let _ = event_tx
441                                        .broadcast_direct(ChainEvent::WinningProbabilityIncreased(new))
442                                        .await;
443                                }
444                                Ordering::Equal => {}
445                            }
446                        }
447                        Ok(SubscribedEventType::TicketPrice((new, old))) => {
448                            tracing::debug!(%new, ?old, "ticket price changed");
449                            let _ = event_tx.broadcast_direct(ChainEvent::TicketPriceChanged(new)).await;
450                        }
451                        Err(error) => {
452                            tracing::error!(%error, "error processing account/graph/ticket params subscription");
453                        }
454                    }
455                }
456            })
457            .await;
458        });
459
460        connection_ready_rx
461            .timeout(futures_time::time::Duration::from(timeout))
462            .map(|res| match res {
463                Ok(Ok(Ok(_))) => Ok(abort_handle),
464                Ok(Ok(Err(error))) => {
465                    abort_handle.abort();
466                    Err(ConnectorError::from(error))
467                }
468                Ok(Err(_)) => {
469                    abort_handle.abort();
470                    Err(ConnectorError::InvalidState("failed to determine connection state"))
471                }
472                Err(_) => {
473                    abort_handle.abort();
474                    tracing::error!(min_accounts, min_channels, "connection timeout when syncing");
475                    Err(ConnectorError::ConnectionTimeout)
476                }
477            })
478            .await
479    }

481    /// Connects to the chain using the underlying client, syncs all on-chain data,
482    /// and subscribes for all future updates.
483    ///
484    /// If the connection does not finish within
485    /// [`BlockchainConnectorConfig::connection_timeout`](BlockchainConnectorConfig)
486    /// the [`ConnectorError::ConnectionTimeout`] error is returned.
487    ///
488    /// Most of the operations with the Connector will fail if it is not connected first.
489    ///
490    /// There are some notable exceptions that DO NOT require a prior call to `connect`:
491    /// - all the [`ChainValues`](hopr_api::chain::ChainValues) methods,
492    /// - all the [`ChainReadSafeOperations`](hopr_api::chain::ChainReadSafeOperations) methods,
493    /// - all the [`ChainWriteSafeOperations`](hopr_api::chain::ChainWriteSafeOperations) methods,
494    /// - [`me`](hopr_api::chain::ChainReadChannelOperations::me)
495    ///
496    /// If you wish to only call operations from the above Chain APIs, consider constructing
497    /// the [`HoprBlockchainReader`](crate::HoprBlockchainReader) instead.
498    pub async fn connect(&mut self) -> Result<(), ConnectorError> {
499        if self
500            .connection_handle
501            .as_ref()
502            .filter(|handle| !handle.is_aborted())
503            .is_some()
504        {
505            return Err(ConnectorError::InvalidState("connector is already connected"));
506        }
507
508        let abort_handle = self
509            .do_connect(self.cfg.connection_sync_timeout.max(MIN_CONNECTION_TIMEOUT))
510            .await?;
511
512        self.connection_handle = Some(abort_handle);
513
514        tracing::info!(node = %self.chain_key.public().to_address(), "connected to chain as node");
515        Ok(())
516    }
517
518    /// Returns the reference to the underlying client.
519    pub fn client(&self) -> &C {
520        self.client.as_ref()
521    }
522
523    /// Checks if the connector is [connected](HoprBlockchainConnector::connect) to the chain.
524    pub fn is_connected(&self) -> bool {
525        self.check_connection_state().is_ok()
526    }
527}

529impl<B, C, P> HoprBlockchainConnector<C, B, P, P::TxRequest>
530where
531    C: BlokliTransactionClient + BlokliQueryClient + Send + Sync + 'static,
532    P: PayloadGenerator + Send + Sync,
533    P::TxRequest: Send + Sync,
534{
535    async fn send_tx<'a>(
536        &'a self,
537        tx_req: P::TxRequest,
538    ) -> Result<impl Future<Output = Result<ChainReceipt, ConnectorError>> + Send + 'a, ConnectorError> {
539        let chain_info = self.query_cached_chain_info().await?;
540        let tx_timeout = TX_TIMEOUT_MULTIPLIER * chain_info.finality * chain_info.expected_block_time;
541        Ok(self
542            .sequencer
543            .enqueue_transaction(tx_req, tx_timeout.max(MIN_TX_CONFIRM_TIMEOUT))
544            .await?
545            .and_then(|tx| {
546                futures::future::ready(
547                    ChainReceipt::from_str(&tx.transaction_hash.0)
548                        .map_err(|_| ConnectorError::TypeConversion("invalid tx hash".into())),
549                )
550            }))
551    }
552}

554impl<B, C, P, R> HoprBlockchainConnector<C, R, B, P> {
555    pub(crate) fn check_connection_state(&self) -> Result<(), ConnectorError> {
556        self.connection_handle
557            .as_ref()
558            .filter(|handle| !handle.is_aborted()) // Do a safety check
559            .ok_or(ConnectorError::InvalidState("connector is not connected"))
560            .map(|_| ())
561    }

563    /// Invalidates all cached on-chain data.
564    pub fn invalidate_caches(&self) {
565        self.channel_by_parties.invalidate_all();
566        self.channel_by_id.invalidate_all();
567        self.packet_to_chain.invalidate_all();
568        self.chain_to_packet.invalidate_all();
569        self.values.invalidate_all();
570    }

573impl<B, C, P, R> Drop for HoprBlockchainConnector<C, R, B, P> {
574    fn drop(&mut self) {
575        self.events.0.close();
576        if let Some(abort_handle) = self.connection_handle.take() {
577            abort_handle.abort();
578        }
579    }
580}

592    }
