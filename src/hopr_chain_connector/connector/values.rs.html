17pub(crate) const CHAIN_INFO_CACHE_KEY: u32 = 0;
18
19impl<B, C, P, R> HoprBlockchainConnector<C, R, B, P>
20where
21    C: BlokliQueryClient + Send + Sync + 'static,
22{
23    pub(crate) async fn query_cached_chain_info(&self) -> Result<ParsedChainInfo, ConnectorError> {
24        Ok(self
25            .values
26            .try_get_with(
27                CHAIN_INFO_CACHE_KEY,
28                self.client
29                    .query_chain_info()
30                    .map_err(ConnectorError::from)
31                    .and_then(|model| futures::future::ready(model_to_chain_info(model))),
32            )
33            .await?)
34    }
35}

84#[cfg(test)]
85mod tests {
86    use std::str::FromStr;
87
88    use hopr_api::chain::ChainValues;
89    use hopr_crypto_types::types::Hash;
90
91    use super::*;
92    use crate::{connector::tests::create_connector, testing::BlokliTestStateBuilder};
93
94    #[tokio::test]
95    async fn connector_should_query_chain_info() -> anyhow::Result<()> {
96        let blokli_client = BlokliTestStateBuilder::default()
97            .with_hopr_network_chain_info("rotsee")
98            .build_static_client();
99
100        let mut connector = create_connector(blokli_client)?;
101        connector.connect().await?;
102
103        let chain_info = connector.chain_info().await?;
104
105        assert_eq!(100, chain_info.chain_id);
106        assert_eq!("rotsee", &chain_info.hopr_network_name);
107
108        assert_eq!(Duration::from_mins(5), connector.channel_closure_notice_period().await?);
109        assert_eq!(HoprBalance::new_base(1), connector.minimum_ticket_price().await?);
110        assert!(WinningProbability::ALWAYS.approx_eq(&connector.minimum_incoming_ticket_win_prob().await?));
111        assert_eq!(Hash::default(), connector.domain_separators().await?.channel);
112        assert_eq!(
113            HoprBalance::from_str("0.01 wxHOPR")?,
114            connector.key_binding_fee().await?
115        );
116
117        Ok(())
118    }
