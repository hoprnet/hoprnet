15/// Compute candidate paths from `src` to `dest` through `graph`.
16///
17/// `length` is the number of edges to traverse (= intermediate hops + 1).
18/// `take` caps the number of candidate paths returned.
19/// The source node is stripped from every returned path; callers receive
20/// `[intermediates..., dest]`.
21fn compute_paths<G, C>(
22    graph: &G,
23    src: &OffchainPublicKey,
24    dest: &OffchainPublicKey,
25    length: std::num::NonZeroUsize,
26    take: usize,
27    cost_fn: C,
28) -> Vec<PathToDestination>
29where
30    G: NetworkGraphTraverse<NodeId = OffchainPublicKey> + NetworkGraphView<NodeId = OffchainPublicKey>,
31    <G as NetworkGraphTraverse>::Observed: EdgeObservableRead + Send + 'static,
32    C: CostFn<Weight = <G as NetworkGraphTraverse>::Observed, Cost = f64>,
33{
34    let raw = graph.simple_paths(src, dest, length.get(), Some(take), cost_fn);
35
36    raw.into_iter()
37        .filter_map(|(path, _, cost)| {
38            tracing::trace!(?path, ?cost, "evaluating candidate path");
39            if cost > 0.0 {
40                // Drop the first element (src) — callers expect [intermediates..., dest].
41                Some(path.into_iter().skip(1).collect::<Vec<_>>())
42            } else {
43                None
44            }
45        })
46        .collect::<Vec<_>>()
47}

49/// Extended forward path search: find shorter paths using [`ForwardPathCostFn`]
50/// and append `dest` to each one.
51///
52/// This handles the case where the last edge (relay -> dest) has no graph edge
53/// (e.g. no payment channel) but the path planner can still assume the last hop
54/// is reachable. Paths already found by Phase 1 are excluded via `existing`.
55fn compute_extended_forward_paths<G>(
56    graph: &G,
57    src: &OffchainPublicKey,
58    dest: &OffchainPublicKey,
59    shorter_length: std::num::NonZeroUsize,
60    take: usize,
61    existing: &[PathToDestination],
62) -> Vec<PathToDestination>
63where
64    G: NetworkGraphTraverse<NodeId = OffchainPublicKey> + NetworkGraphView<NodeId = OffchainPublicKey>,
65    <G as NetworkGraphTraverse>::Observed: EdgeObservableRead + Send + 'static,
66{
67    let raw = graph.simple_paths_from(src, shorter_length.get(), Some(take), ForwardPathCostFn::new());
68
69    raw.into_iter()
70        .filter_map(|(path, _, cost)| {
71            if cost <= 0.0 {
72                return None;
73            }
74
75            // Strip source and append dest.
76            let mut candidate: Vec<_> = path.into_iter().skip(1).collect();
77
78            // Ensure dest is not already in the path (would create a repeated node).
79            if candidate.contains(dest) {
80                return None;
81            }
82
83            candidate.push(*dest);
84
85            // Skip paths already found by Phase 1.
86            if existing.contains(&candidate) {
87                return None;
88            }
89
90            tracing::trace!(?candidate, ?cost, "extended forward path candidate");
91            Some(candidate)
92        })
93        .take(take)
94        .collect()
95}

97/// A lightweight graph-backed path selector.
98///
99/// Returns all candidate paths for a `(src, dest, hops)` query directly from
100/// the network graph — no caching is performed here.  The caller (typically
101/// [`crate::planner::PathPlanner`]) is responsible for caching, TTL management,
102/// background refresh, and final path selection.
103///
104/// Stores the planner's own identity (`me`) so that it can choose the
105/// appropriate cost function:
106/// - forward path (`src == me`): [`HoprForwardCostFn`]
107/// - return path (`dest == me`): [`HoprReturnCostFn`]
108#[derive(Clone)]
109pub struct HoprGraphPathSelector<G> {
110    me: OffchainPublicKey,
111    graph: G,
112    max_paths: usize,
113}
114
115impl<G> HoprGraphPathSelector<G>
116where
117    G: NetworkGraphTraverse<NodeId = OffchainPublicKey>
118        + NetworkGraphView<NodeId = OffchainPublicKey>
119        + Clone
120        + Send
121        + Sync
122        + 'static,
123    <G as NetworkGraphTraverse>::Observed: EdgeObservableRead + Send + 'static,
124{
125    /// Create a new selector.
126    ///
127    /// * `me` – the planner's own offchain public key, used to determine path direction.
128    /// * `graph` – the network graph to query.
129    /// * `max_paths` – maximum number of candidate paths to return per query.
130    pub fn new(me: OffchainPublicKey, graph: G, max_paths: usize) -> Self {
131        Self { me, graph, max_paths }
132    }
133}
134
135impl<G> PathSelector for HoprGraphPathSelector<G>
136where
137    G: NetworkGraphTraverse<NodeId = OffchainPublicKey>
138        + NetworkGraphView<NodeId = OffchainPublicKey>
139        + Clone
140        + Send
141        + Sync
142        + 'static,
143    <G as NetworkGraphTraverse>::Observed: EdgeObservableRead + Send + 'static,
144{
145    /// Return all candidate paths from `src` to `dest` via `hops` relays.
146    ///
147    /// Each returned `Vec<OffchainPublicKey>` has length `hops + 1` and contains
148    /// `[intermediates..., dest]`; `src` is excluded.
149    ///
150    /// Returns `Err(PathNotFound)` when the graph yields no positive-cost paths.
151    ///
152    /// The function has a potential to run expensive operations, it should be benchmarked
153    /// in a production environment and possibly guarded (e.g. by offloading the long execution
154    /// in an async executor to avoid blocking the caller).
155    #[tracing::instrument(level = "trace", skip(self), fields(src = %src, dest = %dest, hops), ret, err)]
156    fn select_path(
157        &self,
158        src: OffchainPublicKey,
159        dest: OffchainPublicKey,
160        hops: usize,
161    ) -> Result<Vec<Vec<OffchainPublicKey>>> {
162        tracing::trace!(%src, %dest, hops, "computing paths from graph");
163        let length = std::num::NonZeroUsize::new(hops + 1)
164            .expect("can never fail, it is physically at least 1 after the addition");
165
166        let paths = if src == self.me {
167            // Phase 1: search for full-length forward paths to dest.
168            let mut found = compute_paths(
169                &self.graph,
170                &src,
171                &dest,
172                length,
173                self.max_paths,
174                HoprForwardCostFn::new(length),
175            );
176
177            // Phase 2: if not enough paths, do an extended search with ForwardPathCostFn
178            // for (length - 1) edges and assume the last hop can be done by anybody.
179            if found.len() < self.max_paths
180                && let Some(shorter) = std::num::NonZeroUsize::new(length.get() - 1)
181            {
182                let remaining = self.max_paths - found.len();
183                let extended = compute_extended_forward_paths(&self.graph, &src, &dest, shorter, remaining, &found);
184                found.extend(extended);
185            }
186
187            found
188        } else {
189            compute_paths(
190                &self.graph,
191                &src,
192                &dest,
193                length,
194                self.max_paths,
195                HoprReturnCostFn::new(length),
196            )
197        };
198
199        // no need to deduplicate for now
200
201        if paths.is_empty() {
202            Err(PathPlannerError::Path(PathError::PathNotFound(
203                hops,
204                src.to_string(),
205                dest.to_string(),
206            )))
207        } else {
208            Ok(paths)
209        }
210    }
211}

581            assert_eq!(path.last(), Some(&dest));
582            assert!(!path.contains(&me));
583        }
584
585        let rev = selector
586            .select_path(dest, me, RoutingOptions::MAX_INTERMEDIATE_HOPS)
587            .context("reverse 3-hop path")?;
588        assert!(!rev.is_empty());
589        for path in &rev {
590            assert_eq!(path.len(), 4, "reverse: [c, b, a, me]");
591            assert_eq!(path.last(), Some(&me));
592            assert!(!path.contains(&dest));
593        }
594
595        Ok(())
596    }
