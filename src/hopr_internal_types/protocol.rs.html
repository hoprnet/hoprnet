38/// Represents packet acknowledgement
39#[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]
40pub struct Acknowledgement {
41    ack_signature: OffchainSignature,
42    pub ack_key_share: HalfKey,
43    validated: bool,
44}
45
46impl Acknowledgement {
47    pub fn new(ack_key_share: HalfKey, node_keypair: &OffchainKeypair) -> Self {
48        Self {
49            ack_signature: OffchainSignature::sign_message(ack_key_share.as_ref(), node_keypair),
50            ack_key_share,
51            validated: true,
52        }
53    }
54
55    /// Generates random, but still a valid acknowledgement.
56    pub fn random(offchain_keypair: &OffchainKeypair) -> Self {
57        Self::new(HalfKey::random(), offchain_keypair)
58    }
59
60    /// Validates the acknowledgement.
61    ///
62    /// Must be called immediately after deserialization, or otherwise
63    /// any operations with the deserialized acknowledgement will panic.
64    #[tracing::instrument(level = "debug", skip(self, sender_node_key))]
65    pub fn validate(&mut self, sender_node_key: &OffchainPublicKey) -> bool {
66        self.validated = self
67            .ack_signature
68            .verify_message(self.ack_key_share.as_ref(), sender_node_key);
69
70        self.validated
71    }
72
73    /// Obtains the acknowledged challenge out of this acknowledgment.
74    pub fn ack_challenge(&self) -> HalfKeyChallenge {
75        assert!(self.validated, "acknowledgement not validated");
76        self.ack_key_share.to_challenge()
77    }
78}

91/// Bloom filter for packet tags to detect packet replays.
92///
93/// In addition, this structure also holds the number of items in the filter
94/// to determine if the filter needs to be refreshed. Once this happens, packet replays
95/// of past packets might be possible.
96#[derive(Debug, Clone, Serialize, Deserialize)]
97pub struct TagBloomFilter {
98    bloom: Bloom<PacketTag>,
99    count: usize,
100    capacity: usize,
101}
102
103impl TagBloomFilter {
104    // Allowed false positive rate. This amounts to 0.001% chance
105    const FALSE_POSITIVE_RATE: f64 = 0.00001_f64;
106
107    // The default maximum number of packet tags this Bloom filter can hold.
108    // After these many packets, the Bloom filter resets and packet replays are possible.
109    const DEFAULT_MAX_ITEMS: usize = 10_000_000;
110
111    /// Returns the current number of items in this Bloom filter.
112    pub fn count(&self) -> usize {
113        self.count
114    }
115
116    pub fn capacity(&self) -> usize {
117        self.capacity
118    }
119
120    /// Puts a packet tag into the Bloom filter
121    pub fn set(&mut self, tag: &PacketTag) {
122        if self.count == self.capacity {
123            warn!("maximum number of items in the Bloom filter reached!");
124            self.bloom.clear();
125            self.count = 0;
126        }
127
128        self.bloom.set(tag);
129        self.count += 1;
130    }
131
132    /// Check if the packet tag is in the Bloom filter.
133    /// False positives are possible.
134    pub fn check(&self, tag: &PacketTag) -> bool {
135        self.bloom.check(tag)
136    }
137
138    /// Checks and sets a packet tag (if not present) in a single operation.
139    pub fn check_and_set(&mut self, tag: &PacketTag) -> bool {
140        // If we're at full capacity, we do only "check" and conditionally reset with the new entry
141        if self.count == self.capacity {
142            let is_present = self.bloom.check(tag);
143            if !is_present {
144                // There cannot be false negatives, so we can reset the filter
145                warn!("maximum number of items in the Bloom filter reached!");
146                self.bloom.clear();
147                self.bloom.set(tag);
148                self.count = 1;
149            }
150            is_present
151        } else {
152            // If not at full capacity, we can do check_and_set
153            let was_present = self.bloom.check_and_set(tag);
154            if !was_present {
155                self.count += 1;
156            }
157            was_present
158        }
159    }
160
161    /// Deserializes the filter from the given byte array.
162    pub fn from_bytes(data: &[u8]) -> Result<Self> {
163        bincode::deserialize(data).map_err(|e| CoreTypesError::ParseError(e.to_string()))
164    }
165
166    /// Serializes the filter to the given byte array.
167    pub fn to_bytes(&self) -> Box<[u8]> {
168        bincode::serialize(&self)
169            .expect("serialization must not fail")
170            .into_boxed_slice()
171    }
172
173    fn with_capacity(size: usize) -> Self {
174        Self {
175            bloom: Bloom::new_for_fp_rate_with_seed(size, Self::FALSE_POSITIVE_RATE, &random_bytes()),
176            count: 0,
177            capacity: size,
178        }
179    }
180}
181
182impl Default for TagBloomFilter {
183    fn default() -> Self {
184        Self::with_capacity(Self::DEFAULT_MAX_ITEMS)
185    }
186}

188/// Represents the received decrypted packet carrying the application-layer data.
189#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
190pub struct ApplicationData {
191    // TODO: 3.0: Remove the Option and replace with the Tag.
192    pub application_tag: Option<Tag>,
193    #[serde(with = "serde_bytes")]
194    pub plain_text: Box<[u8]>,
195}
196
197impl ApplicationData {
198    pub fn new(application_tag: Option<Tag>, plain_text: &[u8]) -> Result<Self> {
199        if plain_text.len() <= PAYLOAD_SIZE - Self::SIZE {
200            Ok(Self {
201                application_tag,
202                plain_text: plain_text.into(),
203            })
204        } else {
205            Err(PayloadSizeExceeded)
206        }
207    }
208
209    pub fn new_from_owned(application_tag: Option<Tag>, plain_text: Box<[u8]>) -> Result<Self> {
210        if plain_text.len() <= PAYLOAD_SIZE - Self::SIZE {
211            Ok(Self {
212                application_tag,
213                plain_text,
214            })
215        } else {
216            Err(PayloadSizeExceeded)
217        }
218    }
219}
220
221impl Display for ApplicationData {
222    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
223        write!(
224            f,
225            "({}): {}",
226            self.application_tag.unwrap_or(DEFAULT_APPLICATION_TAG),
227            hex::encode(&self.plain_text)
228        )
229    }
230}
231
232impl ApplicationData {
233    const SIZE: usize = 2; // minimum size
234
235    pub fn from_bytes(data: &[u8]) -> hopr_primitive_types::errors::Result<Self> {
236        if data.len() <= PAYLOAD_SIZE && data.len() >= Self::SIZE {
237            let mut tag = [0u8; 2];
238            tag.copy_from_slice(&data[0..2]);
239            let tag = u16::from_be_bytes(tag);
240            Ok(Self {
241                application_tag: if tag != DEFAULT_APPLICATION_TAG {
242                    Some(tag)
243                } else {
244                    None
245                },
246                plain_text: (&data[2..]).into(),
247            })
248        } else {
249            Err(GeneralError::ParseError("ApplicationData".into()))
250        }
251    }
252
253    pub fn to_bytes(&self) -> Box<[u8]> {
254        let mut buf = Vec::with_capacity(Self::SIZE + self.plain_text.len());
255        let tag = self.application_tag.unwrap_or(DEFAULT_APPLICATION_TAG);
256        buf.extend_from_slice(&tag.to_be_bytes());
257        buf.extend_from_slice(&self.plain_text);
258        buf.into_boxed_slice()
259    }
260}

309    fn test_packet_tag_bloom_filter() -> anyhow::Result<()> {
310        let mut filter1 = TagBloomFilter::default();
311
312        let items = (0..10_000)
313            .map(|i| {
314                let mut ret = random_bytes::<{ hopr_crypto_types::types::PACKET_TAG_LENGTH }>();
315                ret[i % hopr_crypto_types::types::PACKET_TAG_LENGTH] = 0xaa; // ensure it is not completely just zeroes
316                ret
317            })
318            .collect::<Vec<_>>();
319
320        // Insert items into the BF
321        items.iter().for_each(|item| filter1.set(item));
322
323        assert_eq!(items.len(), filter1.count(), "invalid number of items in bf");
324
325        //let len = filter1.to_bytes().len();
326
327        // Count the number of items in the BF (incl. false positives)
328        let match_count_1 = items.iter().filter(|item| filter1.check(item)).count();
329
330        let filter2 = TagBloomFilter::from_bytes(&filter1.to_bytes())?;
331
332        // Count the number of items in the BF (incl. false positives)
333        let match_count_2 = items.iter().filter(|item| filter2.check(item)).count();
334
335        assert_eq!(
336            match_count_1, match_count_2,
337            "the number of false positives must be equal"
338        );
339        assert_eq!(filter1.count(), filter2.count(), "the number of items must be equal");
340
341        // All zeroes must be present in neither BF; we ensured we never insert a zero tag
342        assert!(!filter1.check(&ZEROS_TAG), "bf 1 must not contain zero tag");
343        assert!(!filter2.check(&ZEROS_TAG), "bf 2 must not contain zero tag");
344
345        Ok(())

364    #[test]
365    fn tag_bloom_filter_wrap_around() {
366        let mut filter = TagBloomFilter::with_capacity(1000);
367        for _ in 1..filter.capacity() {
368            let mut tag: PacketTag = hopr_crypto_random::random_bytes();
369            tag[0] = 0xaa; // ensure it's not all zeroes
370            assert!(!filter.check_and_set(&tag));
371        }
372        // Not yet at full capacity
373        assert_eq!(filter.capacity() - 1, filter.count());
374
375        // This entry is not there yet
376        assert!(!filter.check_and_set(&ZEROS_TAG));
377
378        // Now the filter is at capacity and contains the previously inserted entry
379        assert_eq!(filter.capacity(), filter.count());
380        assert!(filter.check(&ZEROS_TAG));
381
382        // This will not clear out the filter, since the entry is there
383        assert!(filter.check_and_set(&ZEROS_TAG));
384        assert_eq!(filter.capacity(), filter.count());
385
386        // This will clear out the filter, since this other entry is definitely not there
387        assert!(!filter.check_and_set(&ONES_TAG));
388        assert_eq!(1, filter.count());
389        assert!(filter.check(&ONES_TAG));
390    }
