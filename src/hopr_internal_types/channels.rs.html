7/// Describes status of a channel
8#[derive(Copy, Clone, Debug, smart_default::SmartDefault, Serialize, Deserialize, strum::Display)]
9#[strum(serialize_all = "PascalCase")]
10pub enum ChannelStatus {
11    /// The channel is closed.
12    #[default]
13    Closed,
14    /// The channel is opened.
15    Open,
16    /// The channel is pending to be closed.
17    /// The timestamp marks the *earliest* possible time when the channel can transition into the `Closed` state.
18    #[strum(serialize = "PendingToClose")]
19    PendingToClose(SystemTime),
20}
21
22// Cannot use #[repr(u8)] due to PendingToClose
23impl From<ChannelStatus> for i8 {
24    fn from(value: ChannelStatus) -> Self {
25        match value {
26            ChannelStatus::Closed => 0,
27            ChannelStatus::Open => 1,
28            ChannelStatus::PendingToClose(_) => 2,
29        }
30    }
31}
32
33// Manual implementation of PartialEq, because we need only precision up to seconds in PendingToClose
34impl PartialEq for ChannelStatus {
35    fn eq(&self, other: &Self) -> bool {
36        // Use pattern matching to avoid recursion
37        match (self, other) {
38            (Self::Open, Self::Open) => true,
39            (Self::Closed, Self::Closed) => true,
40            (Self::PendingToClose(ct_1), Self::PendingToClose(ct_2)) => {
41                let diff = ct_1.max(ct_2).saturating_sub(*ct_1.min(ct_2));
42                diff.as_secs() == 0
43            }
44            _ => false,
45        }
46    }
47}
48impl Eq for ChannelStatus {}

50/// Describes a direction of node's own channel.
51/// The direction of a channel that is not own is undefined.
52#[repr(u8)]
53#[derive(Clone, Copy, Debug, PartialEq, Eq, strum::Display, strum::EnumString)]
54#[strum(serialize_all = "lowercase")]
55pub enum ChannelDirection {
56    /// The other party is initiator of the channel.
57    Incoming = 0,
58    /// Our own node is the initiator of the channel.
59    Outgoing = 1,
60}

62/// Overall description of a channel
63#[derive(Copy, Clone, Debug, PartialEq, Serialize, Deserialize)]
64pub struct ChannelEntry {
65    pub source: Address,
66    pub destination: Address,
67    pub balance: Balance,
68    pub ticket_index: U256,
69    pub status: ChannelStatus,
70    pub channel_epoch: U256,
71    id: Hash,
72}
73
74impl ChannelEntry {
75    pub fn new(
76        source: Address,
77        destination: Address,
78        balance: Balance,
79        ticket_index: U256,
80        status: ChannelStatus,
81        channel_epoch: U256,
82    ) -> Self {
83        assert_eq!(BalanceType::HOPR, balance.balance_type(), "invalid balance currency");
84        ChannelEntry {
85            source,
86            destination,
87            balance,
88            ticket_index,
89            status,
90            channel_epoch,
91            id: generate_channel_id(&source, &destination),
92        }
93    }
94
95    /// Generates the channel ID using the source and destination address
96    pub fn get_id(&self) -> Hash {
97        self.id
98    }
99
100    /// Checks if the closure time of this channel has passed.
101    /// Also returns `false` if the channel closure has not been initiated (it is in `Open` state).
102    /// Returns also `true`, if the channel is in `Closed` state.
103    pub fn closure_time_passed(&self, current_time: SystemTime) -> bool {
104        match self.status {
105            ChannelStatus::Open => false,
106            ChannelStatus::PendingToClose(closure_time) => closure_time <= current_time,
107            ChannelStatus::Closed => true,
108        }
109    }
110
111    /// Calculates the remaining channel closure grace period.
112    /// Returns `None` if the channel closure has not been initiated yet (channel is in `Open` state).
113    pub fn remaining_closure_time(&self, current_time: SystemTime) -> Option<Duration> {
114        match self.status {
115            ChannelStatus::Open => None,
116            ChannelStatus::PendingToClose(closure_time) => Some(closure_time.saturating_sub(current_time)),
117            ChannelStatus::Closed => Some(Duration::ZERO),
118        }
119    }
120
121    /// Returns the earliest time the channel can transition from `PendingToClose` into `Closed`.
122    /// If the channel is not in `PendingToClose` state, returns `None`.
123    pub fn closure_time_at(&self) -> Option<SystemTime> {
124        match self.status {
125            ChannelStatus::PendingToClose(ct) => Some(ct),
126            _ => None,
127        }
128    }
129
130    /// Determines the channel direction given the self address.
131    /// Returns `None` if neither source nor destination are equal to `me`.
132    pub fn direction(&self, me: &Address) -> Option<ChannelDirection> {
133        if self.source.eq(me) {
134            Some(ChannelDirection::Outgoing)
135        } else if self.destination.eq(me) {
136            Some(ChannelDirection::Incoming)
137        } else {
138            None
139        }
140    }
141
142    /// Determines the channel's direction and counterparty relative to `me`.
143    /// Returns `None` if neither source nor destination are equal to `me`.
144    pub fn orientation(&self, me: &Address) -> Option<(ChannelDirection, Address)> {
145        if self.source.eq(me) {
146            Some((ChannelDirection::Outgoing, self.destination))
147        } else if self.destination.eq(me) {
148            Some((ChannelDirection::Incoming, self.source))
149        } else {
150            None
151        }
152    }
153}

155impl Display for ChannelEntry {
156    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
157        write!(f, "{} channel {}", self.status, self.get_id(),)
158    }
159}

161/// Generates channel ID hash from `source` and `destination` addresses.
162pub fn generate_channel_id(source: &Address, destination: &Address) -> Hash {
163    Hash::create(&[source.as_ref(), destination.as_ref()])
164}

166/// Enumerates possible changes on a channel entry update
167#[derive(Clone, Copy, Debug)]
168pub enum ChannelChange {
169    /// Channel status has changed
170    Status { left: ChannelStatus, right: ChannelStatus },
171
172    /// Channel balance has changed
173    CurrentBalance { left: Balance, right: Balance },
174
175    /// Channel epoch has changed
176    Epoch { left: u32, right: u32 },
177
178    /// Ticket index has changed
179    TicketIndex { left: u64, right: u64 },
180}
181
182impl Display for ChannelChange {
183    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
184        match self {
185            ChannelChange::Status { left, right } => {
186                write!(f, "Status: {left} -> {right}")
187            }
188
189            ChannelChange::CurrentBalance { left, right } => {
190                write!(f, "Balance: {left} -> {right}")
191            }
192
193            ChannelChange::Epoch { left, right } => {
194                write!(f, "Epoch: {left} -> {right}")
195            }
196
197            ChannelChange::TicketIndex { left, right } => {
198                write!(f, "TicketIndex: {left} -> {right}")
199            }
200        }
201    }
202}
203
204impl ChannelChange {
205    /// Compares the two given channels and returns a vector of `ChannelChange`s
206    /// Both channels must have the same ID (source,destination and direction) to be comparable using this function.
207    /// The function panics if `left` and `right` do not have equal ids.
208    /// Note that only some fields are tracked for changes, and therefore an empty vector returned
209    /// does not imply the two `ChannelEntry` instances are equal.
210    pub fn diff_channels(left: &ChannelEntry, right: &ChannelEntry) -> Vec<Self> {
211        assert_eq!(left.id, right.id, "must have equal ids"); // misuse
212        let mut ret = Vec::with_capacity(4);
213        if left.status != right.status {
214            ret.push(ChannelChange::Status {
215                left: left.status,
216                right: right.status,
217            });
218        }
219
220        if left.balance != right.balance {
221            ret.push(ChannelChange::CurrentBalance {
222                left: left.balance,
223                right: right.balance,
224            });
225        }
226
227        if left.channel_epoch != right.channel_epoch {
228            ret.push(ChannelChange::Epoch {
229                left: left.channel_epoch.as_u32(),
230                right: right.channel_epoch.as_u32(),
231            });
232        }
233
234        if left.ticket_index != right.ticket_index {
235            ret.push(ChannelChange::TicketIndex {
236                left: left.ticket_index.as_u64(),
237                right: right.ticket_index.as_u64(),
238            })
239        }
240
241        ret
242    }
243}
