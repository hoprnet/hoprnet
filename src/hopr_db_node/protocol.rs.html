42    /// Validates a ticket from a forwarded packet and replaces it with a new ticket for the next hop.
43    #[instrument(level = "trace", skip_all, err)]
44    async fn validate_and_replace_ticket<R>(
45        &self,
46        mut fwd: HoprForwardedPacket,
47        resolver: &R,
48        outgoing_ticket_win_prob: WinningProbability,
49        outgoing_ticket_price: HoprBalance,
50    ) -> Result<HoprForwardedPacket, NodeDbError>
51    where
52        R: ChainReadChannelOperations + ChainKeyOperations + ChainValues + Send + Sync,
53    {
54        let previous_hop_addr = resolver
55            .packet_key_to_chain_key(&fwd.previous_hop)
56            .await
57            .map_err(|e| NodeDbError::Other(e.into()))?
58            .ok_or_else(|| {
59                NodeDbError::LogicalError(format!(
60                    "failed to find channel key for packet key {} on previous hop",
61                    fwd.previous_hop.to_peerid_str()
62                ))
63            })?;
64
65        let next_hop_addr = resolver
66            .packet_key_to_chain_key(&fwd.outgoing.next_hop)
67            .await
68            .map_err(|e| NodeDbError::Other(e.into()))?
69            .ok_or_else(|| {
70                NodeDbError::LogicalError(format!(
71                    "failed to find channel key for packet key {} on next hop",
72                    fwd.outgoing.next_hop.to_peerid_str()
73                ))
74            })?;
75
76        let incoming_channel = resolver
77            .channel_by_parties(&previous_hop_addr, &self.me_address)
78            .await
79            .map_err(|e| NodeDbError::Other(e.into()))?
80            .ok_or_else(|| {
81                NodeDbError::LogicalError(format!(
82                    "no channel found for previous hop address '{previous_hop_addr}'"
83                ))
84            })?;
85
86        // Check: is the ticket in the packet really for the given channel?
87        if !fwd.outgoing.ticket.channel_id.eq(&incoming_channel.get_id()) {
88            return Err(NodeDbError::LogicalError("invalid ticket for channel".into()));
89        }
90
91        let domain_separator = resolver
92            .domain_separators()
93            .await
94            .map_err(|e| NodeDbError::Other(e.into()))?
95            .channel;
96
97        // The ticket price from the oracle times my node's position on the
98        // path is the acceptable minimum
99        let minimum_ticket_price = resolver
100            .minimum_ticket_price()
101            .await
102            .map_err(|e| NodeDbError::Other(e.into()))?
103            .mul(U256::from(fwd.path_pos));
104
105        #[cfg(all(feature = "prometheus", not(test)))]
106        METRIC_INCOMING_WIN_PROB.observe(fwd.outgoing.ticket.win_prob().as_f64());
107
108        let remaining_balance = incoming_channel
109            .balance
110            .sub(self.ticket_manager.unrealized_value((&incoming_channel).into()).await?);
111
112        // Here also the signature on the ticket gets validated,
113        // so afterward we are sure the source of the `channel`
114        // (which is equal to `previous_hop_addr`) has issued this
115        // ticket.
116        let start = std::time::Instant::now();
117        let win_prob = resolver
118            .minimum_incoming_ticket_win_prob()
119            .await
120            .map_err(|e| NodeDbError::Other(e.into()))?;
121        let verified_incoming_ticket = spawn_fifo_blocking(move || {
122            validate_unacknowledged_ticket(
123                fwd.outgoing.ticket,
124                &incoming_channel,
125                minimum_ticket_price,
126                win_prob,
127                remaining_balance,
128                &domain_separator,
129            )
130        })
131        .await?;
132
133        if start.elapsed().as_millis() > SLOW_OP_MS {
134            warn!("validate_unacknowledged_ticket took {} ms", start.elapsed().as_millis());
135        }
136
137        // We currently take the maximum of the win prob from the incoming ticket
138        // and the one configured on this node.
139        // Therefore, the winning probability can only increase along the path.
140        let outgoing_ticket_win_prob = outgoing_ticket_win_prob.max(&verified_incoming_ticket.win_prob());
141
142        // The ticket is now validated, let's place it into the acknowledgement waiting queue
143        self.caches
144            .unacked_tickets
145            .insert(
146                fwd.outgoing.ack_challenge,
147                PendingAcknowledgement::WaitingAsRelayer(
148                    verified_incoming_ticket.into_unacknowledged(fwd.own_key).into(),
149                ),
150            )
151            .await;
152
153        // NOTE: that the path position according to the ticket value
154        // may no longer match the path position from the packet header,
155        // because the price of the ticket may be set higher be the ticket
156        // issuer.
157
158        // Create the new ticket for the new packet
159        let ticket_builder = if fwd.path_pos > 1 {
160            // There must be a channel to the next node if it's not the final hop
161            let outgoing_channel = resolver
162                .channel_by_parties(&self.me_address, &next_hop_addr)
163                .await
164                .map_err(|e| NodeDbError::Other(e.into()))?
165                .ok_or(NodeDbError::LogicalError(format!(
166                    "channel to '{next_hop_addr}' not found",
167                )))?;
168
169            self.create_multihop_ticket(
170                outgoing_channel,
171                next_hop_addr,
172                fwd.path_pos,
173                outgoing_ticket_win_prob,
174                outgoing_ticket_price,
175            )
176            .await?
177        } else {
178            TicketBuilder::zero_hop().direction(&self.me_address, &next_hop_addr)
179        };
180
181        // Finally, replace the ticket in the outgoing packet with a new one
182        let ticket_builder = ticket_builder.eth_challenge(fwd.next_challenge);
183        let me_on_chain = self.me_onchain.clone();
184        fwd.outgoing.ticket = spawn_fifo_blocking(move || ticket_builder.build_signed(&me_on_chain, &domain_separator))
185            .await?
186            .leak();
187
188        Ok(fwd)
189    }

191    #[instrument(level = "trace", skip(self, ack, resolver), err)]
192    async fn find_ticket_to_acknowledge<R>(
193        &self,
194        ack: &VerifiedAcknowledgement,
195        resolver: &R,
196    ) -> Result<ResolvedAcknowledgement, NodeDbError>
197    where
198        R: ChainReadChannelOperations + ChainValues + Send + Sync,
199    {
200        let ack_half_key = *ack.ack_key_share();
201        let challenge = hopr_parallelize::cpu::spawn_blocking(move || ack_half_key.to_challenge()).await?;
202
203        let pending_ack = self.caches.unacked_tickets.remove(&challenge).await.ok_or_else(|| {
204            NodeDbError::AcknowledgementValidationError(format!(
205                "received unexpected acknowledgement for half key challenge {}",
206                ack.ack_key_share().to_hex()
207            ))
208        })?;
209
210        match pending_ack {
211            PendingAcknowledgement::WaitingAsSender => {
212                trace!("received acknowledgement as sender: first relayer has processed the packet");
213                Ok(ResolvedAcknowledgement::Sending(*ack))
214            }
215
216            PendingAcknowledgement::WaitingAsRelayer(unacknowledged) => {
217                let maybe_channel_with_issuer = resolver
218                    .channel_by_parties(unacknowledged.ticket.verified_issuer(), &self.me_address)
219                    .await
220                    .map_err(|e| NodeDbError::Other(e.into()))?;
221
222                // Issuer's channel must have an epoch matching with the unacknowledged ticket
223                if maybe_channel_with_issuer
224                    .is_some_and(|c| c.channel_epoch.as_u32() == unacknowledged.verified_ticket().channel_epoch)
225                {
226                    let domain_separator = resolver
227                        .domain_separators()
228                        .await
229                        .map_err(|e| NodeDbError::Other(e.into()))?
230                        .channel;
231
232                    let chain_key = self.me_onchain.clone();
233                    hopr_parallelize::cpu::spawn_blocking(move || {
234                        // This explicitly checks whether the acknowledgement
235                        // solves the challenge on the ticket. It must be done before we
236                        // check that the ticket is winning, which is a lengthy operation
237                        // and should not be done for bogus unacknowledged tickets
238                        let ack_ticket = unacknowledged.acknowledge(&ack_half_key)?;
239
240                        if ack_ticket.is_winning(&chain_key, &domain_separator) {
241                            trace!("Found a winning ticket");
242                            Ok(ResolvedAcknowledgement::RelayingWin(ack_ticket))
243                        } else {
244                            trace!("Found a losing ticket");
245                            Ok(ResolvedAcknowledgement::RelayingLoss(
246                                ack_ticket.ticket.verified_ticket().channel_id,
247                            ))
248                        }
249                    })
250                    .await
251                } else {
252                    Err(NodeDbError::LogicalError(format!(
253                        "no channel found for  address '{}' with a matching epoch",
254                        unacknowledged.ticket.verified_issuer()
255                    )))
256                }
257            }
258        }
259    }

266    #[instrument(level = "trace", skip(self, ack, resolver), err(Debug), ret)]
267    async fn handle_acknowledgement<R>(&self, ack: VerifiedAcknowledgement, resolver: &R) -> Result<(), NodeDbError>
268    where
269        R: ChainReadChannelOperations + ChainValues + Send + Sync,
270    {
271        let result = self.find_ticket_to_acknowledge(&ack, resolver).await?;
272        match &result {
273            ResolvedAcknowledgement::RelayingWin(ack_ticket) => {
274                // If the ticket was a win, store it
275                self.ticket_manager.insert_ticket(ack_ticket.clone()).await?;
276
277                #[cfg(all(feature = "prometheus", not(test)))]
278                {
279                    METRIC_RECEIVED_ACKS.increment(&["true"]);
280                    METRIC_TICKETS_COUNT.increment(&["winning"]);
281
282                    let verified_ticket = ack_ticket.ticket.verified_ticket();
283                    crate::tickets::METRIC_HOPR_TICKETS_INCOMING_STATISTICS.set(
284                        &["unredeemed"],
285                        self.ticket_manager
286                            .unrealized_value(TicketSelector::new(
287                                verified_ticket.channel_id,
288                                verified_ticket.channel_epoch,
289                            ))
290                            .await?
291                            .amount()
292                            .as_u128() as f64,
293                    );
294                    crate::tickets::METRIC_HOPR_TICKETS_INCOMING_STATISTICS.increment(&["winning_count"], 1.0f64);
295                }
296            }
297            ResolvedAcknowledgement::RelayingLoss(_channel) => {
298                #[cfg(all(feature = "prometheus", not(test)))]
299                {
300                    METRIC_RECEIVED_ACKS.increment(&["true"]);
301                    METRIC_TICKETS_COUNT.increment(&["losing"]);
302                    crate::tickets::METRIC_HOPR_TICKETS_INCOMING_STATISTICS.increment(&["losing_count"], 1.0f64);
303                }
304            }
305            ResolvedAcknowledgement::Sending(_) => {
306                #[cfg(all(feature = "prometheus", not(test)))]
307                METRIC_RECEIVED_ACKS.increment(&["true"]);
308            }
309        };
310
311        Ok(())
312    }

314    #[tracing::instrument(level = "trace", skip(self, matcher), err)]
315    async fn find_surb(&self, matcher: SurbMatcher) -> Result<FoundSurb, NodeDbError> {
316        let pseudonym = matcher.pseudonym();
317        let surbs_for_pseudonym = self
318            .caches
319            .surbs_per_pseudonym
320            .get(&pseudonym)
321            .await
322            .ok_or(NodeDbError::NoSurbAvailable("pseudonym not found".into()))?;
323
324        match matcher {
325            SurbMatcher::Pseudonym(_) => Ok(surbs_for_pseudonym.pop_one().map(|popped_surb| FoundSurb {
326                sender_id: HoprSenderId::from_pseudonym_and_id(&pseudonym, popped_surb.id),
327                surb: popped_surb.surb,
328                remaining: popped_surb.remaining,
329            })?),
330            // The following code intentionally only checks the first SURB in the ring buffer
331            // and does not search the entire RB.
332            // This is because the exact match use-case is suited only for situations
333            // when there is a single SURB in the RB.
334            SurbMatcher::Exact(id) => Ok(surbs_for_pseudonym
335                .pop_one_if_has_id(&id.surb_id())
336                .map(|popped_surb| FoundSurb {
337                    sender_id: HoprSenderId::from_pseudonym_and_id(&pseudonym, popped_surb.id),
338                    surb: popped_surb.surb,
339                    remaining: popped_surb.remaining, // = likely 0
340                })?),
341        }
342    }

352    #[tracing::instrument(level = "trace", skip(self, data, resolver))]
353    async fn to_send_no_ack<R>(
354        &self,
355        data: Box<[u8]>,
356        destination: OffchainPublicKey,
357        resolver: &R,
358    ) -> Result<OutgoingPacket, NodeDbError>
359    where
360        R: ChainKeyOperations + ChainValues + Send + Sync,
361    {
362        let next_peer = resolver
363            .packet_key_to_chain_key(&destination)
364            .await
365            .map_err(|e| NodeDbError::Other(e.into()))?
366            .ok_or_else(|| {
367                NodeDbError::LogicalError(format!(
368                    "failed to find chain key for packet key {} on previous hop",
369                    destination.to_peerid_str()
370                ))
371            })?;
372
373        // No-ack packets are always sent as zero-hops with a random pseudonym
374        let pseudonym = HoprPseudonym::random();
375        let next_ticket = TicketBuilder::zero_hop().direction(&self.me_address, &next_peer);
376        let domain_separator = resolver
377            .domain_separators()
378            .await
379            .map_err(|e| NodeDbError::Other(e.into()))?
380            .channel;
381
382        // Construct the outgoing packet
383        let chain_key = self.me_onchain.clone();
384        let mapper = resolver.key_id_mapper_ref().clone();
385        let (packet, _) = spawn_fifo_blocking(move || {
386            HoprPacket::into_outgoing(
387                &data,
388                &pseudonym,
389                PacketRouting::NoAck::<ValidatedPath>(destination),
390                &chain_key,
391                next_ticket,
392                &mapper,
393                &domain_separator,
394                None, // NoAck messages currently do not have signals
395            )
396            .map_err(|e| {
397                NodeDbError::LogicalError(format!("failed to construct chain components for a no-ack packet: {e}"))
398            })
399        })
400        .await?;
401
402        if let Some(out) = packet.try_as_outgoing() {
403            let mut transport_payload = Vec::with_capacity(HoprPacket::SIZE);
404            transport_payload.extend_from_slice(out.packet.as_ref());
405            transport_payload.extend_from_slice(&out.ticket.into_encoded());
406
407            #[cfg(all(feature = "prometheus", not(test)))]
408            METRIC_SENT_ACKS.increment();
409
410            Ok(OutgoingPacket {
411                next_hop: out.next_hop,
412                ack_challenge: out.ack_challenge,
413                data: transport_payload.into_boxed_slice(),
414            })
415        } else {
416            Err(NodeDbError::LogicalError("must be an outgoing packet".into()))
417        }
418    }

420    #[tracing::instrument(level = "trace", skip(self, data, routing, resolver))]
421    async fn to_send<R>(
422        &self,
423        data: Box<[u8]>,
424        routing: ResolvedTransportRouting,
425        outgoing_ticket_win_prob: Option<WinningProbability>,
426        outgoing_ticket_price: Option<HoprBalance>,
427        signals: PacketSignals,
428        resolver: &R,
429    ) -> Result<OutgoingPacket, NodeDbError>
430    where
431        R: ChainReadChannelOperations + ChainKeyOperations + ChainValues + Send + Sync,
432    {
433        // Get necessary packet routing values
434        let (next_peer, num_hops, pseudonym, routing) = match routing {
435            ResolvedTransportRouting::Forward {
436                pseudonym,
437                forward_path,
438                return_paths,
439            } => (
440                forward_path[0],
441                forward_path.num_hops(),
442                pseudonym,
443                PacketRouting::ForwardPath {
444                    forward_path,
445                    return_paths,
446                },
447            ),
448            ResolvedTransportRouting::Return(sender_id, surb) => {
449                let next = resolver
450                    .key_id_mapper_ref()
451                    .map_id_to_public(&surb.first_relayer)
452                    .ok_or(NodeDbError::LogicalError(format!(
453                        "failed to find public key for key id {}",
454                        surb.first_relayer
455                    )))?;
456
457                (
458                    next,
459                    surb.additional_data_receiver.proof_of_relay_values().chain_length() as usize,
460                    sender_id.pseudonym(),
461                    PacketRouting::Surb(sender_id.surb_id(), surb),
462                )
463            }
464        };
465
466        let next_peer = resolver
467            .packet_key_to_chain_key(&next_peer)
468            .await
469            .map_err(|e| NodeDbError::Other(e.into()))?
470            .ok_or_else(|| {
471                NodeDbError::LogicalError(format!(
472                    "failed to find chain key for packet key {} on previous hop",
473                    next_peer.to_peerid_str()
474                ))
475            })?;
476
477        // Decide whether to create a multi-hop or a zero-hop ticket
478        let next_ticket = if num_hops > 1 {
479            let channel = resolver
480                .channel_by_parties(&self.me_address, &next_peer)
481                .await
482                .map_err(|e| NodeDbError::Other(e.into()))?
483                .ok_or(NodeDbError::LogicalError(format!("channel to '{next_peer}' not found")))?;
484
485            let (outgoing_ticket_win_prob, outgoing_ticket_price) =
486                determine_network_config(outgoing_ticket_win_prob, outgoing_ticket_price, resolver).await?;
487
488            self.create_multihop_ticket(
489                channel,
490                next_peer,
491                num_hops as u8,
492                outgoing_ticket_win_prob,
493                outgoing_ticket_price,
494            )
495            .await?
496        } else {
497            TicketBuilder::zero_hop().direction(&self.me_address, &next_peer)
498        };
499
500        let domain_separator = resolver
501            .domain_separators()
502            .await
503            .map_err(|e| NodeDbError::Other(e.into()))?
504            .channel;
505
506        // Construct the outgoing packet
507        let chain_key = self.me_onchain.clone();
508        let mapper = resolver.key_id_mapper_ref().clone();
509        let (packet, openers) = spawn_fifo_blocking(move || {
510            HoprPacket::into_outgoing(
511                &data,
512                &pseudonym,
513                routing,
514                &chain_key,
515                next_ticket,
516                &mapper,
517                &domain_separator,
518                signals,
519            )
520            .map_err(|e| NodeDbError::LogicalError(format!("failed to construct chain components for a packet: {e}")))
521        })
522        .await?;
523
524        // Store the reply openers under the given SenderId
525        // This is a no-op for reply packets
526        openers.into_iter().for_each(|(surb_id, opener)| {
527            self.caches
528                .insert_pseudonym_opener(HoprSenderId::from_pseudonym_and_id(&pseudonym, surb_id), opener)
529        });
530
531        if let Some(out) = packet.try_as_outgoing() {
532            self.caches
533                .unacked_tickets
534                .insert(out.ack_challenge, PendingAcknowledgement::WaitingAsSender)
535                .await;
536
537            let mut transport_payload = Vec::with_capacity(HoprPacket::SIZE);
538            transport_payload.extend_from_slice(out.packet.as_ref());
539            transport_payload.extend_from_slice(&out.ticket.into_encoded());
540
541            Ok(OutgoingPacket {
542                next_hop: out.next_hop,
543                ack_challenge: out.ack_challenge,
544                data: transport_payload.into_boxed_slice(),
545            })
546        } else {
547            Err(NodeDbError::LogicalError("must be an outgoing packet".into()))
548        }
549    }

551    #[tracing::instrument(level = "trace", skip_all, fields(sender = %sender), err)]
552    async fn from_recv<R>(
553        &self,
554        data: Box<[u8]>,
555        pkt_keypair: &OffchainKeypair,
556        sender: OffchainPublicKey,
557        outgoing_ticket_win_prob: Option<WinningProbability>,
558        outgoing_ticket_price: Option<HoprBalance>,
559        resolver: &R,
560    ) -> Result<IncomingPacket, IncomingPacketError<NodeDbError>>
561    where
562        R: ChainReadChannelOperations + ChainKeyOperations + ChainValues + Send + Sync,
563    {
564        let offchain_keypair = pkt_keypair.clone();
565        let caches = self.caches.clone();
566        let start = std::time::Instant::now();
567        let mapper = resolver.key_id_mapper_ref().clone();
568        let packet = spawn_fifo_blocking(move || {
569            HoprPacket::from_incoming(&data, &offchain_keypair, sender, &mapper, |p| {
570                caches.extract_pseudonym_opener(p)
571            })
572            .map_err(|error| match &error {
573                PacketError::PacketDecodingError(_) | PacketError::SphinxError(_) => {
574                    // Such packets should not be acknowledged as this might indicate an adversarial behavior
575                    // or incorrect protocol format
576                    IncomingPacketError::Undecodable(NodeDbError::PacketError(error))
577                }
578                _ => IncomingPacketError::ProcessingError(NodeDbError::PacketError(error)),
579            })
580        })
581        .await?;
582        if start.elapsed().as_millis() > SLOW_OP_MS {
583            warn!(
584                peer = sender.to_peerid_str(),
585                "from_incoming took {} ms",
586                start.elapsed().as_millis()
587            );
588        }
589
590        match packet {
591            HoprPacket::Final(incoming) => {
592                // Extract additional information from the packet that will be passed upwards
593                let info = AuxiliaryPacketInfo {
594                    packet_signals: incoming.signals,
595                    num_surbs: incoming.surbs.len(),
596                };
597
598                // Store all incoming SURBs if any
599                if !incoming.surbs.is_empty() {
600                    self.caches
601                        .surbs_per_pseudonym
602                        .entry_by_ref(&incoming.sender)
603                        .or_insert_with(futures::future::lazy(|_| {
604                            SurbRingBuffer::new(self.cfg.surb_ring_buffer_size)
605                        }))
606                        .await
607                        .value()
608                        .push(incoming.surbs)
609                        .map_err(IncomingPacketError::ProcessingError)?;
610
611                    trace!(pseudonym = %incoming.sender, num_surbs = info.num_surbs, "stored incoming surbs for pseudonym");
612                }
613
614                Ok(match incoming.ack_key {
615                    None => {
616                        // The contained payload represents an Acknowledgement
617                        IncomingPacket::Acknowledgement {
618                            packet_tag: incoming.packet_tag,
619                            previous_hop: incoming.previous_hop,
620                            ack: incoming
621                                .plain_text
622                                .as_ref()
623                                .try_into()
624                                .map_err(|e| IncomingPacketError::ProcessingError(NodeDbError::TypeError(e)))?,
625                        }
626                    }
627                    Some(ack_key) => IncomingPacket::Final {
628                        packet_tag: incoming.packet_tag,
629                        previous_hop: incoming.previous_hop,
630                        sender: incoming.sender,
631                        plain_text: incoming.plain_text,
632                        ack_key,
633                        info,
634                    },
635                })
636            }
637            HoprPacket::Forwarded(fwd) => {
638                let (outgoing_ticket_win_prob, outgoing_ticket_price) =
639                    determine_network_config(outgoing_ticket_win_prob, outgoing_ticket_price, resolver)
640                        .await
641                        .map_err(IncomingPacketError::ProcessingError)?;
642
643                let start = std::time::Instant::now();
644                let validation_res = self
645                    .validate_and_replace_ticket(*fwd, resolver, outgoing_ticket_win_prob, outgoing_ticket_price)
646                    .await;
647                if start.elapsed().as_millis() > SLOW_OP_MS {
648                    warn!(
649                        peer = sender.to_peerid_str(),
650                        "validate_and_replace_ticket took {} ms",
651                        start.elapsed().as_millis()
652                    );
653                }
654                match validation_res {
655                    Ok(fwd) => {
656                        let mut payload = Vec::with_capacity(HoprPacket::SIZE);
657                        payload.extend_from_slice(fwd.outgoing.packet.as_ref());
658                        payload.extend_from_slice(&fwd.outgoing.ticket.into_encoded());
659
660                        Ok(IncomingPacket::Forwarded {
661                            packet_tag: fwd.packet_tag,
662                            previous_hop: fwd.previous_hop,
663                            next_hop: fwd.outgoing.next_hop,
664                            data: payload.into_boxed_slice(),
665                            ack_key: fwd.ack_key,
666                        })
667                    }
668                    Err(NodeDbError::TicketValidationError(boxed_error)) => {
669                        let (rejected_ticket, error) = *boxed_error;
670                        let rejected_value = rejected_ticket.amount;
671                        warn!(%rejected_ticket, %rejected_value, %error, "failure to validate during forwarding");
672
673                        self.mark_unsaved_ticket_rejected(&rejected_ticket)
674                            .await
675                            .map_err(|e| {
676                                NodeDbError::TicketValidationError(Box::new((
677                                    rejected_ticket.clone(),
678                                    format!("during validation error '{error}' update another error occurred: {e}"),
679                                )))
680                            })
681                            .map_err(IncomingPacketError::ProcessingError)?;
682
683                        #[cfg(all(feature = "prometheus", not(test)))]
684                        METRIC_TICKETS_COUNT.increment(&["rejected"]);
685
686                        Err(IncomingPacketError::ProcessingError(
687                            NodeDbError::TicketValidationError(Box::new((rejected_ticket, error))),
688                        ))
689                    }
690                    Err(e) => Err(IncomingPacketError::ProcessingError(e)),
691                }
692            }
693            HoprPacket::Outgoing(_) => Err(IncomingPacketError::ProcessingError(NodeDbError::LogicalError(
694                "cannot receive an outgoing packet".into(),
695            ))),
696        }
697    }
698}
699
700async fn determine_network_config<R>(
701    configure_outgoing_wp: Option<WinningProbability>,
702    configured_outgoing_price: Option<HoprBalance>,
703    resolver: &R,
704) -> Result<(WinningProbability, HoprBalance), NodeDbError>
705where
706    R: ChainValues,
707{
708    // This operation hits the cache unless the new value is fetched for the first time
709    // NOTE: as opposed to the winning probability, the ticket price does not have
710    // a reasonable default, and therefore the operation fails
711    let network_ticket_price = resolver
712        .minimum_ticket_price()
713        .await
714        .map_err(|e| NodeDbError::LogicalError(format!("failed to determine current network ticket price: {e}")))?;
715    let outgoing_ticket_price = configured_outgoing_price.unwrap_or(network_ticket_price);
716
717    // This operation hits the cache unless the new value is fetched for the first time
718    let network_win_prob = resolver
719        .minimum_incoming_ticket_win_prob()
720        .await
721        .inspect_err(|error| tracing::error!(%error, "failed to determine current network winning probability"))
722        .ok();
723
724    // If no explicit winning probability is configured, use the network value
725    // or 1 if the network value was not determined.
726    // This code does not take the max from those, as it is the upper layer's responsibility
727    // to ensure the configured value is not smaller than the network value.
728    let outgoing_ticket_win_prob = configure_outgoing_wp.or(network_win_prob).unwrap_or_default(); // Absolute default WinningProbability is 1.0
729
730    Ok((outgoing_ticket_win_prob, outgoing_ticket_price))
731}
732
733impl HoprNodeDb {
734    #[tracing::instrument(level = "trace", skip(self, channel))]
735    async fn create_multihop_ticket(
736        &self,
737        channel: ChannelEntry,
738        destination: Address,
739        current_path_pos: u8,
740        winning_prob: WinningProbability,
741        ticket_price: HoprBalance,
742    ) -> Result<TicketBuilder, NodeDbError> {
743        // The next ticket is worth: price * remaining hop count / winning probability
744        let amount = HoprBalance::from(
745            ticket_price
746                .amount()
747                .mul(U256::from(current_path_pos - 1))
748                .div_f64(winning_prob.into())
749                .map_err(|e| {
750                    NodeDbError::LogicalError(format!(
751                        "winning probability outside of the allowed interval (0.0, 1.0]: {e}"
752                    ))
753                })?,
754        );
755
756        if channel.balance.lt(&amount) {
757            return Err(NodeDbError::LogicalError(format!(
758                "out of funds: {} with counterparty {destination} has balance {} < {amount}",
759                channel.get_id(),
760                channel.balance
761            )));
762        }
763
764        let ticket_builder = TicketBuilder::default()
765            .direction(&self.me_address, &destination)
766            .balance(amount)
767            .index(self.increment_outgoing_ticket_index(channel.get_id()).await?)
768            .win_prob(winning_prob)
769            .channel_epoch(channel.channel_epoch.as_u32());
770
771        Ok(ticket_builder)
772    }
773}
