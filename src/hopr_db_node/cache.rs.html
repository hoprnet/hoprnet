18/// Represents a single SURB along with its ID popped from the [`SurbRingBuffer`].
19#[derive(Debug, Clone)]
20pub struct PoppedSurb<S> {
21    /// Complete SURB sender ID.
22    pub id: HoprSurbId,
23    /// The popped SURB.
24    pub surb: S,
25    /// Number of SURBs left in the RB after the pop.
26    pub remaining: usize,
27}
28
29/// Ring buffer containing SURBs along with their IDs.
30///
31/// All these SURBs usually belong to the same pseudonym and are therefore identified
32/// only by the [`HoprSurbId`].
33#[derive(Clone, Debug)]
34pub(crate) struct SurbRingBuffer<S>(Arc<parking_lot::Mutex<AllocRingBuffer<(HoprSurbId, S)>>>);
35
36impl<S> Default for SurbRingBuffer<S> {
37    fn default() -> Self {
38        // With the current packet size, this is almost 10 MB of data budget in SURBs
39        Self::new(10_000)
40    }
41}
42
43impl<S> SurbRingBuffer<S> {
44    pub fn new(capacity: usize) -> Self {
45        Self(Arc::new(parking_lot::Mutex::new(AllocRingBuffer::new(capacity))))
46    }
47
48    /// Push all SURBs with their IDs into the RB.
49    ///
50    /// Returns the total number of elements in the RB after the push.
51    pub fn push<I: IntoIterator<Item = (HoprSurbId, S)>>(&self, surbs: I) -> Result<usize, NodeDbError> {
52        let mut rb = self.0.lock();
53
54        rb.extend(surbs);
55        Ok(rb.len())
56    }
57
58    /// Pop the latest SURB and its IDs from the RB.
59    pub fn pop_one(&self) -> Result<PoppedSurb<S>, NodeDbError> {
60        let mut rb = self.0.lock();
61
62        let (id, surb) = rb
63            .dequeue()
64            .ok_or(NodeDbError::NoSurbAvailable("no more surbs".into()))?;
65        Ok(PoppedSurb {
66            id,
67            surb,
68            remaining: rb.len(),
69        })
70    }
71
72    /// Check if the next SURB has the given ID and pop it from the RB.
73    pub fn pop_one_if_has_id(&self, id: &HoprSurbId) -> Result<PoppedSurb<S>, NodeDbError> {
74        let mut rb = self.0.lock();
75
76        if rb.peek().is_some_and(|(surb_id, _)| surb_id == id) {
77            let (id, surb) = rb
78                .dequeue()
79                .ok_or(NodeDbError::NoSurbAvailable("no more surbs".into()))?;
80            Ok(PoppedSurb {
81                id,
82                surb,
83                remaining: rb.len(),
84            })
85        } else {
86            Err(NodeDbError::NoSurbAvailable("surb does not match the given id".into()))
87        }
88    }
89}

99/// Contains all caches used by the [crate::db::HoprDb].
100#[derive(Debug)]
101pub struct NodeDbCaches {
102    pub(crate) unacked_tickets: Cache<HalfKeyChallenge, PendingAcknowledgement>,
103    pub(crate) ticket_index: Cache<Hash, Arc<AtomicU64>>,
104    // key is (channel_id, channel_epoch) to ensure calculation of unrealized value does not
105    // include tickets from other epochs
106    pub(crate) unrealized_value: Cache<(Hash, U256), HoprBalance>,
107    pseudonym_openers: moka::sync::Cache<HoprPseudonym, moka::sync::Cache<HoprSurbId, ReplyOpener>>,
108    pub(crate) surbs_per_pseudonym: Cache<HoprPseudonym, SurbRingBuffer<HoprSurb>>,
109}
110
111impl Default for NodeDbCaches {
112    fn default() -> Self {
113        Self {
114            unacked_tickets: Cache::builder()
115                .time_to_live(Duration::from_secs(30))
116                .max_capacity(1_000_000_000)
117                .build(),
118            ticket_index: Cache::builder().expire_after(ExpiryNever).max_capacity(10_000).build(),
119            unrealized_value: Cache::builder().expire_after(ExpiryNever).max_capacity(10_000).build(),
120            // Reply openers are indexed by entire Sender IDs (Pseudonym + SURB ID)
121            // in a cascade fashion, allowing the entire batches (by Pseudonym) to be evicted
122            // if not used.
123            pseudonym_openers: moka::sync::Cache::builder()
124                .time_to_idle(Duration::from_secs(600))
125                .eviction_policy(moka::policy::EvictionPolicy::lru())
126                .eviction_listener(|sender_id, _reply_opener, cause| {
127                    tracing::warn!(?sender_id, ?cause, "evicting reply opener for pseudonym");
128                })
129                .max_capacity(10_000)
130                .build(),
131            // SURBs are indexed only by Pseudonyms, which have longer lifetimes.
132            // For each Pseudonym, there's an RB of SURBs and their IDs.
133            surbs_per_pseudonym: Cache::builder()
134                .time_to_idle(Duration::from_secs(600))
135                .eviction_policy(moka::policy::EvictionPolicy::lru())
136                .eviction_listener(|pseudonym, _reply_opener, cause| {
137                    tracing::warn!(%pseudonym, ?cause, "evicting surb for pseudonym");
138                })
139                .max_capacity(10_000)
140                .build(),
141        }
142    }
143}

145impl NodeDbCaches {
146    pub(crate) fn insert_pseudonym_opener(&self, sender_id: HoprSenderId, opener: ReplyOpener) {
147        self.pseudonym_openers
148            .get_with(sender_id.pseudonym(), move || {
149                moka::sync::Cache::builder()
150                    .time_to_live(Duration::from_secs(3600))
151                    .eviction_listener(move |id: Arc<HoprSurbId>, _, cause| {
152                        if cause != RemovalCause::Explicit {
153                            tracing::warn!(pseudonym = %sender_id.pseudonym(), surb_id = hex::encode(id.as_slice()), ?cause, "evicting reply opener for sender id");
154                        }
155                    })
156                    .max_capacity(100_000)
157                    .build()
158            })
159            .insert(sender_id.surb_id(), opener);
160    }
161
162    pub(crate) fn extract_pseudonym_opener(&self, sender_id: &HoprSenderId) -> Option<ReplyOpener> {
163        self.pseudonym_openers
164            .get(&sender_id.pseudonym())
165            .and_then(|cache| cache.remove(&sender_id.surb_id()))
166    }
167
168    // For future use by the SessionManager
169    #[allow(dead_code)]
170    pub(crate) fn invalidate_pseudonym_openers(&self, pseudonym: &HoprPseudonym) {
171        self.pseudonym_openers.invalidate(pseudonym);
172    }
173
174    /// Invalidates all caches.
175    pub fn invalidate_all(&self) {
176        self.unacked_tickets.invalidate_all();
177        self.unrealized_value.invalidate_all();
178        self.surbs_per_pseudonym.invalidate_all();
179        self.pseudonym_openers.invalidate_all();
180    }
181}

187    #[test]
188    fn surb_ring_buffer_must_drop_items_when_capacity_is_reached() -> anyhow::Result<()> {
189        let rb = SurbRingBuffer::new(3);
190        rb.push([([1u8; 8], 0)])?;
191        rb.push([([2u8; 8], 0)])?;
192        rb.push([([3u8; 8], 0)])?;
193        rb.push([([4u8; 8], 0)])?;
194
195        let popped = rb.pop_one()?;
196        assert_eq!([2u8; 8], popped.id);
197        assert_eq!(2, popped.remaining);
198
199        let popped = rb.pop_one()?;
200        assert_eq!([3u8; 8], popped.id);
201        assert_eq!(1, popped.remaining);
202
203        let popped = rb.pop_one()?;
204        assert_eq!([4u8; 8], popped.id);
205        assert_eq!(0, popped.remaining);
206
207        assert!(rb.pop_one().is_err());
208
209        Ok(())
210    }

217        assert_eq!(1, len);

224        assert_eq!(1, popped.remaining);

228        assert_eq!(0, popped.remaining);
